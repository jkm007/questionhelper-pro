# 刷题练习系统设计文档

**项目名称**：题小助 (QuestionHelper)
**版本**：V2.0
**日期**：2026-05-30
**模块**：刷题练习系统

---

## 一、功能概述

| 功能 | 说明 |
|------|------|
| 多种练习模式 | 顺序练习、随机练习、错题重练、收藏练习、模拟考试、闯关模式、计时模式 |
| 练习管理 | 暂停/继续、练习计划、每日推荐、练习提醒 |
| 社交激励 | 练习打卡、排行榜、连续打卡奖励、题目分享 |
| 智能推荐 | 题目难度自适应、练习报告、练习统计、相关题目推荐 |
| 资源访问 | 根据用户角色权限访问题目资源（公共资源/私有资源/班级资源） |
| 分类浏览 | 按分类树浏览题目、分类树展开/折叠、分类树搜索、最近浏览记录 |
| 即时评判 | 答题后即时显示对错 |
| 学习辅助 | 题目笔记、难度评价、题目质量评分、题目纠错 |
| 管理后台 | 练习数据查看、用户练习统计、热门题目、正确率分析 |

## 二、数据库设计

### 2.1 练习会话表 (practice_sessions)

```sql
CREATE TABLE `practice_sessions` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `user_id` BIGINT UNSIGNED NOT NULL,
  `session_id` VARCHAR(36) NOT NULL COMMENT '会话UUID',
  `mode` TINYINT NOT NULL DEFAULT 1 COMMENT '练习模式:1=随机练习,2=顺序练习,3=错题重练,4=收藏练习,5=模拟考试,6=闯关模式,7=计时模式',
  `category_id` BIGINT UNSIGNED DEFAULT NULL COMMENT '分类ID',
  `question_ids` JSON DEFAULT NULL COMMENT '题目ID列表(有序)',
  `current_index` INT DEFAULT 0 COMMENT '当前答题索引',
  `total_count` INT DEFAULT 0 COMMENT '总题数',
  `correct_count` INT DEFAULT 0 COMMENT '正确数',
  `accuracy` DECIMAL(5,2) DEFAULT 0 COMMENT '正确率',
  `duration` INT DEFAULT 0 COMMENT '练习时长(秒)',
  `score` DECIMAL(10,2) DEFAULT 0 COMMENT '得分(模拟考试)',
  `status` TINYINT DEFAULT 0 COMMENT '状态:0=进行中,1=已暂停,2=已完成',
  `started_at` DATETIME DEFAULT NULL COMMENT '开始时间',
  `paused_at` DATETIME DEFAULT NULL COMMENT '暂停时间',
  `completed_at` DATETIME DEFAULT NULL COMMENT '完成时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `deleted_at` DATETIME DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_session_id` (`session_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_user_mode` (`user_id`, `mode`),
  KEY `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='练习会话表';
```

## 三、缓存设计

### 3.1 Redis Key 设计

| 场景 | Key 格式 | 数据结构 | TTL | 说明 |
|------|---------|---------|-----|------|
| 暂停会话 | `qh:practice:session:{session_id}` | Hash | 7 天 | 暂停的会话状态缓存 |
| 每日练习 | `qh:practice:daily:{uid}:{date}` | String(JSON) | 48 小时 | 当日推荐题目缓存 |
| 排行榜 | `qh:practice:rank:{period}:{date}` | ZSet | 25 小时(日)/8天(周)/35天(月) | 排行榜实时排名 |
| 打卡连续天数 | `qh:practice:streak:{uid}` | String | 90 天 | 连续打卡天数缓存 |
| 用户能力 | `qh:practice:ability:{uid}` | Hash | 24 小时 | 用户能力等级缓存 |
| 练习统计 | `qh:practice:stats:{uid}` | Hash | 1 小时 | 练习统计数据缓存 |
| 模拟考试会话 | `qh:practice:mock:{session_id}` | Hash | 考试时长+10分钟 | 模拟考试进行中状态 |
| 最近浏览分类 | `qh:practice:recent_categories:{uid}` | ZSet | 30 天 | 记录最近浏览的分类，score为时间戳，最多保留50条 |
| 题目分享链接 | `qh:practice:share:{share_code}` | String(JSON) | 7 天 | 题目分享码对应的题目信息缓存 |

| 方法 | 路径 | 说明 |

| POST | `/api/v1/practice/:id/resume` | 继续练习 |

```go
type StartPracticeRequest struct {
    Mode         int8   `json:"mode" binding:"required,oneof=1 2 3 4"` // 1=随机 2=顺序 3=错题 4=收藏
    CategoryID   *uint  `json:"category_id"`
    QuestionType *int8  `json:"question_type"`
    Difficulty   *int8  `json:"difficulty"`
    Count        int    `json:"count" binding:"required,min=1,max=200"`
}

// StartPractice 开始练习
func (s *PracticeService) StartPractice(req *StartPracticeRequest, userID uint) (*dto.PracticeSessionInfo, error) {
    var questions []model.Question
    var err error

    switch req.Mode {
    case 1: // 随机练习
        questions, err = s.getRandomQuestions(req, userID)
    case 2: // 顺序练习
        questions, err = s.getSequentialQuestions(req, userID)
    case 3: // 错题重练
        questions, err = s.getWrongQuestions(req, userID)
    case 4: // 收藏练习
        questions, err = s.getFavoriteQuestions(req, userID)
    }

    if err != nil {
        return nil, err
    }
    if len(questions) == 0 {
        return nil, errors.New("没有符合条件的题目")
    }

    // 构建题目ID列表
    questionIDs := make([]uint, len(questions))
    for i, q := range questions {
        questionIDs[i] = q.ID
    }

    // 创建练习会话
    sessionUUID := uuid.New().String()
    session := &model.PracticeSession{
        UserID:      userID,
        SessionID:   sessionUUID,
        Mode:        req.Mode,
        CategoryID:  req.CategoryID,
        QuestionIDs: questionIDs,
        CurrentIndex: 0,
        TotalCount:  len(questions),
        Status:      0,
        StartedAt:   time.Now(),
    }

    if err := s.practiceRepo.CreateSession(session); err != nil {
        return nil, err
    }

    return &dto.PracticeSessionInfo{
        ID:          session.ID,
        SessionID:   sessionUUID,
        Mode:        session.Mode,
        TotalCount:  session.TotalCount,
        Status:      session.Status,
        FirstQuestion: s.buildQuestionDTO(questions[0]),
    }, nil
}
```

```go
// SubmitAnswer 提交答案并评判
func (s *PracticeService) SubmitAnswer(sessionID string, req *dto.SubmitPracticeAnswerRequest, userID uint) (*dto.PracticeRecordInfo, error) {
    // 获取会话
    session, err := s.practiceRepo.FindBySessionID(sessionID)
    if err != nil {
        return nil, errors.New("练习会话不存在")
    }
    if session.UserID != userID {
        return nil, errors.New("无权操作此练习")
    }
    if session.Status == 2 {
        return nil, errors.New("练习已结束")
    }

    // 获取题目正确答案
    question, err := s.questionRepo.FindByID(req.QuestionID)
    if err != nil {
        return nil, err
    }

    // 判断答案是否正确
    isCorrect := s.checkAnswer(question, req.Answer)

    // 保存练习记录
    record := &model.PracticeRecord{
        SessionID:  session.ID,
        QuestionID: req.QuestionID,
        Answer:     req.Answer,
        IsCorrect:  isCorrect,
        Duration:   req.Duration,
        AnsweredAt: time.Now(),
    }

    if err := s.practiceRepo.CreateRecord(record); err != nil {
        return nil, err
    }

    // 更新会话统计
    s.updateSessionStats(session.ID)

    // 答错自动加入错题本
    if !isCorrect {
        s.wrongRepo.AddOrUpdate(userID, req.QuestionID, 1, session.ID, req.Answer)
    } else {
        // 答对则检查错题本中是否需要更新
        s.wrongRepo.UpdateCorrectStreak(userID, req.QuestionID)
    }

    // 更新用户能力模型(异步)
    go s.abilityService.UpdateAbility(userID, session.CategoryID, req.QuestionID, isCorrect, req.Duration)

    // 获取正确答案和解析
    correctAnswer := s.getCorrectAnswer(question)

    return &dto.PracticeRecordInfo{
        ID:            record.ID,
        SessionID:     record.SessionID,
        QuestionID:    record.QuestionID,
        Answer:        record.Answer,
        IsCorrect:     record.IsCorrect,
        Duration:      record.Duration,
        CorrectAnswer: correctAnswer,
        Analysis:      question.Analysis,
    }, nil
}

// checkAnswer 检查答案
func (s *PracticeService) checkAnswer(question *model.Question, userAnswer string) bool {
    correctOptions, _ := s.questionRepo.GetCorrectOptions(question.ID)

    switch question.Type {
    case 1: // 单选
        return len(correctOptions) == 1 && userAnswer == correctOptions[0].Label
    case 2: // 多选
        userLabels := strings.Split(userAnswer, ",")
        correctLabels := make([]string, len(correctOptions))
        for i, opt := range correctOptions {
            correctLabels[i] = opt.Label
        }
        return s.compareLabels(userLabels, correctLabels)
    case 3: // 判断
        return userAnswer == correctOptions[0].Label
    default:
        return false
    }
}

// compareLabels 比较选项标签集合
func (s *PracticeService) compareLabels(userLabels, correctLabels []string) bool {
    if len(userLabels) != len(correctLabels) {
        return false
    }
    sort.Strings(userLabels)
    sort.Strings(correctLabels)
    for i := range userLabels {
        if userLabels[i] != correctLabels[i] {
            return false
        }
    }
    return true
}
```

```go
// service/practice/practice_service.go

// PausePractice 暂停练习
func (s *PracticeService) PausePractice(sessionID string, userID uint) error {
    session, err := s.practiceRepo.FindBySessionID(sessionID)
    if err != nil {
        return errors.New("练习会话不存在")
    }
    if session.UserID != userID {
        return errors.New("无权操作此练习")
    }
    if session.Status != 0 {
        return errors.New("只能暂停进行中的练习")
    }

    now := time.Now()
    session.Status = 1 // 已暂停
    session.PausedAt = &now
    // 累计时长
    if session.StartedAt != nil {
        session.Duration += int(now.Sub(*session.StartedAt).Seconds())
    }

    if err := s.practiceRepo.UpdateSession(session); err != nil {
        return err
    }

    // 将会话状态缓存到Redis
    s.cacheSession(session)

    return nil
}

// ResumePractice 继续练习
func (s *PracticeService) ResumePractice(sessionID string, userID uint) (*dto.PracticeSessionInfo, error) {
    session, err := s.practiceRepo.FindBySessionID(sessionID)
    if err != nil {
        // 尝试从Redis缓存恢复
        session, err = s.restoreSessionFromCache(sessionID, userID)
        if err != nil {
            return nil, errors.New("练习会话不存在或已过期")
        }
    }

    if session.UserID != userID {
        return nil, errors.New("无权操作此练习")
    }
    if session.Status != 1 {
        return nil, errors.New("只能继续已暂停的练习")
    }

    now := time.Now()
    session.Status = 0 // 恢复进行中
    session.StartedAt = &now
    session.PausedAt = nil

    if err := s.practiceRepo.UpdateSession(session); err != nil {
        return nil, err
    }

    // 清除Redis缓存
    s.redisClient.Del(context.Background(), fmt.Sprintf("qh:practice:session:%s", sessionID))

    // 获取当前题目
    questionIDs := session.QuestionIDs
    currentIndex := session.CurrentIndex
    if currentIndex >= len(questionIDs) {
        currentIndex = len(questionIDs) - 1
    }

    question, _ := s.questionRepo.FindByID(questionIDs[currentIndex])

    return &dto.PracticeSessionInfo{
        ID:           session.ID,
        SessionID:    session.SessionID,
        Mode:         session.Mode,
        TotalCount:   session.TotalCount,
        CurrentIndex: currentIndex,
        CorrectCount: session.CorrectCount,
        Accuracy:     session.Accuracy,
        Duration:     session.Duration,
        Status:       session.Status,
        CurrentQuestion: s.buildQuestionDTO(question),
    }, nil
}

// GetResumeList 获取可继续的练习列表
func (s *PracticeService) GetResumeList(userID uint) ([]dto.PracticeSessionBrief, error) {
    sessions, err := s.practiceRepo.FindByUserAndStatus(userID, 1) // status=1 已暂停
    if err != nil {
        return nil, err
    }

    var result []dto.PracticeSessionBrief
    for _, s := range sessions {
        result = append(result, dto.PracticeSessionBrief{

            Mode:        s.Mode,
            TotalCount:  s.TotalCount,
            CurrentIndex: s.CurrentIndex,
            Progress:    float64(s.CurrentIndex) / float64(s.TotalCount) * 100,

        })
    }

    return result, nil
}

// cacheSession 缓存会话到Redis
func (s *PracticeService) cacheSession(session *model.PracticeSession) {
    key := fmt.Sprintf("qh:practice:session:%s", session.SessionID)
    data, _ := json.Marshal(session)
    s.redisClient.Set(context.Background(), key, data, 7*24*time.Hour)
}

// restoreSessionFromCache 从Redis恢复会话
func (s *PracticeService) restoreSessionFromCache(sessionID string, userID uint) (*model.PracticeSession, error) {
    key := fmt.Sprintf("qh:practice:session:%s", sessionID)
    data, err := s.redisClient.Get(context.Background(), key).Bytes()
    if err != nil {
        return nil, err
    }

    var session model.PracticeSession
    if err := json.Unmarshal(data, &session); err != nil {
        return nil, err
    }

    if session.UserID != userID {
        return nil, errors.New("无权操作此练习")
    }

    return &session, nil
}
```

```go
// getAdaptiveQuestions 自适应获取题目
func (s *MockExamService) getAdaptiveQuestions(req *dto.StartMockExamRequest, userID uint) ([]model.Question, error) {
    // 获取用户能力等级
    profile, _ := s.abilityService.GetProfile(userID, req.CategoryID)
    difficultyScore := profile.DifficultyScore

    // 按能力分数分布获取题目
    // 60%题目接近用户水平, 20%略高, 20%略低
    count := req.QuestionCount
    nearCount := int(float64(count) * 0.6)
    hardCount := int(float64(count) * 0.2)
    easyCount := count - nearCount - hardCount

    var allQuestions []model.Question

    // 接近水平的题目
    near, _ := s.questionRepo.FindRandom(&question.RandomQuery{
        CategoryID: req.CategoryID,
        Difficulty: s.difficultyFromScore(difficultyScore),
        Count:      nearCount,
        UserID:     userID,
    })
    allQuestions = append(allQuestions, near...)

    // 略高难度
    hardDiff := s.difficultyFromScore(math.Min(difficultyScore+1, 5.0))
    hard, _ := s.questionRepo.FindRandom(&question.RandomQuery{
        CategoryID: req.CategoryID,
        Difficulty: hardDiff,
        Count:      hardCount,
        UserID:     userID,
    })
    allQuestions = append(allQuestions, hard...)

    // 略低难度
    easyDiff := s.difficultyFromScore(math.Max(difficultyScore-1, 1.0))
    easy, _ := s.questionRepo.FindRandom(&question.RandomQuery{
        CategoryID: req.CategoryID,
        Difficulty: easyDiff,
        Count:      easyCount,
        UserID:     userID,
    })
    allQuestions = append(allQuestions, easy...)

    return allQuestions, nil
}

    session, err := s.practiceRepo.FindBySessionID(sessionID)
    if err != nil {
        return nil, errors.New("练习会话不存在")
    }
    if session.UserID != userID {
        return nil, errors.New("无权操作此练习")
    }
    if session.Status == 2 {
        return nil, errors.New("已经交卷")
    }

    // 获取分值配置
    examKey := fmt.Sprintf("qh:practice:mock:%s", sessionID)
    passScoreStr, _ := s.redisClient.HGet(context.Background(), examKey, "pass_score").Result()
    totalScoreStr, _ := s.redisClient.HGet(context.Background(), examKey, "total_score").Result()
    scorePerStr, _ := s.redisClient.HGet(context.Background(), examKey, "score_per_question").Result()

    passScore, _ := strconv.ParseFloat(passScoreStr, 64)
    totalScore, _ := strconv.ParseFloat(totalScoreStr, 64)
    scorePer, _ := strconv.ParseFloat(scorePerStr, 64)

    // 统计得分
    records, _ := s.practiceRepo.GetRecordsBySessionID(session.ID)
    var correctCount int
    var score float64
    for _, r := range records {
        if r.IsCorrect == 1 {
            correctCount++
            score += scorePer
        }
    }

    accuracy := float64(correctCount) / float64(session.TotalCount) * 100

    // 更新会话
    now := time.Now()
    session.Status = 2
    session.CompletedAt = &now
    session.CorrectCount = correctCount
    session.Accuracy = decimal.NewFromFloat(accuracy)
    session.Score = decimal.NewFromFloat(score)
    s.practiceRepo.UpdateSession(session)

    // 清除Redis缓存
    s.redisClient.Del(context.Background(), examKey)

    return &dto.MockExamResult{
        SessionID:    sessionID,
        TotalCount:   session.TotalCount,
        CorrectCount: correctCount,
        Score:        score,
        TotalScore:   totalScore,
        PassScore:    passScore,
```

```go
// calculateLevel 计算能力等级
func (s *AbilityService) calculateLevel(accuracy float64, totalQuestions int) int8 {
    // 基于正确率和练习量综合评估
    switch {
    case accuracy >= 90 && totalQuestions >= 500:
        return 5 // 精通
    case accuracy >= 80 && totalQuestions >= 200:
        return 4 // 高级
    case accuracy >= 70 && totalQuestions >= 100:
        return 3 // 中级
    case accuracy >= 60 && totalQuestions >= 50:
        return 2 // 初级
    default:
        return 1 // 入门
    }
}

// calculateDifficultyScore 计算推荐难度分数
func (s *AbilityService) calculateDifficultyScore(profile *model.UserAbilityProfile) decimal.Decimal {
    accuracy := profile.Accuracy.InexactFloat64()

    // 基础难度基于正确率
    baseDifficulty := accuracy / 20 // 0-5

    // 练习量越多，推荐难度越准确
    confidence := math.Min(float64(profile.TotalQuestions)/100, 1.0)

    // 最终难度 = 基础难度 * 置信度 + 默认难度 * (1-置信度)
    finalDifficulty := baseDifficulty*confidence + 3.0*(1-confidence)

    // 限制在1-5范围内
    finalDifficulty = math.Max(1, math.Min(5, finalDifficulty))

    return decimal.NewFromFloat(math.Round(finalDifficulty*10) / 10)
}
```

```go
}

// GenerateDailyReport 生成每日报告
func (s *ReportService) GenerateDailyReport(userID uint, date string) (*dto.DailyReport, error) {
    report := &dto.DailyReport{
        Date:   date,
        UserID: userID,
    }

    // 当日练习统计
    sessions, _ := s.practiceRepo.GetSessionsByUserAndDate(userID, date)
    report.SessionCount = len(sessions)

    var totalQuestions, correctCount, totalDuration int
    for _, sess := range sessions {
        totalQuestions += sess.TotalCount
        correctCount += sess.CorrectCount
        totalDuration += sess.Duration
    }

    report.TotalQuestions = totalQuestions
    report.CorrectCount = correctCount
    report.TotalDuration = totalDuration / 60 // 转为分钟

    if totalQuestions > 0 {
        report.Accuracy = float64(correctCount) / float64(totalQuestions) * 100
    }

    // 错题分析
    wrongQuestions, _ := s.wrongRepo.FindByUserAndDate(userID, date)
    report.WrongCount = len(wrongQuestions)

    // 分类分布
    report.CategoryDistribution, _ = s.practiceRepo.GetCategoryStatsByDate(userID, date)

    // 难度分布
    report.DifficultyDistribution, _ = s.practiceRepo.GetDifficultyStatsByDate(userID, date)

    // 生成建议
    report.Suggestions = s.generateSuggestions(report)

    return report, nil
}

// GenerateWeeklyReport 生成每周报告
func (s *ReportService) GenerateWeeklyReport(userID uint, weekStart string) (*dto.WeeklyReport, error) {
    report := &dto.WeeklyReport{
        WeekStart: weekStart,
        UserID:    userID,
    }

    weekEnd := time.Now().Format("2006-01-02")

    // 周统计
    report.TotalQuestions, _ = s.practiceRepo.CountQuestionsByUserAndDateRange(userID, weekStart, weekEnd)
    report.TotalCorrect, _ = s.practiceRepo.CountCorrectByUserAndDateRange(userID, weekStart, weekEnd)
    report.TotalDuration, _ = s.practiceRepo.SumDurationByUserAndDateRange(userID, weekStart, weekEnd)
    report.TotalDuration /= 60

    if report.TotalQuestions > 0 {
        report.Accuracy = float64(report.TotalCorrect) / float64(report.TotalQuestions) * 100
    }

    // 每日趋势
    report.DailyTrends, _ = s.statsService.GetTrend(userID, 7)

    // 能力变化
    report.AbilityProfile, _ = s.abilityService.GetProfile(userID, nil)

    // 薄弱知识点
    report.WeakPoints, _ = s.wrongRepo.GetWeakPoints(userID, 5)

    // 与上周对比
    prevWeekStart := time.Now().AddDate(0, 0, -14).Format("2006-01-02")
    prevWeekEnd := weekStart
    prevQuestions, _ := s.practiceRepo.CountQuestionsByUserAndDateRange(userID, prevWeekStart, prevWeekEnd)
    if prevQuestions > 0 {
        report.QuestionGrowth = float64(report.TotalQuestions-prevQuestions) / float64(prevQuestions) * 100
    }

    // 生成建议
    report.Suggestions = s.generateWeeklySuggestions(report)

    return report, nil
}

// generateSuggestions 生成每日建议
```

```go
// StartChallenge 开始闯关
func (s *ChallengeService) StartChallenge(challengeID uint, userID uint) (*dto.ChallengeSessionInfo, error) {
    level, err := s.challengeRepo.FindByID(challengeID)
    if err != nil {
        return nil, errors.New("关卡不存在")
    }

    // 检查前置关卡是否已通过
    if level.Level > 1 {
        prevLevel, _ := s.challengeRepo.FindByCategoryAndLevel(level.CategoryID, level.Level-1)
        if prevLevel != nil {
            progress, _ := s.progressRepo.FindByUserAndChallenge(userID, prevLevel.ID)
            if progress == nil || progress.Status != 2 {
                return nil, errors.New("请先通过前置关卡")
            }
        }
    }

    // 获取题目
    questions, err := s.questionRepo.FindRandom(&question.RandomQuery{
        CategoryID: &level.CategoryID,
        Difficulty: &level.Difficulty,
        Count:      level.QuestionCount,
        UserID:     userID,
    })
    if err != nil || len(questions) < level.QuestionCount {
        return nil, errors.New("关卡题目不足")
    }

    // 打乱题目顺序
    rand.Shuffle(len(questions), func(i, j int) {
        questions[i], questions[j] = questions[j], questions[i]
    })

    questionIDs := make([]uint, len(questions))
    for i, q := range questions {
        questionIDs[i] = q.ID
    }

    // 创建或更新闯关进度
    progress, _ := s.progressRepo.FindByUserAndChallenge(userID, challengeID)
    if progress == nil {
        progress = &model.UserChallengeProgress{
            UserID:      userID,
            ChallengeID: challengeID,
            Status:      1,
        }
        s.progressRepo.Create(progress)
    } else {
        progress.Status = 1
        s.progressRepo.Update(progress)
    }
    progress.AttemptCount++

    // 创建练习会话(复用practice_sessions)
    sessionUUID := uuid.New().String()
    session := &model.PracticeSession{
        UserID:       userID,
        SessionID:    sessionUUID,
        Mode:         6, // 闯关模式
        CategoryID:   &level.CategoryID,
        QuestionIDs:  questionIDs,
        CurrentIndex: 0,
        TotalCount:   len(questions),
        Status:       0,
        StartedAt:    time.Now(),
    }
    s.practiceRepo.CreateSession(session)

    // 缓存闯关信息
    challengeKey := fmt.Sprintf("qh:practice:challenge:%s", sessionUUID)
    s.redisClient.HSet(context.Background(), challengeKey,
        "challenge_id", challengeID,
        "pass_rate", level.PassRate,
        "time_limit", level.TimeLimit,
        "reward_points", level.RewardPoints,
    )
    if level.TimeLimit != nil {
        s.redisClient.Expire(context.Background(), challengeKey, time.Duration(*level.TimeLimit+60)*time.Second)
    } else {
        s.redisClient.Expire(context.Background(), challengeKey, 2*time.Hour)
    }

    return &dto.ChallengeSessionInfo{
        SessionID:      sessionUUID,
        Level:          level.Level,
        Title:          level.Title,
        QuestionCount:  len(questions),
        PassRate:       level.PassRate,
        TimeLimit:      level.TimeLimit,
        FirstQuestion:  s.buildQuestionDTO(questions[0]),
    }, nil
}

// SubmitChallenge 提交闯关结果
func (s *ChallengeService) SubmitChallenge(sessionID string, userID uint) (*dto.ChallengeResult, error) {
    session, err := s.practiceRepo.FindBySessionID(sessionID)
    if err != nil {
        return nil, errors.New("闯关会话不存在")
    }

    // 从Redis获取闯关配置
    challengeKey := fmt.Sprintf("qh:practice:challenge:%s", sessionID)
    challengeIDStr, _ := s.redisClient.HGet(context.Background(), challengeKey, "challenge_id").Result()
    passRateStr, _ := s.redisClient.HGet(context.Background(), challengeKey, "pass_rate").Result()
    rewardStr, _ := s.redisClient.HGet(context.Background(), challengeKey, "reward_points").Result()
    challengeID, _ := strconv.ParseUint(challengeIDStr, 10, 64)
    passRate, _ := strconv.ParseFloat(passRateStr, 64)
    rewardPoints, _ := strconv.Atoi(rewardStr)

    // 计算正确率
    records, _ := s.practiceRepo.GetRecordsBySessionID(session.ID)
    correctCount := 0
    for _, r := range records {
        if r.IsCorrect == 1 {
            correctCount++
        }
    }
    accuracy := float64(correctCount) / float64(session.TotalCount) * 100
    passed := accuracy >= passRate

    // 更新闯关进度
    progress, _ := s.progressRepo.FindByUserAndChallenge(userID, uint(challengeID))
    if progress != nil {
        if accuracy > progress.BestAccuracy.InexactFloat64() {
            progress.BestAccuracy = decimal.NewFromFloat(accuracy)
        }
        if passed && progress.Status != 2 {
            progress.Status = 2
            now := time.Now()
            progress.PassedAt = &now
            // 发放奖励积分
            s.pointsService.AddPoints(userID, rewardPoints, "challenge_pass", uint(challengeID))
        }
        s.progressRepo.Update(progress)
    }

    // 更新会话状态
    session.Status = 2
    session.CorrectCount = correctCount
    session.Accuracy = decimal.NewFromFloat(accuracy)
    now := time.Now()
    session.CompletedAt = &now
    s.practiceRepo.UpdateSession(session)

    s.redisClient.Del(context.Background(), challengeKey)

    return &dto.ChallengeResult{
        SessionID:    sessionID,
        Passed:       passed,
        Accuracy:     accuracy,
        CorrectCount: correctCount,
        TotalCount:   session.TotalCount,
        PassRate:     passRate,

    }, nil
```

```go
type RelatedService struct {

// GetRelatedQuestions 获取相关题目推荐
func (s *RelatedService) GetRelatedQuestions(questionID uint, limit int) ([]dto.RelatedQuestionInfo, error) {
    if limit <= 0 || limit > 20 {
        limit = 10
    }

    // 获取当前题目信息
    question, err := s.questionRepo.FindByID(questionID)
    if err != nil {
        return nil, errors.New("题目不存在")
    }

    // 获取当前题目的知识点标签
    tags, _ := s.questionRepo.GetQuestionTags(questionID)
    if len(tags) == 0 {
        // 无知识点标签，回退到同分类同难度推荐
        return s.getBySameCategoryAndDifficulty(question, limit)
    }

    // 基于知识点关联推荐：查找共享知识点标签最多的题目
    tagIDs := make([]uint, len(tags))
    for i, t := range tags {
        tagIDs[i] = t.ID
    }

    relatedQuestions, err := s.questionRepo.FindByTagIDs(tagIDs, questionID, limit*2)
    if err != nil || len(relatedQuestions) == 0 {
        return s.getBySameCategoryAndDifficulty(question, limit)
    }

    // 按标签重合度排序，取前limit条
    type scoredQuestion struct {
        question model.Question
        score    int
    }
    var scored []scoredQuestion
    for _, rq := range relatedQuestions {
        rqTags, _ := s.questionRepo.GetQuestionTags(rq.ID)
        overlap := 0
        for _, rt := range rqTags {
            for _, t := range tagIDs {
                if rt.ID == t {
                    overlap++
                }
            }
        }
        scored = append(scored, scoredQuestion{question: rq, score: overlap})
    }

    sort.Slice(scored, func(i, j int) bool {
        return scored[i].score > scored[j].score
    })

    var result []dto.RelatedQuestionInfo
    for i, sq := range scored {
        if i >= limit {
            break
        }
        result = append(result, dto.RelatedQuestionInfo{
            ID:       sq.question.ID,
            Title:    sq.question.Title,
            Type:     sq.question.Type,
            Difficulty: sq.question.Difficulty,
        })
    }

    return result, nil
}

// getBySameCategoryAndDifficulty 同分类同难度回退推荐
func (s *RelatedService) getBySameCategoryAndDifficulty(question *model.Question, limit int) ([]dto.RelatedQuestionInfo, error) {
    questions, err := s.questionRepo.FindRandom(&question.RandomQuery{

        Difficulty: &question.Difficulty,
        Count:      limit + 1,
        ExcludeIDs: []uint{question.ID},
    })
    if err != nil {
        return nil, err
    }

    var result []dto.RelatedQuestionInfo
    for i, q := range questions {
        if i >= limit {
            break
        }
        result = append(result, dto.RelatedQuestionInfo{
            ID:         q.ID,
            Title:      q.Title,
            Type:       q.Type,
            Difficulty: q.Difficulty,
        })
    }
    return result, nil
}
```

```go
// RecordBrowse 记录分类浏览
func (s *RecentCategoryService) RecordBrowse(userID uint, categoryID uint) {
    key := fmt.Sprintf(recentCategoriesKey, userID)
    member := strconv.Itoa(int(categoryID))
    score := float64(time.Now().Unix())

    s.redisClient.ZAdd(context.Background(), key, &redis.Z{
        Score:  score,
        Member: member,
    })

    // 超过上限时移除最旧的
    count := s.redisClient.ZCard(context.Background(), key).Val()
    if count > maxRecentCategories {
        s.redisClient.ZRemRangeByRank(context.Background(), key, 0, count-maxRecentCategories-1)
    }

    s.redisClient.Expire(context.Background(), key, 30*24*time.Hour)
}

// GetRecentCategories 获取最近浏览的分类
func (s *RecentCategoryService) GetRecentCategories(userID uint, limit int) ([]dto.RecentCategoryInfo, error) {
    if limit <= 0 || limit > 20 {
        limit = 10
    }

    key := fmt.Sprintf(recentCategoriesKey, userID)
    members, err := s.redisClient.ZRevRange(context.Background(), key, 0, int64(limit-1)).Result()
    if err != nil {
        return nil, err
    }

    var result []dto.RecentCategoryInfo
    for _, m := range members {
        categoryID, _ := strconv.ParseUint(m, 10, 64)
        // 此处可查询分类名称，简化为仅返回ID
        result = append(result, dto.RecentCategoryInfo{
            CategoryID: uint(categoryID),
        })
    }

    return result, nil
}
```

```go
// ResourceAccessService 资源访问权限校验服务
type ResourceAccessService struct {
    userRepo     *user.UserRepo
    classRepo    *class.ClassRepo
    questionRepo *question.QuestionRepo
}

// AccessLevel 访问级别
type AccessLevel int8

const (
    AccessDenied   AccessLevel = 0 // 无权限
    AccessPublic   AccessLevel = 1 // 公开资源
    AccessPrivate  AccessLevel = 2 // 私有资源(仅创建者)
    AccessClass    AccessLevel = 3 // 班级资源(班级成员)
    AccessAdmin    AccessLevel = 4 // 管理员
)

// CheckQuestionAccess 校验用户对题目的访问权限
func (s *ResourceAccessService) CheckQuestionAccess(userID uint, questionID uint) (AccessLevel, error) {
    question, err := s.questionRepo.FindByID(questionID)
    if err != nil {
        return AccessDenied, errors.New("题目不存在")
    }

    user, _ := s.userRepo.FindByID(userID)

    // 管理员拥有全部权限
    if user.Role == "admin" || user.Role == "super_admin" {
        return AccessAdmin, nil
    }

    // 公共资源：所有用户可访问
    if question.AccessType == 1 { // 1=公开
        return AccessPublic, nil
    }

    // 私有资源：仅创建者/所属机构可访问
    if question.AccessType == 2 { // 2=私有
        if question.CreatedBy == userID {
            return AccessPrivate, nil
        }
        // 检查是否属于同一机构
        if question.OrgID != nil && user.OrgID == *question.OrgID {
            return AccessPrivate, nil
        }
        return AccessDenied, nil
    }

    // 班级资源：班级成员可访问
    if question.AccessType == 3 { // 3=班级
        if question.ClassID == nil {
            return AccessDenied, nil
        }
        isMember, _ := s.classRepo.IsMember(*question.ClassID, userID)
        if isMember {
            return AccessClass, nil
        }
        // 教师也可以访问
        isTeacher, _ := s.classRepo.IsTeacher(*question.ClassID, userID)
        if isTeacher {
            return AccessClass, nil
        }
        return AccessDenied, nil
    }

    return AccessDenied, nil
}

// FilterAccessibleQuestions 过滤出用户可访问的题目列表
func (s *ResourceAccessService) FilterAccessibleQuestions(userID uint, questions []model.Question) []model.Question {
    user, _ := s.userRepo.FindByID(userID)

    // 管理员返回全部
    if user.Role == "admin" || user.Role == "super_admin" {
        return questions
    }

    // 获取用户所属班级列表
    classIDs, _ := s.classRepo.GetUserClassIDs(userID)

    var accessible []model.Question
    for _, q := range questions {
        switch q.AccessType {
        case 1: // 公开
            accessible = append(accessible, q)
        case 2: // 私有
            if q.CreatedBy == userID || (q.OrgID != nil && user.OrgID == *q.OrgID) {
                accessible = append(accessible, q)
            }
        case 3: // 班级
            if q.ClassID != nil {
                for _, cid := range classIDs {
                    if cid == *q.ClassID {
                        accessible = append(accessible, q)
                        break
                    }
                }
            }
        }
    }

    return accessible
}

// FilterAccessibleCategories 过滤出用户可访问的分类
func (s *ResourceAccessService) FilterAccessibleCategories(userID uint, categories []model.Category) []model.Category {
    user, _ := s.userRepo.FindByID(userID)

    if user.Role == "admin" || user.Role == "super_admin" {
        return categories
    }

    classIDs, _ := s.classRepo.GetUserClassIDs(userID)

    var accessible []model.Category
    for _, c := range categories {
        if c.AccessType == 1 {
            accessible = append(accessible, c)
        } else if c.AccessType == 2 {
            if c.CreatedBy == userID || (c.OrgID != nil && user.OrgID == *c.OrgID) {
                accessible = append(accessible, c)
            }
        } else if c.AccessType == 3 && c.ClassID != nil {
            for _, cid := range classIDs {
                if cid == *c.ClassID {
                    accessible = append(accessible, c)
                    break
                }
            }
        }
    }

    return accessible
}
```

```vue
<!-- components/CategoryTree.vue -->

<template>
  <view class="category-tree">
    <!-- 搜索框 -->
    <view class="search-bar">
      <input
        v-model="searchKeyword"
        placeholder="搜索分类名称"
        confirm-type="search"
        @input="onSearch"
      />
      <text v-if="searchKeyword" class="clear-btn" @tap="clearSearch">清除</text>
    </view>

    <!-- 搜索结果 -->
    <view v-if="searchKeyword && searchResults.length > 0" class="search-results">
      <view
        v-for="item in searchResults"
        :key="item.id"
        class="search-item"
        @tap="selectCategory(item)"
      >
        <text class="name" v-html="highlightText(item.name, searchKeyword)"></text>
        <text class="path">{{ item.breadcrumb }}</text>
      </view>
    </view>

    <!-- 分类树 -->
    <view v-if="!searchKeyword" class="tree-container">
      <CategoryTreeNode
        v-for="node in treeData"
        :key="node.id"
        :node="node"
        :level="0"
        :expanded-ids="expandedIds"
        @toggle="toggleExpand"
        @select="selectCategory"
      />
    </view>

    <!-- 空状态 -->
    <view v-if="searchKeyword && searchResults.length === 0" class="empty">
      <text>未找到匹配的分类</text>
    </view>
  </view>
</template>

<script setup lang="ts">
import { ref, computed, watch } from 'vue'
import CategoryTreeNode from './CategoryTreeNode.vue'

interface CategoryNode {
  id: number
  name: string
  parentId: number | null
  questionCount: number
  children?: CategoryNode[]
  breadcrumb?: string
}

const props = defineProps<{
  categories: CategoryNode[]
}>()

const emit = defineEmits<{
  (e: 'select', category: CategoryNode): void
}>()

const searchKeyword = ref('')
const expandedIds = ref<Set<number>>(new Set())

// 构建树形结构
const treeData = computed(() => {
  return buildTree(props.categories, null)
})

// 搜索结果
const searchResults = computed(() => {
  if (!searchKeyword.value) return []
  const keyword = searchKeyword.value.toLowerCase()
  return props.categories
    .filter(c => c.name.toLowerCase().includes(keyword))
    .map(c => ({
      ...c,
      breadcrumb: buildBreadcrumb(c, props.categories)
    }))
    .slice(0, 20)
})

const buildTree = (items: CategoryNode[], parentId: number | null): CategoryNode[] => {
  return items
    .filter(item => item.parentId === parentId)
    .map(item => ({
      ...item,
      children: buildTree(items, item.id)
    }))
}

const buildBreadcrumb = (category: CategoryNode, all: CategoryNode[]): string => {
  const parts: string[] = [category.name]
  let current = category
  while (current.parentId) {
    const parent = all.find(c => c.id === current.parentId)
    if (!parent) break
    parts.unshift(parent.name)
    current = parent
  }
  return parts.join(' > ')
}

const toggleExpand = (id: number) => {
  if (expandedIds.value.has(id)) {
    expandedIds.value.delete(id)
  } else {
    expandedIds.value.add(id)
  }
}

const selectCategory = (category: CategoryNode) => {
  emit('select', category)
}

const highlightText = (text: string, keyword: string) => {
  const regex = new RegExp(`(${keyword})`, 'gi')
  return text.replace(regex, '<span class="highlight">$1</span>')
}

const clearSearch = () => {
  searchKeyword.value = ''
}

// 默认展开第一级
watch(() => props.categories, (cats) => {
  if (cats) {
    const roots = cats.filter(c => !c.parentId)
    roots.forEach(r => expandedIds.value.add(r.id))
  }
}, { immediate: true })
</script>
```

### 7.2 练习提醒定时任务

```go
// cron/practice_reminder_cron.go

type PracticeReminderCron struct {
    userRepo    *user.UserRepo
    redisClient *redis.Client
}

// SetupPracticeReminderCron 设置练习提醒定时任务
func SetupPracticeReminderCron() {
    c := cron.New()

    // 每天在用户配置的提醒时间检查并发送提醒
    // 默认每小时检查一次，匹配用户配置的提醒时间
    c.AddFunc("0 * * * *", func() {
        practiceReminderCron.SendReminders()
    })

    c.Start()
}

// UserReminderConfig 用户提醒配置(存储在Redis或数据库中)
type UserReminderConfig struct {
    UserID       uint     `json:"user_id"`
    Enabled      bool     `json:"enabled"`
    ReminderTime string   `json:"reminder_time"` // 格式 "HH:MM"
    ReminderDays []int    `json:"reminder_days"` // 0=周日,1=周一 ... 6=周六
    Channel      string   `json:"channel"`       // app/wechat/sms
}

// SendReminders 发送练习提醒
func (c *PracticeReminderCron) SendReminders() {
    currentHour := time.Now().Hour()
    currentMinute := time.Now().Minute()
    currentWeekday := int(time.Now().Weekday())
    timeStr := fmt.Sprintf("%02d:%02d", currentHour, currentMinute)

    // 获取所有启用提醒的用户配置
    configs := c.getAllReminderConfigs()

    for _, config := range configs {
        if !config.Enabled {
            continue
        }

        // 匹配提醒时间
        if config.ReminderTime != timeStr {
            continue
        }

        // 匹配提醒日期
        if len(config.ReminderDays) > 0 {
            matched := false
            for _, day := range config.ReminderDays {
                if day == currentWeekday {
                    matched = true
                    break
                }
            }
            if !matched {
                continue
            }
        }

        // 检查今天是否已练习
        today := time.Now().Format("2006-01-02")
        practicedToday, _ := c.hasPracticedToday(config.UserID, today)
        if practicedToday {
            continue // 今天已练习，不提醒
        }

        // 发送提醒
        c.sendNotification(config)
    }
}

// hasPracticedToday 检查用户今天是否已练习
func (c *PracticeReminderCron) hasPracticedToday(userID uint, date string) (bool, error) {
    key := fmt.Sprintf("qh:practice:daily:%d:%s", userID, date)
    exists, _ := c.redisClient.Exists(context.Background(), key).Result()
    return exists > 0, nil
}

// sendNotification 发送通知
func (c *PracticeReminderCron) sendNotification(config UserReminderConfig) {
    message := fmt.Sprintf("您今天还没有刷题哦，坚持练习才能进步！点击查看今日推荐题目。")

    switch config.Channel {
    case "app":
        // 发送APP推送通知
        pushService.SendAppPush(config.UserID, "刷题提醒", message, map[string]string{
            "type": "practice_reminder",
            "url":  "/pages/practice/daily",
        })
    case "wechat":
        // 发送微信模板消息
        wechatService.SendTemplateMessage(config.UserID, "practice_reminder", message)
    case "sms":
        // 发送短信提醒(仅对VIP用户)
        smsService.SendSMS(config.UserID, message)
    }
}

// getAllReminderConfigs 获取所有用户的提醒配置
func (c *PracticeReminderCron) getAllReminderConfigs() []UserReminderConfig {
    // 从Redis Hash中获取所有用户的提醒配置
    // key: qh:practice:reminder_config:{uid}
    var configs []UserReminderConfig
    pattern := "qh:practice:reminder_config:*"
    keys, _ := c.redisClient.Keys(context.Background(), pattern).Result()

    for _, key := range keys {
        data, err := c.redisClient.Get(context.Background(), key).Bytes()
        if err != nil {
            continue
        }
        var config UserReminderConfig
        json.Unmarshal(data, &config)
        configs = append(configs, config)
    }

    return configs
}
```

提醒配置管理接口：

| 方法 | 路径 | 说明 |
|------|------|------|
| GET | `/api/v1/practice/reminder/config` | 获取用户提醒配置 |
| PUT | `/api/v1/practice/reminder/config` | 更新用户提醒配置(enabled, reminder_time, reminder_days, channel) |

提醒配置存储在Redis中，Key格式：`qh:practice:reminder_config:{uid}`，数据结构为JSON字符串，TTL为永久（用户主动清除或注销时删除）。
