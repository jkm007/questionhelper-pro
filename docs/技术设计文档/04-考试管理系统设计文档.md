### 3.5.6 试卷导出 API（Word/PDF）

**GET /api/v1/admin/papers/:id/export**

**请求参数**：

| 参数 | 类型 | 必填 | 说明 |
|------|------|------|------|
| format | string | 否 | 导出格式: word/pdf，默认word |
| include_answer | int | 否 | 是否包含答案: 0=否,1=是，默认0 |
| include_explanation | int | 否 | 是否包含解析: 0=否,1=是，默认0 |

**后端实现要点**：
- Word 导出使用 `github.com/unidoc/unioffice` 库生成 .docx
- PDF 导出使用 `github.com/jung-kurt/gofpdf` 库
- 导出格式按题型分组，包含题目编号、分值标注
- 答案和解析可选附在试卷末尾或每题下方

```go
// service/paper/export_service.go

type ExportService struct {
    paperRepo *PaperRepo
}

// ExportPaper 导出试卷
func (s *ExportService) ExportPaper(paperID uint64, format string, includeAnswer, includeExplanation bool) ([]byte, string, error) {
    paper, err := s.paperRepo.FindByID(paperID)
    if err != nil {
        return nil, "", fmt.Errorf("试卷不存在")
    }

    questions, err := s.paperRepo.GetQuestionsWithSnapshot(paperID)
    if err != nil {
        return nil, "", err
    }

    switch format {
    case "pdf":
        data, err := s.generatePDF(paper, questions, includeAnswer, includeExplanation)
        return data, "application/pdf", err
    default: // word
        data, err := s.generateWord(paper, questions, includeAnswer, includeExplanation)
        return data, "application/vnd.openxmlformats-officedocument.wordprocessingml.document", err
    }
}

// generateWord 生成 Word 文档
func (s *ExportService) generateWord(paper *Paper, questions []*PaperQuestion, includeAnswer, includeExplanation bool) ([]byte, error) {
    doc := document.New()

    // 添加标题
    titlePara := doc.AddParagraph()
    titleRun := titlePara.AddRun()
    titleRun.SetText(paper.Title)
    titleRun.Properties().SetBold(true)
    titleRun.Properties().SetSize(16)

    // 按题型分组
    groups := groupByType(questions)
    questionNum := 1

    for typeName, qs := range groups {
        // 添加题型标题
        sectionPara := doc.AddParagraph()
        sectionRun := sectionPara.AddRun()
        sectionRun.SetText(typeName)
        sectionRun.Properties().SetBold(true)

        for _, q := range qs {
            // 添加题目
            qPara := doc.AddParagraph()
            qRun := qPara.AddRun()
            snapshot := q.Snapshot
            qRun.SetText(fmt.Sprintf("%d. %s (%.1f分)", questionNum, snapshot.Title, q.Score))
            questionNum++

            // 添加选项（选择题）
            if snapshot.Type <= 3 {
                for _, opt := range snapshot.Options {
                    optPara := doc.AddParagraph()
                    optRun := optPara.AddRun()
                    optRun.SetText(fmt.Sprintf("    %s. %s", opt.Label, opt.Content))
                }
            }

            // 添加答案（可选）
            if includeAnswer && snapshot.CorrectAnswer != "" {
                ansPara := doc.AddParagraph()
                ansRun := ansPara.AddRun()
                ansRun.SetText(fmt.Sprintf("【答案】%s", snapshot.CorrectAnswer))
                ansRun.Properties().SetColor("0000FF")
            }

            // 添加解析（可选）
            if includeExplanation && snapshot.Explanation != "" {
                expPara := doc.AddParagraph()
                expRun := expPara.AddRun()
                expRun.SetText(fmt.Sprintf("【解析】%s", snapshot.Explanation))
                expRun.Properties().SetColor("009900")
            }
        }
    }

    var buf bytes.Buffer
    doc.Write(&buf)
    return buf.Bytes(), nil
}
```

### 3.5.7 试卷发布 API

**PUT /api/v1/admin/papers/:id/publish**

**后端实现要点**：
- 只有草稿状态的试卷才能发布
- 发布时为所有题目生成 `question_snapshot` 快照
- 发布后状态变更为 `status=1`

```go
// service/paper/paper_service.go

// PublishPaper 发布试卷
func (s *PaperService) PublishPaper(paperID, userID uint64) error {
    paper, err := s.paperRepo.FindByID(paperID)
    if err != nil {
        return fmt.Errorf("试卷不存在")
    }

    if paper.CreatorID != userID {
        return fmt.Errorf("无权操作")
    }

    if paper.Status != 0 {
        return fmt.Errorf("只有草稿状态的试卷才能发布")
    }

    // 验证试卷至少包含一道题目
    count, _ := s.paperRepo.CountQuestions(paperID)
    if count == 0 {
        return fmt.Errorf("试卷不能为空，请先添加题目")
    }

    // 为所有题目生成快照
    if err := s.generateSnapshots(paperID); err != nil {
        return fmt.Errorf("生成题目快照失败: %v", err)
    }

    // 更新状态
    paper.Status = 1
    return s.paperRepo.Update(paper)
}

// generateSnapshots 为试卷中所有题目生成快照
func (s *PaperService) generateSnapshots(paperID uint64) error {
    questions, err := s.paperRepo.GetQuestions(paperID)
    if err != nil {
        return err
    }

    for _, pq := range questions {
        question, err := s.questionRepo.FindByID(pq.QuestionID)
        if err != nil {
            continue
        }

        options, _ := s.questionRepo.GetOptions(pq.QuestionID)
        correctAnswer, _ := s.questionRepo.GetCorrectAnswer(pq.QuestionID)

        snapshot := &QuestionSnapshot{
            ID:            question.ID,
            Type:          question.Type,
            Title:         question.Title,
            Options:       options,
            CorrectAnswer: correctAnswer,
            Explanation:   question.Explanation,
            Difficulty:    question.Difficulty,
            Category:      question.Category,
            SnapshotTime:  time.Now(),
        }

        snapshotJSON, _ := json.Marshal(snapshot)
        s.paperRepo.UpdateSnapshot(pq.ID, string(snapshotJSON))
    }

    return nil
}
```

```go
        stats.PassRate = float64(totalPass) / float64(totalStudents) * 100
    }
```

```go
type ImportService struct {
    paperRepo    *PaperRepo

}

// ImportFromWord 从 Word 文件导入试卷
func (s *ImportService) ImportFromWord(file multipart.File, userID uint64, category string) (*ImportResult, error) {
    // 读取 Word 文件内容
    content, err := s.extractWordContent(file)
    if err != nil {
        return nil, fmt.Errorf("读取文件失败: %v", err)
    }

    // 解析题目
    questions, parseErrors := s.parseQuestions(content)

    result := &ImportResult{
        TotalParsed: len(questions),
        Errors:      parseErrors,
    }

    // 如果题目解析成功，创建试卷
    if len(questions) > 0 {
        paper := &Paper{
            Title:      s.extractTitle(content),
            TotalScore: s.calculateTotalScore(questions),
            TotalCount: len(questions),
            Type:       1,
            Status:     0,
            Category:   category,
            CreatorID:  userID,
        }

        if err := s.paperRepo.Create(paper); err != nil {
            return nil, err
        }

        for i, q := range questions {
            pq := &PaperQuestion{
                PaperID: paper.ID,
                Score:   q.Score,
                Sort:    i + 1,
            }
            // 如果题目库中已有相同题目，关联；否则新建
            existing, _ := s.questionRepo.FindByTitle(q.Title)
            if existing != nil {
                pq.QuestionID = existing.ID
            } else {
                newQ := s.createQuestion(q)
                pq.QuestionID = newQ.ID
            }
            s.paperRepo.AddQuestion(pq)
        }

        result.PaperID = paper.ID
        result.PaperTitle = paper.Title
    }

    return result, nil
}

// parseQuestions 解析题目
func (s *ImportService) parseQuestions(content string) ([]*ParsedQuestion, []string) {
    var questions []*ParsedQuestion
    var errors []string

    // 正则匹配题目模式
    // 匹配: 1. / 1、/ （1）/ 一、 等格式
    questionPattern := regexp.MustCompile(`(?m)^(\d+)[.、．]\s*(.+?)(?:\n|$)`)
    optionPattern := regexp.MustCompile(`(?m)^([A-D])[.、．]\s*(.+?)(?:\n|$)`)
    scorePattern := regexp.MustCompile(`[\(（]\s*(\d+(?:\.\d+)?)\s*分\s*[\)）]`)

    matches := questionPattern.FindAllStringSubmatchIndex(content, -1)

    for i, match := range matches {
        start := match[0]
        end := len(content)
        if i+1 < len(matches) {
            end = matches[i+1][0]
        }

        section := content[start:end]
        title := strings.TrimSpace(content[match[2]:match[3]])

        // 提取分值
        score := 5.0 // 默认分值
        if sm := scorePattern.FindStringSubmatch(section); sm != nil {
            score, _ = strconv.ParseFloat(sm[1], 64)
        }

        // 检测题型
        qType := 1 // 默认单选
        options := optionPattern.FindAllStringSubmatch(section, -1)
        if len(options) > 0 {
            // 检查是否为多选（答案包含多个字母）
            qType = 1
        } else if strings.Contains(title, "判断") || strings.Contains(title, "对错") {
            qType = 3
        } else if strings.Contains(title, "填空") || strings.Contains(section, "____") {
            qType = 4
        } else {
            qType = 5 // 简答
        }

        q := &ParsedQuestion{
            Title:  title,
            Type:   qType,
            Score:  score,
        }

        for _, opt := range options {
            q.Options = append(q.Options, &ParsedOption{
                Label:   opt[1],
                Content: opt[2],
            })
        }

        questions = append(questions, q)
    }

    return questions, errors
}
```

```go
// service/paper/auto_compose_service.go

type AutoComposeService struct {
    paperRepo    *PaperRepo
    questionRepo *QuestionRepo
}

// ComposeRule 组卷规则
type ComposeRule struct {
    Type       int       `json:"type"`
    Category   string    `json:"category"`
    Difficulty []int     `json:"difficulty"`
    Count      int       `json:"count"`
    ScoreEach  float64   `json:"score_each"`
}

// ComposeRequest 智能组卷请求
type ComposeRequest struct {
    Title       string         `json:"title" binding:"required"`
    Description string         `json:"description"`
    Rules       []*ComposeRule `json:"rules" binding:"required,min=1"`
    ExcludeIDs  []uint64       `json:"exclude_ids"`
    Category    string         `json:"category"`
}

// AutoCompose 智能组卷
func (s *AutoComposeService) AutoCompose(req *ComposeRequest, userID uint64) (*Paper, error) {
    var allQuestions []*PaperQuestion
    sortIdx := 1

    for _, rule := range req.Rules {
        // 构建查询条件
        query := &QuestionQuery{
            Type:       rule.Type,
            Difficulty: rule.Difficulty,
            ExcludeIDs: req.ExcludeIDs,
        }
        if rule.Category != "" {
            query.Category = rule.Category
        }

        // 从题库中查询符合条件的题目
        candidates, err := s.questionRepo.FindByQuery(query)
        if err != nil {
            return nil, fmt.Errorf("查询题库失败: %v", err)
        }

        if len(candidates) < rule.Count {
            return nil, fmt.Errorf("题型%d的可用题目不足: 需要%d道, 仅有%d道",
                rule.Type, rule.Count, len(candidates))
        }

        // 随机抽取指定数量的题目
        selected := s.randomSelect(candidates, rule.Count, req.ExcludeIDs)

        for _, q := range selected {
            allQuestions = append(allQuestions, &PaperQuestion{
                QuestionID: q.ID,
                Score:      rule.ScoreEach,
                Sort:       sortIdx,
            })
            sortIdx++

            // 记录已选题目，避免重复
            req.ExcludeIDs = append(req.ExcludeIDs, q.ID)
        }
    }

    // 计算总分和总题数
    var totalScore float64
    for _, q := range allQuestions {
        totalScore += q.Score
    }

    // 创建试卷
    paper := &Paper{
        Title:       req.Title,
        Description: req.Description,
        TotalScore:  totalScore,
        TotalCount:  len(allQuestions),
        Type:        2, // 智能组卷
        Status:      0,
        Category:    req.Category,
        CreatorID:   userID,
    }

    if err := s.paperRepo.Create(paper); err != nil {
        return nil, err
    }

    // 添加题目关联
    for _, pq := range allQuestions {
        pq.PaperID = paper.ID
        s.paperRepo.AddQuestion(pq)
    }

    return paper, nil
}

// randomSelect 从候选题目中随机抽取指定数量
func (s *AutoComposeService) randomSelect(candidates []*Question, count int, excludeIDs []uint64) []*Question {
    // 过滤已排除的题目
    excludeMap := make(map[uint64]bool)
    for _, id := range excludeIDs {
        excludeMap[id] = true
    }

    var filtered []*Question
    for _, q := range candidates {
        if !excludeMap[q.ID] {
            filtered = append(filtered, q)
        }
    }

    // Fisher-Yates 洗牌算法
    r := rand.New(rand.NewSource(time.Now().UnixNano()))
    for i := len(filtered) - 1; i > 0; i-- {
        j := r.Intn(i + 1)
        filtered[i], filtered[j] = filtered[j], filtered[i]
    }

    if count > len(filtered) {
        count = len(filtered)
    }
    return filtered[:count]
}

// PreviewComposition 预览智能组卷结果（不实际创建）
func (s *AutoComposeService) PreviewComposition(req *ComposeRequest) (*ComposePreview, error) {
    preview := &ComposePreview{}

    for _, rule := range req.Rules {
        query := &QuestionQuery{
            Type:       rule.Type,
            Difficulty: rule.Difficulty,
            ExcludeIDs: req.ExcludeIDs,
        }
        if rule.Category != "" {
            query.Category = rule.Category
        }

        candidates, _ := s.questionRepo.FindByQuery(query)
        available := len(candidates) - len(req.ExcludeIDs)
        if available < 0 {
            available = 0
        }

        section := &PreviewSection{
            Type:        rule.Type,
            Category:    rule.Category,
            Difficulty:  rule.Difficulty,
            RequestCount: rule.Count,
            Available:   available,
            ScoreEach:   rule.ScoreEach,
            Subtotal:    float64(rule.Count) * rule.ScoreEach,
            Sufficient:  available >= rule.Count,
        }
        preview.Sections = append(preview.Sections, section)
        preview.TotalCount += rule.Count
        preview.TotalScore += section.Subtotal
    }

    return preview, nil
}

type ComposePreview struct {
    Sections   []*PreviewSection `json:"sections"`
    TotalCount int               `json:"total_count"`
    TotalScore float64           `json:"total_score"`
}

type PreviewSection struct {
```

```go
type ExamService struct {

// StartExam 开始考试
func (s *ExamService) StartExam(examID, userID uint) (*dto.ExamRecordInfo, error) {
    exam, err := s.examRepo.FindByID(examID)
    if err != nil {
        return nil, errors.New("考试不存在")
    }
    
    now := time.Now()
    if now.Before(exam.StartTime) {
        return nil, errors.New("考试尚未开始")
    }
    if now.After(exam.EndTime) {
        return nil, errors.New("考试已结束")

    // 检查是否已参加过
    count, _ := s.recordRepo.CountByExamAndUser(examID, userID)
    if count >= exam.MaxAttempts {
        return nil, errors.New("已达最大尝试次数")
    }
    
    // 创建考试记录
    record := &model.ExamRecord{
        ExamID:    examID,
        UserID:    userID,
        Status:    0,
        StartTime: now,
        IP:        "", // 从请求中获取
    }
    
    if err := s.recordRepo.Create(record); err != nil {
        return nil, err
    }
    
    return &dto.ExamRecordInfo{
        ID:        record.ID,
        ExamID:    record.ExamID,
        UserID:    record.UserID,
        Status:    record.Status,
        StartTime: record.StartTime,
    }, nil
}

// SubmitExam 提交考试
func (s *ExamService) SubmitExam(recordID uint, req *dto.SubmitExamRequest) error {
    record, err := s.recordRepo.FindByID(recordID)
    if err != nil {
        return errors.New("考试记录不存在")
    }
    
    if record.Status != 0 {
        return errors.New("已经提交过")
    }
    
    // 保存答案
    for _, answer := range req.Answers {
        answerRecord := &model.AnswerRecord{
            RecordID:   recordID,
            QuestionID: answer.QuestionID,
            Answer:     answer.Answer,
        }
        s.recordRepo.SaveAnswer(answerRecord)
    }
    
    // 自动评分（客观题）
    score := s.autoScore(recordID)
    
    // 更新记录
    now := time.Now()
    record.Status = 1
    record.Score = score
    record.SubmitTime = &now
    s.recordRepo.Update(record)
    
    return nil
}

// autoScore 自动评分
func (s *ExamService) autoScore(recordID uint) float64 {
    answers, _ := s.recordRepo.GetAnswers(recordID)
    var totalScore float64
    
    for _, answer := range answers {
        question, _ := s.examRepo.GetQuestion(answer.QuestionID)
        correctOptions, _ := s.examRepo.GetCorrectOptions(answer.QuestionID)
        
        isCorrect := false
        switch question.Type {
        case 1: // 单选
            isCorrect = len(correctOptions) == 1 && answer.Answer == correctOptions[0].Label
        case 2: // 多选
            isCorrect = s.checkMultipleChoice(answer.Answer, correctOptions)
        case 3: // 判断
            isCorrect = answer.Answer == correctOptions[0].Label
        }
        
        if isCorrect {
            // 获取该题分值
            score, _ := s.examRepo.GetQuestionScore(recordID, answer.QuestionID)
            totalScore += score
            answer.Score = score
            answer.IsCorrect = true
        }
        
        s.recordRepo.UpdateAnswer(&answer)
    }
    
    return totalScore
}
```

## 七、Redis 缓存会话设计

### 7.1 Redis Key 设计

| Key | 数据结构 | TTL | 说明 |
|-----|---------|-----|------|
| `qh:exam:session:{uid}:{eid}` | Hash | 考试时长 | 考试会话：题目顺序、开始时间、状态 |
| `qh:exam:answers:{uid}:{eid}` | Hash | 考试时长 | 用户答案缓存，Key=questionID，Value=答案 |
| `qh:exam:scores:{uid}:{eid}` | Hash | 考试时长 | 每题分值，Key=questionID，Value=score |
| `qh:exam:delay` | ZSet | - | 延迟队列，Score=交卷时间戳，Member=`{uid}:{eid}:{attempt}` |
| `qh:exam:switch:{uid}:{eid}` | List | 考试时长 | 切屏事件记录，Value=JSON(`{time, type}`) |
| `qh:exam:pause:{eid}` | String | - | 考试暂停标记，Value=暂停时间戳 |
| `qh:exam:pause:time:{uid}:{eid}` | String | - | 暂停前剩余秒数 |
| `qh:exam:ranking:{eid}` | ZSet | 7天 | 成绩排名，Score=分数，Member=uid |
| `qh:exam:notice:{nid}` | String | 7天 | 考试公告缓存 |
| `qh:exam:password:{eid}` | String | 24h | 考试密码验证标记，Value=1 |

### 7.2 考试会话数据结构

```
qh:exam:session:{uid}:{eid} (Hash)
├── record_id     : 考试记录ID
├── start_time    : 开始时间戳(秒)
├── duration      : 考试时长(秒)
├── question_order: JSON数组,题目ID乱序排列
├── option_order  : JSON对象,{questionID: [选项乱序映射]}
├── status        : in_progress/paused/submitted
├── attempt       : 第几次尝试
└── ip            : 开始考试时的IP地址
```

```go
// service/exam/session_service.go

package exam

import (
	"context"
	"encoding/json"
	"fmt"
	"time"

	"github.com/redis/go-redis/v9"
)

const (
	sessionKeyFmt  = "qh:exam:session:%d:%d"  // uid:eid
	answersKeyFmt  = "qh:exam:answers:%d:%d"   // uid:eid
	scoresKeyFmt   = "qh:exam:scores:%d:%d"    // uid:eid
	delayKey       = "qh:exam:delay"
	switchKeyFmt   = "qh:exam:switch:%d:%d"    // uid:eid
	pauseKey       = "qh:exam:pause:%d"         // eid
	pauseTimeKeyFmt = "qh:exam:pause:time:%d:%d" // uid:eid
	rankingKeyFmt  = "qh:exam:ranking:%d"       // eid
	noticeKeyFmt   = "qh:exam:notice:%d"        // nid
	passwordKeyFmt = "qh:exam:password:%d:%d"   // uid:eid
)

type SessionService struct {
	rdb *redis.Client
}

// ExamSession 考试会话
type ExamSession struct {
	RecordID     uint64   `json:"record_id"`
	StartTime    int64    `json:"start_time"`
	Duration     int64    `json:"duration"`
	QuestionOrder []uint64 `json:"question_order"`
	OptionOrder  map[string][]string `json:"option_order"`
	Status       string   `json:"status"`
	Attempt      int      `json:"attempt"`
	IP           string   `json:"ip"`
}

// CreateSession 创建考试会话
func (s *SessionService) CreateSession(ctx context.Context, uid, eid uint64, session *ExamSession) error {
	key := fmt.Sprintf(sessionKeyFmt, uid, eid)

	// 构造 Hash 字段
	fields := map[string]interface{}{
		"record_id":      session.RecordID,
		"start_time":     session.StartTime,
		"duration":       session.Duration,
		"question_order": marshalJSON(session.QuestionOrder),
		"option_order":   marshalJSON(session.OptionOrder),
		"status":         session.Status,
		"attempt":        session.Attempt,
		"ip":             session.IP,
	}

	pipe := s.rdb.Pipeline()
	pipe.HSet(ctx, key, fields)
	pipe.Expire(ctx, key, time.Duration(session.Duration)*time.Second)

	// 加入延迟队列：交卷时间 = 开始时间 + 时长
	submitAt := float64(session.StartTime + session.Duration)
	member := fmt.Sprintf("%d:%d:%d", uid, eid, session.Attempt)
	pipe.ZAdd(ctx, delayKey, redis.Z{Score: submitAt, Member: member})

	_, err := pipe.Exec(ctx)
	return err
}

// GetSession 获取考试会话
func (s *SessionService) GetSession(ctx context.Context, uid, eid uint64) (*ExamSession, error) {
	key := fmt.Sprintf(sessionKeyFmt, uid, eid)

	result, err := s.rdb.HGetAll(ctx, key).Result()
	if err != nil || len(result) == 0 {
		return nil, fmt.Errorf("考试会话不存在或已过期")
	}

	session := &ExamSession{}
	fmt.Sscanf(result["record_id"], "%d", &session.RecordID)
	fmt.Sscanf(result["start_time"], "%d", &session.StartTime)
	fmt.Sscanf(result["duration"], "%d", &session.Duration)
	fmt.Sscanf(result["attempt"], "%d", &session.Attempt)
	session.Status = result["status"]
	session.IP = result["ip"]

	json.Unmarshal([]byte(result["question_order"]), &session.QuestionOrder)
	json.Unmarshal([]byte(result["option_order"]), &session.OptionOrder)

	return session, nil
}

// SaveAnswer 保存答案（幂等写入）
func (s *SessionService) SaveAnswer(ctx context.Context, uid, eid uint64, questionID uint64, answer string) error {
	key := fmt.Sprintf(answersKeyFmt, uid, eid)
	return s.rdb.HSet(ctx, key, fmt.Sprintf("%d", questionID), answer).Err()
}

// SaveBatchAnswers 批量保存答案
func (s *SessionService) SaveBatchAnswers(ctx context.Context, uid, eid uint64, answers map[uint64]string) error {
	key := fmt.Sprintf(answersKeyFmt, uid, eid)

	fields := make(map[string]interface{}, len(answers))
	for qid, ans := range answers {
		fields[fmt.Sprintf("%d", qid)] = ans
	}

	return s.rdb.HSet(ctx, key, fields).Err()
}

// GetAllAnswers 获取所有答案
func (s *SessionService) GetAllAnswers(ctx context.Context, uid, eid uint64) (map[uint64]string, error) {

	result, err := s.rdb.HGetAll(ctx, key).Result()
	if err != nil {
		return nil, err
	}

	answers := make(map[uint64]string, len(result))
	for k, v := range result {
		var qid uint64
		fmt.Sscanf(k, "%d", &qid)
		answers[qid] = v
	}

}

// SetScores 设置每题分值
func (s *SessionService) SetScores(ctx context.Context, uid, eid uint64, scores map[uint64]float64) error {
	key := fmt.Sprintf(scoresKeyFmt, uid, eid)

	fields := make(map[string]interface{}, len(scores))
	for qid, score := range scores {
		fields[fmt.Sprintf("%d", qid)] = score
	}

	pipe := s.rdb.Pipeline()
	pipe.HSet(ctx, key, fields)
	pipe.Expire(ctx, key, 24*time.Hour)
	_, err := pipe.Exec(ctx)
	return err
}

// GetScores 获取每题分值
func (s *SessionService) GetScores(ctx context.Context, uid, eid uint64) (map[uint64]float64, error) {
	key := fmt.Sprintf(scoresKeyFmt, uid, eid)

	result, err := s.rdb.HGetAll(ctx, key).Result()
	if err != nil {
		return nil, err
	}

	scores := make(map[uint64]float64, len(result))
	for k, v := range result {
		var qid uint64
		var score float64
		fmt.Sscanf(k, "%d", &qid)
		fmt.Sscanf(v, "%f", &score)
		scores[qid] = score
	}
	return scores, nil
}

// ClearSession 清除所有考试相关缓存
func (s *SessionService) ClearSession(ctx context.Context, uid, eid uint64, attempt int) error {
	keys := []string{
		fmt.Sprintf(sessionKeyFmt, uid, eid),
		fmt.Sprintf(answersKeyFmt, uid, eid),
		fmt.Sprintf(scoresKeyFmt, uid, eid),
		fmt.Sprintf(switchKeyFmt, uid, eid),
		fmt.Sprintf(pauseTimeKeyFmt, uid, eid),
		fmt.Sprintf(passwordKeyFmt, uid, eid),
	}

	member := fmt.Sprintf("%d:%d:%d", uid, eid, attempt)

	pipe := s.rdb.Pipeline()
	pipe.Del(ctx, keys...)
	pipe.ZRem(ctx, delayKey, member)
	_, err := pipe.Exec(ctx)
	return err
}

// RecordSwitchScreen 记录切屏事件
func (s *SessionService) RecordSwitchScreen(ctx context.Context, uid, eid uint64, eventType string) error {
	key := fmt.Sprintf(switchKeyFmt, uid, eid)
	event := fmt.Sprintf(`{"time":%d,"type":"%s"}`, time.Now().Unix(), eventType)
	return s.rdb.RPush(ctx, key, event).Err()
}

// GetRemainingTime 获取剩余时间（秒）
func (s *SessionService) GetRemainingTime(ctx context.Context, uid, eid uint64) (int64, error) {
	session, err := s.GetSession(ctx, uid, eid)
	if err != nil {
		return 0, err
	}

	// 检查是否暂停
	pauseKey := fmt.Sprintf("qh:exam:pause:%d", eid)
	pausedAt, err := s.rdb.Get(ctx, pauseKey).Int64()
	if err == nil && pausedAt > 0 {
		// 已暂停，返回暂停时记录的剩余时间
		pauseTimeKey := fmt.Sprintf(pauseTimeKeyFmt, uid, eid)
		return s.rdb.Get(ctx, pauseTimeKey).Int64()
	}

	elapsed := time.Now().Unix() - session.StartTime
	remaining := session.Duration - elapsed
	if remaining < 0 {
		remaining = 0
	}
	return remaining, nil
}

// GetRanking 获取成绩排名
func (s *SessionService) GetRanking(ctx context.Context, eid uint64, page, pageSize int64) ([]redis.Z, error) {
	key := fmt.Sprintf(rankingKeyFmt, eid)
	start := (page - 1) * pageSize
	stop := start + pageSize - 1

	return s.rdb.ZRevRangeWithScores(ctx, key, start, stop).Result()
}

// SetRanking 设置成绩排名
func (s *SessionService) SetRanking(ctx context.Context, eid, uid uint64, score float64) error {
	key := fmt.Sprintf(rankingKeyFmt, eid)
	return s.rdb.ZAdd(ctx, key, redis.Z{Score: score, Member: uid}).Err()
}

	return string(b)
```

## 八、延迟队列（考试超时自动交卷）

### 8.1 设计方案

使用 Redis ZSet 实现延迟队列：

- **Key**：`qh:exam:delay`
- **Score**：预期交卷时间戳（Unix 秒）
- **Member**：`{uid}:{eid}:{attempt}` 格式

后端定时任务每秒轮询 ZSet，取出 Score <= 当前时间的 Member 执行自动交卷。

### 8.2 定时任务实现

```go
// job/exam_delay_job.go

package job

import (
	"context"
	"fmt"
	"strconv"
	"strings"
	"time"

	"github.com/redis/go-redis/v9"
	"go.uber.org/zap"
)

const delayQueueKey = "qh:exam:delay"

type ExamDelayJob struct {
	rdb          *redis.Client
	examService  ExamSubmitService
	logger       *zap.Logger
}

type ExamSubmitService interface {
	AutoSubmit(ctx context.Context, uid, eid uint64, attempt int) error
}

// Start 启动延迟队列消费者
func (j *ExamDelayJob) Start(ctx context.Context) {
	ticker := time.NewTicker(1 * time.Second)
	defer ticker.Stop()

	j.logger.Info("考试延迟队列消费者启动")

	for {
		select {
		case <-ctx.Done():
			j.logger.Info("考试延迟队列消费者停止")
			return
		case <-ticker.C:
			j.process(ctx)
		}
	}
}

func (j *ExamDelayJob) process(ctx context.Context) {
	now := float64(time.Now().Unix())

	// 取出所有到期的交卷任务（ZRANGEBYSCORE）
	members, err := j.rdb.ZRangeByScore(ctx, delayQueueKey, &redis.ZRangeBy{
		Min: "-inf",
		Max: strconv.FormatFloat(now, 'f', 0, 64),
	}).Result()

	if err != nil || len(members) == 0 {
		return
	}

	for _, member := range members {
		uid, eid, attempt, err := parseDelayMember(member)
		if err != nil {
			j.logger.Error("解析延迟队列成员失败", zap.String("member", member), zap.Error(err))
			j.rdb.ZRem(ctx, delayQueueKey, member)
			continue
		}

		// 先从队列移除，防止重复消费
		j.rdb.ZRem(ctx, delayQueueKey, member)

		// 执行自动交卷
		if err := j.examService.AutoSubmit(ctx, uid, eid, attempt); err != nil {
			j.logger.Error("自动交卷失败",
				zap.Uint64("uid", uid),
				zap.Uint64("eid", eid),
				zap.Error(err),
			)
		} else {
			j.logger.Info("自动交卷成功",
				zap.Uint64("uid", uid),
				zap.Uint64("eid", eid),
			)
		}
	}
}

// parseDelayMember 解析 uid:eid:attempt
func parseDelayMember(member string) (uid, eid uint64, attempt int, err error) {
	parts := strings.Split(member, ":")
	if len(parts) != 3 {
		return 0, 0, 0, fmt.Errorf("invalid member format: %s", member)
	}

	uid, err = strconv.ParseUint(parts[0], 10, 64)
	if err != nil {
		return
	}
	eid, err = strconv.ParseUint(parts[1], 10, 64)
	if err != nil {
		return
	}
	attemptInt, err := strconv.Atoi(parts[2])
	attempt = attemptInt
	return
}
```

### 8.3 自动交卷服务

```go
// service/exam/submit_service.go

package exam

import (
	"context"
	"fmt"
	"time"
)

type SubmitService struct {
	sessionSvc  *SessionService
	recordRepo  *RecordRepo
	answerRepo  *AnswerRepo
	examRepo    *ExamRepo
	wrongRepo   *WrongQuestionRepo
}

// AutoSubmit 自动交卷（超时触发）
func (s *SubmitService) AutoSubmit(ctx context.Context, uid, eid uint64, attempt int) error {
	// 检查会话是否存在（可能已手动交卷）
	session, err := s.sessionSvc.GetSession(ctx, uid, eid)
	if err != nil {
		return nil // 会话已不存在，忽略
	}

	if session.Status == "submitted" {
		return nil // 已提交，忽略
	}

	return s.doSubmit(ctx, uid, eid, attempt, true)
}

// ManualSubmit 手动交卷
func (s *SubmitService) ManualSubmit(ctx context.Context, uid, eid uint64) error {
	session, err := s.sessionSvc.GetSession(ctx, uid, eid)
	if err != nil {
		return fmt.Errorf("考试会话不存在")
	}

	if session.Status == "submitted" {
		return fmt.Errorf("已经提交过")
	}

	return s.doSubmit(ctx, uid, eid, session.Attempt, false)
}

// doSubmit 执行交卷
func (s *SubmitService) doSubmit(ctx context.Context, uid, eid uint64, attempt int, isAuto bool) error {
	// 1. 从 Redis 读取所有答案
	answers, err := s.sessionSvc.GetAllAnswers(ctx, uid, eid)
	if err != nil {
		return err
	}

	// 2. 获取每题分值
	scores, err := s.sessionSvc.GetScores(ctx, uid, eid)
	if err != nil {
		return err
	}

	// 3. 获取考试信息
	exam, err := s.examRepo.FindByID(eid)
	if err != nil {
		return err
	}

	// 4. 获取题目信息，自动评分
	var totalScore, objScore float64
	var answerRecords []*AnswerRecord

	for qid, userAnswer := range answers {
		question, err := s.examRepo.GetQuestion(qid)
		if err != nil {
			continue
		}

		correctOptions, _ := s.examRepo.GetCorrectOptions(qid)
		questionScore := scores[qid]

		isCorrect := 0 // 0=错误
		score := float64(0)

		switch question.Type {
		case 1: // 单选
			if len(correctOptions) == 1 && userAnswer == correctOptions[0] {
				isCorrect = 1
				score = questionScore
			}
		case 2: // 多选
			if checkMultipleChoice(userAnswer, correctOptions) {
				isCorrect = 1
				score = questionScore
			}
		case 3: // 判断
			if userAnswer == correctOptions[0] {
				isCorrect = 1
				score = questionScore
			}
		case 4, 5: // 填空/简答 - 主观题待阅卷
			isCorrect = 2 // 2=待阅卷
			score = 0
		}

		totalScore += score
		if question.Type <= 3 {
			objScore += score
		}

		ar := &AnswerRecord{
			RecordID:   0, // 后续填充
			QuestionID: qid,
			Answer:     userAnswer,
			Score:      score,
			IsCorrect:  isCorrect,
		}
		answerRecords = append(answerRecords, ar)
	}

	// 5. 更新考试记录
	record, err := s.recordRepo.FindByExamAndUser(eid, uid, attempt)
	if err != nil {
		return err
	}

	now := time.Now()
	record.Score = totalScore
	record.ObjScore = objScore
	record.Status = 1 // 已交卷
	record.SubmitTime = &now
	record.DurationUsed = int(now.Sub(*record.StartTime).Seconds())
	if isAuto {
		record.SubmitType = 1 // 自动交卷
	}

	if err := s.recordRepo.Update(record); err != nil {
		return err
	}

	// 6. 批量写入答题记录
	for _, ar := range answerRecords {
		ar.RecordID = record.ID
	}
	if err := s.answerRepo.BatchCreate(answerRecords); err != nil {
		return err
	}

	// 7. 答错题目写入错题本
	for _, ar := range answerRecords {
		if ar.IsCorrect == 0 {
			s.wrongRepo.AddOrUpdate(ctx, uid, ar.QuestionID, "exam", record.ID)
		}
	}

	// 8. 写入排名（如启用）
	if exam.ShowRanking {
		s.sessionSvc.SetRanking(ctx, eid, uid, totalScore)
	}

	// 9. 清除 Redis 缓存
	s.sessionSvc.ClearSession(ctx, uid, eid, attempt)

	return nil
}

// checkMultipleChoice 检查多选题答案
func checkMultipleChoice(userAnswer string, correctOptions []string) bool {
	if len(userAnswer) == 0 || len(correctOptions) == 0 {
		return false
	}

	userSet := make(map[string]bool)
	for _, opt := range splitOptions(userAnswer) {
		userSet[opt] = true
	}

	if len(userSet) != len(correctOptions) {
		return false
	}

	for _, opt := range correctOptions {
		if !userSet[opt] {
			return false
		}
	}
	return true
}

func splitOptions(answer string) []string {
	var result []string
	for _, r := range answer {
		if r >= 'A' && r <= 'Z' {
			result = append(result, string(r))
		}
	}
	return result
}
```

### 9.3 题目/选项乱序

```go
// service/exam/shuffle_service.go

package exam

import (
	"math/rand"
	"time"
)

type ShuffleService struct{}

// ShuffleQuestions 题目乱序（Fisher-Yates 洗牌算法）
func (s *ShuffleService) ShuffleQuestions(questionIDs []uint64) []uint64 {
	result := make([]uint64, len(questionIDs))
	copy(result, questionIDs)

	r := rand.New(rand.NewSource(time.Now().UnixNano()))
	for i := len(result) - 1; i > 0; i-- {
		j := r.Intn(i + 1)
		result[i], result[j] = result[j], result[i]
	}
	return result
}

// ShuffleOptions 选项乱序，返回乱序后的选项列表和映射关系
func (s *ShuffleService) ShuffleOptions(options []string) ([]string, map[int]int) {
	n := len(options)
	indices := make([]int, n)
	for i := range indices {
		indices[i] = i
	}

	r := rand.New(rand.NewSource(time.Now().UnixNano()))
	for i := n - 1; i > 0; i-- {
		j := r.Intn(i + 1)
		indices[i], indices[j] = indices[j], indices[i]
	}

	// 构造映射：原始位置 -> 新位置
	mapping := make(map[int]int, n)
	for newPos, oldPos := range indices {
		mapping[oldPos] = newPos
	}

	shuffled := make([]string, n)
	for newPos, oldPos := range indices {
		shuffled[newPos] = options[oldPos]
	}

	return shuffled, mapping
}

// GenerateOptionOrder 为所有题目生成选项乱序映射
func (s *ShuffleService) GenerateOptionOrder(questions []QuestionInfo) map[string][]string {
	optionOrder := make(map[string][]string, len(questions))

	for _, q := range questions {
		if q.Type > 3 { // 填空/简答不需要乱序
			continue
		}
		if len(q.Options) == 0 {
			continue
		}

		_, mapping := s.ShuffleOptions(q.Options)
		// 存储映射关系：原始选项标签 -> 乱序后的位置
		order := make([]string, len(mapping))
		for oldPos, newPos := range mapping {
			order[newPos] = string(rune('A' + oldPos))
		}
		optionOrder[fmt.Sprintf("%d", q.ID)] = order
	}

	return optionOrder
}
```

```go
// service/exam/ws_monitor_service.go

package exam

import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"sync"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/gorilla/websocket"
	"github.com/redis/go-redis/v9"
	"go.uber.org/zap"
)

var upgrader = websocket.Upgrader{
	CheckOrigin: func(r *http.Request) bool {
		return true // 生产环境应校验 Origin
	},
	ReadBufferSize:  1024,
	WriteBufferSize: 1024,
}

// WsMonitorService WebSocket 考试监控服务
type WsMonitorService struct {
	rdb       *redis.Client
	logger    *zap.Logger
	examRepo  *ExamRepo

	// examID -> 教师连接集合
	clients sync.Map // map[uint64]*sync.Map (uid -> *WsClient)
}

type WsClient struct {
	Conn   *websocket.Conn
	UserID uint64
	ExamID uint64
	Send   chan []byte
}

type MonitorMessage struct {
	Type    string      `json:"type"` // student_join/submit/progress/switch_screen/exam_pause/exam_resume
	ExamID  uint64      `json:"exam_id"`
	UserID  uint64      `json:"user_id,omitempty"`
	Data    interface{} `json:"data"`
	Time    int64       `json:"time"`
}

const monitorChannel = "qh:exam:monitor"

// HandleWebSocket 处理 WebSocket 连接
// GET /ws/exam/:id/monitor
func (s *WsMonitorService) HandleWebSocket(c *gin.Context) {
	examID := parseID(c, "id")
	userID := getUserID(c)

	// 验证教师权限
	if !s.isTeacher(userID, examID) {
		c.JSON(http.StatusForbidden, gin.H{"code": 403, "message": "无权限"})
		return
	}

	conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
	if err != nil {
		s.logger.Error("WebSocket 升级失败", zap.Error(err))
		return
	}

	client := &WsClient{
		Conn:   conn,
		UserID: userID,
		ExamID: examID,
		Send:   make(chan []byte, 256),
	}

	// 注册客户端
	s.addClient(examID, userID, client)

	// 启动读写协程
	go s.writePump(client)
	go s.readPump(client)

	// 发送当前考试状态
	s.sendCurrentStatus(client)
}

// PublishMessage 发布监控消息
func (s *WsMonitorService) PublishMessage(ctx context.Context, msg *MonitorMessage) {
	msg.Time = time.Now().Unix()
	data, _ := json.Marshal(msg)
	s.rdb.Publish(ctx, monitorChannel, data)
}

// SubscribeMessages 订阅 Redis 频道并广播
func (s *WsMonitorService) SubscribeMessages(ctx context.Context) {
	pubsub := s.rdb.Subscribe(ctx, monitorChannel)
	defer pubsub.Close()

	ch := pubsub.Channel()
	for msg := range ch {
		var monitorMsg MonitorMessage
		if err := json.Unmarshal([]byte(msg.Payload), &monitorMsg); err != nil {
			continue
		}
		s.broadcastToExam(monitorMsg.ExamID, []byte(msg.Payload))
	}
}

// sendCurrentStatus 发送当前考试状态给新连接的教师
func (s *WsMonitorService) sendCurrentStatus(client *WsClient) {
	exam, _ := s.examRepo.FindByID(client.ExamID)

	status := map[string]interface{}{
		"type": "init_status",
		"data": map[string]interface{}{
			"exam_id":      client.ExamID,
			"online_count": s.getOnlineCount(client.ExamID),
			"submit_count": s.getSubmitCount(client.ExamID),
			"total_count":  exam.TotalCount,
			"status":       exam.Status,
		},
	}

	data, _ := json.Marshal(status)
	client.Send <- data
}

func (s *WsMonitorService) addClient(examID, uid uint64, client *WsClient) {
	val, _ := s.clients.LoadOrStore(examID, &sync.Map{})
	clients := val.(*sync.Map)
	clients.Store(uid, client)
}

func (s *WsMonitorService) removeClient(examID, uid uint64) {
	if val, ok := s.clients.Load(examID); ok {
		clients := val.(*sync.Map)
		clients.Delete(uid)
	}
}

func (s *WsMonitorService) broadcastToExam(examID uint64, data []byte) {
	val, ok := s.clients.Load(examID)
	if !ok {
		return
	}
	clients := val.(*sync.Map)
	clients.Range(func(key, value interface{}) bool {
		client := value.(*WsClient)
		select {
		case client.Send <- data:
		default:
			// 发送缓冲满，关闭连接
			s.removeClient(examID, client.UserID)
			client.Conn.Close()
		}
		return true
	})
}

func (s *WsMonitorService) readPump(client *WsClient) {
	defer func() {
		s.removeClient(client.ExamID, client.UserID)
		client.Conn.Close()
	}()

	client.Conn.SetReadLimit(512)
	client.Conn.SetReadDeadline(time.Now().Add(60 * time.Second))
	client.Conn.SetPongHandler(func(string) error {
		client.Conn.SetReadDeadline(time.Now().Add(60 * time.Second))
		return nil
	})

	for {
		_, _, err := client.Conn.ReadMessage()
		if err != nil {
			break
		}
	}
}

func (s *WsMonitorService) writePump(client *WsClient) {
	ticker := time.NewTicker(30 * time.Second)
	defer func() {
		ticker.Stop()
		client.Conn.Close()
	}()

	for {
		select {
		case message, ok := <-client.Send:
			if !ok {
				client.Conn.WriteMessage(websocket.CloseMessage, []byte{})
				return
			}
			client.Conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
			if err := client.Conn.WriteMessage(websocket.TextMessage, message); err != nil {
				return
			}
		case <-ticker.C:
			client.Conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
			if err := client.Conn.WriteMessage(websocket.PingMessage, nil); err != nil {
				return
			}
		}
	}
}
```

```go
type ReminderService struct {

	notifySvc    *NotificationService
	logger       *zap.Logger

// CreateReminders 创建考试提醒（创建考试时调用）
func (s *ReminderService) CreateReminders(exam *Exam) error {
	reminders := []*ExamReminder{}

	// 开始前提醒（提前30分钟）
	startRemindTime := exam.StartTime.Add(-30 * time.Minute)
	if startRemindTime.After(time.Now()) {
		reminders = append(reminders, &ExamReminder{
			ExamID:         exam.ID,
			Type:           1,
			RemindTime:     startRemindTime,
			AdvanceMinutes: 30,
			Channel:        "in_app",
		})
	}

	// 结束前提醒（提前15分钟）
	endRemindTime := exam.EndTime.Add(-15 * time.Minute)
	if endRemindTime.After(time.Now()) {
		reminders = append(reminders, &ExamReminder{
			ExamID:         exam.ID,
			Type:           2,
			RemindTime:     endRemindTime,
			AdvanceMinutes: 15,
			Channel:        "in_app",
		})
	}

	return s.reminderRepo.BatchCreate(reminders)
}

// ProcessReminders 处理待发送的提醒（定时任务调用）
func (s *ReminderService) ProcessReminders(ctx context.Context) {
	now := time.Now()
	reminders, err := s.reminderRepo.FindPendingBefore(now)
	if err != nil {
		s.logger.Error("查询待发送提醒失败", zap.Error(err))
		return
	}

	for _, reminder := range reminders {
		exam, err := s.examRepo.FindByID(reminder.ExamID)
		if err != nil {
			continue
		}

		// 获取目标用户列表
		userIDs := s.getTargetUsers(ctx, exam)

		// 构造通知内容
		var title, content string
		if reminder.Type == 1 {
			title = "考试即将开始"
			content = fmt.Sprintf("考试「%s」将在%d分钟后开始，请做好准备", exam.Title, reminder.AdvanceMinutes)
		} else {
			title = "考试即将结束"
			content = fmt.Sprintf("考试「%s」将在%d分钟后结束，请及时提交", exam.Title, reminder.AdvanceMinutes)
		}

		// 发送通知
		for _, uid := range userIDs {
			s.notifySvc.Send(ctx, uid, "exam_remind", title, content)
		}

		// 更新状态
		s.reminderRepo.UpdateStatus(reminder.ID, 1)
	}
}

// getTargetUsers 获取需要提醒的用户
func (s *ReminderService) getTargetUsers(ctx context.Context, exam *Exam) []uint64 {
	if exam.ClassID > 0 {
		// 班级考试：班级成员
		return s.examRepo.GetClassMemberIDs(exam.ClassID)
	}
	// 公共考试：所有已开始但未提交的用户
	return s.recordRepo.GetInProgressUserIDs(exam.ID)
}
```

```go
// service/exam/extend_service.go

package exam

import (
	"context"
	"fmt"
	"time"
)

type ExtendService struct {
	examRepo    *ExamRepo
	extendRepo  *ExtendRepo
	sessionSvc  *SessionService
	rdb         *redis.Client
}

const delayQueueKey = "qh:exam:delay"

// ExtendExam 延长考试截止时间
func (s *ExtendService) ExtendExam(ctx context.Context, examID, operatorID uint64, req *ExtendRequest) error {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return fmt.Errorf("考试不存在")
	}

	if exam.Status == 2 {
		return fmt.Errorf("考试已结束，无法延期")
	}

	// 记录原始截止时间
	originalEndTime := exam.EndTime
	newEndTime := originalEndTime.Add(time.Duration(req.ExtendMinutes) * time.Minute)

	// 更新考试截止时间
	exam.EndTime = newEndTime
	exam.OriginalEndTime = &originalEndTime
	exam.ExtendReason = req.Reason
	if err := s.examRepo.Update(exam); err != nil {
		return err
	}

	// 记录延期操作
	extension := &ExamExtension{
		ExamID:          examID,
		OperatorID:      operatorID,
		OriginalEndTime: originalEndTime,
		NewEndTime:      newEndTime,
		ExtendMinutes:   req.ExtendMinutes,
		Reason:          req.Reason,
	}
	s.extendRepo.Create(extension)

	// 更新延迟队列中未交卷考生的交卷时间
	s.updateDelayQueue(ctx, examID, req.ExtendMinutes)

	return nil
}

// updateDelayQueue 更新延迟队列中相关考生的交卷时间
func (s *ExtendService) updateDelayQueue(ctx context.Context, examID uint64, extendMinutes int) {
	// 遍历延迟队列，找到该考试的所有成员
	members, err := s.rdb.ZRange(ctx, delayQueueKey, 0, -1).Result()
	if err != nil {
		return
	}

	extendSec := float64(extendMinutes * 60)

	for _, member := range members {
		uid, eid, attempt, err := parseDelayMember(member)
		if err != nil || eid != examID {
			continue
		}

		// 获取当前 Score（交卷时间），加上延期秒数
		currentScore, err := s.rdb.ZScore(ctx, delayQueueKey, member).Result()
		if err != nil {
			continue
		}

		// 更新 Score
		newScore := currentScore + extendSec
		s.rdb.ZAdd(ctx, delayQueueKey, redis.Z{Score: newScore, Member: member})

		// 更新 Redis 中的会话时长
		sessionKey := fmt.Sprintf("qh:exam:session:%d:%d", uid, eid)
		s.rdb.HIncrBy(ctx, sessionKey, "duration", int64(extendMinutes)*60)
	}
}

type ExtendRequest struct {
	ExtendMinutes int    `json:"extend_minutes" binding:"required,min=1"`
	Reason        string `json:"reason" binding:"required"`
}
```

```go
// service/exam/pause_service.go

package exam

import (
	"context"
	"fmt"
	"time"
)

type PauseService struct {
	examRepo   *ExamRepo
	pauseRepo  *PauseRepo
	sessionSvc *SessionService
	wsSvc      *WsMonitorService
}

// PauseExam 暂停考试
func (s *PauseService) PauseExam(ctx context.Context, examID, operatorID uint64, reason string) error {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return fmt.Errorf("考试不存在")
	}

	if exam.StatusPause {
		return fmt.Errorf("考试已经暂停")
	}

	now := time.Now()

	// 更新考试状态
	exam.StatusPause = true
	s.examRepo.Update(exam)

	// 记录暂停操作
	pause := &ExamPause{
		ExamID:      examID,
		OperatorID:  operatorID,
		Action:      1,
		PauseAt:     &now,
		Reason:      reason,
	}
	s.pauseRepo.Create(pause)

	// 在 Redis 中记录暂停时间
	pauseKey := fmt.Sprintf("qh:exam:pause:%d", examID)
	s.sessionSvc.rdb.Set(ctx, pauseKey, now.Unix(), 0)

	// 保存所有在线考生的剩余时间
	s.saveRemainingTimeForAll(ctx, examID)

	// 通过 WebSocket 通知所有在线考生
	s.wsSvc.PublishMessage(ctx, &MonitorMessage{
		Type:   "exam_pause",
		ExamID: examID,
		Data: map[string]interface{}{
			"reason": reason,
		},
	})

	return nil
}

// ResumeExam 恢复考试
func (s *PauseService) ResumeExam(ctx context.Context, examID, operatorID uint64) error {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return fmt.Errorf("考试不存在")
	}

	if !exam.StatusPause {
		return fmt.Errorf("考试未暂停")
	}

	now := time.Now()

	// 更新考试状态
	exam.StatusPause = false
	s.examRepo.Update(exam)

	// 更新暂停记录
	pauseRecord, _ := s.pauseRepo.FindLastPause(examID)
	if pauseRecord != nil {
		pauseRecord.ResumeAt = &now
		pauseRecord.PausedSeconds = int(now.Sub(*pauseRecord.PauseAt).Seconds())
		s.pauseRepo.Update(pauseRecord)
	}

	// 清除 Redis 暂停标记
	pauseKey := fmt.Sprintf("qh:exam:pause:%d", examID)
	s.sessionSvc.rdb.Del(ctx, pauseKey)

	// 恢复所有考生的倒计时
	s.restoreRemainingTimeForAll(ctx, examID)

	// 延长延迟队列中的交卷时间
	s.extendDelayQueue(ctx, examID, pauseRecord.PausedSeconds)

	// 通知所有在线考生
	s.wsSvc.PublishMessage(ctx, &MonitorMessage{
		Type:   "exam_resume",
		ExamID: examID,
		Data: map[string]interface{}{
			"paused_seconds": pauseRecord.PausedSeconds,
		},
	})

	return nil
}

// saveRemainingTimeForAll 保存所有在线考生的剩余时间
func (s *PauseService) saveRemainingTimeForAll(ctx context.Context, examID uint64) {
	records, _ := s.recordRepo.FindInProgress(examID)
	for _, record := range records {
		remaining, err := s.sessionSvc.GetRemainingTime(ctx, record.UserID, examID)
		if err != nil {
			continue
		}
		key := fmt.Sprintf("qh:exam:pause:time:%d:%d", record.UserID, examID)
		s.sessionSvc.rdb.Set(ctx, key, remaining, 24*time.Hour)
	}
}

// restoreRemainingTimeForAll 恢复所有考生的倒计时
func (s *PauseService) restoreRemainingTimeForAll(ctx context.Context, examID uint64) {
	records, _ := s.recordRepo.FindInProgress(examID)
	for _, record := range records {
		key := fmt.Sprintf("qh:exam:pause:time:%d:%d", record.UserID, examID)
		remaining, err := s.sessionSvc.rdb.Get(ctx, key).Int64()
		if err != nil {
			continue
		}

		// 更新会话的开始时间（向后推迟暂停时长）
		sessionKey := fmt.Sprintf("qh:exam:session:%d:%d", record.UserID, examID)
		newStartTime := time.Now().Unix() - (s.getDuration(record.UserID, examID) - remaining)
		s.sessionSvc.rdb.HSet(ctx, sessionKey, "start_time", newStartTime)

		// 清除暂停时间记录
		s.sessionSvc.rdb.Del(ctx, key)
	}
}

// extendDelayQueue 延长延迟队列交卷时间
func (s *PauseService) extendDelayQueue(ctx context.Context, examID uint64, pausedSeconds int) {
	members, _ := s.sessionSvc.rdb.ZRange(ctx, delayQueueKey, 0, -1).Result()
	extendSec := float64(pausedSeconds)

	for _, member := range members {
		_, eid, _, err := parseDelayMember(member)
		if err != nil || eid != examID {
			continue
		}
		currentScore, err := s.sessionSvc.rdb.ZScore(ctx, delayQueueKey, member).Result()
		if err != nil {
			continue
		}
		s.sessionSvc.rdb.ZAdd(ctx, delayQueueKey, redis.Z{
			Score:  currentScore + extendSec,
			Member: member,
		})
	}
}
```

```go
// ReviewScore 教师审核复核
func (s *ScoreReviewService) ReviewScore(ctx context.Context, reviewID, reviewerID uint64, req *ReviewRequest) error {
	review, err := s.reviewRepo.FindByID(reviewID)
	if err != nil {
		return fmt.Errorf("复核记录不存在")
	}

	if review.Status != 0 {
		return fmt.Errorf("该复核已处理")
	}

	now := time.Now()
	review.Status = req.Status
	review.ReviewerID = reviewerID
	review.ReviewComment = req.Comment
	review.ReviewedAt = &now

	if req.Status == 1 && req.NewScore != nil {
		review.NewScore = *req.NewScore

		// 更新考试记录分数
		record, _ := s.recordRepo.FindByID(review.RecordID)
		if record != nil {
			record.Score = *req.NewScore
			s.recordRepo.Update(record)
		}
	}

	if err := s.reviewRepo.Update(review); err != nil {
		return err
	}

	// 通知学生
	var notifyContent string
	if req.Status == 1 {

	return nil
}
```

```go
	"context"

	"fmt"
```

```go
type PasswordService struct {
	examRepo    *ExamRepo
	sessionSvc  *SessionService
}

// VerifyPassword 验证考试密码
func (s *PasswordService) VerifyPassword(ctx context.Context, examID, userID uint64, password string) error {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return fmt.Errorf("考试不存在")
	}

	if exam.ExamPassword == "" {
		return nil // 无密码限制
	}

	// 比对密码（哈希比对）
	inputHash := hashPassword(password)
	if inputHash != exam.ExamPassword {
		return fmt.Errorf("密码错误")
	}

	// 验证通过，写入 Redis 标记
	key := fmt.Sprintf(passwordKeyFmt, userID, examID)
	s.sessionSvc.rdb.Set(ctx, key, "1", 24*time.Hour)

	return nil
}

// IsPasswordVerified 检查密码是否已验证
func (s *PasswordService) IsPasswordVerified(ctx context.Context, examID, userID uint64) bool {
	key := fmt.Sprintf(passwordKeyFmt, userID, examID)
	result, _ := s.sessionSvc.rdb.Get(ctx, key).Result()
	return result == "1"
}

// SetExamPassword 设置考试密码（创建/编辑考试时）
func (s *PasswordService) SetExamPassword(examID uint64, password string) error {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return err
	}

	if password == "" {
		exam.ExamPassword = ""
	} else {
		exam.ExamPassword = hashPassword(password)
	}

	return s.examRepo.Update(exam)
}

func hashPassword(password string) string {
	h := sha256.New()
	h.Write([]byte(password))
	return hex.EncodeToString(h.Sum(nil))
}
```

```go
// service/exam/ranking_service.go

package exam

import (
	"context"
	"fmt"
	"time"

	"github.com/redis/go-redis/v9"
)

type RankingService struct {
	rdb         *redis.Client
	rankingRepo *RankingRepo
	examRepo    *ExamRepo
	recordRepo  *RecordRepo
}

const rankingKeyFmt = "qh:exam:ranking:%d"

// GetRanking 获取成绩排名
func (s *RankingService) GetRanking(ctx context.Context, examID uint64, page, pageSize int) ([]*RankingItem, int64, error) {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return nil, 0, err
	}

	if !exam.ShowRanking {
		return nil, 0, fmt.Errorf("该考试未开启排名")
	}

	key := fmt.Sprintf(rankingKeyFmt, examID)
	start := int64((page - 1) * pageSize)
	stop := start + int64(pageSize) - 1

	// 从 Redis ZSet 获取排名
	zs, err := s.rdb.ZRevRangeWithScores(ctx, key, start, stop).Result()
	if err != nil || len(zs) == 0 {
		// Redis 无数据，从数据库查询
		return s.getRankingFromDB(ctx, examID, page, pageSize)
	}

	total, _ := s.rdb.ZCard(ctx, key).Result()

	items := make([]*RankingItem, 0, len(zs))
	for i, z := range zs {
		uid := uint64(z.Member.(uint64))
		user, _ := s.getUserInfo(uid)
		record, _ := s.recordRepo.FindByExamAndUser(examID, uid, 0)

		items = append(items, &RankingItem{
			Rank:         int(start) + i + 1,
			UserID:       uid,
			Username:     user.Username,
			Nickname:     user.Nickname,
			Avatar:       user.Avatar,
			Score:        z.Score,
			DurationUsed: record.DurationUsed,
		})
	}

	return items, total, nil
}

// GetMyRank 查看我的排名
func (s *RankingService) GetMyRank(ctx context.Context, examID, userID uint64) (*MyRankInfo, error) {
	key := fmt.Sprintf(rankingKeyFmt, examID)

	// 获取排名（0-based）
	rank, err := s.rdb.ZRevRank(ctx, key, userID).Result()
	if err != nil {
		return &MyRankInfo{Rank: -1}, nil
	}

	score, _ := s.rdb.ZScore(ctx, key, userID).Result()
	record, _ := s.recordRepo.FindByExamAndUser(examID, userID, 0)

	return &MyRankInfo{
		Rank:         int(rank) + 1,
		Score:        score,
		DurationUsed: record.DurationUsed,
	}, nil
}

// BuildRanking 构建排名（考试结束后调用）
func (s *RankingService) BuildRanking(ctx context.Context, examID uint64) error {
	exam, err := s.examRepo.FindByID(examID)
	if err != nil || !exam.ShowRanking {
		return nil
	}

	// 获取所有已交卷的记录
	records, err := s.recordRepo.FindSubmitted(examID)
	if err != nil {
		return err
	}

	key := fmt.Sprintf(rankingKeyFmt, examID)

	// 清空旧排名
	s.rdb.Del(ctx, key)

	// 写入排名
	for _, record := range records {
		s.rdb.ZAdd(ctx, key, redis.Z{
			Score:  float64(record.Score),
			Member: record.UserID,
		})

		// 持久化到数据库
		s.rankingRepo.Upsert(&ExamRanking{
			ExamID:       examID,
			UserID:       record.UserID,
			Score:        record.Score,
			DurationUsed: record.DurationUsed,
		})
	}

	// 计算排名序号
	s.rankingRepo.CalculateRankNumbers(examID)

	// 设置过期时间
	s.rdb.Expire(ctx, key, 7*24*time.Hour)

	return nil
}

// getRankingFromDB 从数据库获取排名
func (s *RankingService) getRankingFromDB(ctx context.Context, examID uint64, page, pageSize int) ([]*RankingItem, int64, error) {
	rankings, total, err := s.rankingRepo.List(examID, page, pageSize)
	if err != nil {
		return nil, 0, err
	}

	items := make([]*RankingItem, 0, len(rankings))
	for _, r := range rankings {
		user, _ := s.getUserInfo(r.UserID)
		items = append(items, &RankingItem{
			Rank:         r.RankNum,
			UserID:       r.UserID,
			Username:     user.Username,
			Nickname:     user.Nickname,
			Avatar:       user.Avatar,
			Score:        float64(r.Score),
			DurationUsed: r.DurationUsed,
		})
	}

	return items, total, nil
}

type RankingItem struct {
	Rank         int     `json:"rank"`
	UserID       uint64  `json:"user_id"`
	Username     string  `json:"username"`
	Nickname     string  `json:"nickname"`
	Avatar       string  `json:"avatar"`
	Score        float64 `json:"score"`
	DurationUsed int     `json:"duration_used"`
}

type MyRankInfo struct {
	Rank         int     `json:"rank"`
	Score        float64 `json:"score"`
	DurationUsed int     `json:"duration_used"`
}
```

## 十八、断线续考

### 18.1 设计原理

考生意外退出（网络断开、应用崩溃、浏览器关闭）后重新进入考试：

1. 检查 Redis 中是否存在考试会话 `qh:exam:session:{uid}:{eid}`
2. 会话存在：计算剩余时间 = duration - (now - start_time)
3. 从 Redis Hash 读取已保存的答案
4. 返回题目列表（按原始乱序）+ 已选答案 + 剩余时间
5. 会话不存在：检查数据库中是否有进行中的记录，如有则说明超时已自动交卷

**响应**：

```go
// service/exam/resume_service.go

package exam

import (
	"context"
	"fmt"
	"time"
)

type ResumeService struct {
	sessionSvc *SessionService
	examRepo   *ExamRepo
	recordRepo *RecordRepo
}

// ResumeExam 断线续考 - 恢复考试状态
func (s *ResumeService) ResumeExam(ctx context.Context, examID, userID uint64) (*ResumeInfo, error) {
	// 1. 检查 Redis 会话是否存在
	session, err := s.sessionSvc.GetSession(ctx, userID, examID)
	if err != nil {
		// Redis 会话不存在，检查数据库
		return s.handleExpiredSession(ctx, examID, userID)
	}

	if session.Status == "submitted" {
		return nil, fmt.Errorf("考试已提交")
	}

	// 2. 检查考试是否暂停
	exam, err := s.examRepo.FindByID(examID)
	if err != nil {
		return nil, fmt.Errorf("考试不存在")
	}

	isPaused := exam.StatusPause

	// 3. 计算剩余时间
	var remainingSeconds int64
	if isPaused {
		// 暂停中，返回暂停时记录的剩余时间
		remainingSeconds, _ = s.sessionSvc.GetRemainingTime(ctx, userID, examID)
	} else {
		elapsed := time.Now().Unix() - session.StartTime
		remainingSeconds = session.Duration - elapsed
		if remainingSeconds <= 0 {
			// 超时，触发自动交卷
			return nil, fmt.Errorf("考试已超时")
		}
	}

	// 4. 获取已保存的答案
	answers, _ := s.sessionSvc.GetAllAnswers(ctx, userID, examID)

	// 5. 获取题目信息（按原始乱序顺序）
	questions, err := s.getQuestionsInOrder(ctx, examID, session.QuestionOrder, session.OptionOrder)
	if err != nil {
		return nil, err
	}

	// 6. 获取每题分值
	scores, _ := s.sessionSvc.GetScores(ctx, userID, examID)

	// 7. 合并分值到题目信息
	for i := range questions {
		if score, ok := scores[questions[i].ID]; ok {
			questions[i].Score = score
		}
	}

	return &ResumeInfo{
		RecordID:         session.RecordID,
		Exam:             exam,
		RemainingSeconds: remainingSeconds,
		Questions:        questions,
		Answers:          answers,
		IsPaused:         isPaused,
	}, nil
}

// handleExpiredSession 处理已过期的会话
func (s *ResumeService) handleExpiredSession(ctx context.Context, examID, userID uint64) (*ResumeInfo, error) {
	// 查询数据库中是否有进行中的记录
	record, err := s.recordRepo.FindInProgressByExamAndUser(examID, userID)
	if err != nil {
		return nil, fmt.Errorf("没有进行中的考试记录")
	}

	if record.Status == 1 || record.Status == 2 {
		return nil, fmt.Errorf("考试已提交")
	}

	// 记录存在但 Redis 会话已过期（可能超时自动交卷了）
	return nil, fmt.Errorf("考试会话已过期，可能已自动交卷")
}

// getQuestionsInOrder 按乱序顺序获取题目
func (s *ResumeService) getQuestionsInOrder(ctx context.Context, examID uint64, questionOrder []uint64, optionOrder map[string][]string) ([]QuestionInfo, error) {
	// 获取试卷的所有题目
	allQuestions, err := s.examRepo.GetExamQuestions(examID)
	if err != nil {
		return nil, err
	}

	// 按 questionOrder 排序
	questionMap := make(map[uint64]*QuestionInfo, len(allQuestions))
	for i := range allQuestions {
		questionMap[allQuestions[i].ID] = &allQuestions[i]
	}

	result := make([]QuestionInfo, 0, len(questionOrder))
	for _, qid := range questionOrder {
		if q, ok := questionMap[qid]; ok {
			// 应用选项乱序
			qKey := fmt.Sprintf("%d", qid)
			if order, exists := optionOrder[qKey]; exists && q.Type <= 3 {
				q.Options = applyOptionOrder(q.Options, order)
			}
			result = append(result, *q)
		}
	}

	return result, nil
}

// applyOptionOrder 应用选项乱序
func applyOptionOrder(options []string, order []string) []string {
	if len(options) != len(order) {
		return options
	}

	result := make([]string, len(options))
	for i, label := range order {
		idx := int(label[0] - 'A')
		if idx >= 0 && idx < len(options) {
			result[i] = options[idx]
		}
	}
	return result
}

type ResumeInfo struct {
	RecordID         uint64            `json:"record_id"`
	Exam             *Exam             `json:"exam"`
	RemainingSeconds int64             `json:"remaining_seconds"`
	Questions        []QuestionInfo    `json:"questions"`
	Answers          map[uint64]string `json:"answers"`
	IsPaused         bool              `json:"is_paused"`
}
```

## 二十、定时任务汇总

| 任务 | 执行频率 | 说明 |
|------|---------|------|
| 考试超时交卷 | 每秒 | 轮询 `qh:exam:delay` ZSet，处理到期的自动交卷 |
| 考试提醒 | 每分钟 | 查询 `exam_reminders` 表，发送到期提醒 |
| 考试状态更新 | 每分钟 | 更新考试状态（未发布->进行中->已结束） |
| 排名构建 | 考试结束后 | 异步构建成绩排名（Redis ZSet + DB） |
| 异常行为汇总 | 每5分钟 | 汇总切屏、IP变更等异常数据 |
| 过期会话清理 | 每小时 | 清理 Redis 中已过期但未清除的考试会话 |
| 公告缓存刷新 | 公告变更时 | 更新 Redis 中的公告缓存 |

```go
// job/scheduler.go

package job

import (
	"context"
	"time"
)

type Scheduler struct {
	delayJob     *ExamDelayJob
	reminderJob  *ReminderJob
	statusJob    *ExamStatusJob
}

func (s *Scheduler) Start(ctx context.Context) {
	// 考试超时交卷 - 每秒
	go s.delayJob.Start(ctx)

	// 考试提醒 - 每分钟
	go func() {
		ticker := time.NewTicker(1 * time.Minute)
		defer ticker.Stop()
		for {
			select {
			case <-ctx.Done():
				return
			case <-ticker.C:
				s.reminderJob.Process(ctx)
			}
		}
	}()

	// 考试状态更新 - 每分钟
	go func() {
		ticker := time.NewTicker(1 * time.Minute)
		defer ticker.Stop()
		for {
			select {
			case <-ctx.Done():
				return
			case <-ticker.C:
				s.statusJob.UpdateStatus(ctx)
			}
		}
	}()
}
```

```go
// ExportScores 导出成绩
func (s *ScoreExportService) ExportScores(examID uint64, req *ExportScoresRequest) ([]byte, string, error) {
    exam, err := s.examRepo.FindByID(examID)
    if err != nil {
        return nil, "", fmt.Errorf("考试不存在")
    }

    // 获取所有考试记录

    }

// exportSummary 导出汇总成绩
func (s *ScoreExportService) exportSummary(exam *Exam, records []*ExamRecord) ([]byte, string, error) {
    f := excelize.NewFile()
    sheet := "成绩汇总"
    f.NewSheet(sheet)

    // 设置表头
    headers := []string{"序号", "学号", "姓名", "得分", "客观题得分", "用时(分钟)", "交卷方式", "状态", "提交时间"}
    for i, h := range headers {
        cell, _ := excelize.CoordinatesToCellName(i+1, 1)
        f.SetCellValue(sheet, cell, h)
    }

    // 填充数据
    for i, record := range records {
        row := i + 2
        f.SetCellValue(sheet, fmt.Sprintf("A%d", row), i+1)
        f.SetCellValue(sheet, fmt.Sprintf("B%d", row), record.User.StudentID)
        f.SetCellValue(sheet, fmt.Sprintf("C%d", row), record.User.Nickname)
        f.SetCellValue(sheet, fmt.Sprintf("D%d", row), record.Score)
        f.SetCellValue(sheet, fmt.Sprintf("E%d", row), record.ObjScore)
        f.SetCellValue(sheet, fmt.Sprintf("F%d", row), float64(record.DurationUsed)/60.0)

        submitType := "手动交卷"
        if record.SubmitType == 1 {
            submitType = "超时自动交卷"
        }
        f.SetCellValue(sheet, fmt.Sprintf("G%d", row), submitType)

        statusText := "已交卷"
        if record.Status == 2 {
            statusText = "已阅卷"
        }
        f.SetCellValue(sheet, fmt.Sprintf("H%d", row), statusText)

        if record.SubmitTime != nil {
            f.SetCellValue(sheet, fmt.Sprintf("I%d", row), record.SubmitTime.Format("2006-01-02 15:04:05"))
        }
    }

    // 添加统计行
    statsRow := len(records) + 3
    f.SetCellValue(sheet, fmt.Sprintf("A%d", statsRow), "统计")
    f.SetCellValue(sheet, fmt.Sprintf("B%d", statsRow), fmt.Sprintf("参考人数: %d", len(records)))

    var totalScore float64
    var passCount int
    for _, r := range records {
        totalScore += float64(r.Score)
        if r.Score >= exam.PassScore {
            passCount++
        }
    }
    if len(records) > 0 {
        f.SetCellValue(sheet, fmt.Sprintf("C%d", statsRow), fmt.Sprintf("平均分: %.2f", totalScore/float64(len(records))))
        f.SetCellValue(sheet, fmt.Sprintf("D%d", statsRow), fmt.Sprintf("及格率: %.1f%%", float64(passCount)/float64(len(records))*100))
    }

    var buf bytes.Buffer
    f.Write(&buf)
    return buf.Bytes(), "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet", nil
}
```

```go
type ReviewService struct {
    recordRepo  *RecordRepo
    answerRepo  *AnswerRepo
    notifySvc   *NotificationService
}

// ReviewAnswers 阅卷（含评语）
func (s *ReviewService) ReviewAnswers(examID uint64, req *ReviewAnswersRequest, reviewerID uint64) error {
    record, err := s.recordRepo.FindByID(req.RecordID)
    if err != nil {
        return fmt.Errorf("考试记录不存在")
    }

    var totalScore float64
    for _, ans := range req.Answers {
        answerRecord, err := s.answerRepo.FindByRecordAndQuestion(req.RecordID, ans.QuestionID)
        if err != nil {
            continue
        }

        answerRecord.Score = ans.Score
        answerRecord.Comment = ans.Comment
        answerRecord.ReviewerID = &reviewerID
        now := time.Now()
        answerRecord.ReviewedAt = &now

        // 判断是否正确
        if ans.Score > 0 {
            answerRecord.IsCorrect = 1
        }

        s.answerRepo.Update(answerRecord)
        totalScore += ans.Score
    }

    // 更新总分（客观题 + 主观题）
    record.Score = record.ObjScore + totalScore
    record.Status = 2 // 已阅卷
    record.OverallComment = req.OverallComment
    s.recordRepo.Update(record)

    // 通知学生
    s.notifySvc.Send(context.Background(), record.UserID, "exam_graded",
        "试卷已批阅",
        fmt.Sprintf("您的试卷已批阅完成，得分: %.2f", record.Score))

    return nil
}

type ReviewAnswersRequest struct {
    RecordID       uint64            `json:"record_id" binding:"required"`
    Answers        []*ReviewAnswer   `json:"answers" binding:"required"`
    OverallComment string            `json:"overall_comment"`
}

type ReviewAnswer struct {
    QuestionID uint64  `json:"question_id" binding:"required"`
    Score      float64 `json:"score" binding:"min=0"`
```

```go
// GetReviewProgress 获取阅卷进度
func (s *ReviewProgressService) GetReviewProgress(examID uint64) (*ReviewProgress, error) {
    // 获取所有已交卷的记录
    records, err := s.recordRepo.FindSubmitted(examID)
    if err != nil {
        return nil, err
    }

    progress := &ReviewProgress{
        Total: len(records),
    }

    reviewedCount := 0
    questionStats := make(map[uint64]*QuestionReviewStat)
    reviewerStats := make(map[uint64]*ReviewerStat)

    for _, record := range records {
        if record.Status == 2 {
            reviewedCount++
        }

        // 获取该记录的所有答题
        answers, _ := s.answerRepo.FindByRecordID(record.ID)
        for _, answer := range answers {
            if answer.IsCorrect == 2 { // 待阅卷
                qs, ok := questionStats[answer.QuestionID]
                if !ok {
                    qs = &QuestionReviewStat{QuestionID: answer.QuestionID}
                    questionStats[answer.QuestionID] = qs
                }
                qs.Total++
                if answer.ReviewerID != nil {
                    qs.Rereviewed++
                } else {
                    qs.Unreviewed++
                }
            }

            // 统计阅卷人工作量
            if answer.ReviewerID != nil {
                rs, ok := reviewerStats[*answer.ReviewerID]
                if !ok {
                    rs = &ReviewerStat{ReviewerID: *answer.ReviewerID}
                    reviewerStats[*answer.ReviewerID] = rs
                }
                rs.ReviewedCount++
            }
        }
    }

    progress.Reviewed = reviewedCount
    progress.Unreviewed = progress.Total - reviewedCount
    if progress.Total > 0 {
        progress.Progress = float64(reviewedCount) / float64(progress.Total) * 100
    }

    return progress, nil
}
```

```go
// GetExamAnalysis 获取考试分析报告
func (s *AnalysisService) GetExamAnalysis(examID uint64) (*ExamAnalysis, error) {
    exam, err := s.examRepo.FindByID(examID)
    if err != nil {
        return nil, fmt.Errorf("考试不存在")
    }

    records, err := s.recordRepo.FindSubmitted(examID)
    if err != nil {
        return nil, err
    }

    analysis := &ExamAnalysis{
        ExamID:    examID,
        ExamTitle: exam.Title,
    }

    if len(records) == 0 {
        return analysis, nil
    }

    // 计算汇总统计
    scores := make([]float64, 0, len(records))
    passCount, excellentCount := 0, 0
    for _, r := range records {
        scores = append(scores, float64(r.Score))
        if r.Score >= exam.PassScore {
            passCount++
        }
        if r.Score >= exam.TotalScore*0.9 {
            excellentCount++
        }
    }

    sort.Float64s(scores)
    n := float64(len(scores))
    var sum float64
    for _, s := range scores {
        sum += s
    }
    avg := sum / n

    // 标准差
    var variance float64
    for _, s := range scores {
        variance += (s - avg) * (s - avg)
    }
    stdDev := math.Sqrt(variance / n)

    analysis.Summary = &ExamSummary{
        TotalStudents: len(records),
        Submitted:     len(records),
        AvgScore:      avg,
        MedianScore:   calculateMedian(scores),
        MaxScore:      scores[len(scores)-1],
        MinScore:      scores[0],
        StdDeviation:  stdDev,
        PassRate:      float64(passCount) / n * 100,
        ExcellentRate: float64(excellentCount) / n * 100,
    }

    // 分数段分布
    analysis.ScoreDistribution = calculateDistribution(scores, exam.TotalScore)

    // 各题分析
    analysis.QuestionAnalysis = s.analyzeQuestions(examID, records)

    return analysis, nil
}

// analyzeQuestions 分析每道题
func (s *AnalysisService) analyzeQuestions(examID uint64, records []*ExamRecord) []*QuestionAnalysis {
    var analyses []*QuestionAnalysis

    // 获取试卷所有题目
    questions, _ := s.examRepo.GetExamQuestions(examID)

    for _, q := range questions {
        qa := &QuestionAnalysis{
            QuestionID:    q.ID,
            QuestionTitle: q.Title,
            Type:          q.Type,
            FullScore:     q.Score,
        }

        // 获取所有该题的答题记录
        answers, _ := s.answerRepo.FindByQuestionAndExam(q.ID, examID)

        correctCount := 0
        var totalScore float64
        answerDist := make(map[string]int)

        for _, ans := range answers {
            totalScore += float64(ans.Score)
            if ans.IsCorrect == 1 {
                correctCount++
            }
            answerDist[ans.Answer]++
        }

        total := len(answers)
        if total > 0 {
            qa.CorrectCount = correctCount
            qa.WrongCount = total - correctCount
            qa.CorrectRate = float64(correctCount) / float64(total) * 100
            qa.AvgScore = totalScore / float64(total)
            qa.DifficultyCoefficient = qa.AvgScore / q.Score

            // 难度等级
            switch {
            case qa.DifficultyCoefficient >= 0.8:
                qa.DifficultyLevel = "简单"
            case qa.DifficultyCoefficient >= 0.5:
                qa.DifficultyLevel = "中等"
            default:
                qa.DifficultyLevel = "困难"
            }

            // 选项分布（选择题）
            if q.Type <= 3 {
                for opt, count := range answerDist {
                    qa.AnswerDistribution = append(qa.AnswerDistribution, &AnswerDistItem{
                        Option:     opt,
                        Count:      count,
                        Percentage: float64(count) / float64(total) * 100,
                    })
                }
            }
        }

        analyses = append(analyses, qa)
    }

    return analyses
}

func calculateMedian(sorted []float64) float64 {
    n := len(sorted)
    if n == 0 {
        return 0
    }
    if n%2 == 0 {
        return (sorted[n/2-1] + sorted[n/2]) / 2
    }
    return sorted[n/2]
}
```

```go
// CreateRemindersWithCustom 创建自定义提醒
func (s *ReminderService) CreateRemindersWithCustom(exam *Exam, customReminders []*CustomReminder) error {
    reminders := []*ExamReminder{}

    if len(customReminders) > 0 {
        // 使用教师自定义的提醒配置
        for _, cr := range customReminders {
            var remindTime time.Time
            if cr.Type == 1 {
                remindTime = exam.StartTime.Add(-time.Duration(cr.AdvanceMinutes) * time.Minute)
            } else {
                remindTime = exam.EndTime.Add(-time.Duration(cr.AdvanceMinutes) * time.Minute)
            }

            if remindTime.After(time.Now()) {
                reminders = append(reminders, &ExamReminder{
                    ExamID:         exam.ID,
                    Type:           cr.Type,
                    RemindTime:     remindTime,
                    AdvanceMinutes: cr.AdvanceMinutes,
                    Channel:        cr.Channel,
                })
            }
        }
    } else {
        // 使用默认提醒配置
        return s.CreateReminders(exam)
    }

    return s.reminderRepo.BatchCreate(reminders)
}
```

```vue
<script setup lang="ts">
import { ref, watch } from 'vue'

export function useCountdownAlert(remainingSeconds: Ref<number>) {
  const alertTriggered = ref(new Set<number>())

  const alertMilestones = [15 * 60, 5 * 60, 60, 30] // 15分钟、5分钟、1分钟、30秒

  watch(remainingSeconds, (newVal) => {
    for (const milestone of alertMilestones) {
      if (newVal <= milestone && !alertTriggered.value.has(milestone)) {
        alertTriggered.value.add(milestone)
        showAlert(milestone)
      }
    }
  })

  const showAlert = (seconds: number) => {
    let title = '时间提醒'
    let content = ''

    if (seconds >= 60) {
      const minutes = Math.floor(seconds / 60)
      content = `距离考试结束还剩 ${minutes} 分钟，请合理安排时间！`
    } else {
      content = `距离考试结束还剩 ${seconds} 秒，请尽快提交！`
    }

    uni.showModal({
      title,
      content,
      showCancel: false,
      confirmText: '知道了'
    })
  }

  return { alertTriggered }
}
</script>
```

```go
func (s *SessionService) SaveDraft(ctx context.Context, uid, eid uint64, questionID uint64, draft string) error {
    key := fmt.Sprintf("qh:exam:drafts:%d:%d", uid, eid)
    pipe := s.rdb.Pipeline()
    pipe.HSet(ctx, key, fmt.Sprintf("%d", questionID), draft)
    pipe.Expire(ctx, key, 24*time.Hour)
    _, err := pipe.Exec(ctx)
    return err
}

// GetDrafts 获取所有草稿
func (s *SessionService) GetDrafts(ctx context.Context, uid, eid uint64) (map[uint64]string, error) {
    key := fmt.Sprintf("qh:exam:drafts:%d:%d", uid, eid)
    result, err := s.rdb.HGetAll(ctx, key).Result()
    if err != nil {
        return nil, err
    }

    drafts := make(map[uint64]string, len(result))
    for k, v := range result {
        var qid uint64
        fmt.Sscanf(k, "%d", &qid)
        drafts[qid] = v
    }
    return drafts, nil
}

// PromoteDraft 将草稿提升为正式答案（交卷时调用）
func (s *SessionService) PromoteDraft(ctx context.Context, uid, eid uint64, questionID uint64) error {
    draftKey := fmt.Sprintf("qh:exam:drafts:%d:%d", uid, eid)
    draft, err := s.rdb.HGet(ctx, draftKey, fmt.Sprintf("%d", questionID)).Result()
    if err != nil {
        return nil // 没有草稿，跳过
    }

    // 将草稿作为正式答案保存
    return s.SaveAnswer(ctx, uid, eid, questionID, draft)
}
```

### 27.1 关键词子串匹配（填空题）

填空题答案匹配支持多种策略：精确匹配、包含匹配、忽略大小写、忽略空格。

```go
// service/exam/scoring/fill_blank_scorer.go

type FillBlankScorer struct{}

// MatchStrategy 匹配策略
type MatchStrategy int

const (
    MatchExact     MatchStrategy = iota // 精确匹配
    MatchContains                       // 包含匹配
    MatchIgnoreSpace                    // 忽略空格
    MatchIgnoreCase                     // 忽略大小写
    MatchFuzzy                          // 模糊匹配（忽略大小写+空格+标点）
)

// Score 填空题评分
func (s *FillBlankScorer) Score(userAnswer string, keywords []string, strategy MatchStrategy, fullScore float64) (float64, bool) {
    if len(keywords) == 0 {
        return 0, false
    }

    matched := true
    for _, keyword := range keywords {
        if !s.matchKeyword(userAnswer, keyword, strategy) {
            matched = false
            break
        }
    }

    if matched {
        return fullScore, true
    }
    return 0, false
}

// matchKeyword 匹配单个关键词
func (s *FillBlankScorer) matchKeyword(answer, keyword string, strategy MatchStrategy) bool {
    switch strategy {
    case MatchExact:
        return answer == keyword
    case MatchContains:
        return strings.Contains(answer, keyword)
    case MatchIgnoreSpace:
        return strings.ReplaceAll(answer, " ", "") == strings.ReplaceAll(keyword, " ", "")
    case MatchIgnoreCase:
        return strings.EqualFold(answer, keyword)
    case MatchFuzzy:
        normalize := func(s string) string {
            s = strings.ToLower(s)
            s = strings.ReplaceAll(s, " ", "")
            s = strings.ReplaceAll(s, "，", "")
            s = strings.ReplaceAll(s, ",", "")
            s = strings.ReplaceAll(s, ".", "")
            s = strings.ReplaceAll(s, "。", "")
            return s
        }
        return strings.Contains(normalize(answer), normalize(keyword))
    default:
        return answer == keyword
    }
}

// MultiBlankScore 多空填空题评分（每空独立计分）
func (s *FillBlankScorer) MultiBlankScore(userAnswers []string, blanks []*BlankConfig, fullScore float64) (float64, bool) {
    if len(blanks) == 0 {
        return 0, false
    }

    scorePerBlank := fullScore / float64(len(blanks))
    var totalScore float64
    allCorrect := true

    for i, blank := range blanks {
        if i >= len(userAnswers) {
            allCorrect = false
            continue
        }

        matched := false
        for _, keyword := range blank.AcceptedAnswers {
            if s.matchKeyword(userAnswers[i], keyword, blank.Strategy) {
                matched = true
                break
            }
        }

        if matched {
            totalScore += scorePerBlank
        } else {
            allCorrect = false
        }
    }

    return totalScore, allCorrect
}

type BlankConfig struct {
    AcceptedAnswers []string      `json:"accepted_answers"`
    Strategy        MatchStrategy `json:"strategy"`
}
```

### 27.2 部分得分（多选题少选给半分）

```go
// service/exam/scoring/multiple_choice_scorer.go

type MultipleChoiceScorer struct{}

// Score 多选题评分
// 规则：
//   - 全选正确：满分
//   - 少选（选了部分正确选项，没有错误选项）：半分
//   - 多选或选了错误选项：0分
func (s *MultipleChoiceScorer) Score(userAnswer string, correctOptions []string, fullScore float64, allowPartial bool) (float64, bool) {
    userSet := parseOptions(userAnswer)
    correctSet := make(map[string]bool)
    for _, opt := range correctOptions {
        correctSet[opt] = true
    }

    // 检查是否有错误选项
    hasWrong := false
    for opt := range userSet {
        if !correctSet[opt] {
            hasWrong = true
            break
        }
    }

    // 全部正确
    if len(userSet) == len(correctSet) && !hasWrong {
        return fullScore, true
    }

    // 有错误选项，0分
    if hasWrong {
        return 0, false
    }

    // 少选且没有错误选项
    if allowPartial && len(userSet) > 0 && len(userSet) < len(correctSet) {
        return fullScore * 0.5, false // 半分
    }

    return 0, false
}

func parseOptions(answer string) map[string]bool {
    result := make(map[string]bool)
    for _, r := range answer {
        if r >= 'A' && r <= 'Z' {
            result[string(r)] = true
        }
    }
    return result
}
```

### 27.3 自定义评分规则（按关键词命中数给分）

```go
// service/exam/scoring/custom_rule_scorer.go

type CustomRuleScorer struct{}

// KeywordRule 关键词评分规则
type KeywordRule struct {
    Keywords    []string `json:"keywords"`     // 关键词列表
    MatchMode   string   `json:"match_mode"`   // 匹配模式: contains/exact/fuzzy
    ScoreMode   string   `json:"score_mode"`   // 计分模式: per_hit/full/all_required
    ScorePerHit float64  `json:"score_per_hit"` // 每命中一个关键词的分值
    FullScore   float64  `json:"full_score"`    // 全部命中的满分
}

// Score 自定义规则评分
func (s *CustomRuleScorer) Score(userAnswer string, rule *KeywordRule) float64 {
    hitCount := 0

    for _, keyword := range rule.Keywords {
        if s.matchKeyword(userAnswer, keyword, rule.MatchMode) {
            hitCount++
        }
    }

    switch rule.ScoreMode {
    case "per_hit":
        // 每命中一个关键词得固定分值
        score := float64(hitCount) * rule.ScorePerHit
        if score > rule.FullScore {
            score = rule.FullScore
        }
        return score

    case "full":
        // 命中所有关键词才得分
        if hitCount == len(rule.Keywords) {
            return rule.FullScore
        }
        return 0

    case "all_required":
        // 必须全部命中
        if hitCount == len(rule.Keywords) {
            return rule.FullScore
        }
        return 0

    case "proportional":
        // 按比例得分
        if len(rule.Keywords) == 0 {
            return 0
        }
        return rule.FullScore * float64(hitCount) / float64(len(rule.Keywords))

    default:
        return 0
    }
}

func (s *CustomRuleScorer) matchKeyword(answer, keyword, mode string) bool {
    switch mode {
    case "exact":
        return strings.Contains(answer, keyword)
    case "fuzzy":
        normalize := func(s string) string {
            s = strings.ToLower(s)
            s = strings.ReplaceAll(s, " ", "")
            return s
        }
        return strings.Contains(normalize(answer), normalize(keyword))
    default: // contains
        return strings.Contains(answer, keyword)
    }
}
```
