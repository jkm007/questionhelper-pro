## 二、数据库设计

### 2.1 题目表 (questions)

```sql
CREATE TABLE `questions` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '题目ID',
  `title` TEXT NOT NULL COMMENT '题目标题',
  `content` TEXT COMMENT '题目内容(富文本)',
  `type` TINYINT NOT NULL COMMENT '题型:1=单选,2=多选,3=判断,4=填空,5=简答',
  `difficulty` TINYINT DEFAULT 2 COMMENT '难度:1=简单,2=中等,3=困难',
  `answer` TEXT COMMENT '答案',
  `analysis` TEXT COMMENT '解析',
  `category_id` BIGINT UNSIGNED DEFAULT NULL COMMENT '分类ID',
  `visibility` TINYINT DEFAULT 1 COMMENT '可见性:1=公开,2=私有,3=班级',
  `class_id` BIGINT UNSIGNED DEFAULT NULL COMMENT '班级ID(班级题目)',
  `creator_id` BIGINT UNSIGNED NOT NULL COMMENT '创建者ID',
  `content_hash` VARCHAR(64) DEFAULT NULL COMMENT '内容MD5哈希，用于查重',
  `is_subjective` TINYINT(1) DEFAULT 0 COMMENT '是否主观题:0=客观题,1=主观题',
  `status` TINYINT DEFAULT 0 COMMENT '状态:0=draft草稿,1=published已发布,2=archived已下架,3=pending_review待审核,4=revision需修改,5=timeout审核超时',
  `view_count` INT DEFAULT 0 COMMENT '浏览次数',
  `like_count` INT DEFAULT 0 COMMENT '点赞次数',
  `version` INT DEFAULT 1 COMMENT '当前版本号',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `deleted_at` DATETIME DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_category_id` (`category_id`),
  KEY `idx_creator_id` (`creator_id`),
  KEY `idx_visibility` (`visibility`),
  KEY `idx_status` (`status`),
  KEY `idx_deleted_at` (`deleted_at`),
  KEY `idx_content_hash` (`content_hash`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目表';
```

### 2.15 分类path字段说明

categories表的`path`字段用于存储分类的层级路径，格式为`/1/3/5/`，其中数字为分类ID。

**path的优势**：
- 快速查询所有祖先分类：`WHERE '/1/3/5/' LIKE CONCAT(path, '%')`
- 快速查询所有后代分类：`WHERE path LIKE '/1/3/5/%'`
- 避免递归查询，提升树形结构查询性能

**path更新策略**：
1. **创建分类**：path = 父分类path + 当前ID + `/`
2. **移动分类**：更新当前分类及所有子分类的path
3. **删除分类**：软删除，path不变

**示例**：
```
根分类1:     /1/
子分类3:     /1/3/
子分类5:     /1/3/5/
子分类7:     /1/3/5/7/
```

### 2.16 题目状态流转说明

**状态定义**：
- `0` - draft（草稿）：题目创建后的初始状态
- `1` - published（已发布）：审核通过后发布，用户可见
- `2` - archived（已下架）：从发布状态手动下架
- `3` - pending_review（待审核）：提交审核后的状态
- `4` - revision（需修改）：审核提出修改意见后的状态
- `5` - timeout（审核超时）：超过审核时限未处理

**状态流转规则**：
```
草稿(0) --> 待审核(3)：提交审核
待审核(3) --> 已发布(1)：审核通过
待审核(3) --> 草稿(0)：审核驳回
待审核(3) --> 需修改(4)：审核提出修改意见
待审核(3) --> 审核超时(5)：超过48小时未处理
需修改(4) --> 待审核(3)：回复修改意见后重新提交审核
已发布(1) --> 已下架(2)：手动下架
已下架(2) --> 待审核(3)：重新提交审核
审核超时(5) --> 待审核(3)：重新提交审核
```

### 2.17 知识点关联限制说明

每道题目最多关联3个知识点，最少可以不关联知识点。

**校验逻辑**：
1. 创建/更新题目时，检查`knowledge_ids`数组长度
2. 如果长度大于3，返回错误"每道题目最多关联3个知识点"
3. 验证所有知识点ID是否有效
4. 清除旧的关联关系，建立新的关联关系

```go
// CreateQuestion 创建题目
func (s *QuestionService) CreateQuestion(req *dto.CreateQuestionRequest, userID uint) (*dto.QuestionInfo, error) {
    // 1. 敏感词过滤
    if err := s.sensitiveFilter.Check(req.Title); err != nil {
        return nil, errors.New("题目标题包含敏感词: " + err.Error())
    }
    if req.Content != "" {
        if err := s.sensitiveFilter.Check(req.Content); err != nil {
            return nil, errors.New("题目内容包含敏感词: " + err.Error())
        }
    }

    // 2. XSS消毒
    req.Title = s.sanitizerSvc.Sanitize(req.Title)
    if req.Content != "" {
        req.Content = s.sanitizerSvc.Sanitize(req.Content)
    }
    if req.Analysis != "" {
        req.Analysis = s.sanitizerSvc.Sanitize(req.Analysis)
    }

    // 3. 计算内容哈希用于查重
    contentHash := s.duplicateSvc.CalculateHash(req.Title, req.Content, req.Type, req.Options)

    // 4. 检查重复
    duplicate, err := s.duplicateSvc.CheckDuplicate(contentHash, userID)
    if err != nil {
        return nil, err
    }
    if duplicate != nil {
        return nil, fmt.Errorf("题目已存在(重复题目ID: %d)", duplicate.ID)
    }

    // 5. 创建题目
    question := &model.Question{
        Title:        req.Title,
        Content:      req.Content,
        Type:         req.Type,
        Difficulty:   req.Difficulty,
        Answer:       req.Answer,
        Analysis:     req.Analysis,
        CategoryID:   req.CategoryID,
        Visibility:   req.Visibility,
        IsSubjective: req.IsSubjective,
        CreatorID:    userID,
        ContentHash:  contentHash,
        Version:      1,
        Status:       0, // 草稿状态
    }

    if err := s.questionRepo.Create(question); err != nil {
        return nil, err
    }

    // 6. 创建选项
    if len(req.Options) > 0 {
        options := make([]model.Option, len(req.Options))
        for i, opt := range req.Options {
            opt.Content = s.sanitizerSvc.Sanitize(opt.Content)
            options[i] = model.Option{
                QuestionID: question.ID,
                Label:      opt.Label,
                Content:    opt.Content,
                IsCorrect:  opt.IsCorrect,
                Sort:       i,
            }
        }
        s.questionRepo.CreateOptions(options)
    }

    // 7. 关联知识点
    if len(req.KnowledgeIDs) > 0 {
        // 校验知识点关联数量（1-3个）
        if len(req.KnowledgeIDs) > 3 {
            return nil, errors.New("每道题目最多关联3个知识点")
        }
        // 验证知识点是否存在
        var count int64
        s.questionRepo.db.Model(&model.KnowledgePoint{}).Where("id IN ?", req.KnowledgeIDs).Count(&count)
        if int(count) != len(req.KnowledgeIDs) {
            return nil, errors.New("部分知识点不存在")
        }
        s.questionRepo.AssociateKnowledge(question.ID, req.KnowledgeIDs)
    }

    // 8. 创建版本快照
    s.versionService.CreateSnapshot(question.ID, 1, userID, "创建题目", req)

    return s.GetQuestionDetail(question.ID, userID)
}

func (s *QuestionService) UpdateQuestion(id uint, req *dto.UpdateQuestionRequest, userID uint) (*dto.QuestionInfo, error) {
    question, err := s.questionRepo.FindByID(id)
    if err != nil {
        return nil, err
    }

    // 敏感词过滤
    if err := s.sensitiveFilter.Check(req.Title); err != nil {
        return nil, errors.New("题目标题包含敏感词: " + err.Error())
    }

    // XSS消毒
    req.Title = s.sanitizerSvc.Sanitize(req.Title)
    req.Content = s.sanitizerSvc.Sanitize(req.Content)
    req.Analysis = s.sanitizerSvc.Sanitize(req.Analysis)

    // 计算新的内容哈希
    newHash := s.duplicateSvc.CalculateHash(req.Title, req.Content, req.Type, req.Options)

    // 更新题目
    question.Title = req.Title
    question.Content = req.Content
    question.Type = req.Type
    question.Difficulty = req.Difficulty
    question.Answer = req.Answer
    question.Analysis = req.Analysis
    question.CategoryID = req.CategoryID
    question.ContentHash = newHash
    question.Version = question.Version + 1

    if err := s.questionRepo.Update(question); err != nil {
        return nil, err
    }

    // 更新选项
    if len(req.Options) > 0 {
        s.questionRepo.DeleteOptionsByQuestionID(id)
        options := make([]model.Option, len(req.Options))
        for i, opt := range req.Options {
            opt.Content = s.sanitizerSvc.Sanitize(opt.Content)
            options[i] = model.Option{
                QuestionID: id,
                Label:      opt.Label,
                Content:    opt.Content,
                IsCorrect:  opt.IsCorrect,
                Sort:       i,
            }
        }
        s.questionRepo.CreateOptions(options)
    }

    // 创建版本快照
    s.versionService.CreateSnapshot(id, question.Version, userID, req.ChangeSummary, req)

    return s.GetQuestionDetail(id, userID)
}

// BatchPublish 批量发布题目
func (s *QuestionService) BatchPublish(ids []uint, userID uint) (*dto.BatchResult, error) {
	result := &dto.BatchResult{Total: len(ids)}

	for _, id := range ids {
		question, err := s.questionRepo.FindByID(id)
		if err != nil {
			result.Failed++
			result.Errors = append(result.Errors, dto.BatchError{ID: id, Reason: "题目不存在"})
			continue
		}

		// 只有草稿和已下架状态可以发布
		if question.Status != 0 && question.Status != 2 {
			result.Failed++
			result.Errors = append(result.Errors, dto.BatchError{ID: id, Reason: "当前状态不允许发布"})
			continue
		}

		if err := s.db.Model(question).Update("status", 1).Error; err != nil {
			result.Failed++
			result.Errors = append(result.Errors, dto.BatchError{ID: id, Reason: err.Error()})
		} else {
			result.Success++
		}
	}

	return result, nil
}
```

```go
func (s *VersionService) Rollback(questionID uint, targetVersion int, userID uint) (*dto.QuestionInfo, error) {
    // 1. 获取目标版本快照
    versionRecord, err := s.versionRepo.FindByQuestionIDAndVersion(questionID, targetVersion)
    if err != nil {
        return nil, fmt.Errorf("版本 %d 不存在", targetVersion)
    }

    // 2. 获取当前题目
    question, err := s.questionRepo.FindByID(questionID)
    if err != nil {
        return nil, err
    }

    // 3. 还原题目数据
    question.Title = versionRecord.Title
    question.Content = versionRecord.Content
    question.Type = versionRecord.Type
    question.Difficulty = versionRecord.Difficulty
    question.Answer = versionRecord.Answer
    question.Analysis = versionRecord.Analysis
    question.CategoryID = versionRecord.CategoryID
    question.Version = question.Version + 1

    if err := s.questionRepo.Update(question); err != nil {
        return nil, err
    }

    // 4. 还原选项
    var options []model.Option
    var optionSnapshots []map[string]interface{}
    json.Unmarshal([]byte(versionRecord.OptionsSnapshot), &optionSnapshots)

    s.questionRepo.DeleteOptionsByQuestionID(questionID)
    for i, opt := range optionSnapshots {
        options = append(options, model.Option{
            QuestionID: questionID,
            Label:      opt["label"].(string),
            Content:    opt["content"].(string),
            IsCorrect:  int(opt["is_correct"].(float64)),
            Sort:       i,
        })
    }
    if len(options) > 0 {
        s.questionRepo.CreateOptions(options)
    }

    // 5. 还原知识点关联
    var knowledgeSnapshots []map[string]interface{}
    json.Unmarshal([]byte(versionRecord.KnowledgeSnapshot), &knowledgeSnapshots)

    s.questionRepo.DeleteKnowledgeByQuestionID(questionID)
    var knowledgeIDs []uint
    for _, kp := range knowledgeSnapshots {
        knowledgeIDs = append(knowledgeIDs, uint(kp["id"].(float64)))
    }
    if len(knowledgeIDs) > 0 {
        s.questionRepo.AssociateKnowledge(questionID, knowledgeIDs)
    }

    // 6. 记录回滚操作为新版本
    rollbackSummary := fmt.Sprintf("回滚到版本 %d", targetVersion)
    s.CreateSnapshot(questionID, question.Version, userID, rollbackSummary, nil)

    return nil, nil // 由controller层调用GetQuestionDetail返回
}
```

```go
// CalculateHash 计算题目内容哈希
func (s *DuplicateService) CalculateHash(title, content string, qType int, options []dto.OptionRequest) string {
    // 构建标准化内容字符串
    var builder strings.Builder
    builder.WriteString(strings.TrimSpace(title))
    builder.WriteString("|")
    builder.WriteString(strings.TrimSpace(content))
    builder.WriteString("|")
    builder.WriteString(strconv.Itoa(qType))

    // 选项排序后拼接，确保顺序不影响哈希
    if len(options) > 0 {
        sortedOpts := make([]string, len(options))
        for i, opt := range options {
            sortedOpts[i] = opt.Label + ":" + strings.TrimSpace(opt.Content)
        }
        sort.Strings(sortedOpts)
        builder.WriteString("|")
        builder.WriteString(strings.Join(sortedOpts, ","))
    }

    hash := md5.Sum([]byte(builder.String()))
    return hex.EncodeToString(hash[:])
}
```

```go
// GetShareByCode 通过分享码获取题目
func (s *ShareService) GetShareByCode(code string) (*dto.ShareQuestionInfo, error) {
    share, err := s.shareRepo.FindByCode(code)
    if err != nil {
        return nil, errors.New("分享链接不存在")
    }

    // 检查状态
    if share.Status == 0 {
        return nil, errors.New("分享链接已失效")
    }

    // 检查过期
    if share.ExpireAt != nil && share.ExpireAt.Before(time.Now()) {
        s.shareRepo.UpdateStatus(share.ID, 0)
        return nil, errors.New("分享链接已过期")
    }

    // 检查访问次数
    if share.MaxAccess > 0 && share.AccessCount >= share.MaxAccess {
        s.shareRepo.UpdateStatus(share.ID, 0)
        return nil, errors.New("分享链接访问次数已达上限")
    }

    // 增加访问计数
    s.shareRepo.IncrementAccessCount(share.ID)

    // 获取题目信息
    question, err := s.questionRepo.FindByID(share.QuestionID)
    if err != nil {
        return nil, errors.New("题目不存在")
    }

    return &dto.ShareQuestionInfo{
        Question:   question,
        AccessType: share.AccessType,
    }, nil
}
```

```go
// ImportFromWord 从Word文档导入题目
func (s *WordImportService) ImportFromWord(file io.Reader, req *dto.WordImportRequest, userID uint) (*dto.ImportResult, error) {
    // 1. 解析Word文档
    doc, err := document.Open(file)
    if err != nil {
        return nil, errors.New("无法解析Word文档，请检查文件格式")
    }
    defer doc.Close()

    // 2. 提取题目结构
    items, err := s.parseQuestions(doc)
    if err != nil {
        return nil, fmt.Errorf("解析题目结构失败: %v", err)
    }

    if len(items) == 0 {
        return nil, errors.New("未识别到任何题目")
    }

    // 3. 校验题目
    errors_list := s.validateQuestions(items)

    // 4. 查重
    duplicates, _ := s.duplicateSvc.CheckDuplicateBatch(items)

    // 5. 过滤重复题目
    duplicateMap := make(map[int]bool)
    for _, d := range duplicates {
        duplicateMap[d.Index] = true
    }

    // 6. 批量入库
    var successCount, failCount int
    for i, item := range items {
        if duplicateMap[i] {
            failCount++
            errors_list = append(errors_list, dto.ImportError{
                Row:    i + 1,
                Reason: "重复题目",
            })
            continue
        }

        // XSS消毒
        item.Title = s.sanitizerSvc.Sanitize(item.Title)
        item.Content = s.sanitizerSvc.Sanitize(item.Content)

        contentHash := s.duplicateSvc.CalculateHash(item.Title, item.Content, item.Type, item.Options)

        question := &model.Question{
            Title:       item.Title,
            Content:     item.Content,
            Type:        item.Type,
            Difficulty:  req.DefaultDifficulty,
            Answer:      item.Answer,
            Analysis:    item.Analysis,
            CategoryID:  req.CategoryID,
            CreatorID:   userID,
            ContentHash: contentHash,
            Version:     1,
            Status:      0, // 草稿
        }

        if err := s.questionRepo.Create(question); err != nil {
            failCount++
            errors_list = append(errors_list, dto.ImportError{
                Row:    i + 1,
                Reason: "入库失败: " + err.Error(),
            })
            continue
        }

        // 创建选项
        if len(item.Options) > 0 {
            options := make([]model.Option, len(item.Options))
            for j, opt := range item.Options {
                opt.Content = s.sanitizerSvc.Sanitize(opt.Content)
                options[j] = model.Option{
                    QuestionID: question.ID,
                    Label:      opt.Label,
                    Content:    opt.Content,
                    IsCorrect:  opt.IsCorrect,
                    Sort:       j,
                }
            }
            s.questionRepo.CreateOptions(options)
        }

        successCount++
    }

    return &dto.ImportResult{
        Total:      len(items),
        Success:    successCount,
        Failed:     failCount,
        Duplicates: len(duplicates),
        Errors:     errors_list,
    }, nil
}

// parseQuestions 解析Word文档中的题目结构
func (s *WordImportService) parseQuestions(doc *document.Doc) ([]dto.ImportItem, error) {
    var items []dto.ImportItem
    var currentItem *dto.ImportItem

    for _, para := range doc.Paragraphs() {
        text := strings.TrimSpace(para.Text())
        if text == "" {
            continue
        }

        // 检测题号标记（如 "1." "1、" "题目1" 等）
        if s.isQuestionStart(text) {
            if currentItem != nil {
                items = append(items, *currentItem)
            }
            currentItem = &dto.ImportItem{
                Title: s.extractQuestionTitle(text),
            }
            continue
        }

        if currentItem == nil {
            continue
        }

        // 检测题型标记
        if s.isTypeMark(text) {
            currentItem.Type = s.parseType(text)
            continue
        }

        // 检测选项标记（A. B. C. D.）
        if s.isOptionMark(text) {
            label, content := s.parseOption(text)
            currentItem.Options = append(currentItem.Options, dto.OptionRequest{
                Label:   label,
                Content: content,
            })
            continue
        }

        // 检测答案标记
        if s.isAnswerMark(text) {
            currentItem.Answer = s.extractAnswer(text)
            continue
        }

        // 检测解析标记
        if s.isAnalysisMark(text) {
            currentItem.Analysis = s.extractAnalysis(text)
            continue
        }

        // 默认追加到题目内容
        currentItem.Content += text + "\n"
    }

    // 添加最后一道题
    if currentItem != nil {
        items = append(items, *currentItem)
    }

    return items, nil
}

// isQuestionStart 检测是否为题目开始标记
func (s *WordImportService) isQuestionStart(text string) bool {
    patterns := []string{
        `^\d+[.、．]`,          // 1. 1、1．
        `^第\d+题`,             // 第1题
        `^题目\d+`,             // 题目1
        `^\(\d+\)`,             // (1)
        `^（\d+）`,             // （1）
        `^Q\d+`,               // Q1
    }
    for _, pattern := range patterns {
        matched, _ := regexp.MatchString(pattern, text)
        if matched {
            return true
        }
    }
    return false
}

// parseType 从文本中识别题型
func (s *WordImportService) parseType(text string) int {
    text = strings.ToLower(text)
    if strings.Contains(text, "单选") || strings.Contains(text, "single") {
        return 1
    }
    if strings.Contains(text, "多选") || strings.Contains(text, "multiple") {
        return 2
    }
    if strings.Contains(text, "判断") || strings.Contains(text, "true") || strings.Contains(text, "false") {
        return 3
    }
    if strings.Contains(text, "填空") || strings.Contains(text, "fill") {
        return 4
    }
    if strings.Contains(text, "简答") || strings.Contains(text, "short") {
        return 5
    }
    return 1 // 默认单选
}

// validateQuestions 校验题目数据
func (s *WordImportService) validateQuestions(items []dto.ImportItem) []dto.ImportError {
    var errors []dto.ImportError

    for i, item := range items {
        if strings.TrimSpace(item.Title) == "" {
            errors = append(errors, dto.ImportError{Row: i + 1, Reason: "题目内容为空"})
        }
        if item.Type == 0 {
            item.Type = 1 // 默认单选题
        }
        if item.Type == 1 || item.Type == 2 {
            if len(item.Options) < 2 {
                errors = append(errors, dto.ImportError{Row: i + 1, Reason: "选择题至少需要2个选项"})
            }
            hasCorrect := false
            for _, opt := range item.Options {
                if opt.IsCorrect {
                    hasCorrect = true
                    break
                }
            }
            if !hasCorrect && item.Answer != "" {
                // 根据答案标记正确选项
                for j, opt := range item.Options {
                    if strings.Contains(item.Answer, opt.Label) {
                        items[i].Options[j].IsCorrect = true
                    }
                }
            }
        }
    }

    return errors
}
```

```go
		FileName:    fileName,
		CategoryID:  req.CategoryID,
		DefaultDiff: req.DefaultDifficulty,
		UserID:      userID,
		FileData:    fileData,
		Errors:      make([]ImportError, 0),
	}

	// 存储任务
	s.taskStore.Store(taskID, task)

	// 异步执行导入
	go s.processImport(task)

	return taskID, nil
}

// processImport 处理导入任务
func (s *ImportService) processImport(task *ImportTask) {
	task.mu.Lock()
	task.Status = "processing"
	task.mu.Unlock()

	// 打开Excel文件
	f, err := excelize.OpenReader(bytes.NewReader(task.FileData))
	if err != nil {
		task.mu.Lock()
		task.Status = "failed"
		task.Errors = append(task.Errors, ImportError{Row: 0, Reason: "无法解析Excel文件: " + err.Error()})
		now := time.Now()
		task.CompletedAt = &now
		task.Duration = int64(now.Sub(task.StartedAt).Seconds())
		task.mu.Unlock()
		return
	}
	defer f.Close()

	// 获取题目导入Sheet
	rows, err := f.GetRows("题目导入")
	if err != nil || len(rows) < 2 {
		task.mu.Lock()
		task.Status = "failed"
		task.Errors = append(task.Errors, ImportError{Row: 0, Reason: "未找到'题目导入'工作表或数据为空"})
		now := time.Now()
		task.CompletedAt = &now
		task.Duration = int64(now.Sub(task.StartedAt).Seconds())
		task.mu.Unlock()
		return
	}

	// 解析题目数据
	items := s.parseExcelRows(rows, task)

	// 更新总数
	task.mu.Lock()
	task.Total = len(items)
	task.mu.Unlock()

	// 批量入库
	ctx := context.Background()
	for i, item := range items {
		// 检查上下文是否取消
		select {
		case <-ctx.Done():
			task.mu.Lock()
			task.Status = "failed"
			task.Errors = append(task.Errors, ImportError{Row: i + 2, Reason: "任务被取消"})
			task.mu.Unlock()
			return
		default:
		}

		// 处理单条记录
		err := s.processSingleItem(item, i+2, task)
		if err != nil {
			task.mu.Lock()
			task.Failed++
			task.Errors = append(task.Errors, ImportError{Row: i + 2, Reason: err.Error(), Data: fmt.Sprintf("%v", item)})
			task.mu.Unlock()
		} else {
			task.mu.Lock()
			task.Success++
			task.mu.Unlock()
		}

		// 更新进度
		task.mu.Lock()
		task.Processed++
		task.CurrentRow = i + 2
		task.Progress = float64(task.Processed) / float64(task.Total) * 100
		task.mu.Unlock()

		// 每处理100条休眠一下，避免数据库压力过大
		if (i+1)%100 == 0 {
			time.Sleep(100 * time.Millisecond)
		}
	}

	// 完成任务
	task.mu.Lock()
	task.Status = "completed"
	now := time.Now()
	task.CompletedAt = &now
	task.Duration = int64(now.Sub(task.StartedAt).Seconds())
	task.mu.Unlock()
}

// parseExcelRows 解析Excel行数据
func (s *ImportService) parseExcelRows(rows [][]string, task *ImportTask) []dto.ImportItem {
	var items []dto.ImportItem

	for i, row := range rows {
		// 跳过表头
		if i == 0 {
			continue
		}

		// 跳过空行
		if len(row) == 0 || strings.TrimSpace(row[0]) == "" {
			continue
		}

		item := dto.ImportItem{
			Title: strings.TrimSpace(row[0]),
		}

		// 解析题型
		if len(row) > 1 {
			item.Type = s.parseTypeString(row[1])
		}

		// 解析选项
		if len(row) > 2 {
			for j := 2; j < 8 && j < len(row); j++ {
				if strings.TrimSpace(row[j]) != "" {
					label := string(rune('A' + j - 2))
					item.Options = append(item.Options, dto.OptionRequest{
						Label:   label,
						Content: strings.TrimSpace(row[j]),
					})
				}
			}
		}

		// 解析正确答案
		if len(row) > 8 {
			item.Answer = strings.TrimSpace(row[8])
			// 标记正确选项
			for j, opt := range item.Options {
				if strings.Contains(item.Answer, opt.Label) {
					item.Options[j].IsCorrect = true
				}
			}
		}

		// 解析解析
		if len(row) > 9 {
			item.Analysis = strings.TrimSpace(row[9])
		}

		// 解析难度
		if len(row) > 10 && strings.TrimSpace(row[10]) != "" {
			item.Difficulty = s.parseDifficultyString(row[10])
		} else {
			item.Difficulty = task.DefaultDiff
		}

		// 解析知识点
		if len(row) > 11 {
			item.KnowledgeNames = strings.Split(strings.TrimSpace(row[11]), ",")
		}

		// 解析是否主观题
		if len(row) > 12 && strings.TrimSpace(row[12]) == "主观题" {
			item.IsSubjective = 1
		}

		items = append(items, item)
	}

	return items
}

// processSingleItem 处理单条导入记录
func (s *ImportService) processSingleItem(item dto.ImportItem, row int, task *ImportTask) error {
	// 1. 敏感词过滤
	if err := s.sensitiveFilt.Check(item.Title); err != nil {
		return fmt.Errorf("标题包含敏感词: %v", err)
	}

	// 2. XSS消毒
	item.Title = s.sanitizerSvc.Sanitize(item.Title)
	item.Analysis = s.sanitizerSvc.Sanitize(item.Analysis)

	// 3. 计算内容哈希
	contentHash := s.duplicateSvc.CalculateHash(item.Title, "", item.Type, item.Options)

	// 4. 检查重复
	duplicate, err := s.duplicateSvc.CheckDuplicate(contentHash, task.UserID)
	if err != nil {
		return fmt.Errorf("查重失败: %v", err)
	}
	if duplicate != nil {
		task.mu.Lock()
		task.Duplicates++
		task.mu.Unlock()
		return fmt.Errorf("重复题目，已存在ID: %d", duplicate.ID)
	}

	// 5. 创建题目
	question := &model.Question{
		Title:        item.Title,
		Content:      s.buildContentJSON(item),
		Type:         item.Type,
		Difficulty:   item.Difficulty,
		Answer:       item.Answer,
		Analysis:     item.Analysis,
		CategoryID:   task.CategoryID,
		IsSubjective: item.IsSubjective,
		CreatorID:    task.UserID,
		ContentHash:  contentHash,
		Version:      1,
		Status:       0, // 草稿状态
	}

	// 6. 入库
	if err := s.db.Create(question).Error; err != nil {
		return fmt.Errorf("入库失败: %v", err)
	}

	// 7. 创建选项
	if len(item.Options) > 0 {
		options := make([]model.Option, len(item.Options))
		for j, opt := range item.Options {
			options[j] = model.Option{
				QuestionID: question.ID,
				Label:      opt.Label,
				Content:    opt.Content,
				IsCorrect:  opt.IsCorrect,
				Sort:       j,
			}
		}
		s.db.CreateInBatches(options, 100)
	}

	return nil
}

	go s.processImport(retryTask)

	return string(jsonBytes)
}
```

```go
// ExportToWord 导出题目到Word文档

	// 查询题目列表

	titleRun.SetText("题库导出报告")

	titleRun.Properties().SetSize(24)

	infoPara := doc.AddParagraph()
	infoRun := infoPara.AddRun()
	infoRun.SetText(fmt.Sprintf("导出时间：%s\n题目数量：%d", time.Now().Format("2006-01-02 15:04:05"), len(questions)))
	infoRun.Properties().SetSize(12)

	// 添加分隔线
	doc.AddParagraph()

	// 按题型分组
	typeGroups := s.groupByType(questions)

	typeNames := map[int]string{
		1: "单选题",
		2: "多选题",
		3: "判断题",
		4: "填空题",
		5: "简答题",
	}

	// 遍历每种题型
	for qType := 1; qType <= 5; qType++ {
		items, ok := typeGroups[qType]
		if !ok || len(items) == 0 {
			continue
		}

		// 添加题型标题
		typeTitle := doc.AddParagraph()
		typeRun := typeTitle.AddRun()
		typeRun.SetText(fmt.Sprintf("%s（共%d题）", typeNames[qType], len(items)))
		typeRun.Properties().SetBold(true)
		typeRun.Properties().SetSize(18)
		typeRun.Properties().SetColor(0x2E74B5)

		// 添加题目
		for i, q := range items {
			s.addQuestionToDoc(doc, q, i+1)
		}

		// 添加分页符
		doc.AddParagraph().AddRun().AddBreak()
	}

	// 保存到缓冲区
	buf := new(bytes.Buffer)
	if _, err := doc.WriteTo(buf); err != nil {
		return nil, fmt.Errorf("生成Word文档失败: %v", err)
	}

	return buf, nil
}

// queryQuestions 查询题目
func (s *WordExportService) queryQuestions(req *ExportRequest, userID uint) ([]model.Question, error) {
	query := s.db.Model(&model.Question{}).
		Preload("Options").
		Preload("KnowledgePoints").
		Where("deleted_at IS NULL")

	if req.CategoryID != nil {
		query = query.Where("category_id = ?", *req.CategoryID)
	}
	if req.Type != nil {
		query = query.Where("type = ?", *req.Type)
	}
	if req.Difficulty != nil {
		query = query.Where("difficulty = ?", *req.Difficulty)
	}
	if req.Status != nil {
		query = query.Where("status = ?", *req.Status)
	}
	if req.Keyword != "" {
		query = query.Where("title LIKE ?", "%"+req.Keyword+"%")
	}

	var questions []model.Question
	if err := query.Order("type, id").Find(&questions).Error; err != nil {
		return nil, err
	}

	return questions, nil
}
```

```go
// BatchImportAttachments 批量导入附件
func (s *AttachmentService) BatchImportAttachments(questionID uint, files []*multipart.FileHeader, userID uint) (*dto.AttachmentImportResult, error) {
    // 验证题目存在
    question, err := s.questionRepo.FindByID(questionID)
    if err != nil {
        return nil, errors.New("题目不存在")
    }
    _ = question

    var successCount, failCount int

    for _, fileHeader := range files {
        attachment, err := s.importSingleAttachment(questionID, fileHeader, userID)
        if err != nil {
            failCount++
            continue
        }
        results = append(results, *attachment)
        successCount++
    }

    return &dto.AttachmentImportResult{
        Total:       len(files),
        Success:     successCount,
        Failed:      failCount,
        Attachments: results,
    }, nil
}

// importSingleAttachment 导入单个附件
func (s *AttachmentService) importSingleAttachment(questionID uint, fileHeader *multipart.FileHeader, userID uint) (*dto.AttachmentInfo, error) {
    // 1. 校验文件类型
    contentType := fileHeader.Header.Get("Content-Type")
    attachmentType, ok := allowedAttachmentTypes[contentType]
    if !ok {
        return nil, fmt.Errorf("不支持的文件类型: %s", contentType)
    }

    // 2. 校验文件大小
    sizeLimit := attachmentSizeLimits[attachmentType]
    if fileHeader.Size > sizeLimit {
        return nil, fmt.Errorf("文件大小超过限制(最大 %dMB)", sizeLimit/(1024*1024))
    }

    // 3. 打开文件
    file, err := fileHeader.Open()
    if err != nil {
        return nil, err
    }
    defer file.Close()

    // 4. 上传到文件存储
    ext := filepath.Ext(fileHeader.Filename)
    filePath, err := s.fileService.Upload(file, fmt.Sprintf("question/%d/", questionID), ext)
    if err != nil {
        return nil, fmt.Errorf("文件上传失败: %v", err)
    }

    // 5. 获取音视频时长和图片尺寸
    var duration, width, height int
    if attachmentType == 2 || attachmentType == 3 {
        duration = s.getMediaDuration(filePath)
    }
    if attachmentType == 1 || attachmentType == 3 {
        width, height = s.getMediaDimensions(filePath)
    }

    // 6. 保存附件记录
    attachment := &model.QuestionAttachment{
        QuestionID:     questionID,
        FileName:       fileHeader.Filename,
        FilePath:       filePath,
        FileSize:       fileHeader.Size,
        FileType:       contentType,
        FileExt:        ext,
        AttachmentType: attachmentType,
        Duration:       duration,
        Width:          width,
        Height:         height,
        CreatorID:      userID,
    }

    if err := s.attachmentRepo.Create(attachment); err != nil {
        return nil, err
    }

    return &dto.AttachmentInfo{
        ID:             attachment.ID,
        FileName:       attachment.FileName,
        FilePath:       attachment.FilePath,
        FileType:       attachment.FileType,
        AttachmentType: attachment.AttachmentType,
        FileSize:       attachment.FileSize,
    }, nil
```

### 4.10 敏感词过滤（DFA算法）

```go
// pkg/sensitive/dfa.go

package sensitive

import (
    "strings"
    "sync"
)

// DFANode DFA树节点
type DFANode struct {
    Children map[rune]*DFANode
    IsEnd    bool
    Word     string // 完整敏感词（仅在结束节点有值）
}

// DFATrie DFA敏感词树
type DFATrie struct {
    root *DFANode
    mu   sync.RWMutex
}

// NewDFATrie 创建DFA敏感词树
func NewDFATrie() *DFATrie {
    return &DFATrie{
        root: &DFANode{
            Children: make(map[rune]*DFANode),
        },
    }
}

// AddWord 添加敏感词
func (t *DFATrie) AddWord(word string) {
    t.mu.Lock()
    defer t.mu.Unlock()

    node := t.root
    for _, ch := range word {
        if _, ok := node.Children[ch]; !ok {
            node.Children[ch] = &DFANode{
                Children: make(map[rune]*DFANode),
            }
        }
        node = node.Children[ch]
    }
    node.IsEnd = true
    node.Word = word
}

// AddWords 批量添加敏感词
func (t *DFATrie) AddWords(words []string) {
    for _, word := range words {
        t.AddWord(word)
    }
}

// Match 检测文本中是否包含敏感词，返回第一个匹配的敏感词
func (t *DFATrie) Match(text string) (string, bool) {
    t.mu.RLock()
    defer t.mu.RUnlock()

    runes := []rune(text)
    length := len(runes)

    for i := 0; i < length; i++ {
        node := t.root
        j := i
        for j < length {
            if child, ok := node.Children[runes[j]]; ok {
                node = child
                if node.IsEnd {
                    return node.Word, true
                }
                j++
            } else {
                break
            }
        }
    }

    return "", false
}

// MatchAll 检测文本中所有匹配的敏感词
func (t *DFATrie) MatchAll(text string) []string {
    t.mu.RLock()
    defer t.mu.RUnlock()

    var matches []string
    runes := []rune(text)
    length := len(runes)
    matched := make(map[string]bool)

    for i := 0; i < length; i++ {
        node := t.root
        j := i
        for j < length {
            if child, ok := node.Children[runes[j]]; ok {
                node = child
                if node.IsEnd && !matched[node.Word] {
                    matches = append(matches, node.Word)
                    matched[node.Word] = true
                }
                j++
            } else {
                break
            }
        }
    }

    return matches
}

// Replace 将文本中的敏感词替换为指定字符
func (t *DFATrie) Replace(text string, replaceChar rune) string {
    t.mu.RLock()
    defer t.mu.RUnlock()

    runes := []rune(text)
    length := len(runes)
    result := make([]rune, length)
    copy(result, runes)

    for i := 0; i < length; i++ {
        node := t.root
        j := i
        matchEnd := -1

        for j < length {
            if child, ok := node.Children[runes[j]]; ok {
                node = child
                if node.IsEnd {
                    matchEnd = j
                }
                j++
            } else {
                break
            }
        }

        if matchEnd >= 0 {
            for k := i; k <= matchEnd; k++ {
                result[k] = replaceChar
            }
        }
    }

    return string(result)
}

// Clear 清空敏感词树
func (t *DFATrie) Clear() {
    t.mu.Lock()
    defer t.mu.Unlock()

    t.root = &DFANode{
        Children: make(map[rune]*DFANode),
    }
}

// WordCount 返回敏感词数量
func (t *DFATrie) WordCount() int {
    t.mu.RLock()
    defer t.mu.RUnlock()

    count := 0
    var walk func(node *DFANode)
    walk = func(node *DFANode) {
        if node.IsEnd {
            count++
        }
        for _, child := range node.Children {
            walk(child)
        }
    }
    walk(t.root)
    return count
}
```

```go
    loadedAt time.Time

        trie: NewDFATrie(),
        db:   db,
    }

    return f
}

// LoadFromDB 从数据库加载敏感词
func (f *Filter) LoadFromDB() error {
    f.mu.Lock()
    defer f.mu.Unlock()

    var words []struct {
        Word string
    }

    if err := f.db.Table("sensitive_words").
        Where("status = 1").
        Pluck("word", &words).
        Error; err != nil {
        return err
    }

    f.trie.Clear()
    for _, w := range words {
        f.trie.AddWord(w.Word)
    }

    f.loadedAt = time.Now()
    return nil
}

// Check 检查文本是否包含敏感词
func (f *Filter) Check(text string) error {
    f.mu.RLock()
    defer f.mu.RUnlock()

    word, found := f.trie.Match(text)
    if found {
        return errors.New("包含敏感词: " + word)
    }
    return nil
}

// CheckAll 检查文本中所有敏感词
func (f *Filter) CheckAll(text string) []string {
    f.mu.RLock()
    defer f.mu.RUnlock()

    return f.trie.MatchAll(text)
}

// Replace 将文本中的敏感词替换为*
func (f *Filter) Replace(text string) string {
    f.mu.RLock()
    defer f.mu.RUnlock()

    return f.trie.Replace(text, '*')
}

// Reload 重新加载敏感词库
func (f *Filter) Reload() error {
    return f.f.LoadFromDB()
}

// GetLoadedAt 获取最后加载时间
func (f *Filter) GetLoadedAt() time.Time {
    f.mu.RLock()
    defer f.mu.RUnlock()

}
```

```go
				Update("category_id", req.TargetID).Error; err != nil {
				return err
			}
			result.MovedKnowledgePoints += int(kpCount)

			// 移动子分类到目标分类下
			var childCount int64
			if err := tx.Model(&model.Category{}).Where("parent_id = ? AND deleted_at IS NULL", sourceID).
				Count(&childCount).Error; err != nil {

			}

			tx.Where("parent_id = ? AND deleted_at IS NULL", req.TargetID).Find(&children)
			for _, child := range children {
				newPath := fmt.Sprintf("%s%d/", target.Path, child.ID)
				tx.Model(&child).Update("path", newPath)
			}

			// 软删除源分类
			if err := tx.Delete(source).Error; err != nil {
				return err
			}

			result.MergedCount++
		}

		return nil
	})

	if err != nil {
		return nil, err
	}

	return result, nil
}
```

```go
	groupBy := "DATE(created_at)"
	switch granularity {
	case "week":
		dateFormat = "2006-W01"
		groupBy = "YEARWEEK(created_at)"
	case "month":
		dateFormat = "2006-01"
		groupBy = "DATE_FORMAT(created_at, '%Y-%m')"

		Date  string
		Count int64

		SELECT

			COUNT(*) as count

		WHERE created_at >= ? AND created_at <= ? AND deleted_at IS NULL
		GROUP BY %s
		ORDER BY date
	`, groupBy, groupBy), startDate, endDate+" 23:59:59").Scan(&createdStats)

	// 查询发布数量
	var publishedStats []struct {
		Date  string
		Count int64
	}
	s.db.Raw(fmt.Sprintf(`
		SELECT 
			%s as date,
			COUNT(*) as count
		FROM questions
		WHERE status = 1 AND created_at >= ? AND created_at <= ? AND deleted_at IS NULL
		GROUP BY %s
		ORDER BY date
	`, groupBy, groupBy), startDate, endDate+" 23:59:59").Scan(&publishedStats)

	// 构建数据
	createdMap := make(map[string]int64)
	for _, stat := range createdStats {
		createdMap[stat.Date] = stat.Count
		data.Labels = append(data.Labels, stat.Date)
	}

	publishedMap := make(map[string]int64)
	for _, stat := range publishedStats {

		publishedDataset.Data = append(publishedDataset.Data, publishedMap[label])
	}

	data.Datasets = append(data.Datasets, createdDataset, publishedDataset)
```

```go
    return func(c *gin.Context) {
        // 仅对POST/PUT请求进行敏感词检测
        if c.Request.Method != http.MethodPost && c.Request.Method != http.MethodPut {
            c.Next()
            return
        }

        // 从请求上下文中获取需要检查的字段
        // 由具体的handler在绑定参数后调用filter.Check()
        c.Set("sensitiveFilter", filter)
        c.Next()
    }
}
```

```vue
      <view class="question-type">
        <uni-tag :text="typeMap[question.type]" type="primary" />

      <!-- 选项 -->

        <view
          v-for="opt in question.options"

import { ref, onMounted } from 'vue'

const loading = ref(true)
const error = ref('')
const accessType = ref(1)
const selectedOptions = ref<string[]>([])
const showAnswer = ref(false)

const typeMap: Record<number, string> = {
  1: '单选题', 2: '多选题', 3: '判断题', 4: '填空题', 5: '简答题',
}

const selectOption = (label: string) => {
  if (accessType.value !== 2 || showAnswer.value) return
  if (question.value.type === 1) {
    selectedOptions.value = [label]
  } else {
    const index = selectedOptions.value.indexOf(label)
    if (index >= 0) {
      selectedOptions.value.splice(index, 1)
    } else {
      selectedOptions.value.push(label)
    }
  }
}

const submitAnswer = () => {
  showAnswer.value = true
}

onMounted(async () => {
  const pages = getCurrentPages()
  const page = pages[pages.length - 1]
  shareCode.value = page.options?.code || ''

  if (!shareCode.value) {
    error.value = '无效的分享链接'
    loading.value = false
    return
  }

  try {
    const res = await getShareQuestion(shareCode.value)
    question.value = res.data.question
    accessType.value = res.data.access_type
  } catch (e: any) {
    error.value = e.message || '获取题目失败'
  } finally {
    loading.value = false
  }
})
```

```go
// 在Service层使用Redis缓存

func (s *QuestionService) GetQuestionDetail(id uint, userID uint) (*dto.QuestionInfo, error) {
    cacheKey := fmt.Sprintf("qh:question:%d", id)

    // 尝试从缓存读取
    cached, err := s.redis.Get(cacheKey)
    if err == nil {
        var info dto.QuestionInfo
        json.Unmarshal([]byte(cached), &info)
        return &info, nil
    }

    // 从数据库读取
    question, err := s.questionRepo.FindByIDWithRelations(id)
    if err != nil {
        return nil, err
    }

    info := s.toQuestionInfo(question, userID)

    // 写入缓存
    data, _ := json.Marshal(info)
    s.redis.Set(cacheKey, string(data), 10*time.Minute)

    return info, nil
}
```
