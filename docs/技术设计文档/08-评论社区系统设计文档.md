## 一、功能概述

### 1.1 功能清单

| 功能 | 说明 |
|------|------|
| 题目评论 | 对题目发表评论，支持嵌套回复（楼中楼，最多2层） |
| 评论编辑 | 用户可编辑自己发布的评论 |
| 评论图片 | 支持在评论中添加图片（最多9张） |
| 评论@功能 | 支持@其他用户，被@用户收到通知 |
| 评论表情 | 支持插入系统表情包 |
| 评论置顶 | 管理员/教师可置顶重要评论 |
| 评论精选 | 标记精选评论，优先展示 |
| 评论互动 | 点赞、回复 |
| 举报管理 | 举报违规评论，管理员处理 |
| 评论黑名单 | 禁止特定用户发表评论 |
| 评论审核 | 敏感词自动审核、新用户评论审核、关键词触发审核、高频发评论审核 |
| 评论统计 | 评论数量、活跃度、热评话题、小时分布统计 |
| 评论导出 | 导出评论数据为Excel |
| 通知系统 | 站内消息通知，WebSocket实时推送 |
| 通知模板 | 通知模板管理，支持变量替换 |
| 通知群发 | 群发通知给指定用户/角色/全体 |
| 通知撤回 | 撤回已发送的通知 |
| 通知定时发送 | 创建定时通知任务，到时自动发送 |
| 通知统计 | 通知发送量、已读率、点击率统计 |
| 通知推送 | 短信推送、微信模板消息推送 |
| 通知设置 | 用户自定义通知偏好 |
| 通知免打扰 | 设置免打扰时间段 |

### 1.2 技术栈

| 组件 | 技术选型 |
|------|---------|
| 后端框架 | Go 1.21+ + Gin 1.9+ |
| ORM | GORM 1.25+ |
| 数据库 | MySQL 8.0 |
| 缓存 | Redis 7.x |
| WebSocket | Gorilla WebSocket |
| 消息广播 | Redis Pub/Sub |
| 文件存储 | MinIO / 阿里云 OSS |
| 敏感词过滤 | DFA 算法自定义实现 |

## 二、数据库设计

### 2.1 评论表 (comments)

```sql
CREATE TABLE `comments` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `target_type` TINYINT NOT NULL COMMENT '目标类型:1=题目,2=考试,3=班级',
  `target_id` BIGINT UNSIGNED NOT NULL COMMENT '目标ID',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `content` TEXT NOT NULL COMMENT '评论内容',
  `images` JSON DEFAULT NULL COMMENT '图片列表(JSON数组，最多9张)',
  `mentions` JSON DEFAULT NULL COMMENT '@用户列表(JSON数组，存储被@用户ID)',
  `parent_id` BIGINT UNSIGNED DEFAULT NULL COMMENT '父评论ID',
  `reply_to_user_id` BIGINT UNSIGNED DEFAULT NULL COMMENT '回复目标用户ID',
  `like_count` INT DEFAULT 0 COMMENT '点赞数',
  `is_pinned` TINYINT(1) DEFAULT 0 COMMENT '是否置顶:0=否,1=是',
  `is_featured` TINYINT(1) DEFAULT 0 COMMENT '是否精选:0=否,1=是',
  `is_official` TINYINT(1) DEFAULT 0 COMMENT '是否官方解答:0=否,1=是',
  `status` TINYINT DEFAULT 1 COMMENT '状态:0=隐藏,1=正常,2=待审核,3=已举报',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `deleted_at` DATETIME DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_target` (`target_type`, `target_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_parent_id` (`parent_id`),
  KEY `idx_status` (`status`),
  KEY `idx_pinned` (`is_pinned`),
  KEY `idx_featured` (`is_featured`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='评论表';
```

package model

import (

// JSONInt64Slice 自定义JSON类型，存储整数数组
type JSONInt64Slice []uint

func (j JSONInt64Slice) Value() (driver.Value, error) {
    return json.Marshal(j)
}

func (j *JSONInt64Slice) Scan(value interface{}) error {
    if value == nil {
        return nil
    }
    bytes, ok := value.([]byte)
    if !ok {
        return nil
    }
    return json.Unmarshal(bytes, j)
}

// JSONStringSlice 自定义JSON类型，存储字符串数组
type JSONStringSlice []string

func (j JSONStringSlice) Value() (driver.Value, error) {
    return json.Marshal(j)
}

func (j *JSONStringSlice) Scan(value interface{}) error {
    if value == nil {
        return nil
    }
    bytes, ok := value.([]byte)
    if !ok {
        return nil
    }
    return json.Unmarshal(bytes, j)
}

type Comment struct {

// CreateComment 发表评论
func (s *CommentService) CreateComment(req *dto.CreateCommentRequest, userID uint) (*dto.CommentInfo, error) {
    // 1. 检查是否被禁言
    blacklisted, err := s.blacklistRepo.IsBlacklisted(userID, req.TargetType, req.TargetID)
    if err != nil {
        return nil, err
    }
    if blacklisted {
        return nil, errors.New("您已被禁言，无法发表评论")
    }

    // 2. 敏感词过滤
    if sensitive.HasSensitive(req.Content) {
        return nil, errors.New("评论包含敏感词，请修改后重试")
    }

    // 3. 校验图片数量
    if len(req.Images) > 9 {
        return nil, errors.New("评论最多添加9张图片")
    }

    // 4. 校验@用户是否存在
    if len(req.Mentions) > 0 {
        validUsers, err := s.userRepo.FindByIDs(req.Mentions)
        if err != nil {
            return nil, err
        }
        if len(validUsers) != len(req.Mentions) {
            return nil, errors.New("部分@用户不存在")
        }
    }

    // 5. 创建评论
    comment := &model.Comment{
        TargetType:    req.TargetType,
        TargetID:      req.TargetID,
        UserID:        userID,
        Content:       req.Content,
        Images:        req.Images,
        Mentions:      req.Mentions,
        ParentID:      req.ParentID,
        ReplyToUserID: req.ReplyToUserID,
        Status:        1,
    }

    if err := s.commentRepo.Create(comment); err != nil {
        return nil, err
    }

    // 6. 发送通知 - 回复通知
    if req.ParentID != nil {
        parentComment, _ := s.commentRepo.FindByID(*req.ParentID)
        if parentComment != nil && parentComment.UserID != userID {
            s.notificationSvc.SendNotification(&notification.SendNotificationParam{
                UserID:     parentComment.UserID,
                Type:       "comment_reply",
                Title:      "有人回复了你的评论",
                Content:    req.Content,
                TargetType: "comment",
                TargetID:   comment.ID,
                SenderID:   userID,
            })
        }
    }

    // 7. 发送通知 - @提及通知
    if len(req.Mentions) > 0 {
        for _, mentionedUserID := range req.Mentions {
            if mentionedUserID == userID {
                continue
            }
            s.notificationSvc.SendNotification(&notification.SendNotificationParam{
                UserID:     mentionedUserID,
                Type:       "comment_mention",
                Title:      fmt.Sprintf("有人在评论中@了你"),
                Content:    req.Content,
                TargetType: "comment",
                TargetID:   comment.ID,
                SenderID:   userID,
            })
        }
    }

    return s.GetCommentDetail(comment.ID)
}

// EditComment 编辑评论
func (s *CommentService) EditComment(commentID uint, req *dto.EditCommentRequest, userID uint) (*dto.CommentInfo, error) {
    // 1. 查找评论
    comment, err := s.commentRepo.FindByID(commentID)
    if err != nil {
        return nil, errors.New("评论不存在")
    }

    // 2. 权限校验：只有评论作者可以编辑
    if comment.UserID != userID {
        return nil, errors.New("只能编辑自己的评论")
    }

    // 3. 状态校验：已删除或已隐藏的评论不可编辑
    if comment.Status == 0 {
        return nil, errors.New("该评论已被隐藏，无法编辑")
    }

    // 4. 敏感词过滤
    if sensitive.HasSensitive(req.Content) {
        return nil, errors.New("评论包含敏感词，请修改后重试")
    }

    // 5. 校验图片数量
    if len(req.Images) > 9 {
        return nil, errors.New("评论最多添加9张图片")
    }

    // 6. 校验@用户是否存在
    if len(req.Mentions) > 0 {
        validUsers, err := s.userRepo.FindByIDs(req.Mentions)
        if err != nil {
            return nil, err
        }
        if len(validUsers) != len(req.Mentions) {
            return nil, errors.New("部分@用户不存在")
        }
    }

    // 7. 更新评论内容
    comment.Content = req.Content
    comment.Images = req.Images
    comment.Mentions = req.Mentions

    // 编辑后重新进入审核状态（如果配置了编辑需审核）
    comment.Status = 1 // 默认直接通过，可由审核规则决定

    if err := s.commentRepo.Update(comment); err != nil {
        return nil, err
    }

    // 8. 处理新增的@用户通知
    if len(req.Mentions) > 0 {
        oldMentions := comment.Mentions
        for _, mentionedUserID := range req.Mentions {
            // 跳过已通知过的用户和自己
            if mentionedUserID == userID {
                continue
            }
            alreadyNotified := false
            for _, oldID := range oldMentions {
                if oldID == mentionedUserID {
                    alreadyNotified = true
                    break
                }
            }
            if !alreadyNotified {
                s.notificationSvc.SendNotification(&notification.SendNotificationParam{
                    UserID:     mentionedUserID,
                    Type:       "comment_mention",
                    Title:      "有人在评论中@了你",
                    Content:    req.Content,
                    TargetType: "comment",
                    TargetID:   comment.ID,
                    SenderID:   userID,
                })
            }
        }
    }

    return s.GetCommentDetail(comment.ID)
}

// GetCommentList 获取评论列表（置顶+精选优先）
func (s *CommentService) GetCommentList(req *dto.CommentListRequest) (*dto.CommentListResponse, error) {
    // 构建查询
    query := &comment.ListQuery{
        TargetType: req.TargetType,
        TargetID:   req.TargetID,
        ParentID:   nil, // 顶级评论
        Status:     1,
        Page:       req.Page,
        PageSize:   req.PageSize,
    }

    // 排序逻辑：置顶 > 精选+官方 > 按时间/点赞排序
    items, total, err := s.commentRepo.FindTopLevelWithPriority(query, req.Sort)
    if err != nil {
        return nil, err
    }

    // 加载子评论
    for i := range items {
        children, err := s.commentRepo.FindChildren(items[i].ID, 3)
        if err == nil {
            items[i].Children = children
        }
    }

    return &dto.CommentListResponse{
        Items:    items,
        Total:    total,
        Page:     req.Page,
        PageSize: req.PageSize,
    }, nil
}

// LikeComment 点赞/取消点赞评论
func (s *CommentService) LikeComment(commentID, userID uint) (bool, error) {
    // 检查是否已点赞
    liked, _ := s.commentRepo.IsLiked(commentID, userID)

    if liked {
        // 取消点赞
        if err := s.commentRepo.DeleteLike(commentID, userID); err != nil {
            return false, err
        }
        s.commentRepo.DecrementLikeCount(commentID)
        return false, nil
    }

    // 创建点赞
    like := &model.CommentLike{
        CommentID: commentID,
        UserID:    userID,
    }
    if err := s.commentRepo.CreateLike(like); err != nil {
        return false, err
    }
    s.commentRepo.IncrementLikeCount(commentID)

    // 发送点赞通知
    comment, _ := s.commentRepo.FindByID(commentID)
    if comment != nil && comment.UserID != userID {
        s.notificationSvc.SendNotification(&notification.SendNotificationParam{
            UserID:     comment.UserID,
            Type:       "comment_like",
            Title:      "有人赞了你的评论",
            Content:    comment.Content,
            TargetType: "comment",
            TargetID:   commentID,
            SenderID:   userID,
        })
    }

    return true, nil
}

// ReportComment 举报评论
func (s *CommentReportService) ReportComment(commentID, userID uint, req *dto.ReportCommentRequest) error {
    // 检查评论是否存在
    comment, err := s.commentRepo.FindByID(commentID)
    if err != nil {
        return errors.New("评论不存在")
    }

    // 不能举报自己的评论
    if comment.UserID == userID {
        return errors.New("不能举报自己的评论")
    }

    // 检查是否已举报
    exists, _ := s.reportRepo.HasReported(commentID, userID)
    if exists {
        return errors.New("您已举报过该评论")
    }

    // 创建举报记录
    report := &model.CommentReport{
        CommentID:  commentID,
        UserID:     userID,
        Reason:     req.Reason,
        ReasonType: req.ReasonType,
        Status:     0,
    }

    if err := s.reportRepo.Create(report); err != nil {
        return err
    }

    // 更新评论状态为已举报
    s.commentRepo.UpdateStatus(commentID, 3)

    return nil
}

// HandleReport 处理举报
func (s *CommentReportService) HandleReport(reportID uint, handlerID uint, req *dto.HandleReportRequest) error {
    report, err := s.reportRepo.FindByID(reportID)
    if err != nil {
        return errors.New("举报记录不存在")
    }

    now := time.Now()
    report.Status = req.Status
    report.HandlerID = &handlerID
    report.HandleRemark = req.HandleRemark
    report.HandledAt = &now

    comment, _ := s.commentRepo.FindByID(report.CommentID)

    switch req.Action {
    case "warn":
        // 仅警告：通知被举报用户
        if comment != nil {
            s.notificationSvc.SendNotification(&notification.SendNotificationParam{
                UserID:     comment.UserID,
                Type:       "system",
                Title:      "评论违规警告",
                Content:    fmt.Sprintf("您的评论「%s」被举报，原因：%s。请注意遵守社区规范。", truncate(comment.Content, 20), report.Reason),
                TargetType: "comment",
                TargetID:   comment.ID,
            })
        }
        report.Status = 1

    case "mute":
        // 禁言：加入评论黑名单（默认7天）
        if comment != nil {
            expireAt := time.Now().AddDate(0, 0, 7)
            blacklist := &model.CommentBlacklist{
                UserID:     comment.UserID,
                OperatorID: handlerID,
                Reason:     fmt.Sprintf("举报处理：%s", report.Reason),
                BanType:    1, // 全局禁言
                ExpireAt:   &expireAt,
            }
            s.blacklistRepo.Create(blacklist)

            s.notificationSvc.SendNotification(&notification.SendNotificationParam{
                UserID:     comment.UserID,
                Type:       "system",
                Title:      "您已被禁言",
                Content:    fmt.Sprintf("因评论违规，您已被禁言至 %s。原因：%s", expireAt.Format("2006-01-02 15:04"), report.Reason),
                TargetType: "comment",
                TargetID:   comment.ID,
            })
        }
        report.Status = 2

    case "delete":
        // 删除评论
        if comment != nil {
            s.commentRepo.Delete(comment.ID)
            s.notificationSvc.SendNotification(&notification.SendNotificationParam{
                UserID:     comment.UserID,
                Type:       "system",
                Title:      "评论已被删除",
                Content:    fmt.Sprintf("您的评论因违规已被删除。原因：%s", report.Reason),
                TargetType: "comment",
                TargetID:   comment.ID,
            })
        }
        report.Status = 3

    case "reject":
        // 驳回举报
        report.Status = 4
    }

    return s.reportRepo.Update(report)
}

func truncate(s string, maxLen int) string {
    runes := []rune(s)
    if len(runes) <= maxLen {
        return s
    }
    return string(runes[:maxLen]) + "..."
}

type SendNotificationParam struct {
    UserID     uint
    Type       string
    Title      string
    Content    string
    TargetType string
    TargetID   uint
    SenderID   uint
    Extra      map[string]interface{}
}

func NewNotificationService() *NotificationService {
    return &NotificationService{
        notificationRepo: notification.NewNotificationRepo(),
        settingsRepo:     notification.NewSettingsRepo(),
        cache:            cache.NewRedisCache(),
        wsHub:            wsService.GetHub(),
    }
}

// SendNotification 发送通知
func (s *NotificationService) SendNotification(param *SendNotificationParam) error {
    // 1. 检查用户通知设置
    settings, err := s.settingsRepo.FindByUserID(param.UserID)
    if err != nil {
        // 没有设置则使用默认设置
        settings = &model.NotificationSetting{
            UserID:         param.UserID,
            CommentReply:   true,
            CommentMention: true,
            CommentLike:    true,
            System:         true,
            Security:       true,
        }
    }

    // 2. 根据通知类型检查是否开启
    if !s.isNotificationEnabled(settings, param.Type) {
        return nil
    }

    // 3. 检查免打扰
    if s.isDNDActive(settings) {
        // 安全类通知不受免打扰影响
        if !isSecurityType(param.Type) {
            return nil
        }
    }

    // 4. 创建通知记录
    notification := &model.Notification{
        UserID:     param.UserID,
        Type:       param.Type,
        Title:      param.Title,
        Content:    param.Content,
        TargetType: param.TargetType,
        TargetID:   &param.TargetID,
        SenderID:   &param.SenderID,
        Extra:      param.Extra,
        IsRead:     false,
    }

    if err := s.notificationRepo.Create(notification); err != nil {
        return err
    }

    // 5. WebSocket 实时推送
    s.wsHub.SendToUser(param.UserID, &wsService.WSMessage{
        Type: "notification",
        Data: map[string]interface{}{

// isDNDActive 检查当前是否处于免打扰时段
func (s *NotificationService) isDNDActive(settings *model.NotificationSetting) bool {
    if !settings.DNDEnabled || settings.DNDStartTime == nil || settings.DNDEndTime == nil {
        return false
    }

    now := time.Now()
    currentTime := now.Hour()*3600 + now.Minute()*60 + now.Second()

    startParts := settings.DNDStartTime
    endParts := settings.DNDEndTime
    startSeconds := startParts.Hour()*3600 + startParts.Minute()*60 + startParts.Second()
    endSeconds := endParts.Hour()*3600 + endParts.Minute()*60 + endParts.Second()

    if startSeconds <= endSeconds {
        // 同一天内，如 08:00 - 22:00
        return currentTime >= startSeconds && currentTime <= endSeconds
    }
    // 跨天，如 22:00 - 08:00
    return currentTime >= startSeconds || currentTime <= endSeconds
}

// isSecurityType 是否为安全类通知
func isSecurityType(notifType string) bool {
    securityTypes := []string{"security_login", "security_password", "security_bind", "security_logout"}
    for _, t := range securityTypes {
        if t == notifType {
            return true
        }
    }
    return false
}

// CheckAudit 检查评论是否需要审核，返回审核结果
// 返回值: status(1=通过,2=待审核), needAudit(是否需要人工审核)
func (s *CommentAuditService) CheckAudit(userID uint, content string, targetType int8, targetID uint) (status int8, needAudit bool, err error) {
    rules, err := s.auditRuleRepo.FindEnabled()
    if err != nil {
        // 查询失败默认放行
        return 1, false, nil
    }

    // 按优先级排序逐条检查
    for _, rule := range rules {
        matched := false
        switch rule.RuleType {
        case "sensitive_word":
            // 敏感词匹配
            for _, word := range rule.Config.SensitiveWords {
                if sensitive.ContainsWord(content, word) {
                    matched = true
                    break
                }
            }

        case "new_user":
            // 新用户检查：注册天数小于阈值
            isNew, _ := s.isNewUser(userID, rule.Config.NewUserDays)
            if isNew {
                matched = true
            }

        case "keyword":
            // 关键词触发
            for _, kw := range rule.Config.Keywords {
                if sensitive.ContainsWord(content, kw) {
                    matched = true
                    break
                }
            }

        case "frequency":
            // 高频发评论检查
            isHighFreq, _ := s.isHighFrequency(userID, rule.Config.MaxCommentsPerHour)
            if isHighFreq {
                matched = true
            }
        }

        if matched {
            switch rule.Action {
            case 1: // 标记待审核
                return 2, true, nil
            case 2: // 自动隐藏
                return 0, false, nil
            case 3: // 自动删除
                return -1, false, nil
            }
        }
    }

    return 1, false, nil // 默认通过
}

// BatchSend 群发通知
func (s *NotificationEnhancedService) BatchSend(req *dto.BatchSendRequest, senderID uint) (*dto.BatchSendResponse, error) {
    batchID := uuid.New().String()

    // 确定目标用户列表
    var userIDs []uint
    switch req.TargetType {
    case "user":
        userIDs = req.TargetIDs
    case "role":
        ids, err := s.userRepo.FindIDsByRoles(req.TargetRoles)
        if err != nil {
            return nil, err
        }
        userIDs = ids
    case "all":
        ids, err := s.userRepo.FindAllIDs()
        if err != nil {
            return nil, err
        }
        userIDs = ids
    default:
        return nil, errors.New("无效的target_type")
    }

    if len(userIDs) == 0 {
        return nil, errors.New("没有目标用户")
    }

    // 异步发送（放入后台任务队列）
    go func() {
        for _, uid := range userIDs {
            for _, channel := range req.Channels {
                notif := &model.Notification{
                    UserID:     uid,
                    Type:       req.Type,
                    Title:      req.Title,
                    Content:    req.Content,
                    TargetType: req.TargetType,
                    SenderID:   &senderID,
                    Channel:    channel,
                    BatchID:    batchID,
                    IsRead:     false,
                }

                if err := s.notificationRepo.Create(notif); err != nil {
                    continue
                }

                // 站内推送
                if channel == "site" || channel == "all" {
                    s.wsHub.SendToUser(uid, &wsService.WSMessage{
                        Type: "notification",
                        Data: map[string]interface{}{
                            "id":      notif.ID,
                            "type":    req.Type,
                            "title":   req.Title,
                            "content": req.Content,
                        },
                    })
                    s.incrementUnreadCount(uid)
                }

                // 外部推送
                if channel == "sms" || channel == "wechat" {
                    s.pushSvc.Push(uid, channel, req.Title, req.Content)
                }
            }
        }
    }()

    return &dto.BatchSendResponse{
        BatchID: batchID,
        Total:   len(userIDs),
        Status:  "sending",
    }, nil
}

// RecallNotification 撤回通知
func (s *NotificationEnhancedService) RecallNotification(notificationID uint, operatorID uint) error {
    notif, err := s.notificationRepo.FindByID(notificationID)
    if err != nil {
        return errors.New("通知不存在")
    }

    if notif.IsRecalled {
        return errors.New("该通知已被撤回")
    }

    now := time.Now()
    notif.IsRecalled = true
    notif.RecalledAt = &now

    if err := s.notificationRepo.Update(notif); err != nil {
        return err
    }

    // 通过WebSocket通知客户端撤回
    s.wsHub.SendToUser(notif.UserID, &wsService.WSMessage{
        Type: "notification_recall",
        Data: map[string]interface{}{
            "id": notif.ID,
        },
    })

    return nil
}

// GetStats 获取通知统计
func (s *NotificationEnhancedService) GetStats(req *dto.NotificationStatsRequest) (*dto.NotificationStatsResponse, error) {
    startDate := req.StartDate
    endDate := req.EndDate

    // 总发送量
    totalSent, err := s.notificationRepo.CountByDateRange(startDate, endDate)
    if err != nil {
        return nil, err
    }

    // 已读量
    totalRead, err := s.notificationRepo.CountReadByDateRange(startDate, endDate)
    if err != nil {
        return nil, err
    }

    // 点击量（通过extra中的clicked标记统计）
    totalClicked, err := s.notificationRepo.CountClickedByDateRange(startDate, endDate)
    if err != nil {
        return nil, err
    }

    readRate := float64(0)
    if totalSent > 0 {
        readRate = float64(totalRead) / float64(totalSent)
    }
    clickRate := float64(0)
    if totalSent > 0 {
        clickRate = float64(totalClicked) / float64(totalSent)
    }

    // 每日统计
    dailyStats, err := s.notificationRepo.DailyStats(startDate, endDate)
    if err != nil {
        return nil, err
    }

    // 按类型统计
    typeStats, err := s.notificationRepo.TypeStats(startDate, endDate)
    if err != nil {
        return nil, err
    }

    return &dto.NotificationStatsResponse{
        TotalSent:     totalSent,
        TotalRead:     totalRead,
        ReadRate:      readRate,
        TotalClicked:  totalClicked,
        ClickRate:     clickRate,
        DailyStats:    dailyStats,
        TypeStats:     typeStats,
    }, nil
}

// CreateScheduledNotification 创建定时通知
func (s *NotificationEnhancedService) CreateScheduledNotification(req *dto.CreateScheduledNotificationRequest, senderID uint) (*dto.ScheduledNotificationResponse, error) {
    // 将定时任务存储到数据库或Redis
    // 这里使用Redis sorted set，score为执行时间戳
    ctx := context.Background()
    key := "qh:notif:scheduled"

    taskData := map[string]interface{}{
        "title":      req.Title,
        "content":    req.Content,
        "type":       req.Type,
        "target_type": req.TargetType,
        "target_ids": req.TargetIDs,

    s.cache.HSet(ctx, "qh:notif:scheduled:"+taskID, taskData)
    s.cache.ZAdd(ctx, key, float64(req.ScheduledAt.Unix()), taskID)

    return &dto.ScheduledNotificationResponse{
        TaskID:      taskID,
        ScheduledAt: req.ScheduledAt,
        Status:      "pending",

// CancelScheduledNotification 取消定时通知
func (s *NotificationEnhancedService) CancelScheduledNotification(taskID string) error {
    ctx := context.Background()
    key := "qh:notif:scheduled"

    removed, err := s.cache.ZRem(ctx, key, taskID)
    if err != nil || removed == 0 {
        return errors.New("定时任务不存在或已执行")
    }

    s.cache.Del(ctx, "qh:notif:scheduled:"+taskID)
    return nil
}

// GetScheduledNotifications 获取定时通知列表

            })
        }

    // 截取短信内容（短信有长度限制）

    }

    reqBody := SMSRequest{

        SignName:      c.signName,
        TemplateCode:  c.templateID,
        TemplateParam: map[string]string{"content": content},
    }

    body, _ := json.Marshal(reqBody)

    req, err := http.NewRequest("POST", c.endpoint, bytes.NewReader(body))
    if err != nil {
        return err
    }
    req.Header.Set("Content-Type", "application/json")
    // 实际生产环境需要添加阿里云短信API签名认证头

    client := &http.Client{Timeout: 10 * time.Second}
    resp, err := client.Do(req)
    if err != nil {
        logger.Errorf("短信发送失败: %v", err)
        return err
    }
    defer resp.Body.Close()

    var smsResp SMSResponse
    respBody, _ := io.ReadAll(resp.Body)
    if err := json.Unmarshal(respBody, &smsResp); err != nil {
        logger.Errorf("短信响应解析失败: %v", err)

    }

    logger.Infof("短信发送成功: %s, BizID: %s", phone, smsResp.BizID)
    return nil
}

// getAccessToken 获取微信access_token
func (c *WeChatClient) getAccessToken() (string, error) {
    if c.accessToken != "" && time.Now().Before(c.tokenExpireAt) {
        return c.accessToken, nil
    }

    url := fmt.Sprintf("https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential&appid=%s&secret=%s",
        c.appID, c.appSecret)

    client := &http.Client{Timeout: 10 * time.Second}
    resp, err := client.Get(url)
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()

    var result struct {
        AccessToken string `json:"access_token"`
        ExpiresIn   int    `json:"expires_in"`
        ErrCode     int    `json:"errcode"`
        ErrMsg      string `json:"errmsg"`
    }
    body, _ := io.ReadAll(resp.Body)
    if err := json.Unmarshal(body, &result); err != nil {
        return "", err
    }

    if result.ErrCode != 0 {
        return "", fmt.Errorf("获取access_token失败: %s", result.ErrMsg)
    }

    c.accessToken = result.AccessToken
    c.tokenExpireAt = time.Now().Add(time.Duration(result.ExpiresIn-300) * time.Second) // 提前5分钟刷新

    return c.accessToken, nil
}

// SendTemplateMessage 发送微信模板消息
func (c *WeChatClient) SendTemplateMessage(openID string, title string, content string) error {
    token, err := c.getAccessToken()

    msg := WeChatTemplateMsg{
        ToUser:     openID,
        TemplateID: c.templateID,
        Data: map[string]WeChatTemplateDataItem{
            "title":   {Value: title, Color: "#333333"},
            "content": {Value: content, Color: "#666666"},
            "time":    {Value: time.Now().Format("2006-01-02 15:04"), Color: "#999999"},
        },
    }

    body, _ := json.Marshal(msg)
    url := fmt.Sprintf("https://api.weixin.qq.com/cgi-bin/message/template/send?access_token=%s", token)

    req, err := http.NewRequest("POST", url, bytes.NewReader(body))
    if err != nil {
        return err
    }
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{Timeout: 10 * time.Second}
    resp, err := client.Do(req)
    if err != nil {
        logger.Errorf("微信模板消息发送失败: %v", err)
        return err
    }
    defer resp.Body.Close()

    logger.Infof("微信模板消息发送成功: %s", openID)
    return nil
}

    Send   chan []byte

// run 运行Hub
func (h *Hub) run() {
    for {
        select {
        case client := <-h.register:
            h.mu.Lock()
            if _, ok := h.clients[client.UserID]; !ok {
                h.clients[client.UserID] = make(map[*Client]bool)
            }
            h.clients[client.UserID][client] = true
            h.mu.Unlock()

            // 更新用户在线状态
            h.updateOnlineStatus(client.UserID, true)
            log.Printf("WebSocket: 用户 %d 连接, 当前连接数: %d", client.UserID, h.ClientCount())

        case client := <-h.unregister:
            h.mu.Lock()
            if clients, ok := h.clients[client.UserID]; ok {
                delete(clients, client)
                if len(clients) == 0 {
                    delete(h.clients, client.UserID)
                    h.updateOnlineStatus(client.UserID, false)
                }
            }
            close(client.Send)
            h.mu.Unlock()
            log.Printf("WebSocket: 用户 %d 断开, 当前连接数: %d", client.UserID, h.ClientCount())
        }
    }
}

// SendToUser 向指定用户发送消息
func (h *Hub) SendToUser(userID uint, msg *WSMessage) {
    data, err := json.Marshal(msg)
    if err != nil {
        return
    }

    h.mu.RLock()
    defer h.mu.RUnlock()

    if clients, ok := h.clients[userID]; ok {
        for client := range clients {
            select {
            case client.Send <- data:
            default:
                // 发送失败，关闭连接
                close(client.Send)
                delete(clients, client)
            }
        }
    }

    // 同时通过 Redis Pub/Sub 广播（支持多实例部署）
    ctx := context.Background()
    database.RDB.Publish(ctx, "qh:ws:broadcast", string(data))
}

// subscribeRedis 订阅 Redis 消息（支持多实例）
func (h *Hub) subscribeRedis() {
    ctx := context.Background()
    pubsub := database.RDB.Subscribe(ctx, "qh:ws:broadcast")
    defer pubsub.Close()

    for msg := range pubsub.Channel() {
        var wsMsg WSMessage
        if err := json.Unmarshal([]byte(msg.Payload), &wsMsg); err != nil {
            continue
        }

        // 解析目标用户ID并发送
        if data, ok := wsMsg.Data.(map[string]interface{}); ok {
            if uid, ok := data["user_id"].(float64); ok {
                h.mu.RLock()
                if clients, ok := h.clients[uint(uid)]; ok {
                    for client := range clients {
                        select {
                        case client.Send <- []byte(msg.Payload):
                        default:
                        }
                    }
                }
                h.mu.RUnlock()
            }
        }
    }
}

// ClientCount 获取连接数
func (h *Hub) ClientCount() int {
    h.mu.RLock()
    defer h.mu.RUnlock()
    count := 0
    for _, clients := range h.clients {
        count += len(clients)
    }
    return count
}

// updateOnlineStatus 更新在线状态
func (h *Hub) updateOnlineStatus(userID uint, online bool) {
    ctx := context.Background()
    key := "qh:user:online:" + fmt.Sprintf("%d", userID)
    if online {
        database.RDB.Set(ctx, key, "1", 5*time.Minute)
    } else {
        database.RDB.Del(ctx, key)
    }
}

// HandleWebSocket WebSocket 连接处理
func HandleWebSocket(c *gin.Context) {
    userID, exists := c.Get("user_id")
    if !exists {
        c.JSON(http.StatusUnauthorized, gin.H{"code": 401, "message": "未认证"})
        return
    }

    conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
    if err != nil {
        log.Printf("WebSocket升级失败: %v", err)
        return
    }

    client := &Client{
        UserID: userID.(uint),
        Conn:   conn,
        Send:   make(chan []byte, 256),
    }

    GetHub().register <- client

    // 读协程
    go func() {
        defer func() {
            GetHub().unregister <- client
            conn.Close()
        }()

        conn.SetReadLimit(512)
        conn.SetReadDeadline(time.Now().Add(60 * time.Second))
        conn.SetPongHandler(func(string) error {
            conn.SetReadDeadline(time.Now().Add(60 * time.Second))
            return nil
        })

        for {
            _, _, err := conn.ReadMessage()
            if err != nil {
                break
            }
        }
    }()

    // 写协程 + 心跳
    go func() {
        ticker := time.NewTicker(30 * time.Second)
        defer func() {
            ticker.Stop()
            conn.Close()
        }()

        for {
            select {
            case message, ok := <-client.Send:
                conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
                if !ok {
                    conn.WriteMessage(websocket.CloseMessage, []byte{})
                    return
                }
                conn.WriteMessage(websocket.TextMessage, message)

            case <-ticker.C:
                conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
                if err := conn.WriteMessage(websocket.PingMessage, nil); err != nil {
                    return
                }
            }
        }
    }()
}

```typescript
// utils/websocket.ts

type WSMessage = {
  type: string
  data: any
}

type MessageHandler = (data: any) => void

class WSClient {
  private ws: WebSocket | null = null
  private url: string
  private token: string
  private handlers: Map<string, MessageHandler[]> = new Map()
  private reconnectTimer: any = null
  private heartbeatTimer: any = null
  private reconnectAttempts = 0
  private maxReconnectAttempts = 10

  constructor(url: string, token: string) {
    this.url = url
    this.token = token
  }

  connect() {
    if (this.ws?.readyState === WebSocket.OPEN) return

    this.ws = new WebSocket(`${this.url}?token=${this.token}`)

    this.ws.onopen = () => {
      console.log('[WS] 连接成功')
      this.reconnectAttempts = 0
      this.startHeartbeat()
    }

    this.ws.onmessage = (event) => {
      try {
        const msg: WSMessage = JSON.parse(event.data)
        this.dispatch(msg.type, msg.data)
      } catch (e) {
        console.error('[WS] 消息解析失败:', e)
      }
    }

    this.ws.onclose = () => {
      console.log('[WS] 连接关闭')
      this.stopHeartbeat()
      this.tryReconnect()
    }

    this.ws.onerror = (error) => {
      console.error('[WS] 连接错误:', error)
    }
  }

  private startHeartbeat() {
    this.heartbeatTimer = setInterval(() => {
      if (this.ws?.readyState === WebSocket.OPEN) {
        this.ws.send(JSON.stringify({ type: 'ping' }))
      }
    }, 30000)
  }

  private stopHeartbeat() {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer)
      this.heartbeatTimer = null
    }
  }

  private tryReconnect() {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      console.log('[WS] 重连次数超过限制，停止重连')
      return
    }

    const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempts), 30000)
    this.reconnectTimer = setTimeout(() => {
      this.reconnectAttempts++
      console.log(`[WS] 第${this.reconnectAttempts}次重连...`)
      this.connect()
    }, delay)
  }

  on(type: string, handler: MessageHandler) {
    if (!this.handlers.has(type)) {
      this.handlers.set(type, [])
    }
    this.handlers.get(type)!.push(handler)
  }

  off(type: string, handler: MessageHandler) {
    const handlers = this.handlers.get(type)
    if (handlers) {
      const index = handlers.indexOf(handler)
      if (index > -1) {
        handlers.splice(index, 1)
      }
    }
  }

  private dispatch(type: string, data: any) {
    const handlers = this.handlers.get(type) || []
    handlers.forEach((handler) => handler(data))
  }

  disconnect() {
    this.stopHeartbeat()
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer)
    }
    this.ws?.close()
    this.ws = null
  }
}

export default WSClient
```

    }

    wsClient.value?.disconnect()
    wsClient.value = null
  })

    notificationSvc *notification.NotificationEnhancedService
    wsHub           *wsService.Hub
    userRepo        *user.UserRepo
}

func NewScheduledNotificationTask() *ScheduledNotificationTask {
    return &ScheduledNotificationTask{

        notificationSvc: notification.NewNotificationEnhancedService(),
        wsHub:           wsService.GetHub(),

// Run 执行定时通知任务
func (t *ScheduledNotificationTask) Run() {
    ctx := context.Background()
    key := "qh:notif:scheduled"

    now := float64(time.Now().Unix())

    // 获取所有到期的定时任务
    taskIDs, err := t.cache.ZRangeByScore(ctx, key, 0, now)
    if err != nil {
        logger.Errorf("获取定时通知任务失败: %v", err)
        return
    }

    for _, taskID := range taskIDs {
        // 从sorted set中移除（防止重复执行）
        removed, _ := t.cache.ZRem(ctx, key, taskID)
        if removed == 0 {
            continue // 已被其他实例处理
        }

        // 获取任务详情
        taskKey := "qh:notif:scheduled:" + taskID
        taskData, err := t.cache.HGetAll(ctx, taskKey)
        if err != nil || len(taskData) == 0 {
            continue
        }

        // 解析任务数据
        var targetIDs []uint
        var targetRoles []string
        var channels []string
        json.Unmarshal([]byte(taskData["target_ids"]), &targetIDs)
        json.Unmarshal([]byte(taskData["target_roles"]), &targetRoles)
        json.Unmarshal([]byte(taskData["channels"]), &channels)

        senderID := uint(0)
        json.Unmarshal([]byte(taskData["sender_id"]), &senderID)

        // 构建群发请求并执行
        batchReq := &dto.BatchSendRequest{
            Title:       taskData["title"],
            Content:     taskData["content"],
            Type:        taskData["type"],
            TargetType:  taskData["target_type"],
            TargetIDs:   targetIDs,
            TargetRoles: targetRoles,
            Channels:    channels,
        }

        _, err = t.notificationSvc.BatchSend(batchReq, senderID)
        if err != nil {
            logger.Errorf("定时通知发送失败 taskID=%s: %v", taskID, err)
        } else {
            logger.Infof("定时通知已发送 taskID=%s", taskID)
        }

        // 清理任务数据
        t.cache.Del(ctx, taskKey)
    }
}
