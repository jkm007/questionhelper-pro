```typescript
// store/modules/user.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getToken, setToken, removeToken } from '@/utils/auth'

export const useUserStore = defineStore('user', () => {
  const token = ref(getToken())
  const userInfo = ref<any>(null)
  
  const isLoggedIn = computed(() => !!token.value)
  const nickname = computed(() => userInfo.value?.nickname || '')
  const avatar = computed(() => userInfo.value?.avatar || '')
  
  const login = async (username: string, password: string) => {
    const res = await loginApi({ username, password })
    token.value = res.data.token
    setToken(res.data.token)
    userInfo.value = res.data.user
  }
  
  const logout = async () => {
    await logoutApi()
    token.value = ''
    userInfo.value = null
    removeToken()
  }
  
  return { token, userInfo, isLoggedIn, nickname, avatar, login, logout }
})
```

## 六、自定义 Hooks

### 6.1 分页 Hook

```typescript
// hooks/usePagination.ts

import { ref, reactive } from 'vue'

export const usePagination = <T>(fetchFn: (params: any) => Promise<any>) => {
  const list = ref<T[]>([]) as any
  const loading = ref(false)
  const finished = ref(false)
  
  const pagination = reactive({
    page: 1,
    pageSize: 10,
    total: 0
  })
  
  const loadData = async (isRefresh = false) => {
    if (loading.value) return
    
    if (isRefresh) {
      pagination.page = 1
      finished.value = false
    }
    
    loading.value = true
    try {
      const res = await fetchFn({
        page: pagination.page,
        page_size: pagination.pageSize
      })
      
      const { list: newList, total } = res.data
      
      if (isRefresh) {
        list.value = newList || []
      } else {
        list.value = [...list.value, ...(newList || [])]
      }
      
      pagination.total = total || 0
      
      if (list.value.length >= pagination.total) {
        finished.value = true
      } else {
        pagination.page++
      }
    } finally {
      loading.value = false
    }
  }
  
  return { list, loading, finished, pagination, loadData }
}
```

### 7.3 状态管理

```typescript
// store/modules/study-plan.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useStudyPlanStore = defineStore('studyPlan', () => {
  const plans = ref<any[]>([])
  const todayTask = ref<any>(null)

  const activePlans = computed(() =>
    plans.value.filter(p => p.status === 'active')
  )

  const todayProgress = computed(() => {
    if (!todayTask.value) return 0
    return Math.min(
      Math.round((todayTask.value.completed_count / todayTask.value.target_count) * 100),
      100
    )
  })

  const fetchPlans = async () => {
    const res = await uni.request({
      url: '/api/v1/study-plans',
      method: 'GET',
      header: { Authorization: `Bearer ${getToken()}` }
    })
    plans.value = res.data.data || []
  }

  const fetchTodayTask = async () => {
    const res = await uni.request({
      url: '/api/v1/study-plans/today',
      method: 'GET',
      header: { Authorization: `Bearer ${getToken()}` }
    })
    todayTask.value = res.data.data
  }

  const createPlan = async (data: {
    name: string
    daily_target: number
    total_days: number
    start_date: string
  }) => {
    const res = await uni.request({
      url: '/api/v1/study-plans',
      method: 'POST',
      data,
      header: { Authorization: `Bearer ${getToken()}` }
    })
    plans.value.unshift(res.data.data)
    return res.data.data
  }

  const updatePlan = async (id: number, data: Partial<StudyPlan>) => {
    const res = await uni.request({
      url: `/api/v1/study-plans/${id}`,
      method: 'PUT',
      data,
      header: { Authorization: `Bearer ${getToken()}` }
    })
    const index = plans.value.findIndex(p => p.id === id)
    if (index !== -1) plans.value[index] = res.data.data
  }

  const deletePlan = async (id: number) => {
    await uni.request({
      url: `/api/v1/study-plans/${id}`,
      method: 'DELETE',
      header: { Authorization: `Bearer ${getToken()}` }
    })
    plans.value = plans.value.filter(p => p.id !== id)
  }

  return {
    plans, todayTask, activePlans, todayProgress,
    fetchPlans, fetchTodayTask, createPlan, updatePlan, deletePlan
  }
})
```

```vue
  records: CheckinRecord[]
  stats: CheckinStats
}>()

const emit = defineEmits(['makeup'])

const now = new Date()
const currentYear = ref(now.getFullYear())
const currentMonth = ref(now.getMonth() + 1)
const weekdays = ['日', '一', '二', '三', '四', '五', '六']

const calendarCells = computed(() => {
  const year = currentYear.value
  const month = currentMonth.value
  const firstDay = new Date(year, month - 1, 1).getDay()
  const daysInMonth = new Date(year, month, 0).getDate()
  const today = new Date()
  const todayStr = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}-${String(today.getDate()).padStart(2, '0')}`

  const cells: any[] = []

  // 填充月初空白
  for (let i = 0; i < firstDay; i++) {
    cells.push({ day: null })
  }

  // 填充日期
  for (let d = 1; d <= daysInMonth; d++) {
    const dateStr = `${year}-${String(month).padStart(2, '0')}-${String(d).padStart(2, '0'}`
    const record = props.records.find(r => r.date === dateStr)
    const cellDate = new Date(year, month - 1, d)

    cells.push({
      day: d,
      date: dateStr,
      isToday: dateStr === todayStr,
      isChecked: !!record,
      isMakeup: record?.is_makeup || false,
      isFuture: cellDate > today
    })
  }

  return cells
})

const prevMonth = () => {
  if (currentMonth.value === 1) {
    currentMonth.value = 12
    currentYear.value--
  } else {
    currentMonth.value--
  }
}

const nextMonth = () => {
  if (currentMonth.value === 12) {
    currentMonth.value = 1
    currentYear.value++
  } else {
    currentMonth.value++
  }
}

  if (!cell.isChecked && !cell.isFuture) {
    const today = new Date()
    const cellDate = new Date(cell.date)
    const diffDays = Math.floor((today.getTime() - cellDate.getTime()) / 86400000)
    if (diffDays <= 3 && diffDays > 0) {
      emit('makeup', cell.date)
    }
  }
}
```

```typescript
// hooks/useCheckin.ts

import { ref, onMounted } from 'vue'

export const useCheckin = () => {
  const records = ref<CheckinRecord[]>([])
  const stats = ref<CheckinStats>({
    total_days: 0,
    current_streak: 0,
    longest_streak: 0,
    month_days: 0,
    achievements: []
  })
  const loading = ref(false)

  const fetchCheckinData = async (year?: number, month?: number) => {
    loading.value = true
    try {
      const now = new Date()
      const y = year || now.getFullYear()
      const m = month || now.getMonth() + 1

      const res = await uni.request({
        url: `/api/v1/checkin?year=${y}&month=${m}`,
        method: 'GET',
        header: { Authorization: `Bearer ${getToken()}` }
      })

      records.value = res.data.data.records || []
      stats.value = res.data.data.stats || stats.value
    } finally {
      loading.value = false
    }
  }

  const doCheckin = async () => {
    const res = await uni.request({
      url: '/api/v1/checkin',
      method: 'POST',
      header: { Authorization: `Bearer ${getToken()}` }
    })
    if (res.data.code === 0) {
      await fetchCheckinData()
      uni.showToast({ title: '打卡成功', icon: 'success' })
    }
  }

  const makeupCheckin = async (date: string) => {
    const res = await uni.request({
      url: '/api/v1/checkin/makeup',
      method: 'POST',
      data: { date },
      header: { Authorization: `Bearer ${getToken()}` }
    })
    if (res.data.code === 0) {
      await fetchCheckinData()
      uni.showToast({ title: '补卡成功', icon: 'success' })
    }
  }

  return { records, stats, loading, fetchCheckinData, doCheckin, makeupCheckin }
}
```

```typescript
export const usePush = () => {

   * 微信小程序订阅消息

    // #ifdef MP-WEIXIN

          tmplIds,
          success: resolve,
          fail: reject
        })
      })

    } catch (e) {
      console.warn('订阅消息授权失败', e)
    }
    // #endif

   * H5 浏览器通知
   */

    // #ifdef H5
    if (!('Notification' in window)) {
      console.warn('当前浏览器不支持通知')
      return false
    }
    if (Notification.permission === 'granted') return true
    if (Notification.permission === 'denied') return false

    const permission = await Notification.requestPermission()
    return permission === 'granted'
    // #endif
  }

  /**
   * 发送 H5 本地通知
   */
  const sendH5Notification = (title: string, options?: NotificationOptions) => {
    // #ifdef H5
    if (Notification.permission === 'granted') {
      new Notification(title, {
        icon: '/static/logo.png',
        badge: '/static/logo.png',
        ...options
      })
    }
    // #endif
  }

  /**
   * 初始化推送
   */
  const initPush = async () => {
    // #ifdef MP-WEIXIN
    // 小程序订阅消息模板 ID（在微信公众平台配置）
    const tmplIds = [
      'exam_remind_tmpl_id',     // 考试提醒
      'homework_tmpl_id',        // 作业通知

    ]
    await requestSubscribeMessage(tmplIds)
    // #endif

    // #ifdef H5
    await requestNotificationPermission()
    // #endif
  }

  }
```

```typescript
// utils/cache.ts

interface CacheItem<T> {
  data: T
  expire: number    // 过期时间戳（毫秒）
  version: string   // 数据版本
}

const CACHE_VERSION = 'v2.0'

export const cacheManager = {
  /**
   * 设置缓存
   */
  set<T>(key: string, data: T, ttl: number = 3600000): void {
    const item: CacheItem<T> = {
      data,
      expire: Date.now() + ttl,
      version: CACHE_VERSION
    }
    try {
      uni.setStorageSync(`qh_cache_${key}`, JSON.stringify(item))
    } catch (e) {
      // 存储空间不足时清理过期缓存
      this.clearExpired()
      try {
        uni.setStorageSync(`qh_cache_${key}`, JSON.stringify(item))
      } catch (e2) {
        console.error('缓存写入失败', e2)
      }
    }
  },

  /**
   * 获取缓存
   */
  get<T>(key: string): T | null {
    try {
      const raw = uni.getStorageSync(`qh_cache_${key}`)
      if (!raw) return null

      const item: CacheItem<T> = JSON.parse(raw)

      // 检查版本
      if (item.version !== CACHE_VERSION) {
        this.remove(key)
        return null
      }

      // 检查过期
      if (Date.now() > item.expire) {
        this.remove(key)
        return null
      }

      return item.data
    } catch (e) {
      return null
    }
  },

  /**
   * 删除缓存
   */
  remove(key: string): void {
    uni.removeStorageSync(`qh_cache_${key}`)
  },

  /**
   * 清除所有过期缓存
   */
  clearExpired(): void {
    try {
      const res = uni.getStorageInfoSync()
      const keys = res.keys.filter(k => k.startsWith('qh_cache_'))
      keys.forEach(key => {
        try {
          const raw = uni.getStorageSync(key)
          if (raw) {
            const item = JSON.parse(raw)
            if (Date.now() > item.expire || item.version !== CACHE_VERSION) {
              uni.removeStorageSync(key)
            }
          }
        } catch (e) {
          uni.removeStorageSync(key)
        }
      })
    } catch (e) {
      console.error('清理过期缓存失败', e)
    }
  },

  /**
   * 清除所有缓存
   */
  clearAll(): void {
    try {
      const res = uni.getStorageInfoSync()
      res.keys.filter(k => k.startsWith('qh_cache_')).forEach(key => {
        uni.removeStorageSync(key)
      })
    } catch (e) {
      console.error('清除缓存失败', e)
    }
  },

  /**
   * 获取缓存大小（KB）
   */
  getSize(): number {
    try {
      const res = uni.getStorageInfoSync()
      return res.currentSize || 0
    } catch (e) {
      return 0
    }
  }
}
```

### 14.3 离线练习记录同步

```typescript
// utils/offline-sync.ts

interface PendingRecord {
  id: string
  type: 'practice' | 'exam'
  data: any
  timestamp: number
  retryCount: number
}

export const offlineSync = {
  STORAGE_KEY: 'qh_offline_pending',

  /**
   * 添加待同步记录
   */
  addRecord(type: 'practice' | 'exam', data: any): void {
    const pending = this.getPending()
    const record: PendingRecord = {
      id: `${Date.now()}_${Math.random().toString(36).slice(2, 8)}`,
      type,
      data,
      timestamp: Date.now(),
      retryCount: 0
    }
    pending.push(record)
    uni.setStorageSync(this.STORAGE_KEY, JSON.stringify(pending))
  },

  /**
   * 获取待同步记录
   */
  getPending(): PendingRecord[] {
    try {
      const raw = uni.getStorageSync(this.STORAGE_KEY)
      return raw ? JSON.parse(raw) : []
    } catch (e) {
      return []
    }
  },

  /**
   * 同步所有待上传记录
   */
  async syncAll(): Promise<{ success: number; failed: number }> {
    const pending = this.getPending()
    if (pending.length === 0) return { success: 0, failed: 0 }

    let success = 0
    let failed = 0
    const remaining: PendingRecord[] = []

    for (const record of pending) {
      try {
        const url = record.type === 'practice'
          ? '/api/v1/practice/records/sync'
          : '/api/v1/exam/answers/sync'

        await uni.request({
          url,
          method: 'POST',
          data: record.data,
          header: { Authorization: `Bearer ${getToken()}` }
        })
        success++
      } catch (e) {
        record.retryCount++
        if (record.retryCount < 3) {
          remaining.push(record)
        }
        failed++
      }
    }

    uni.setStorageSync(this.STORAGE_KEY, JSON.stringify(remaining))
    return { success, failed }
  },

  /**
   * 检查网络状态并自动同步
   */
  initAutoSync(): void {
    uni.onNetworkStatusChange((res) => {
      if (res.isConnected) {
        this.syncAll().then(({ success, failed }) => {
          if (success > 0) {
```

### 15.2 下拉刷新 Hook

```typescript
// hooks/useRefresh.ts

import { ref } from 'vue'

export const useRefresh = (fetchFn: () => Promise<void>) => {
  const refreshing = ref(false)

  const onRefresh = async () => {
    if (refreshing.value) return
    refreshing.value = true
    try {
      await fetchFn()
    } finally {
      refreshing.value = false
    }
  }

  return { refreshing, onRefresh }
}
```

## 十六、上拉加载

### 16.1 功能概述

| 功能点 | 功能描述 |
|-------|---------|
| 滚动加载更多 | 列表滚动到底部自动加载下一页 |
| 加载状态 | 显示加载中/加载完成/没有更多状态 |
| 分页参数 | page + page_size 统一分页 |
| 防重复请求 | 加载中禁止重复触发 |
| 空数据处理 | 无数据时显示空状态占位图 |

### 16.2 增强版分页 Hook

```typescript
// hooks/useLoadMore.ts

import { ref, reactive, computed } from 'vue'

export interface LoadMoreOptions {
  pageSize?: number
  immediate?: boolean
}

export const useLoadMore = <T>(
  fetchFn: (params: { page: number; page_size: number }) => Promise<any>,
  options: LoadMoreOptions = {}
) => {
  const { pageSize = 10, immediate = true } = options

  const list = ref<T[]>([]) as any
  const loading = ref(false)
  const refreshing = ref(false)
  const finished = ref(false)
  const error = ref(false)

  const pagination = reactive({
    page: 1,
    pageSize,
    total: 0
  })

  const isEmpty = computed(() => !loading.value && list.value.length === 0)

  /**
   * 加载数据
   * @param isRefresh 是否刷新（重置列表）
   */
  const loadData = async (isRefresh = false) => {
    if (loading.value) return

    if (isRefresh) {
      pagination.page = 1
      finished.value = false
      error.value = false
      refreshing.value = true
    }

    if (finished.value && !isRefresh) return

    loading.value = true
    try {
      const res = await fetchFn({
        page: pagination.page,
        page_size: pagination.pageSize
      })

      const { list: newList, total } = res.data.data || res.data

      if (isRefresh) {
        list.value = newList || []
      } else {
        list.value = [...list.value, ...(newList || [])]
      }

      pagination.total = total || 0

      if (list.value.length >= pagination.total) {
        finished.value = true
      } else {
        pagination.page++
      }
    } catch (e) {
      error.value = true
      console.error('加载数据失败', e)
    } finally {
      loading.value = false
      refreshing.value = false
    }
  }

  /**
   * 下拉刷新
   */
  const onRefresh = async () => {
    await loadData(true)
  }

  /**
   * 上拉加载更多
   */
  const onLoadMore = () => {
    if (!loading.value && !finished.value && !error.value) {
      loadData()
    }
  }

  /**
   * 重试（加载失败时）
   */
  const retry = () => {
    error.value = false
    loadData()
  }

  return {
    list,
    loading,
    refreshing,
    finished,
    error,
    isEmpty,
    pagination,
    loadData,
    onRefresh,
    onLoadMore,
    retry
  }
}
```

```typescript
// utils/anti-cheat.ts

export class ScreenMonitor {
  private switchCount = 0
  private examId: string
  
  constructor(examId: string) {
    this.examId = examId
  }
  
  start() {
    // #ifdef H5
    document.addEventListener('visibilitychange', () => {
      if (document.hidden) {
        this.onScreenSwitch()
      }
    })
    // #endif
    
    // #ifdef APP-PLUS
    uni.onAppHide(() => this.onScreenSwitch())
    // #endif
    
    // #ifdef MP-WEIXIN
    wx.onHide(() => this.onScreenSwitch())
    // #endif
  }
  
  private onScreenSwitch() {
    this.switchCount++
    this.reportSwitch()
  }
  
  private async reportSwitch() {
    await uni.request({
      url: `/api/v1/exam/${this.examId}/warning`,
      method: 'POST',
      data: {
        type: 'screen_switch',
        count: this.switchCount,
        timestamp: Date.now()
      }
    })
  }
  
  getSwitchCount() {
    return this.switchCount
  }
}
```

### 21.2 断线续考

```typescript
// utils/resume-exam.ts

export class ExamResumer {
  private examId: string
  
  constructor(examId: string) {
    this.examId = examId
  }
  
  // 保存答题进度到本地
  saveProgress(answers: Record<number, string>, currentIndex: number) {
    const data = {
      exam_id: this.examId,
      answers,
      currentIndex,
      savedAt: Date.now()
    }
    uni.setStorageSync(`exam_progress_${this.examId}`, JSON.stringify(data))
  }
  
  // 恢复答题进度
  restoreProgress(): { answers: Record<number, string>; currentIndex: number } | null {
    try {
      const raw = uni.getStorageSync(`exam_progress_${this.examId}`)
      if (!raw) return null
      const data = JSON.parse(raw)
      // 检查是否在有效期内（考试时长内）
      if (Date.now() - data.savedAt > 24 * 60 * 60 * 1000) {
        this.clearProgress()
        return null
      }
      return { answers: data.answers, currentIndex: data.currentIndex }
    } catch {
      return null
    }
  }
  
  // 清除进度
  clearProgress() {
    uni.removeStorageSync(`exam_progress_${this.examId}`)
  }
  
  // 从服务器恢复（断线后重新进入）
  async resumeFromServer(): Promise<{
    answers: Record<number, string>
    remainingTime: number
    currentIndex: number
  } | null> {
    try {
      const res = await uni.request({
        url: `/api/v1/exam/${this.examId}/resume`
      })
      if (res.data.can_resume) {
        return {
          answers: res.data.answers || {},
          remainingTime: res.data.remaining_seconds,
          currentIndex: res.data.current_index || 0
        }
      }
      return null
    } catch {
      return null
    }
  }
}
```

```vue
<!-- pages/exam/answer.vue 中的集成示例 -->
<script setup lang="ts">
import { onMounted, onUnmounted } from 'vue'
import { ScreenMonitor } from '@/utils/anti-cheat'
import { ExamResumer } from '@/utils/resume-exam'

let screenMonitor: ScreenMonitor
let examResumer: ExamResumer

onMounted(async () => {
  const examId = examId.value
  
  // 初始化防作弊
  screenMonitor = new ScreenMonitor(examId)
  screenMonitor.start()
  
  // 初始化断线续考
  examResumer = new ExamResumer(examId)
  
  // 尝试恢复进度
  const serverProgress = await examResumer.resumeFromServer()
  if (serverProgress) {
    answers.value = serverProgress.answers
    currentIndex.value = serverProgress.currentIndex
    remainingTime.value = serverProgress.remainingTime
    uni.showToast({ title: '已恢复答题进度', icon: 'success' })
  }
})

// 每次答题保存进度
const saveAnswer = async (questionId: number, answer: string) => {
  answers.value[questionId] = answer
  examResumer.saveProgress(answers.value, currentIndex.value)
  
  // 同步到服务器
  await uni.request({
    url: `/api/v1/exam/${examId.value}/save-answer`,
    method: 'POST',
    data: { question_id: questionId, answer }
  })
}

onUnmounted(() => {
  examResumer?.clearProgress()
})
</script>
```
