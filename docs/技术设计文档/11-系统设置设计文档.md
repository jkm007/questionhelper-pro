```go
// service/system/settings_service.go

type SettingsService struct {
    settingsRepo *system.SettingsRepo
    cache        cache.Cache
}

// GetAll 获取所有设置
func (s *SettingsService) GetAll() (map[string]string, error) {
    // 先从缓存获取
    ctx := context.Background()
    cached, err := s.cache.Get(ctx, "system:settings")
    if err == nil {
        var settings map[string]string
        json.Unmarshal([]byte(cached), &settings)
        return settings, nil
    }
    
    // 从数据库获取
    settings, err := s.settingsRepo.GetAll()
    if err != nil {
        return nil, err
    }
    
    // 缓存
    data, _ := json.Marshal(settings)
    s.cache.Set(ctx, "system:settings", string(data), time.Hour)
    
    return settings, nil
}

// Update 更新设置
func (s *SettingsService) Update(settings map[string]string) error {
    for key, value := range settings {
        if err := s.settingsRepo.Set(key, value); err != nil {
            return err
        }
    }
    
    // 清除缓存
    ctx := context.Background()
    s.cache.Delete(ctx, "system:settings")
    
    return nil
}
```

```go
func OperationLogMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        startTime := time.Now()
        
        // 读取请求体
        var body []byte
        if c.Request.Body != nil {
            body, _ = io.ReadAll(c.Request.Body)
            c.Request.Body = io.NopCloser(bytes.NewBuffer(body))
        }
        
        // 处理请求
        c.Next()
        
        // 记录日志
        latency := time.Since(startTime).Milliseconds()
        
        log := &model.OperationLog{
            UserID:    getUserID(c),
            Username:  getUsername(c),
            Action:    getAction(c),
            Resource:  getResource(c),
            Method:    c.Request.Method,
            Path:      c.Request.URL.Path,
            IP:        c.ClientIP(),
            UserAgent: c.Request.UserAgent(),
            Status:    c.Writer.Status(),
            Latency:   latency,
        }
        
        // 异步写入数据库
        go func() {
            database.DB.Create(log)
        }()
    }
}
```

```go
func (s *LogExportService) ExportLogs(logType string, req interface{}) (string, error) {
    var records []map[string]interface{}
    var headers []string

    switch logType {
    case "login":
        result, _ := s.logService.QueryLoginLogs(req.(*dto.LoginLogQuery))
        headers = []string{"ID", "用户ID", "登录方式", "IP", "地点", "设备", "状态", "失败原因", "时间"}

                "ID": log.ID, "用户ID": log.UserID, "登录方式": log.LoginType,
                "IP": log.IP, "地点": log.Location, "设备": log.Device,
                "状态": log.Status, "失败原因": log.FailReason, "时间": log.CreatedAt,
            })
        }
    case "system":
        result, _ := s.logService.QuerySystemLogs(req.(*dto.SystemLogQuery))
        headers = []string{"ID", "级别", "模块", "消息", "时间"}
        for _, log := range result.List.([]model.SystemLog) {
            records = append(records, map[string]interface{}{
                "ID": log.ID, "级别": log.Level, "模块": log.Module,
                "消息": log.Message, "时间": log.CreatedAt,
            })
        }
    case "error":
        result, _ := s.logService.QueryErrorLogs(req.(*dto.ErrorLogQuery))
        headers = []string{"ID", "错误码", "模块", "消息", "时间"}
        for _, log := range result.List.([]model.ErrorLog) {
            records = append(records, map[string]interface{}{
                "ID": log.ID, "错误码": log.ErrorCode, "模块": log.Module,
                "消息": log.Message, "时间": log.CreatedAt,
            })
        }
    }

    // 生成CSV文件
    fileName := fmt.Sprintf("logs_%s_%s.csv", logType, time.Now().Format("20060102_150405"))
    filePath := filepath.Join("/tmp/exports", fileName)

    file, _ := os.Create(filePath)
    defer file.Close()

    writer := csv.NewWriter(file)
    writer.Write(headers)
    for _, record := range records {
        row := make([]string, len(headers))
        for i, h := range headers {
            row[i] = fmt.Sprintf("%v", record[h])
        }
        writer.Write(row)
    }
    writer.Flush()

    return filePath, nil
}
```

```go
// ScheduleCleanup 调度日志清理任务
func (t *LogCleanupTask) ScheduleCleanup() {
    // 每天凌晨3点执行
    cron.AddFunc("0 3 * * *", func() {
        t.CleanupExpiredLogs()
    })
}

// CleanupExpiredLogs 清理过期日志（默认保留90天）
func (t *LogCleanupTask) CleanupExpiredLogs() error {
    retentionDays := 90
    cutoff := time.Now().AddDate(0, 0, -retentionDays)

    // 清理登录日志
    result := t.db.Where("created_at < ?", cutoff).Delete(&model.LoginLog{})
    log.Printf("清理登录日志 %d 条", result.RowsAffected)

    // 清理系统日志
    result = t.db.Where("created_at < ?", cutoff).Delete(&model.SystemLog{})
    log.Printf("清理系统日志 %d 条", result.RowsAffected)

    // 清理错误日志（保留更长时间，180天）
    errorCutoff := time.Now().AddDate(0, 0, -180)
    result = t.db.Where("created_at < ?", errorCutoff).Delete(&model.ErrorLog{})
    log.Printf("清理错误日志 %d 条", result.RowsAffected)

    // 清理操作日志
    result = t.db.Where("created_at < ?", cutoff).Delete(&model.OperationLog{})
    log.Printf("清理操作日志 %d 条", result.RowsAffected)

    return nil
}
```

```go
// service/system/log_stats_service.go

type LogStatsService struct {
    db *gorm.DB
}

type LogStats struct {
    TotalCount    int64            `json:"total_count"`
    ErrorCount    int64            `json:"error_count"`
    ErrorRate     float64          `json:"error_rate"`
    ModuleStats   []ModuleStat     `json:"module_stats"`
    DailyStats    []DailyLogStat   `json:"daily_stats"`
    LevelStats    []LevelStat      `json:"level_stats"`
}

type ModuleStat struct {
    Module string `json:"module"`
    Count  int64  `json:"count"`
}

type DailyLogStat struct {
    Date  string `json:"date"`
    Count int64  `json:"count"`
}

type LevelStat struct {
    Level string `json:"level"`
    Count int64  `json:"count"`
}

// GetLogStats 获取日志统计
func (s *LogStatsService) GetLogStats(days int) (*LogStats, error) {
    startTime := time.Now().AddDate(0, 0, -days)
    stats := &LogStats{}

    // 总日志数
    s.db.Model(&model.SystemLog{}).Where("created_at >= ?", startTime).Count(&stats.TotalCount)

    // 错误日志数
    s.db.Model(&model.SystemLog{}).Where("created_at >= ? AND level = ?", startTime, "error").Count(&stats.ErrorCount)

    // 错误率
    if stats.TotalCount > 0 {
        stats.ErrorRate = float64(stats.ErrorCount) / float64(stats.TotalCount) * 100
    }

    // 模块分布
    s.db.Model(&model.SystemLog{}).
        Select("module, COUNT(*) as count").
        Where("created_at >= ?", startTime).
        Group("module").
        Order("count DESC").
        Find(&stats.ModuleStats)

    // 每日统计
    s.db.Model(&model.SystemLog{}).
        Select("DATE(created_at) as date, COUNT(*) as count").
        Where("created_at >= ?", startTime).
        Group("DATE(created_at)").
        Order("date").
        Find(&stats.DailyStats)

    // 级别分布
    s.db.Model(&model.SystemLog{}).
        Select("level, COUNT(*) as count").
        Where("created_at >= ?", startTime).
        Group("level").
        Find(&stats.LevelStats)

    return stats, nil
}
```

```go
        }
        if req.StartTime != "" {
            query = query.Where("created_at >= ?", req.StartTime)
        }
        if req.EndTime != "" {
            query = query.Where("created_at <= ?", req.EndTime)
        }
        query.Count(&total)
        query.Order("created_at DESC").Offset((req.Page - 1) * req.PageSize).Limit(req.PageSize).Find(&logs)
        result.Type = "error"
        result.Total = total
        result.List = logs
    }

    return result, nil
```

```go
    MaxLength    int  `json:"max_length"`

    RequireNumber bool `json:"require_number"`
    HistoryCount int  `json:"history_count"`
}

func (s *SecurityService) GetPasswordPolicy() (*PasswordPolicy, error) {
    value, err := s.repo.Get("password_policy")
    if err != nil {
        return nil, err
    }
    
    var policy PasswordPolicy
    json.Unmarshal([]byte(value), &policy)
    return &policy, nil
}

// ValidatePassword 验证密码是否符合策略
func (s *SecurityService) ValidatePassword(password string) error {
    policy, err := s.GetPasswordPolicy()
    if err != nil {
        return err
    }
    
    if len(password) < policy.MinLength {
        return fmt.Errorf("密码长度不能少于%d位", policy.MinLength)
    }
    
    if len(password) > policy.MaxLength {
        return fmt.Errorf("密码长度不能超过%d位", policy.MaxLength)
    }
    
    if policy.RequireLetter && !strings.ContainsAny(password, "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ") {
        return errors.New("密码必须包含字母")
    }
    
    if policy.RequireNumber && !strings.ContainsAny(password, "0123456789") {
        return errors.New("密码必须包含数字")
    }
    
    return nil
}
```

```go
// CheckRateLimit 检查是否超过频率限制
func (s *NotificationService) CheckRateLimit(channel, scene string) (bool, error) {
    limit, err := s.rateLimitRepo.FindByChannelAndScene(channel, scene)
    if err != nil || !limit.Enabled {
        return true, nil
    }

    cacheKey := fmt.Sprintf("notify_rate:%s:%s", channel, scene)
    count, _ := s.cache.Incr(cacheKey)
    if count == 1 {
        s.cache.Expire(cacheKey, time.Duration(limit.Period)*time.Second)
    }

    return count <= int64(limit.MaxCount), nil
}

// SendNotification 发送通知
func (s *NotificationService) SendNotification(req *dto.SendNotificationRequest) error {
    // 检查频率限制
    allowed, err := s.CheckRateLimit(req.Channel, req.Scene)
    if err != nil {
        return err
    }
    if !allowed {
        return errors.New("发送频率超过限制")
    }

    // 检查渠道是否启用
    channel, err := s.channelRepo.FindByChannel(req.Channel)
    if err != nil || !channel.Enabled {
        return errors.New("通知渠道未启用")
    }
```

```go
// CreateBackup 创建备份
func (s *BackupService) CreateBackup() error {
    // 获取配置
    config, _ := s.getConfig()
    
    // 生成备份文件名
    fileName := fmt.Sprintf("backup_%s.sql", time.Now().Format("20060102_150405"))
    filePath := filepath.Join(config.StoragePath, fileName)
    
    // 执行MySQL备份
    cmd := exec.Command("mysqldump",
        "-h", "localhost",
        "-u", "root",
        "--all-databases",
        "--result-file="+filePath,
    )
    
    if err := cmd.Run(); err != nil {
        return err
    }
    
    // 压缩
    if config.CompressEnabled {
        compressedPath := filePath + ".gz"
        if err := compressFile(filePath, compressedPath); err != nil {
            return err
        }
        os.Remove(filePath)
        filePath = compressedPath
    }
    
    // 加密
    if config.EncryptEnabled && config.EncryptKey != "" {
        encryptedPath := filePath + ".enc"
        if err := encryptFile(filePath, encryptedPath, config.EncryptKey); err != nil {
            return err
        }
        os.Remove(filePath)
        filePath = encryptedPath
    }
    
    // 记录备份
    fileInfo, _ := os.Stat(filePath)
    record := &model.BackupRecord{
        Type:       "full",
        FileName:   fileName,
        FilePath:   filePath,
        FileSize:   fileInfo.Size(),
        Encrypted:  config.EncryptEnabled,
        Compressed: config.CompressEnabled,
        Status:     1,
    }
    
    return s.recordRepo.Create(record)
}

// RestoreBackup 恢复备份
func (s *BackupService) RestoreBackup(recordID uint) error {
    record, err := s.recordRepo.FindByID(recordID)
    if err != nil {
        return err
    }
    
    filePath := record.FilePath
    
    // 解密
    if record.Encrypted {
        config, _ := s.getConfig()
        decryptedPath := filePath + ".dec"
        if err := decryptFile(filePath, decryptedPath, config.EncryptKey); err != nil {
            return err
        }
        defer os.Remove(decryptedPath)
        filePath = decryptedPath
    }
    
    // 解压
    if record.Compressed {
        decompressedPath := strings.TrimSuffix(filePath, ".gz")
        if err := decompressFile(filePath, decompressedPath); err != nil {
            return err
        }
        defer os.Remove(decompressedPath)
        filePath = decompressedPath
    }
    
    // 执行恢复
    cmd := exec.Command("mysql",
        "-h", "localhost",
        "-u", "root",
        "-e", "source "+filePath,
    )
    
    return cmd.Run()
}
```

```go
type BackupStatus struct {

// GetBackupStatus 获取备份状态监控
func (s *BackupMonitorService) GetBackupStatus() (*BackupStatus, error) {
    status := &BackupStatus{}

    // 最近一次备份
    lastRecord, err := s.recordRepo.FindLatest()
    if err == nil {
        status.LastBackupTime = &lastRecord.CreatedAt
        if lastRecord.Status == 1 {
            status.LastBackupStatus = "success"
        } else {
            status.LastBackupStatus = "failed"
        }
    }

    // 统计总数
    s.recordRepo.DB().Model(&model.BackupRecord{}).Count(&status.TotalBackups)
    s.recordRepo.DB().Model(&model.BackupRecord{}).Where("status = 1").Count(&status.SuccessCount)
    s.recordRepo.DB().Model(&model.BackupRecord{}).Where("status = 0").Count(&status.FailedCount)

    // 总大小
    s.recordRepo.DB().Model(&model.BackupRecord{}).Select("COALESCE(SUM(file_size), 0)").Scan(&status.TotalSize)

    // 下次计划时间
    config, _ := s.configRepo.Get("backup_schedule")
    status.NextScheduled = config

    return status, nil
}

// CheckBackupHealth 检查备份健康状态
func (s *BackupMonitorService) CheckBackupHealth() error {
    status, _ := s.GetBackupStatus()

    // 检查最近备份是否成功
    if status.LastBackupStatus == "failed" {
        s.notifier.SendToAdmin("备份失败告警", "最近一次备份失败，请及时检查。")
    }

    // 检查备份是否超时（超过24小时未备份）
    if status.LastBackupTime != nil && time.Since(*status.LastBackupTime) > 24*time.Hour {
        s.notifier.SendToAdmin("备份超时告警", "超过24小时未进行备份，请检查备份任务。")
```

```go
// RunDrill 执行恢复演练
func (t *RestoreDrillTask) RunDrill() error {
    startTime := time.Now()

    // 获取最近一次成功的备份
    record, err := t.recordRepo.FindLatestSuccess()
    if err != nil {
        t.notifier.SendToAdmin("恢复演练失败", "没有可用的成功备份记录。")
        return err
    }

    // 执行恢复演练（恢复到测试数据库）
    err = t.backupService.RestoreToTestDB(record.ID)
    duration := time.Since(startTime).Seconds()

    drillRecord := &RestoreDrillRecord{
        BackupID:  record.ID,
        Duration:  int64(duration),
        CreatedAt: time.Now(),
    }

    if err != nil {
        drillRecord.Status = "failed"
        drillRecord.ErrorMsg = err.Error()
        t.notifier.SendToAdmin("恢复演练失败",
            fmt.Sprintf("备份ID:%d 恢复演练失败，耗时:%.0f秒，错误:%s", record.ID, duration, err.Error()))
    } else {
        drillRecord.Status = "success"
        t.notifier.SendToAdmin("恢复演练成功",
            fmt.Sprintf("备份ID:%d 恢复演练成功，耗时:%.0f秒", record.ID, duration))
    }

    // 保存演练记录
    t.recordRepo.CreateDrillRecord(drillRecord)

    return err
}
```
