## 一、功能概述

| 功能 | 说明 |
|------|------|
| 错题收集 | 自动收集答错的题目，支持手动添加 |
| 错题管理 | 查看、搜索、排序、筛选、复习、移除错题，支持批量操作 |
| 错题分类 | 根据知识点和题型自动分类 |
| 错题分析 | 薄弱知识点分析、错题趋势、错题对比 |
| 错题复习 | 间隔重复复习，连续答对N次自动移除 |
| 错题导出 | 导出、打印、分享错题 |

```go
type WrongService struct {
    wrongRepo    *wrong.WrongRepo

// AddWrongQuestion 添加错题
func (s *WrongService) AddWrongQuestion(userID, questionID uint, source int8, sourceID uint, answer string) error {
    // 检查是否已存在
    existing, err := s.wrongRepo.FindByUserAndQuestion(userID, questionID)
    if err == nil {
        // 已存在，增加错误次数
        existing.WrongCount++
        existing.LastAnswer = answer
        existing.Mastered = false
        return s.wrongRepo.Update(existing)
    }
    
    // 新增错题
    wrong := &model.WrongQuestion{
        UserID:     userID,
        QuestionID: questionID,
        Source:     source,
        SourceID:   sourceID,
        WrongCount: 1,
        LastAnswer: answer,
        Mastered:   false,
    }
    
    return s.wrongRepo.Create(wrong)
}

// ReviewWrongQuestion 复习错题
func (s *WrongService) ReviewWrongQuestion(wrongID uint, answer string) (*dto.WrongReviewResult, error) {
    wrong, err := s.wrongRepo.FindByID(wrongID)
    if err != nil {
        return nil, err
    }
    
    // 检查答案是否正确
    question, _ := s.questionRepo.FindByID(wrong.QuestionID)
    isCorrect := s.checkAnswer(question, answer)
    
    if isCorrect {
        // 标记为已掌握
        wrong.Mastered = true
        s.wrongRepo.Update(wrong)
    }
    
    return &dto.WrongReviewResult{
        IsCorrect:    isCorrect,
        CorrectAnswer: s.getCorrectAnswer(question),
        Analysis:     question.Analysis,
    }, nil
}

// GetWrongAnalysis 获取错题分析
func (s *WrongService) GetWrongAnalysis(userID uint) (*dto.WrongAnalysisInfo, error) {
    // 总错题数
    totalCount, _ := s.wrongRepo.CountByUser(userID)
    
    // 已掌握数
    masteredCount, _ := s.wrongRepo.CountMasteredByUser(userID)
    
    // 按分类统计
    byCategory, _ := s.wrongRepo.GroupByCategory(userID)
    
    // 按题型统计
    byType, _ := s.wrongRepo.GroupByType(userID)
    
    // 按难度统计
    byDifficulty, _ := s.wrongRepo.GroupByDifficulty(userID)
    
    return &dto.WrongAnalysisInfo{
        TotalCount:    totalCount,
        MasteredCount: masteredCount,
        ByCategory:    byCategory,
        ByType:        byType,
        ByDifficulty:  byDifficulty,
    }, nil
}
```

```go
// ManualAddWrongQuestion 手动添加错题
func (s *WrongService) ManualAddWrongQuestion(userID uint, req *dto.ManualAddWrongRequest) (*model.WrongQuestion, error) {
    // 验证题目是否存在
    question, err := s.questionRepo.FindByID(req.QuestionID)
    if err != nil {
        return nil, errors.New("题目不存在")
    }

    // 检查是否已在错题本中
    existing, err := s.wrongRepo.FindByUserAndQuestion(userID, req.QuestionID)
    if err == nil {
        // 已存在，增加错误次数
        existing.WrongCount++
        existing.Mastered = false
        existing.CorrectStreak = 0
        if req.Answer != "" {
            existing.LastAnswer = req.Answer
        }
        if err := s.wrongRepo.Update(existing); err != nil {
            return nil, err
        }
        return existing, nil
    }

    // 自动分类
    categoryID := s.autoClassify(question)

    // 新增错题
    wrong := &model.WrongQuestion{
        UserID:       userID,
        QuestionID:   req.QuestionID,
        Source:       3, // 3=手动添加
        SourceID:     0,
        WrongCount:   1,
        LastAnswer:   req.Answer,
        Mastered:     false,
        CorrectStreak: 0,
    }

    if err := s.wrongRepo.Create(wrong); err != nil {
        return nil, err
    }

    // 如果有标签，自动打标签
    if categoryID > 0 {
        s.tagRepo.AddRelation(wrong.ID, categoryID)
    }

    return wrong, nil
}

// ManualAddWrongRequest 手动添加错题请求
type ManualAddWrongRequest struct {
    QuestionID uint   `json:"question_id" binding:"required"`
    Answer     string `json:"answer"`
    Remark     string `json:"remark"`
}
```

```go
// AutoClassify 根据知识点和题型自动分类
func (s *WrongClassifyService) AutoClassify(question *model.Question) []string {
    var tags []string

    // 1. 根据题型分类
    switch question.Type {
    case 1:
        tags = append(tags, "单选题")
    case 2:
        tags = append(tags, "多选题")
    case 3:
        tags = append(tags, "判断题")
    case 4:
        tags = append(tags, "填空题")
    case 5:
        tags = append(tags, "简答题")
    }

    // 2. 根据知识点分类
    for _, kp := range question.KnowledgePoints {
        tags = append(tags, kp.Name)
    }

    // 3. 根据难度分类
    switch question.Difficulty {
    case 1:
        tags = append(tags, "简单")
    case 2:
        tags = append(tags, "中等")
    case 3:
        tags = append(tags, "困难")
    }

    // 4. 根据错误模式智能分类（可选）
    // 通过分析答案特征判断错误类型
    // 例如：答案为空 -> 不会；答案与正确答案相近 -> 粗心；答案与某个干扰项一致 -> 混淆

    return tags
}

// ClassifyAndTag 对错题进行自动分类并打标签
func (s *WrongClassifyService) ClassifyAndTag(userID, wrongID uint, question *model.Question) error {
    tags := s.AutoClassify(question)

    for _, tagName := range tags {
        // 查找或创建标签
        tag, err := s.tagRepo.FindByName(userID, tagName)
        if err != nil {
            // 标签不存在，创建系统预置标签
            tag = &model.WrongQuestionTag{
                UserID: 0, // 系统标签
                Name:   tagName,
                Color:  s.getTagColor(tagName),
            }
            s.tagRepo.Create(tag)
        }

        // 关联标签
        s.tagRepo.AddRelation(wrongID, tag.ID)
    }

    return nil
}

// getTagColor 根据标签名获取颜色
func (s *WrongClassifyService) getTagColor(tagName string) string {
    colorMap := map[string]string{
        "单选题": "#1890ff",
        "多选题": "#722ed1",
        "判断题": "#52c41a",
        "填空题": "#fa8c16",
        "简答题": "#f5222d",
        "简单":   "#52c41a",
        "中等":   "#faad14",
        "困难":   "#f5222d",
    }
    if color, ok := colorMap[tagName]; ok {
        return color
    }
    return "#1890ff"
}
```

```go
type WrongSearchService struct {
    wrongRepo *wrong.WrongRepo

// Search 错题搜索
func (s *WrongSearchService) Search(userID uint, req *dto.WrongSearchRequest) ([]model.WrongQuestion, int64, error) {
    query := s.wrongRepo.DB().
        Where("user_id = ? AND deleted_at IS NULL", userID)

    // 关键词搜索（匹配题目内容、答案、解析）
    if req.Keyword != "" {
        keyword := "%" + req.Keyword + "%"
        query = query.Joins("LEFT JOIN questions ON wrong_questions.question_id = questions.id").
            Where("questions.title LIKE ? OR questions.answer LIKE ? OR questions.analysis LIKE ?",
                keyword, keyword, keyword)
    }

    // 标签筛选
    if req.TagID > 0 {
        query = query.Joins("LEFT JOIN wrong_question_tag_relations ON wrong_questions.id = wrong_question_tag_relations.wrong_id").
            Where("wrong_question_tag_relations.tag_id = ?", req.TagID)
    }

    // 来源筛选
    if req.Source > 0 {
        query = query.Where("wrong_questions.source = ?", req.Source)
    }

    // 状态筛选
    if req.Mastered != nil {
        query = query.Where("wrong_questions.mastered = ?", *req.Mastered)
    }

    // 时间范围筛选
    if req.StartDate != "" {
        query = query.Where("wrong_questions.created_at >= ?", req.StartDate)
    }
    if req.EndDate != "" {
        query = query.Where("wrong_questions.created_at <= ?", req.EndDate)
    }

    // 错误次数筛选
    if req.MinWrongCount > 0 {
        query = query.Where("wrong_questions.wrong_count >= ?", req.MinWrongCount)
    }
    if req.MaxWrongCount > 0 {
        query = query.Where("wrong_questions.wrong_count <= ?", req.MaxWrongCount)
    }

    // 难度筛选
    if req.Difficulty > 0 {
        query = query.Where("questions.difficulty = ?", req.Difficulty)
    }

    // 排序
    switch req.SortBy {
    case "wrong_count":
        query = query.Order("wrong_questions.wrong_count DESC")
    case "created_at":
        query = query.Order("wrong_questions.created_at DESC")
    case "updated_at":
        query = query.Order("wrong_questions.updated_at DESC")
    case "difficulty":
        query = query.Order("questions.difficulty DESC")
    default:
        query = query.Order("wrong_questions.created_at DESC")
    }

    // 分页
    var total int64
    query.Count(&total)

    var wrongs []model.WrongQuestion
    offset := (req.Page - 1) * req.PageSize
    query.Offset(offset).Limit(req.PageSize).Find(&wrongs)

    return wrongs, total, nil
}

// WrongSearchRequest 错题搜索请求
type WrongSearchRequest struct {
    Keyword       string `form:"keyword"`
    TagID         uint   `form:"tag_id"`
    Source        int8   `form:"source"`
    Mastered      *bool  `form:"mastered"`
    StartDate     string `form:"start_date"`
    EndDate       string `form:"end_date"`
    MinWrongCount int    `form:"min_wrong_count"`
    MaxWrongCount int    `form:"max_wrong_count"`
    Difficulty    int8   `form:"difficulty"`
    SortBy        string `form:"sort_by"` // wrong_count, created_at, updated_at, difficulty
    SortOrder     string `form:"sort_order"` // asc, desc
    Page          int    `form:"page" binding:"required,min=1"`
    PageSize      int    `form:"page_size" binding:"required,min=1,max=100"`
}
```

### 4.5 自动移除逻辑完善

```go
// service/wrong/auto_remove_service.go

type WrongAutoRemoveService struct {
    wrongRepo    *wrong.WrongRepo
    notifService *notification.NotificationService
}

// CheckAndAutoRemove 检查并自动移除已掌握的错题
// 当 correct_streak 达到阈值（默认3次）后自动从错题本移除
func (s *WrongAutoRemoveService) CheckAndAutoRemove(wrongID uint) error {
    wrong, err := s.wrongRepo.FindByID(wrongID)
    if err != nil {
        return err
    }

    // 连续答对3次自动标记为已掌握
    const masterThreshold = 3
    // 连续答对5次自动从错题本移除（软删除）
    const removeThreshold = 5

    if wrong.CorrectStreak >= removeThreshold {
        // 软删除：设置 deleted_at
        now := time.Now()
        wrong.DeletedAt = &now
        wrong.Mastered = true
        if err := s.wrongRepo.Update(wrong); err != nil {
            return err
        }

        // 发送通知
        s.notifService.Send(wrong.UserID, "wrong_auto_removed",
            fmt.Sprintf("恭喜！您已连续答对%d次，错题已自动移除", removeThreshold))

        return nil
    }

    if wrong.CorrectStreak >= masterThreshold {
        wrong.Mastered = true
        return s.wrongRepo.Update(wrong)
    }

    return nil
}

// ReviewWithAutoRemove 复习错题并检查自动移除
func (s *WrongAutoRemoveService) ReviewWithAutoRemove(wrongID uint, isCorrect bool) error {
    wrong, err := s.wrongRepo.FindByID(wrongID)
    if err != nil {
        return err
    }

    wrong.ReviewCount++
    now := time.Now()
    wrong.LastReviewAt = &now

    if isCorrect {
        wrong.CorrectStreak++

        // 计算下次复习间隔
        interval := calculateInterval(wrong.CorrectStreak)
        nextReview := now.AddDate(0, 0, interval)
        wrong.NextReviewAt = &nextReview

        // 检查自动移除
        if err := s.CheckAndAutoRemove(wrongID); err != nil {
            return err
        }
    } else {
        // 答错，重置连续正确次数
        wrong.CorrectStreak = 0
        wrong.Mastered = false
        wrong.WrongCount++

        // 明天再复习
        nextReview := now.AddDate(0, 0, 1)
        wrong.NextReviewAt = &nextReview
    }

    return s.wrongRepo.Update(wrong)
}

// calculateInterval 计算复习间隔
func calculateInterval(streak int) int {
    intervals := []int{1, 3, 7, 14, 30, 60}
    if streak <= 0 {
        return 1
    }
    if streak > len(intervals) {
        return intervals[len(intervals)-1]
    }
    return intervals[streak-1]
}
```

```go
// calculateForgetRate 计算遗忘率
func calculateForgetRate(days float64) float64 {
    // 艾宾浩斯遗忘曲线公式
    // R = e^(-t/S) 其中 R=记忆保留率, t=时间, S=稳定性
    S := 1.0 // 稳定性系数
    retention := math.Exp(-days / S)
    return 1 - retention
}
```

```go
// service/wrong/spaced_repetition.go

type SpacedRepetitionService struct {
    wrongRepo *wrong.WrongRepo
}

// Review 复习错题（间隔重复算法）
func (s *SpacedRepetitionService) Review(wrongID uint, isCorrect bool) error {
    wrong, err := s.wrongRepo.FindByID(wrongID)
    if err != nil {
        return err
    }
    
    // 更新复习次数
    wrong.ReviewCount++
    wrong.LastReviewAt = time.Now()
    
    if isCorrect {
        // 连续正确次数+1
        wrong.CorrectStreak++
        
        // 根据连续正确次数计算下次复习间隔
        interval := s.calculateInterval(wrong.CorrectStreak)
        nextReview := time.Now().AddDate(0, 0, interval)
        wrong.NextReviewAt = &nextReview
        
        // 连续正确3次标记为已掌握
        if wrong.CorrectStreak >= 3 {
            wrong.Mastered = true
        }
    } else {
        // 重置连续正确次数
        wrong.CorrectStreak = 0
        wrong.Mastered = false
        
        // 明天再复习
        nextReview := time.Now().AddDate(0, 0, 1)
        wrong.NextReviewAt = &nextReview
    }
    
    return s.wrongRepo.Update(wrong)
}

// calculateInterval 计算复习间隔（天）
func (s *SpacedRepetitionService) calculateInterval(streak int) int {
    // 间隔重复算法：1, 3, 7, 14, 30, 60...
    intervals := []int{1, 3, 7, 14, 30, 60}
    
    if streak <= 0 {
        return 1
    }
    
    if streak > len(intervals) {
        return intervals[len(intervals)-1]
    }
    
    return intervals[streak-1]
}
```

```go
            continue
        }

        favorite := &model.WrongQuestionFavorite{
            UserID:  userID,
            WrongID: wrongID,
        }
        if err := s.favoriteRepo.Create(favorite); err != nil {
            failCount++
            continue
        }

        // 更新收藏状态
        s.wrongRepo.UpdateFavoriteStatus(wrongID, true)
        successCount++
    }

    return &dto.BatchResult{
        Total:   len(wrongIDs),
        Success: successCount,
        Fail:    failCount,
    }, nil
}

// BatchUnfavorite 批量取消收藏
func (s *WrongBatchService) BatchUnfavorite(userID uint, wrongIDs []uint) (*dto.BatchResult, error) {
    var successCount, failCount int
    for _, wrongID := range wrongIDs {
        if err := s.favoriteRepo.Delete(userID, wrongID); err != nil {
            failCount++
            continue
        }
        s.wrongRepo.UpdateFavoriteStatus(wrongID, false)
        successCount++
    }

    return &dto.BatchResult{
        Total:   len(wrongIDs),
        Success: successCount,
        Fail:    failCount,
    }, nil
}

    var successCount, failCount int
    for _, wrongID := range wrongIDs {
        if err := s.tagRepo.AddRelation(wrongID, tagID); err != nil {

            continue
        }
        successCount++
    }

    return &dto.BatchResult{
        Total:   len(wrongIDs),
        Success: successCount,
        Fail:    failCount,
    }, nil
```
```go
}

// Compare 对比不同时间段的错题情况
func (s *WrongCompareService) Compare(userID uint, req *dto.CompareRequest) (*dto.CompareResult, error) {
    // 解析时间段
    period1Start, period1End := parsePeriod(req.Period1)
    period2Start, period2End := parsePeriod(req.Period2)

    // 获取时间段1的错题统计
    period1Stats, err := s.getPeriodStats(userID, period1Start, period1End)
    if err != nil {
        return nil, err
    }

    // 获取时间段2的错题统计
    period2Stats, err := s.getPeriodStats(userID, period2Start, period2End)
    if err != nil {
        return nil, err
    }

    // 计算变化
    result := &dto.CompareResult{
        Period1: dto.PeriodInfo{
            Start: period1Start,
            End:   period1End,
            Stats: period1Stats,
        },
        Period2: dto.PeriodInfo{
            Start: period2Start,
            End:   period2End,
            Stats: period2Stats,
        },
        Changes: dto.CompareChanges{
            TotalCountChange:    period2Stats.TotalCount - period1Stats.TotalCount,
            MasteredCountChange: period2Stats.MasteredCount - period1Stats.MasteredCount,
            MasteryRateChange:   period2Stats.MasteryRate - period1Stats.MasteryRate,
        },
    }

    // 计算趋势
    if result.Changes.TotalCountChange < 0 {
        result.Changes.Trend = "improving" // 错题减少，进步
    } else if result.Changes.TotalCountChange > 0 {
        result.Changes.Trend = "declining" // 错题增加，退步
    } else {
        result.Changes.Trend = "stable" // 稳定
    }

    return result, nil
}

// getPeriodStats 获取时间段内的错题统计
func (s *WrongCompareService) getPeriodStats(userID uint, start, end time.Time) (*dto.PeriodStats, error) {
    total, _ := s.wrongRepo.CountByUserAndDateRange(userID, start, end)
    mastered, _ := s.wrongRepo.CountMasteredByUserAndDateRange(userID, start, end)
    byCategory, _ := s.wrongRepo.GroupByCategoryAndDateRange(userID, start, end)
    byType, _ := s.wrongRepo.GroupByTypeAndDateRange(userID, start, end)

    masteryRate := float64(0)
    if total > 0 {
        masteryRate = float64(mastered) / float64(total) * 100
    }

    return &dto.PeriodStats{
        TotalCount:    total,
        MasteredCount: mastered,
        MasteryRate:   masteryRate,
        ByCategory:    byCategory,
        ByType:        byType,
    }, nil
}

// parsePeriod 解析时间段
func parsePeriod(period string) (time.Time, time.Time) {
    // 支持格式: "2026-01", "2026-01-01~2026-01-31", "last7days", "last30days"
    now := time.Now()

    switch period {
    case "last7days":
        return now.AddDate(0, 0, -7), now
    case "last30days":
        return now.AddDate(0, 0, -30), now
    case "last90days":
        return now.AddDate(0, 0, -90), now
    default:
        // 尝试解析月份格式
        if t, err := time.Parse("2006-01", period); err == nil {
            start := t
            end := t.AddDate(0, 1, -1)
            return start, end
        }
        // 尝试解析日期范围
        parts := strings.Split(period, "~")
        if len(parts) == 2 {
            start, _ := time.Parse("2006-01-02", parts[0])
            end, _ := time.Parse("2006-01-02", parts[1])
            return start, end
        }
        return now, now
    }
}
```

```go
    // 尝试从缓存获取

    cacheKey := fmt.Sprintf("wrong:ranking:%s:%d", rankingType, limit)

        var items []dto.RankingItem

    }

    var items []dto.RankingItem

        items, err = s.wrongRepo.GetRankingByCount(limit)
    case "mastery":
        // 掌握率排行榜（掌握率最高的用户）
        items, err = s.wrongRepo.GetRankingByMasteryRate(limit)
    case "improvement":
        // 进步排行榜（近期错题减少最多的用户）
        items, err = s.wrongRepo.GetRankingByImprovement(limit)
    default:
        return nil, errors.New("无效的排行榜类型")
    }

    if err != nil {
        return nil, err
    }

    // 缓存结果，有效期1小时
    data, _ := json.Marshal(items)
    s.cache.Set(ctx, cacheKey, string(data), time.Hour)

    return items, nil
}
```

```vue
      page: page.value,
      page_size: pageSize
    }

    if (filterSource.value > 0) params.source = filterSource.value
    if (filterMastered.value !== null) params.mastered = filterMastered.value
    if (filterWrongCount.value > 0) params.min_wrong_count = filterWrongCount.value
    if (filterDifficulty.value > 0) params.difficulty = filterDifficulty.value
    if (filterTimeRange.value) {
      const now = new Date()

      const start = new Date(now.getTime() - days * 24 * 60 * 60 * 1000)
      params.start_date = start.toISOString().split('T')[0]
    }

    const res = await getWrongList(params)
    const items = res.data.items || []

    if (reset) {
      wrongList.value = items
    } else {

    }

    total.value = res.data.total

    favoriteCount.value = res.data.favorite_count || 0

    if (items.length < pageSize) {
      noMore.value = true
    }

    page.value++
  } finally {
    loading.value = false
    refreshing.value = false
  }
}
```

### 11.1 间隔重复提醒

```go
// internal/task/wrong_review_remind.go

package task

import (
    "questionhelper-server/internal/repository/wrong"
    "questionhelper-server/internal/service/notification"
)

// WrongReviewRemindTask 错题复习提醒任务
type WrongReviewRemindTask struct {
    wrongRepo *wrong.WrongRepo
    notifService *notification.NotificationService
}

// Run 执行提醒
func (t *WrongReviewRemindTask) Run() {
    // 获取今天需要复习的错题
    wrongs, err := t.wrongRepo.FindForReviewToday()
    if err != nil {
        return
    }
    
    // 按用户分组
    userWrongs := make(map[uint][]model.WrongQuestion)
    for _, wrong := range wrongs {
        userWrongs[wrong.UserID] = append(userWrongs[wrong.UserID], wrong)
    }
    
    // 发送提醒
    for userID, items := range userWrongs {
        content := fmt.Sprintf("您有 %d 道错题需要复习", len(items))
        t.notifService.Send(userID, "wrong_review_remind", content)
    }
}
```
