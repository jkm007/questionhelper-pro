# 文件管理设计文档

**项目名称**：题小助 (QuestionHelper)
**版本**：V2.0
**日期**：2026-05-30
**模块**：文件管理

---

## 一、功能概述

| 功能 | 说明 |
|------|------|
| 文件上传 | 图片、文档、音视频上传，支持批量和拖拽 |
| 文件存储 | MinIO/OSS 存储，按业务类型分目录 |
| 文件访问 | CDN 域名加速静态资源访问 |
| 缩略图生成 | 图片上传自动生成多尺寸缩略图 |
| 文件去重 | 基于文件 MD5 去重，相同文件只存储一份 |
| 文件管理 | 文件列表、详情、删除、搜索 |
| 引用计数 | 记录文件被引用次数，防止误删正在使用的文件 |
| 防盗链 | 配置 Referer 白名单，防止外部站点盗链 |
| 病毒扫描 | 上传文件自动进行病毒扫描 |
| 敏感内容检测 | 图片上传自动进行敏感内容检测（色情、暴恐、广告等） |
| 水印功能 | 图片可配置自动添加文字/图片水印 |
| 访问日志 | 记录文件访问日志（访问时间、IP、文件路径、用户） |
| 孤立文件清理 | 定时清理未被引用的临时文件 |

#### 3.2.1 批量删除 API 详细设计

**POST `/api/v1/admin/files/batch/delete`**

请求体：

```json
{
  "file_ids": [101, 102, 103]
}
```

| 字段 | 类型 | 必填 | 说明 |
|------|------|------|------|
| file_ids | array\<uint64\> | 是 | 文件ID列表，单次最多100个 |

响应体：

```json
{
  "code": 0,
  "message": "success",
  "data": {
    "success": [101, 103],
    "failed": [
      {"id": 102, "reason": "文件正在被引用，无法删除"}
    ],
    "total": 3,
    "success_count": 2,
    "failed_count": 1
  }
}
```

后端实现要点：
- 逐个校验文件是否存在、是否被引用（`reference_count > 0` 的文件跳过并记录原因）
- 逐个软删除数据库记录
- 异步清理 OSS 中的实际文件（可选，或留给孤立文件清理任务）
- 操作记录写入审计日志

后端控制器代码：

```go
// BatchDeleteRequest 批量删除请求
type BatchDeleteRequest struct {
    FileIDs []uint64 `json:"file_ids" binding:"required,min=1,max=100"`
}

// BatchDeleteResult 批量删除结果
type BatchDeleteResult struct {
    Success      []uint64          `json:"success"`
    Failed       []BatchDeleteFail `json:"failed"`
    Total        int               `json:"total"`
    SuccessCount int               `json:"success_count"`
    FailedCount  int               `json:"failed_count"`
}

type BatchDeleteFail struct {
    ID     uint64 `json:"id"`
    Reason string `json:"reason"`
}

// BatchDelete 批量删除文件
func (c *FileController) BatchDelete(ctx *gin.Context) {
    var req dto.BatchDeleteRequest
    if err := ctx.ShouldBindJSON(&req); err != nil {
        response.Error(ctx, 400, "参数错误")
        return
    }

    userID := ctx.MustGet("user_id").(uint)
    isAdmin := ctx.GetBool("is_admin")

    result := &dto.BatchDeleteResult{
        Total: len(req.FileIDs),
    }

    for _, fileID := range req.FileIDs {
        err := c.fileService.DeleteFile(ctx, fileID, userID, isAdmin)
        if err != nil {
            result.Failed = append(result.Failed, dto.BatchDeleteFail{
                ID:     fileID,
                Reason: err.Error(),
            })
            result.FailedCount++
        } else {
            result.Success = append(result.Success, fileID)
            result.SuccessCount++
        }
    }

    response.Success(ctx, result)
}
```

```go
    return &CDNManager{config: config}

// GetURL 获取文件的 CDN 访问地址
func (m *CDNManager) GetURL(path string) string {
    if !m.config.Enabled || m.config.Domain == "" {
        return path
    }

    protocol := "http"
    if m.config.HTTPS {
        protocol = "https"
    }

    // 移除路径开头的斜杠
    path = strings.TrimPrefix(path, "/")

    return fmt.Sprintf("%s://%s/%s", protocol, m.config.Domain, path)
}

// GetImageURL 获取图片的 CDN 访问地址（支持图片处理参数）
func (m *CDNManager) GetImageURL(path string, style string) string {
    baseURL := m.GetURL(path)

    if style == "" {
        return baseURL
    }

    // 阿里云 OSS 图片处理样式
    if styleParams, ok := m.config.ImageStyle[style]; ok {
        return baseURL + styleParams
    }

    return baseURL
}

// GetThumbnailURL 获取缩略图地址
func (m *CDNManager) GetThumbnailURL(path string, size string) string {
    return m.GetImageURL(path, "thumbnail_"+size)
}

// MinIO 兼容的图片处理（不依赖云端图片处理）
func (m *CDNManager) GetLocalThumbnailURL(originalPath, suffix string) string {
    // 对于 MinIO 等不支持云端图片处理的存储，使用本地生成的缩略图
    ext := ""
    dotIndex := strings.LastIndex(originalPath, ".")
    if dotIndex >= 0 {
        ext = originalPath[dotIndex:]
        originalPath = originalPath[:dotIndex]
    }
    return m.GetURL(originalPath + "_" + suffix + ext)
```

```go
// service/file/thumbnail_service.go

package file

import (
    "bytes"
    "context"
    "fmt"
    "image"
    "image/gif"
    "image/jpeg"
    "image/png"
    "path/filepath"
    "strings"

    "go.uber.org/zap"

    "questionhelper-pro/internal/model"
    "questionhelper-pro/internal/repository/file"
    "questionhelper-pro/pkg/upload"
)

// ThumbnailSize 缩略图尺寸配置
type ThumbnailSize struct {
    Name   string
    Width  int
    Height int
}

// 预定义缩略图尺寸
var ThumbnailSizes = []ThumbnailSize{
    {Name: "small", Width: 100, Height: 100},
    {Name: "medium", Width: 300, Height: 300},
    {Name: "large", Width: 800, Height: 800},
}

type ThumbnailService struct {
    fileRepo    *file.FileRepo
    ossClient   *upload.OSSClient
    cdnManager  *upload.CDNManager
    logger      *zap.Logger
}

func NewThumbnailService(
    fileRepo *file.FileRepo,
    ossClient *upload.OSSClient,
    cdnManager *upload.CDNManager,
    logger *zap.Logger,
) *ThumbnailService {
    return &ThumbnailService{
        fileRepo:   fileRepo,
        ossClient:  ossClient,
        cdnManager: cdnManager,
        logger:     logger,
    }
}

// GenerateThumbnails 为图片生成多尺寸缩略图
func (s *ThumbnailService) GenerateThumbnails(ctx context.Context, fileID uint64) error {
    // 获取原始文件信息
    file, err := s.fileRepo.GetByID(fileID)
    if err != nil {
        return fmt.Errorf("获取文件信息失败: %w", err)
    }

    // 检查是否为图片类型
    if !isImageType(file.Type) {
        return nil // 非图片不生成缩略图
    }

    // 从 OSS 下载原始文件
    fileData, err := s.ossClient.Download(ctx, file.Path)
    if err != nil {
        return fmt.Errorf("下载原始文件失败: %w", err)
    }

    // 解码图片
    img, format, err := image.Decode(bytes.NewReader(fileData))
    if err != nil {
        return fmt.Errorf("解码图片失败: %w", err)
    }

    var thumbnailIDs []string

    // 生成各尺寸缩略图
    for _, size := range ThumbnailSizes {
        // 如果原图比目标尺寸小，跳过
        if img.Bounds().Dx() <= size.Width && img.Bounds().Dy() <= size.Height {
            continue
        }

        // 生成缩略图
        thumbnail := imaging.Fit(img, size.Width, size.Height, imaging.Lanczos)

        // 编码缩略图
        var buf bytes.Buffer
        switch format {
        case "jpeg":
            err = jpeg.Encode(&buf, thumbnail, &jpeg.Options{Quality: 85})
        case "png":
            err = png.Encode(&buf, thumbnail)
        case "gif":
            err = gif.Encode(&buf, thumbnail, nil)
        default:
            err = jpeg.Encode(&buf, thumbnail, &jpeg.Options{Quality: 85})
        }
        if err != nil {
            s.logger.Error("编码缩略图失败", zap.String("size", size.Name), zap.Error(err))
            continue
        }

        // 生成缩略图文件名
        ext := filepath.Ext(file.Name)
        thumbnailName := strings.TrimSuffix(file.Name, ext) + "_" + size.Name + ext

        // 上传缩略图到 OSS
        thumbnailPath := file.Path[:strings.LastIndex(file.Path, "/")+1] + thumbnailName
        thumbnailURL, err := s.ossClient.UploadBytes(ctx, buf.Bytes(), thumbnailPath, file.Type)
        if err != nil {
            s.logger.Error("上传缩略图失败", zap.String("size", size.Name), zap.Error(err))
            continue
        }

        // 创建缩略图文件记录
        thumbnailRecord := &model.File{
            Name:           thumbnailName,
            Original:       file.Original,
            Path:           thumbnailPath,
            URL:            thumbnailURL,
            Size:           int64(buf.Len()),
            Type:           file.Type,
            Extension:      file.Extension,
            MD5:            calculateMD5(buf.Bytes()),
            BusinessType:   "thumbnail",
            StorageType:    file.StorageType,
            Bucket:         file.Bucket,
            Width:          uintPtr(uint(size.Width)),
            Height:         uintPtr(uint(size.Height)),
            OriginalFileID: &fileID,
            IsPublic:       1,
            UploaderID:     file.UploaderID,
            Status:         1,
        }

        if err := s.fileRepo.Create(thumbnailRecord); err != nil {
            s.logger.Error("保存缩略图记录失败", zap.Error(err))
            continue
        }

        thumbnailIDs = append(thumbnailIDs, fmt.Sprintf("%d:%s", thumbnailRecord.ID, size.Name))
    }

    // 更新原始文件的缩略图 ID 列表
    if len(thumbnailIDs) > 0 {
        file.ThumbnailIDs = strings.Join(thumbnailIDs, ",")
        if err := s.fileRepo.Update(file); err != nil {
            s.logger.Error("更新文件缩略图ID失败", zap.Error(err))
        }
    }

    return nil
}

// GetThumbnailURL 根据尺寸获取缩略图 URL
func (s *ThumbnailService) GetThumbnailURL(fileID uint64, size string) (string, error) {
    file, err := s.fileRepo.GetByID(fileID)
    if err != nil {
        return "", err
    }

    if file.ThumbnailIDs == "" {
        return s.cdnManager.GetImageURL(file.Path, size), nil
    }

    // 解析缩略图 ID 列表
    pairs := strings.Split(file.ThumbnailIDs, ",")
    for _, pair := range pairs {
        parts := strings.SplitN(pair, ":", 2)
        if len(parts) == 2 && parts[1] == size {
            thumbnailID := parts[0]
            thumbnail, err := s.fileRepo.GetByID(parseUint64(thumbnailID))
            if err == nil {
                return s.cdnManager.GetURL(thumbnail.Path), nil
            }
        }
    }

    // 未找到对应尺寸，返回原图 URL
    return s.cdnManager.GetImageURL(file.Path, size), nil
}

// isImageType 检查是否为图片类型
```

```go
// CheckDuplicate 检查文件是否重复（基于 MD5）
func (s *DedupService) CheckDuplicate(fileData []byte) (*DedupResult, error) {
    md5Hash := CalculateMD5(fileData)

    // 查询是否存在相同 MD5 的文件
    existingFile, err := s.fileRepo.GetByMD5(md5Hash)
    if err != nil {
        // 查询失败（非记录不存在的情况）
        if err.Error() != "record not found" {
            return nil, fmt.Errorf("查询文件MD5失败: %w", err)
        }
        // 记录不存在，不是重复文件
        return &DedupResult{IsDuplicate: false}, nil
    }

    // 找到重复文件
    s.logger.Info("发现重复文件",
        zap.String("md5", md5Hash),
        zap.Uint64("existing_file_id", existingFile.ID),
    )

    return &DedupResult{
        IsDuplicate: true,
        File:        existingFile,
    }, nil
}

// UploadWithDedup 上传文件（带去重逻辑）
func (s *DedupService) UploadWithDedup(ctx context.Context, fileData []byte, fileName, mimeType, objectName string, userID uint) (*model.File, bool, error) {
    // 计算文件 MD5
    md5Hash := CalculateMD5(fileData)

    // 检查是否已存在相同文件
    existingFile, err := s.fileRepo.GetByMD5(md5Hash)
    if err == nil && existingFile != nil {
        // 文件已存在，返回已有文件，增加引用计数
        s.logger.Info("文件已存在，复用已有文件",
            zap.String("md5", md5Hash),
            zap.Uint64("existing_id", existingFile.ID),
        )
        return existingFile, true, nil
    }

    // 文件不存在，正常上传
    url, err := s.ossClient.UploadBytes(ctx, fileData, objectName, mimeType)
    if err != nil {
        return nil, false, fmt.Errorf("上传文件到OSS失败: %w", err)
    }

    // 创建文件记录
    fileRecord := &model.File{
        Name:     objectName,
        Original: fileName,
        Path:     objectName,
        URL:      url,
        Size:     int64(len(fileData)),
        Type:     mimeType,
        Extension: getFileExtension(fileName),
        MD5:      md5Hash,
        IsPublic: 1,
        UploaderID: &userID,
        Status:   1,
    }

    if err := s.fileRepo.Create(fileRecord); err != nil {
        return nil, false, fmt.Errorf("保存文件记录失败: %w", err)
    }

    return fileRecord, false, nil
}

// CalculateMD5 计算文件 MD5
func CalculateMD5(data []byte) string {
    hash := md5.Sum(data)
    return fmt.Sprintf("%x", hash)
}

// CalculateMD5FromReader 从 Reader 计算 MD5
func CalculateMD5FromReader(reader io.Reader) (string, error) {
    hash := md5.New()
    if _, err := io.Copy(hash, reader); err != nil {
        return "", err
    }
    return fmt.Sprintf("%x", hash.Sum(nil)), nil
}
```

```go
// CheckReferer 检查 Referer 是否允许访问
func (s *HotlinkService) CheckReferer(referer string) (allowed bool, isHotlink bool) {
    s.mu.RLock()
    defer s.mu.RUnlock()

    // 没有配置规则，允许所有访问
    if len(s.rules) == 0 {
        return true, false
    }

    // 提取 Referer 域名
    refererDomain := extractDomain(referer)
    if refererDomain == "" {
        // 空 Referer（直接访问），允许
        return true, false
    }

    // 检查白名单
    for _, rule := range s.rules {
        if rule.Status != 1 {
            continue
        }
        if matchDomain(refererDomain, rule.Domain) {
            if rule.IsAllowed == 1 {
                return true, false // 白名单命中，允许
            }
            return false, true // 黑名单命中，拒绝（盗链）
        }
    }

    // 默认允许（白名单模式下，未匹配的域名放行；黑名单模式下，未匹配的域名拒绝）
    // 这里采用白名单模式：配置了规则但未匹配，视为盗链
    return false, true
}

// extractDomain 从 URL 中提取域名
func extractDomain(url string) string {
    // 移除协议前缀
    url = strings.TrimPrefix(url, "http://")
    url = strings.TrimPrefix(url, "https://")

    // 提取域名部分
    if idx := strings.Index(url, "/"); idx >= 0 {
        url = url[:idx]
    }
    // 移除端口号
    if idx := strings.Index(url, ":"); idx >= 0 {
        url = url[:idx]
    }

    return strings.ToLower(url)
}

// matchDomain 匹配域名（支持通配符 *.example.com）
func matchDomain(domain, pattern string) bool {
    pattern = strings.ToLower(pattern)

    if pattern == domain {
        return true
    }

    // 通配符匹配
    if strings.HasPrefix(pattern, "*.") {
        suffix := pattern[1:] // .example.com
        return strings.HasSuffix(domain, suffix) || domain == suffix[1:]
    }

    return false
}
```

```go
    if err != nil {
        return fmt.Errorf("解码原始图片失败: %w", err)
    }

    watermarkImg, _, err := image.Decode(bytes.NewReader(watermarkData))
    if err != nil {
        return fmt.Errorf("解码水印图片失败: %w", err)
    }

    // 添加图片水印
    watermarked := s.drawImageWatermark(img, watermarkImg, config)

    // 编码图片
    var buf bytes.Buffer
    switch format {
    case "jpeg":
        err = jpeg.Encode(&buf, watermarked, &jpeg.Options{Quality: 90})
    case "png":
        err = png.Encode(&buf, watermarked)
    default:
        err = jpeg.Encode(&buf, watermarked, &jpeg.Options{Quality: 90})
    }
    if err != nil {
        return fmt.Errorf("编码图片失败: %w", err)
    }

    // 上传水印图片
    watermarkPath := strings.TrimSuffix(fileRecord.Path, getFileExtension(fileRecord.Path)) + "_wm" + getFileExtension(fileRecord.Path)
    _, err = s.ossClient.UploadBytes(ctx, buf.Bytes(), watermarkPath, fileRecord.Type)
    if err != nil {
        return fmt.Errorf("上传水印图片失败: %w", err)
    }

    // 更新文件记录
    fileRecord.WatermarkApplied = 1
    return s.fileRepo.Update(fileRecord)
}

// drawTextWatermark 绘制文字水印
func (s *WatermarkService) drawTextWatermark(img image.Image, config *model.WatermarkConfig) image.Image {
    bounds := img.Bounds()
    watermarked := imaging.Clone(img)

    // 设置字体和大小
    fontSize := 24.0
    if config.FontSize != nil {
        fontSize = float64(*config.FontSize)
    }

    // 设置颜色
    c := color.RGBA{R: 255, G: 255, B: 255, A: uint8(config.Opacity * 255)}

    // 计算水印位置
    x, y := s.calculatePosition(bounds.Dx(), bounds.Dy(), config.Position,
        int(config.MarginX), int(config.MarginY), fontSize)

    // 使用 freetype 绘制文字
    // 注意：实际项目中需要加载字体文件
    // 这里使用简化的绘制方式
    dc := imaging.Clone(img)
    ft := freetype.NewContext()
    ft.SetDPI(72)
    ft.SetFontSize(fontSize)
    ft.SetDst(dc)
    ft.SetSrc(image.NewUniform(c))
    ft.DrawString(config.Content, freetype.Pt(x, y))

    return dc
}

// drawImageWatermark 绘制图片水印
func (s *WatermarkService) drawImageWatermark(img, watermark image.Image, config *model.WatermarkConfig) image.Image {
    bounds := img.Bounds()

    // 调整水印图片大小
    wmWidth := watermark.Bounds().Dx()
    wmHeight := watermark.Bounds().Dy()
    if config.ImageWidth != nil {
        wmWidth = int(*config.ImageWidth)
    }
    if config.ImageHeight != nil {
        wmHeight = int(*config.ImageHeight)
    }

    resizedWatermark := imaging.Resize(watermark, wmWidth, wmHeight, imaging.Lanczos)

    // 计算水印位置
    x, y := s.calculatePosition(bounds.Dx(), bounds.Dy(), config.Position,
        int(config.MarginX), int(config.MarginY), 0)

    // 创建带透明度的水印层
    watermarkLayer := imaging.New(bounds.Dx(), bounds.Dy(), color.Transparent)
    watermarkLayer = imaging.Paste(watermarkLayer, resizedWatermark, image.Pt(x, y))

    // 设置透明度
    opacity := config.Opacity
    watermarkLayer = imaging.Opacity(watermarkLayer, opacity)

    // 合并图层
    result := imaging.Clone(img)
    draw.Draw(result, bounds, watermarkLayer, image.Point{}, draw.Over)

    return result
}

// calculatePosition 计算水印位置
func (s *WatermarkService) calculatePosition(imgWidth, imgHeight int, position string, marginX, marginY int, fontSize float64) (int, int) {
    textWidth := len("watermark") * int(fontSize) / 2 // 简化计算
    textHeight := int(fontSize)

    switch position {
    case "top-left":
        return marginX, marginY + textHeight
    case "top-right":
        return imgWidth - textWidth - marginX, marginY + textHeight
    case "bottom-left":
        return marginX, imgHeight - marginY
    case "bottom-right":
        return imgWidth - textWidth - marginX, imgHeight - marginY
    case "center":
        return (imgWidth - textWidth) / 2, (imgHeight + textHeight) / 2
    default:
        return imgWidth - textWidth - marginX, imgHeight - marginY
    }
}

    var watermarked image.Image
```

```go
            c.Next()

        }

                return
            }

        if uid, exists := c.Get("user_id"); exists {
            id := uid.(uint)
            userID = &id
        }

        // 从路径提取文件ID
        fileID := extractFileID(path)
        if fileID == 0 {
            return
        }

        // 判断访问类型
        accessType := "view"
        if isDownloadPath(path) {
            accessType = "download"
        }

        // 创建访问日志
        log := &model.FileAccessLog{

            UserID:       userID,

            UserAgent:    c.Request.UserAgent(),

            ResponseCode: intPtr(c.Writer.Status()),
```

```go
    TotalSize  int64  `json:"total_size"`
    Duration   int64  `json:"duration_ms"`
    Errors     []string `json:"errors,omitempty"`

// CleanupOrphanFiles 清理孤立文件
func (s *CleanupService) CleanupOrphanFiles(ctx context.Context) (*CleanupResult, error) {
    if !s.config.Enabled {
        return nil, fmt.Errorf("孤立文件清理功能未启用")
    }

    batchNo := uuid.New().String()
    startedAt := time.Now()

    s.logger.Info("开始清理孤立文件", zap.String("batch_no", batchNo))

    // 创建清理记录
    cleanupLog := &model.FileCleanupLog{
        BatchNo:   batchNo,
        Status:    0, // 进行中
        StartedAt: startedAt,
        Operator:  "system",
    }
    if err := s.fileRepo.CreateCleanupLog(cleanupLog); err != nil {
        return nil, fmt.Errorf("创建清理记录失败: %w", err)
    }

    // 计算过期时间
    ttl := s.config.TempFileTTL
    if ttl <= 0 {
        ttl = 24
    }
    olderThan := time.Now().Add(-time.Duration(ttl) * time.Hour)

    // 获取孤立文件
    orphanFiles, err := s.fileRepo.GetOrphanFiles(olderThan)
    if err != nil {
        s.updateCleanupLog(cleanupLog, 2, fmt.Sprintf("获取孤立文件失败: %s", err.Error()))
        return nil, fmt.Errorf("获取孤立文件失败: %w", err)
    }

    result := &CleanupResult{
        BatchNo: batchNo,
    }

    var errors []string
    var totalSize int64

    // 分批处理
    for i := 0; i < len(orphanFiles); i += s.config.BatchSize {
        end := i + s.config.BatchSize
        if end > len(orphanFiles) {
            end = len(orphanFiles)
        }
        batch := orphanFiles[i:end]

        for _, f := range batch {
            if s.config.DryRun {
                s.logger.Info("[试运行] 将删除孤立文件",
                    zap.Uint64("file_id", f.ID),
                    zap.String("path", f.Path),
                    zap.Int64("size", f.Size),
                )
                totalSize += f.Size
                result.FileCount++
                continue
            }

            // 从 OSS 删除文件
            if err := s.ossClient.Delete(ctx, f.Path); err != nil {
                s.logger.Error("删除OSS文件失败",
                    zap.Uint64("file_id", f.ID),
                    zap.String("path", f.Path),
                    zap.Error(err),
                )
                errors = append(errors, fmt.Sprintf("file_%d: %s", f.ID, err.Error()))
                continue
            }

            // 软删除数据库记录
            if err := s.fileRepo.Delete(f.ID); err != nil {
                s.logger.Error("删除文件记录失败",
                    zap.Uint64("file_id", f.ID),
                    zap.Error(err),
                )
                errors = append(errors, fmt.Sprintf("file_%d: %s", f.ID, err.Error()))
                continue
            }

            totalSize += f.Size
            result.FileCount++

            s.logger.Debug("删除孤立文件",
                zap.Uint64("file_id", f.ID),
                zap.String("path", f.Path),
            )
        }
    }

    result.TotalSize = totalSize
    result.Duration = time.Since(startedAt).Milliseconds()
    result.Errors = errors

    // 更新清理记录
    status := int8(1) // 完成
    if len(errors) > 0 {
        status = 2 // 有错误
    }
    errorMsg := ""
    if len(errors) > 0 {
        errorBytes, _ := json.Marshal(errors)
        errorMsg = string(errorBytes)
    }
    cleanupLog.FileCount = uint(result.FileCount)
    cleanupLog.TotalSize = uint64(totalSize)
    s.updateCleanupLog(cleanupLog, status, errorMsg)

    s.logger.Info("孤立文件清理完成",
        zap.String("batch_no", batchNo),
        zap.Int("file_count", result.FileCount),
        zap.Int64("total_size", totalSize),
        zap.Int("errors", len(errors)),
    )

    return result, nil
}
```

```go
// Upload 上传文件
func (s *FileService) Upload(ctx context.Context, fileHeader *multipart.FileHeader, userID uint, businessType string) (*dto.FileInfo, error) {
    // 1. 验证文件
    if err := upload.ValidateFile(fileHeader, 20*1024*1024); err != nil {
        return nil, err
    }

    // 2. 读取文件内容
    file, err := fileHeader.Open()
    if err != nil {
        return nil, fmt.Errorf("打开文件失败: %w", err)
    }
    defer file.Close()

    fileData, err := io.ReadAll(file)
    if err != nil {
        return nil, fmt.Errorf("读取文件失败: %w", err)
    }

    // 3. 计算 MD5 并检查去重
    md5Hash := fmt.Sprintf("%x", md5.Sum(fileData))
    existingFile, err := s.fileRepo.GetByMD5(md5Hash)
    if err == nil && existingFile != nil {
        // 文件已存在，复用
        s.logger.Info("文件已存在，复用已有文件",
            zap.String("md5", md5Hash),
            zap.Uint64("existing_id", existingFile.ID),
        )
        return &dto.FileInfo{
            ID:     existingFile.ID,
            URL:    s.cdnManager.GetURL(existingFile.Path),
            Name:   existingFile.Original,
            MD5:    existingFile.MD5,
            Size:   existingFile.Size,
            IsDup:  true,
        }, nil
    }

    // 4. 生成文件名
    ext := filepath.Ext(fileHeader.Filename)
    objectName := upload.GenerateFilename(ext)

    // 5. 设置业务目录
    if businessType == "" {
        businessType = "temp"
    }
    objectPath := fmt.Sprintf("%s/%s", businessType, objectName)

    // 6. 上传到 OSS
    url, err := s.ossClient.UploadBytes(ctx, fileData, objectPath, fileHeader.Header.Get("Content-Type"))
    if err != nil {
        return nil, fmt.Errorf("上传到OSS失败: %w", err)
    }

    // 7. 创建文件记录
    fileRecord := &model.File{
        Name:         objectName,
        Original:     fileHeader.Filename,
        Path:         objectPath,
        URL:          url,
        Size:         fileHeader.Size,
        Type:         fileHeader.Header.Get("Content-Type"),
        Extension:    ext,
        MD5:          md5Hash,
        BusinessType: businessType,
        StorageType:  "oss",
        IsPublic:     1,
        UploaderID:   &userID,
        Status:       1,
    }

    if err := s.fileRepo.Create(fileRecord); err != nil {
        return nil, fmt.Errorf("保存文件记录失败: %w", err)
    }

    // 8. 异步处理
    // 病毒扫描
    s.virusScanService.ScanFileAsync(fileRecord.ID)

    return &dto.FileInfo{
        ID:     fileRecord.ID,
        URL:    s.cdnManager.GetURL(fileRecord.Path),
        Name:   fileRecord.Original,
        MD5:    fileRecord.MD5,
        Size:   fileRecord.Size,
        IsDup:  false,
    }, nil
}

func (s *FileService) UploadImage(ctx context.Context, fileHeader *multipart.FileHeader, userID uint, businessType string) (*dto.FileInfo, error) {
    // 验证图片类型
    ext := filepath.Ext(fileHeader.Filename)
    allowedExts := map[string]bool{".jpg": true, ".jpeg": true, ".png": true, ".gif": true, ".webp": true}
    if !allowedExts[strings.ToLower(ext)] {
        return nil, fmt.Errorf("不支持的图片格式")
    }

    // 上传文件
    result, err := s.Upload(ctx, fileHeader, userID, businessType)
    if err != nil {

    }

            s.logger.Error("生成缩略图失败", zap.Uint64("file_id", result.ID), zap.Error(err))

    // 2. 敏感内容检测
```

### 4.13 文件上传校验增强

#### 4.13.1 每日上传限制校验

同一用户每日上传文件数限制为 500 个。使用 Redis 计数器实现，避免频繁查询数据库。

```go
// service/file/upload_limit_service.go

package file

import (
    "context"
    "fmt"
    "time"

    "github.com/redis/go-redis/v9"
    "go.uber.org/zap"
)

// UploadLimitConfig 上传限制配置
type UploadLimitConfig struct {
    DailyMaxFiles int `yaml:"daily_max_files"` // 每日最大上传文件数，默认500
}

type UploadLimitService struct {
    rdb    *redis.Client
    config UploadLimitConfig
    logger *zap.Logger
}

func NewUploadLimitService(rdb *redis.Client, config UploadLimitConfig, logger *zap.Logger) *UploadLimitService {
    if config.DailyMaxFiles <= 0 {
        config.DailyMaxFiles = 500
    }
    return &UploadLimitService{
        rdb:    rdb,
        config: config,
        logger: logger,
    }
}

// CheckDailyLimit 检查用户每日上传数量限制
// 返回值：remaining = 剩余可上传数量，err != nil 表示已超出限制
func (s *UploadLimitService) CheckDailyLimit(ctx context.Context, userID uint) (remaining int, err error) {
    key := fmt.Sprintf("file:upload:daily:%d:%s", userID, time.Now().Format("20060102"))

    count, err := s.rdb.Get(ctx, key).Int()
    if err != nil && err != redis.Nil {
        // Redis 异常时放行（降级策略）
        s.logger.Warn("查询每日上传计数失败，放行上传", zap.Error(err))
        return s.config.DailyMaxFiles, nil
    }

    if count >= s.config.DailyMaxFiles {
        return 0, fmt.Errorf("今日上传文件数已达上限(%d个)，请明天再试", s.config.DailyMaxFiles)
    }

    return s.config.DailyMaxFiles - count, nil
}

// IncrDailyCount 上传成功后递增计数
func (s *UploadLimitService) IncrDailyCount(ctx context.Context, userID uint) {
    key := fmt.Sprintf("file:upload:daily:%d:%s", userID, time.Now().Format("20060102"))
    pipe := s.rdb.Pipeline()
    pipe.Incr(ctx, key)
    pipe.Expire(ctx, key, 25*time.Hour) // 25小时过期，确保跨天不丢失
    if _, err := pipe.Exec(ctx); err != nil {
        s.logger.Error("递增每日上传计数失败", zap.Error(err))
    }
}
```

```go
func ValidateFilename(filename string) error {
    if len(filename) == 0 {
        return fmt.Errorf("文件名不能为空")
    }
    // 按 UTF-8 字符数计算（而非字节数）
    if utf8.RuneCountInString(filename) > MaxFilenameLength {

    for _, ch := range illegalChars {
        if strings.Contains(filename, ch) {
            return fmt.Errorf("文件名包含非法字符: %s", ch)
        }
    }
    return nil
}
```

#### 4.13.3 MIME 类型校验（Magic Bytes）

不仅校验文件扩展名，还校验文件头 magic bytes 以防止伪造扩展名的恶意文件。

```go
// pkg/upload/mime_validate.go

package upload

import (
    "bytes"
    "fmt"
    "net/http"
    "path/filepath"
    "strings"
)

// magicSignature 文件头魔数签名
type magicSignature struct {
    offset int
    bytes  []byte
    mime   string
}

// 已知文件类型的 magic bytes 映射
var magicSignatures = []magicSignature{
    // 图片
    {0, []byte{0xFF, 0xD8, 0xFF}, "image/jpeg"},
    {0, []byte{0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A}, "image/png"},
    {0, []byte{0x47, 0x49, 0x46, 0x38}, "image/gif"},               // GIF87a / GIF89a
    {0, []byte{0x52, 0x49, 0x46, 0x46}, "image/webp"},              // RIFF container, 需进一步检查WEBP
    {0, []byte{0x42, 0x4D}, "image/bmp"},

    // 文档
    {0, []byte{0x25, 0x50, 0x44, 0x46}, "application/pdf"},          // %PDF
    {0, []byte{0x50, 0x4B, 0x03, 0x04}, "application/zip"},          // ZIP/DOCX/XLSX/PPTX
    {0, []byte{0xD0, 0xCF, 0x11, 0xE0, 0xA1, 0xB1, 0x1A, 0xE1}, "application/msword"}, // OLE2 (doc/xls/ppt)

    // 音视频
    {0, []byte{0x49, 0x44, 0x33}, "audio/mpeg"},                     // ID3 (MP3)
    {0, []byte{0xFF, 0xFB}, "audio/mpeg"},                           // MP3 frame sync
    {4, []byte{0x66, 0x74, 0x79, 0x70}, "video/mp4"},                // ftyp (MP4)
    {0, []byte{0x1A, 0x45, 0xDF, 0xA3}, "video/webm"},               // WebM/MKV
    {0, []byte{0x4F, 0x67, 0x67, 0x53}, "audio/ogg"},               // OggS

    // 压缩包
    {0, []byte{0x1F, 0x8B}, "application/gzip"},
    {0, []byte{0x37, 0x7A, 0xBC, 0xAF, 0x27, 0x1C}, "application/x-7z-compressed"},
    {0, []byte{0x52, 0x61, 0x72, 0x21, 0x1A, 0x07}, "application/x-rar-compressed"},
}

// ValidateMIMEType 校验文件MIME类型
// 同时校验扩展名白名单和文件头 magic bytes
func ValidateMIMEType(fileData []byte, filename string, allowedTypes []string) (string, error) {
    // 1. 扩展名校验
    ext := strings.ToLower(filepath.Ext(filename))
    extMIME := mime.TypeByExtension(ext)
    if extMIME == "" {
        return "", fmt.Errorf("不支持的文件扩展名: %s", ext)
    }

    // 2. Magic bytes 检测实际MIME类型
    detectedMIME := detectMIMEByMagicBytes(fileData)
    if detectedMIME == "" {
        // magic bytes 无法识别时，回退到 http.DetectContentType
        detectedMIME = http.DetectContentType(fileData)
    }

    // 3. 一致性校验：扩展名对应的MIME与文件头检测到的MIME必须一致
    //    允许 application/zip 与 office 格式的交叉（docx/xlsx 本质是 zip）
    if !isMIMECompatible(extMIME, detectedMIME) {
        return "", fmt.Errorf("文件扩展名(%s)与实际内容(%s)不匹配，拒绝上传", ext, detectedMIME)
    }

    // 4. 白名单校验
    if len(allowedTypes) > 0 {
        allowed := false
        for _, t := range allowedTypes {
            if strings.HasPrefix(detectedMIME, t) {
                allowed = true
                break
            }
        }
        if !allowed {
            return "", fmt.Errorf("不允许的文件类型: %s", detectedMIME)
        }
    }

    return detectedMIME, nil
}

// detectMIMEByMagicBytes 通过文件头魔数检测MIME类型
func detectMIMEByMagicBytes(data []byte) string {
    for _, sig := range magicSignatures {
        end := sig.offset + len(sig.bytes)
        if len(data) >= end {
            if bytes.Equal(data[sig.offset:end], sig.bytes) {
                // 额外检查 WEBP（RIFF 容器中需确认子类型）
                if sig.mime == "image/webp" {
                    if len(data) >= 12 && string(data[8:12]) == "WEBP" {
                        return "image/webp"
                    }
                    continue // RIFF 但不是 WEBP
                }
                return sig.mime
            }
        }
    }
    return ""
}

// isMIMECompatible 检查两个MIME类型是否兼容
func isMIMECompatible(extMIME, detectedMIME string) bool {
    // 精确匹配
    if extMIME == detectedMIME {
        return true
    }
    // 同一大类匹配（如 image/* 匹配 image/*）
    extParts := strings.SplitN(extMIME, "/", 2)
    detParts := strings.SplitN(detectedMIME, "/", 2)
    if len(extParts) == 2 && len(detParts) == 2 {
        if extParts[0] == detParts[0] {
            return true
        }
    }
    // Office 格式特例：docx/xlsx/pptx 的 magic bytes 是 application/zip
    zipCompat := map[string]bool{
        "application/vnd.openxmlformats-officedocument.wordprocessingml.document":   true,
        "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet":         true,
        "application/vnd.openxmlformats-officedocument.presentationml.presentation": true,
    }
    if detectedMIME == "application/zip" && zipCompat[extMIME] {
        return true
    }
    return false
}
```

#### 4.13.4 图片自动压缩（超宽缩放）

上传图片宽度超过 1920px 时，等比缩放至最大宽度 1920px，同时保持宽高比。

```go
// service/file/image_compress.go

package file

import (
    "bytes"
    "fmt"
    "image"
    "image/gif"
    "image/jpeg"
    "image/png"

    "github.com/disintegration/imaging"
)

const MaxImageWidth = 1920

// CompressResult 图片压缩结果
type CompressResult struct {
    Data      []byte
    Width     int
    Height    int
    Compressed bool // 是否进行了压缩
}

// CompressImage 图片自动压缩
// 当图片宽度超过 MaxImageWidth 时等比缩放
func CompressImage(fileData []byte, mimeType string, maxWidth int) (*CompressResult, error) {
    if maxWidth <= 0 {
        maxWidth = MaxImageWidth
    }

    img, format, err := image.Decode(bytes.NewReader(fileData))
    if err != nil {
        return nil, fmt.Errorf("解码图片失败: %w", err)
    }

    bounds := img.Bounds()
    origWidth := bounds.Dx()
    origHeight := bounds.Dy()

    // 宽度未超过限制，不压缩
    if origWidth <= maxWidth {
        return &CompressResult{
            Data:       fileData,
            Width:      origWidth,
            Height:     origHeight,
            Compressed: false,
        }, nil
    }

    // 等比缩放：宽度缩放到 maxWidth，高度按比例计算
    ratio := float64(maxWidth) / float64(origWidth)
    newHeight := int(float64(origHeight) * ratio)

    // 使用 Lanczos 滤镜进行高质量缩放
    resized := imaging.Resize(img, maxWidth, newHeight, imaging.Lanczos)

    // 编码
    var buf bytes.Buffer
    switch format {
    case "jpeg":
        err = jpeg.Encode(&buf, resized, &jpeg.Options{Quality: 92})
    case "png":
        err = png.Encode(&buf, resized)
    case "gif":
        err = gif.Encode(&buf, resized, nil)
    default:
        // 默认用 JPEG
        err = jpeg.Encode(&buf, resized, &jpeg.Options{Quality: 92})
    }
    if err != nil {
        return nil, fmt.Errorf("编码压缩图片失败: %w", err)
    }

    return &CompressResult{
        Data:       buf.Bytes(),
        Width:      maxWidth,
        Height:     newHeight,
        Compressed: true,
    }, nil
}
```

```go
func (c *FileUploadConfig) loadFromDB() {
    // 加载允许的扩展名
    var extConfig string
    if err := c.db.Table("system_configs").Where("`key` = ?", "file.allowed_extensions").
        Select("value").Scan(&extConfig).Error; err == nil && extConfig != "" {
        var exts []string
        if json.Unmarshal([]byte(extConfig), &exts) == nil {
            c.mu.Lock()
            c.allowedExtensions = exts
            c.mu.Unlock()
        }
    }

    // 加载各业务类型大小限制
    type ConfigRow struct {
        Key   string `gorm:"column:key"`
        Value string `gorm:"column:value"`
    }
    var rows []ConfigRow
    c.db.Table("system_configs").Where("`key` LIKE ?", "file.max_size.%").Find(&rows)

    c.mu.Lock()
    for _, row := range rows {
        businessType := row.Key[len("file.max_size."):]
        var maxSize int64
        if _, err := fmt.Sscanf(row.Value, "%d", &maxSize); err == nil {
            c.maxSizeByType[businessType] = maxSize
        }
    }
    c.mu.Unlock()
}

// autoRefresh 定时刷新配置
func (c *FileUploadConfig) autoRefresh() {
    ticker := time.NewTicker(c.refreshInterval)
    defer ticker.Stop()
    for range ticker.C {
        c.loadFromDB()
    }
}

// IsAllowedExtension 检查扩展名是否允许
func (c *FileUploadConfig) IsAllowedExtension(ext string) bool {
    c.mu.RLock()
    defer c.mu.RUnlock()
    if len(c.allowedExtensions) == 0 {
        return true // 未配置则不限制
    }
    for _, allowed := range c.allowedExtensions {
        if allowed == ext {
            return true
        }
    }
    return false
}

// GetMaxSize 获取指定业务类型的最大文件大小
func (c *FileUploadConfig) GetMaxSize(businessType string) int64 {
    c.mu.RLock()
    defer c.mu.RUnlock()
    if maxSize, ok := c.maxSizeByType[businessType]; ok {
        return maxSize
    }
    // 回退到默认值
    if defaultSize, ok := c.maxSizeByType["default"]; ok {
        return defaultSize
    }
    return 20 * 1024 * 1024 // 20MB 兜底默认值
}
```

| Key 模式 | 说明 | 数据类型 | TTL | 备注 |
|-----------|------|----------|-----|------|
| `file:info:{file_id}` | 文件基本信息缓存 | Hash | 30min | 文件详情页频繁查询 |
| `file:upload:daily:{user_id}:{date}` | 用户每日上传计数 | String (counter) | 25h | 用于每日上传限制校验 |
| `file:hotlink:rules` | 防盗链规则列表 | String (JSON) | 10min | 规则变更时主动刷新 |
| `file:config:upload` | 上传配置（格式/大小限制） | Hash | 5min | 与DB同步刷新 |
| `file:config:storage` | 存储配置（OSS/CDN参数） | Hash | 1h | 变更频率低 |
| `file:config:watermark:default` | 默认水印配置 | String (JSON) | 30min | 水印配置变更时刷新 |
| `file:access:count:{file_id}:{date}` | 文件每日访问计数 | String (counter) | 25h | 用于热门文件统计 |
| `file:lock:upload:{user_id}` | 上传并发锁（防重复提交） | String | 30s | 同一文件短时间内不重复处理 |

缓存更新策略：

```go
// cache/file_cache.go

package cache

import (
    "context"
    "encoding/json"
    "fmt"
    "time"

    "github.com/redis/go-redis/v9"
    "questionhelper-pro/internal/model"
)

type FileCache struct {
    rdb *redis.Client
}

func NewFileCache(rdb *redis.Client) *FileCache {
    return &FileCache{rdb: rdb}
}

const (
    fileInfoKeyPrefix    = "file:info:"
    hotlinkRulesKey      = "file:hotlink:rules"
    uploadConfigKey      = "file:config:upload"
    storageConfigKey     = "file:config:storage"
    defaultWatermarkKey  = "file:config:watermark:default"

    fileInfoTTL    = 30 * time.Minute
    hotlinkTTL     = 10 * time.Minute
    uploadCfgTTL   = 5 * time.Minute
    storageCfgTTL  = 1 * time.Hour
    watermarkTTL   = 30 * time.Minute
)

// GetFileInfo 获取文件信息缓存
func (c *FileCache) GetFileInfo(ctx context.Context, fileID uint64) (*model.File, error) {
    key := fmt.Sprintf("%s%d", fileInfoKeyPrefix, fileID)
    data, err := c.rdb.Get(ctx, key).Bytes()
    if err != nil {
        return nil, err
    }
    var file model.File
    if err := json.Unmarshal(data, &file); err != nil {
        return nil, err
    }
    return &file, nil
}

// SetFileInfo 设置文件信息缓存
func (c *FileCache) SetFileInfo(ctx context.Context, file *model.File) {
    key := fmt.Sprintf("%s%d", fileInfoKeyPrefix, file.ID)
    data, _ := json.Marshal(file)
    c.rdb.Set(ctx, key, data, fileInfoTTL)
}

// InvalidateFileInfo 删除文件信息缓存（文件更新/删除时调用）
func (c *FileCache) InvalidateFileInfo(ctx context.Context, fileID uint64) {
    key := fmt.Sprintf("%s%d", fileInfoKeyPrefix, fileID)
    c.rdb.Del(ctx, key)
}

// GetHotlinkRules 获取防盗链规则缓存
func (c *FileCache) GetHotlinkRules(ctx context.Context) (string, error) {
    return c.rdb.Get(ctx, hotlinkRulesKey).Result()
}

// SetHotlinkRules 设置防盗链规则缓存
func (c *FileCache) SetHotlinkRules(ctx context.Context, rulesJSON string) {
    c.rdb.Set(ctx, hotlinkRulesKey, rulesJSON, hotlinkTTL)
}

// RefreshHotlinkRules 主动刷新防盗链规则缓存（规则变更时调用）
func (c *FileCache) RefreshHotlinkRules(ctx context.Context, rulesJSON string) {
    c.rdb.Set(ctx, hotlinkRulesKey, rulesJSON, hotlinkTTL)
}

// GetUploadConfig 获取上传配置缓存
func (c *FileCache) GetUploadConfig(ctx context.Context) (string, error) {
    return c.rdb.Get(ctx, uploadConfigKey).Result()
}

// SetUploadConfig 设置上传配置缓存
func (c *FileCache) SetUploadConfig(ctx context.Context, configJSON string) {
    c.rdb.Set(ctx, uploadConfigKey, configJSON, uploadCfgTTL)
}

// GetStorageConfig 获取存储配置缓存
func (c *FileCache) GetStorageConfig(ctx context.Context) (string, error) {
    return c.rdb.Get(ctx, storageConfigKey).Result()
}

// SetStorageConfig 设置存储配置缓存
func (c *FileCache) SetStorageConfig(ctx context.Context, configJSON string) {
    c.rdb.Set(ctx, storageConfigKey, configJSON, storageCfgTTL)
}

// GetDefaultWatermark 获取默认水印配置缓存
func (c *FileCache) GetDefaultWatermark(ctx context.Context) (string, error) {
    return c.rdb.Get(ctx, defaultWatermarkKey).Result()
}

// SetDefaultWatermark 设置默认水印配置缓存
func (c *FileCache) SetDefaultWatermark(ctx context.Context, configJSON string) {
    c.rdb.Set(ctx, defaultWatermarkKey, configJSON, watermarkTTL)
}

// InvalidateDefaultWatermark 删除默认水印缓存
func (c *FileCache) InvalidateDefaultWatermark(ctx context.Context) {
    c.rdb.Del(ctx, defaultWatermarkKey)
}
```

```vue
const progressText = ref('')
const uploadSpeed = ref('')
const remainingTime = ref('')

const uploadUrl = computed(() => {
  const base = import.meta.env.VITE_API_BASE_URL
  return props.type === 'image'

  // 文件大小校验
  const isLtMaxSize = file.size / 1024 / 1024 < props.maxSize
  if (!isLtMaxSize) {
    ElMessage.error(`文件大小不能超过 ${props.maxSize}MB`)
    return false
  }

  // 文件名长度校验
  if (file.name.length > 200) {
    ElMessage.error('文件名长度不能超过200个字符')
    return false
  }

  // 图片类型校验
  if (props.type === 'image') {
    const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp']
    if (!allowedTypes.includes(file.type)) {
      ElMessage.error('只支持 jpg/png/gif/webp 格式的图片')
      return false
    }
  }

  // 初始化进度
  showProgress.value = true
  uploadProgress.value = 0
  uploadStatus.value = ''
  progressText.value = '准备上传...'
  uploadSpeed.value = ''
  remainingTime.value = ''

  currentFileSize.value = file.size

  lastTime = Date.now()

  // 拖拽模式下的额外处理

const handleProgress = (event: any) => {
  const now = Date.now()
  const loaded = event.loaded || 0
  const total = event.total || 1
  const percent = Math.round(event.percent || 0)

  uploadProgress.value = percent

  // 计算上传速度（每秒字节数）
  const timeDiff = (now - lastTime) / 1000 // 秒
  if (timeDiff > 0.5) { // 每0.5秒更新一次速度
    const loadedDiff = loaded - lastLoaded
    const speed = loadedDiff / timeDiff // bytes/s

    uploadSpeed.value = formatSpeed(speed)

    // 计算剩余时间
    if (speed > 0) {
      const remaining = (total - loaded) / speed
      remainingTime.value = formatDuration(remaining)
    }

    lastLoaded = loaded
    lastTime = now
  }

  progressText.value = `已上传 ${percent}%`
}

const handleSuccess = (response: any) => {
  showProgress.value = false
  if (response.code === 0) {
    emit('success', response.data)
    ElMessage.success('上传成功')
  } else {
    uploadStatus.value = 'exception'
    progressText.value = response.message || '上传失败'
    emit('error', response.message)
    ElMessage.error(response.message)
  }
}

  showProgress.value = false
  uploadStatus.value = 'exception'
  progressText.value = '上传失败'
  emit('error', '上传失败')
  ElMessage.error('上传失败')
}

  if (bytes < 1024) return bytes + ' B'
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB'
  return (bytes / 1024 / 1024).toFixed(1) + ' MB'
}

// 格式化上传速度
const formatSpeed = (bytesPerSec: number): string => {
  if (bytesPerSec < 1024) return bytesPerSec.toFixed(0) + ' B/s'
  if (bytesPerSec < 1024 * 1024) return (bytesPerSec / 1024).toFixed(1) + ' KB/s'
  return (bytesPerSec / 1024 / 1024).toFixed(1) + ' MB/s'
}

// 格式化时长
const formatDuration = (seconds: number): string => {
  if (seconds < 60) return Math.ceil(seconds) + ' 秒'
  if (seconds < 3600) return Math.ceil(seconds / 60) + ' 分钟'
  return Math.ceil(seconds / 3600) + ' 小时'
}
```

```typescript
// src/api/file.ts

import request from '@/utils/request'

export const fileApi = {
  // 获取文件列表
  list(params: any) {
    return request.get('/api/v1/file/list', { params })
  },

  // 获取文件详情
  getDetail(id: number) {
    return request.get(`/api/v1/file/${id}`)
  },

  // 删除文件
  delete(id: number) {
    return request.delete(`/api/v1/file/${id}`)
  },

  // 批量删除
  batchDelete(fileIds: number[]) {
    return request.post('/api/v1/admin/files/batch/delete', { file_ids: fileIds })
  },

  // 获取引用列表
  getReferences(fileId: number) {
    return request.get(`/api/v1/file/${fileId}/references`)
  },

  // 触发病毒扫描
  triggerVirusScan(fileId: number) {
    return request.post(`/api/v1/file/${fileId}/virus-scan`)
  },

  // 触发内容检测
  triggerContentCheck(fileId: number) {
    return request.post(`/api/v1/file/${fileId}/content-check`)
  },

  // 下载文件
  download(fileId: number) {
    return request.get(`/api/v1/file/${fileId}/download`, { responseType: 'blob' })
  },
}
```
