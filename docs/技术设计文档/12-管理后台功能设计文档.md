## 三、菜单布局设计

### 3.1 菜单树形结构

管理后台采用左侧边栏 + 顶部导航的经典布局。菜单基于 RBAC 权限模型动态加载，不同角色看到的菜单项不同。

**设计原则**：
- 用户管理、权限管理、系统设置为平级的系统管理模块
- 菜单管理属于权限系统核心，独立于用户管理
- 消息中心、文件管理作为独立的一级菜单
- 数据统计包含基础统计和高级分析两个层级
- 审核功能统一归入「审批管理」或各业务模块内

### 3.2 菜单加载流程

```
用户登录 → 获取Token → 获取用户菜单 → 动态添加路由 → 渲染侧边栏
```

### 3.3 核心代码实现

```typescript
// src/composables/useMenu.ts

import { ref } from 'vue'
import { RouteRecordRaw } from 'vue-router'
import { getUserMenus } from '@/api/system/menu'
import router from '@/router'

export const useMenu = () => {
  const menus = ref<RouteRecordRaw[]>([])
  const loading = ref(false)
  
  // 动态导入组件
  const modules = import.meta.glob('../views/**/*.vue')
  
  // 加载用户菜单
  const loadMenus = async () => {
    loading.value = true
    try {
      const res = await getUserMenus()
      const menuData = res.data
      
      // 转换为路由格式
      const routes = transformRoutes(menuData)
      
      // 动态添加路由
      routes.forEach(route => {
        router.addRoute('Layout', route)
      })
      
      menus.value = menuData
    } catch (e) {
      console.error('加载菜单失败', e)
    } finally {
      loading.value = false
    }
  }
  
  // 转换菜单为路由
  const transformRoutes = (menus: any[]): RouteRecordRaw[] => {
    return menus
      .filter(menu => menu.type !== 3) // 过滤按钮
      .map(menu => {
        const route: RouteRecordRaw = {
          path: menu.path,
          name: menu.name,
          component: loadComponent(menu.component),
          meta: {
            title: menu.title,
            icon: menu.icon,
            hidden: menu.hidden
          },
          children: menu.children ? transformRoutes(menu.children) : []
        }
        return route
      })
  }
  
  // 动态加载组件
  const loadComponent = (component: string) => {
    if (!component) return undefined
    return modules[`../views/${component}.vue`]
  }
  
  return {
    menus,
    loading,
    loadMenus
  }
}
```

### 3.4 按钮权限指令

```typescript
// src/directive/permission.ts

import { Directive, DirectiveBinding } from 'vue'
import { usePermissionStore } from '@/store/modules/permission'

export const vPermission: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding<string>) {
    const permissionStore = usePermissionStore()
    const requiredPermission = binding.value
    
    if (!requiredPermission) return
    
    // 管理员拥有所有权限
    if (permissionStore.permissions.includes('*:*')) {
      return
    }
    
    // 检查权限
    if (!permissionStore.permissions.includes(requiredPermission)) {
      el.parentNode?.removeChild(el)
    }
  }
}

// 注册指令
// main.ts
app.directive('permission', vPermission)

// 使用示例
// <el-button v-permission="'user:create'">新增用户</el-button>
// <el-button v-permission="'user:edit'">编辑</el-button>
// <el-button v-permission="'user:delete'">删除</el-button>
```

### 4.1 文件导入组件

```vue
<!-- src/components/FileUpload/ExcelUpload.vue -->

<template>
  <el-upload
    ref="uploadRef"
    :action="uploadUrl"
    :headers="headers"
    :before-upload="beforeUpload"
    :on-success="handleSuccess"
    :on-error="handleError"
    accept=".xlsx,.xls"
  >
    <el-button type="primary">选择文件</el-button>
    <template #tip>
      <div class="el-upload__tip">只能上传 xlsx/xls 文件</div>
    </template>

<script setup lang="ts">
import { computed } from 'vue'
import { getToken } from '@/utils/auth'

const props = defineProps<{
  action: string
}>()

const emit = defineEmits(['success', 'error'])

const uploadUrl = computed(() => import.meta.env.VITE_API_BASE_URL + props.action)

const headers = computed(() => ({
  Authorization: `Bearer ${getToken()}`
}))

const beforeUpload = (file: File) => {
  const isExcel = file.type === 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' ||
                  file.type === 'application/vnd.ms-excel'
  if (!isExcel) {
    ElMessage.error('只能上传 Excel 文件')
    return false
  }
  return true
}

const handleSuccess = (response: any) => {
  if (response.code === 0) {
    emit('success', response.data)
    ElMessage.success('导入成功')
  } else {
    ElMessage.error(response.message)
  }
}

const handleError = () => {
  ElMessage.error('上传失败')
}
</script>
```

```typescript
// src/utils/export.ts

import axios from 'axios'
import { getToken } from './auth'

export const exportExcel = async (url: string, params: any, filename: string) => {
  try {
    const response = await axios({
      url: import.meta.env.VITE_API_BASE_URL + url,
      method: 'get',
      params,
      responseType: 'blob',
      headers: {
        Authorization: `Bearer ${getToken()}`
      }
    })
    
    const blob = new Blob([response.data], {
      type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
    })
    
    const link = document.createElement('a')
    link.href = URL.createObjectURL(blob)
    link.download = filename
    link.click()
    URL.revokeObjectURL(link.href)
  } catch (e) {
    ElMessage.error('导出失败')
  }
}
```

### 6.2 连接管理

```typescript
// src/composables/useWebSocket.ts

import { ref, onMounted, onUnmounted } from 'vue'
import { getToken } from '@/utils/auth'
import { useUserStore } from '@/store/modules/user'

export interface WsMessage {
  type: string
  data: any
  timestamp: number
}

type MessageHandler = (data: any) => void

export const useWebSocket = () => {
  const ws = ref<WebSocket | null>(null)
  const connected = ref(false)
  const reconnectCount = ref(0)
  const maxReconnect = 5
  const handlers = new Map<string, MessageHandler[]>()
  let reconnectTimer: ReturnType<typeof setTimeout> | null = null

  // 建立连接
  const connect = () => {
    const token = getToken()
    if (!token) return

    const wsUrl = `${import.meta.env.VITE_WS_URL}/ws?token=${token}`
    ws.value = new WebSocket(wsUrl)

    ws.value.onopen = () => {
      connected.value = true
      reconnectCount.value = 0
      console.log('[WebSocket] 连接成功')
      // 启动心跳
      startHeartbeat()
    }

    ws.value.onmessage = (event) => {
      try {
        const message: WsMessage = JSON.parse(event.data)
        handleMessage(message)
      } catch (e) {
        console.error('[WebSocket] 消息解析失败', e)
      }
    }

    ws.value.onclose = () => {
      connected.value = false
      stopHeartbeat()
      // 自动重连
      if (reconnectCount.value < maxReconnect) {
        const delay = Math.min(1000 * Math.pow(2, reconnectCount.value), 30000)
        reconnectTimer = setTimeout(() => {
          reconnectCount.value++
          connect()
        }, delay)
      }
    }

    ws.value.onerror = (err) => {
      console.error('[WebSocket] 连接错误', err)
    }
  }

  // 心跳机制
  let heartbeatTimer: ReturnType<typeof setInterval> | null = null

  const startHeartbeat = () => {
    heartbeatTimer = setInterval(() => {
      if (ws.value?.readyState === WebSocket.OPEN) {
        ws.value.send(JSON.stringify({ type: 'ping' }))
      }
    }, 30000)
  }

  const stopHeartbeat = () => {
    if (heartbeatTimer) {
      clearInterval(heartbeatTimer)
      heartbeatTimer = null
    }
  }

  // 消息分发
  const handleMessage = (message: WsMessage) => {
    const list = handlers.get(message.type)
    if (list) {
      list.forEach(fn => fn(message.data))
    }
    // 通配符处理
    const allList = handlers.get('*')
    if (allList) {
      allList.forEach(fn => fn(message))
    }
  }

  // 注册消息处理器
  const on = (type: string, handler: MessageHandler) => {
    if (!handlers.has(type)) {
      handlers.set(type, [])
    }
    handlers.get(type)!.push(handler)
  }

  // 移除消息处理器
  const off = (type: string, handler: MessageHandler) => {
    const list = handlers.get(type)
    if (list) {
      const idx = list.indexOf(handler)
      if (idx !== -1) list.splice(idx, 1)
    }
  }

  // 发送消息
  const send = (type: string, data: any) => {
    if (ws.value?.readyState === WebSocket.OPEN) {
      ws.value.send(JSON.stringify({ type, data }))
    }
  }

  // 断开连接
  const disconnect = () => {
    stopHeartbeat()
    if (reconnectTimer) {
      clearTimeout(reconnectTimer)
      reconnectTimer = null
    }
    ws.value?.close()
    ws.value = null
    connected.value = false
  }

  onMounted(() => connect())
  onUnmounted(() => disconnect())

  return { connected, connect, disconnect, send, on, off }
}
```

### 8.1 权限Store

```typescript
// src/store/modules/permission.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getUserMenus } from '@/api/system/menu'
import type { MenuData } from '@/api/system/menu'

export const usePermissionStore = defineStore('permission', () => {
  const menus = ref<MenuData[]>([])
  const permissions = ref<string[]>([])

  // 加载用户权限
  const loadPermissions = async () => {
    const res = await getUserMenus()
    menus.value = res.data
    // 提取所有按钮权限
    permissions.value = extractPermissions(res.data)
  }

  // 递归提取权限标识
  const extractPermissions = (menuList: MenuData[]): string[] => {
    const result: string[] = []
    const traverse = (list: MenuData[]) => {
      list.forEach(item => {
        if (item.type === 'button' && item.permission) {
          result.push(item.permission)
        }
        if (item.children?.length) {
          traverse(item.children)
        }
      })
    }
    traverse(menuList)
    return result
  }

  // 检查是否有某权限
  const hasPermission = (permission: string): boolean => {
    // 超级管理员拥有所有权限
    if (permissions.value.includes('*:*')) return true
    return permissions.value.includes(permission)
  }

  // 检查是否有任一权限
  const hasAnyPermission = (perms: string[]): boolean => {
    if (permissions.value.includes('*:*')) return true
    return perms.some(p => permissions.value.includes(p))
  }

  // 检查是否有全部权限
  const hasAllPermissions = (perms: string[]): boolean => {
    if (permissions.value.includes('*:*')) return true
    return perms.every(p => permissions.value.includes(p))
  }

  // 清空权限
  const resetPermissions = () => {
    menus.value = []
    permissions.value = []
  }

  return {
    menus,
    permissions,
    loadPermissions,
    hasPermission,
    hasAnyPermission,
    hasAllPermissions,
    resetPermissions
  }
})
```

```typescript
 *   <el-button v-permission.all="['user:create', 'user:edit']">操作</el-button>
 */
export const vPermission: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding<string | string[]>) {
    const permissionStore = usePermissionStore()
    const value = binding.value

    if (!value || (Array.isArray(value) && value.length === 0)) return

    let hasPerm = false

    if (typeof value === 'string') {
      // 单个权限
      hasPerm = permissionStore.hasPermission(value)
    } else if (Array.isArray(value)) {
      if (binding.modifiers.all) {
        // 全部满足
        hasPerm = permissionStore.hasAllPermissions(value)
      } else {
        // 任一满足
        hasPerm = permissionStore.hasAnyPermission(value)
      }
    }

    if (!hasPerm) {
      el.parentNode?.removeChild(el)
    }
  }
}

/**
 * v-role 指令：基于角色控制元素显示
 *
 * 用法：
 *   <el-button v-role="'admin'">管理员操作</el-button>
 *   <el-button v-role="['admin', 'teacher']">操作</el-button>
 */
export const vRole: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding<string | string[]>) {
    const value = binding.value
    if (!value) return

    // 从 user store 获取角色
    const userStore = useUserStore()
    const userRoles = userStore.roles

    let hasRole = false
    if (typeof value === 'string') {
      hasRole = userRoles.includes(value)
    } else if (Array.isArray(value)) {
      hasRole = value.some(r => userRoles.includes(r))
    }

    if (!hasRole) {
      el.parentNode?.removeChild(el)
    }
  }
}
```

```typescript
import axios from 'axios'
import { getToken } from './auth'

export interface ExportOptions {
  url: string
  params?: Record<string, any>
  filename: string
  method?: 'get' | 'post'
}

/**
 * 导出Excel文件
 */
export const exportExcel = async (options: ExportOptions) => {
  const { url, params, filename, method = 'get' } = options

  try {
    const response = await axios({
      url: import.meta.env.VITE_API_BASE_URL + url,
      method,
      params: method === 'get' ? params : undefined,
      data: method === 'post' ? params : undefined,
      responseType: 'blob',
      headers: {
        Authorization: `Bearer ${getToken()}`
      }
    })

    downloadBlob(response.data, filename, 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet')
    ElMessage.success('导出成功')
  } catch (e) {
    ElMessage.error('导出失败')
    throw e
  }
}

/**
 * 导出CSV文件
 */
export const exportCSV = async (options: ExportOptions) => {
  const { url, params, filename, method = 'get' } = options

  try {
    const response = await axios({
      url: import.meta.env.VITE_API_BASE_URL + url,
      method,
      params: method === 'get' ? params : undefined,
      data: method === 'post' ? params : undefined,
      responseType: 'blob',
      headers: {
        Authorization: `Bearer ${getToken()}`
      }
    })

    downloadBlob(response.data, filename.endsWith('.csv') ? filename : `${filename}.csv`, 'text/csv')
    ElMessage.success('导出成功')
  } catch (e) {
    ElMessage.error('导出失败')
    throw e
  }
}

/**
 * 前端表格数据直接导出为CSV
 */
export const exportTableToCSV = (data: any[], columns: { key: string; title: string }[], filename: string) => {
  // BOM头解决中文乱码
  const BOM = '﻿'
  const header = columns.map(c => c.title).join(',')
  const rows = data.map(row =>
    columns.map(c => {
      const value = row[c.key]
      // 处理包含逗号、换行的字段
      if (typeof value === 'string' && (value.includes(',') || value.includes('\n'))) {
        return `"${value.replace(/"/g, '""')}"`
      }
      return value ?? ''
    }).join(',')
  )

  const csv = BOM + [header, ...rows].join('\n')
  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8' })
  downloadBlob(blob, filename.endsWith('.csv') ? filename : `${filename}.csv`, 'text/csv')
}

/**
 * 下载Blob文件
 */
const downloadBlob = (blob: Blob, filename: string, mimeType: string) => {
  const file = new Blob([blob], { type: mimeType })
  const link = document.createElement('a')
  link.href = URL.createObjectURL(file)
  link.download = filename
  link.click()
  URL.revokeObjectURL(link.href)
}
```

```typescript
// src/composables/useShortcut.ts

import { onMounted, onUnmounted } from 'vue'

export interface ShortcutConfig {
  key: string
  ctrl?: boolean
  alt?: boolean
  shift?: boolean
  meta?: boolean
  description: string
  handler: (event: KeyboardEvent) => void
}

// 全局快捷键注册表
const globalShortcuts = new Map<string, ShortcutConfig>()

// 生成快捷键唯一标识
const getShortcutKey = (config: ShortcutConfig): string => {
  const parts: string[] = []
  if (config.ctrl) parts.push('ctrl')
  if (config.alt) parts.push('alt')
  if (config.shift) parts.push('shift')
  if (config.meta) parts.push('meta')
  parts.push(config.key.toLowerCase())
  return parts.join('+')
}

// 全局键盘事件监听
const handleKeyDown = (event: KeyboardEvent) => {
  // 忽略输入框中的快捷键
  const target = event.target as HTMLElement
  if (['INPUT', 'TEXTAREA', 'SELECT'].includes(target.tagName) || target.isContentEditable) {
    return
  }

  const pressed = getShortcutKey({
    key: event.key,
    ctrl: event.ctrlKey || event.metaKey,
    alt: event.altKey,
    shift: event.shiftKey,
    meta: event.metaKey,
    description: '',
    handler: () => {}
  })

  const config = globalShortcuts.get(pressed)
  if (config) {
    event.preventDefault()
    config.handler(event)
  }
}

/**
 * 注册全局快捷键
 */
export const useShortcut = (shortcuts: ShortcutConfig[]) => {
  onMounted(() => {
    shortcuts.forEach(config => {
      const key = getShortcutKey(config)
      globalShortcuts.set(key, config)
    })
  })

  onUnmounted(() => {
    shortcuts.forEach(config => {
      const key = getShortcutKey(config)
      globalShortcuts.delete(key)
    })
  })
}

/**
 * 初始化全局快捷键监听
 */
export const initShortcutListener = () => {
  document.addEventListener('keydown', handleKeyDown)
}

/**
 * 获取所有已注册的快捷键
 */
export const getAllShortcuts = (): ShortcutConfig[] => {
  return Array.from(globalShortcuts.values())
}
```

```typescript
// src/store/modules/app.ts

import { defineStore } from 'pinia'
import { ref, watch } from 'vue'

export type ThemeMode = 'light' | 'dark'

export const useAppStore = defineStore('app', () => {
  const theme = ref<ThemeMode>((localStorage.getItem('theme') as ThemeMode) || 'light')
  const sidebarCollapsed = ref(false)
  const searchDialogVisible = ref(false)
  const shortcutHelpVisible = ref(false)

  // 切换主题
  const toggleTheme = () => {
    theme.value = theme.value === 'light' ? 'dark' : 'light'
  }

  const setTheme = (mode: ThemeMode) => {
    theme.value = mode
  }

  // 监听主题变化，更新DOM
  watch(theme, (val) => {
    document.documentElement.setAttribute('data-theme', val)
    document.documentElement.classList.toggle('dark', val === 'dark')
    localStorage.setItem('theme', val)
  }, { immediate: true })

  // 侧边栏
  const toggleSidebar = () => {
    sidebarCollapsed.value = !sidebarCollapsed.value
  }

  // 搜索弹窗
  const toggleSearchDialog = () => {
    searchDialogVisible.value = !searchDialogVisible.value
  }

  // 快捷键帮助
  const toggleShortcutHelp = () => {
    shortcutHelpVisible.value = !shortcutHelpVisible.value
  }

  // 关闭顶层弹窗
  const closeTopDialog = () => {
    if (searchDialogVisible.value) {
      searchDialogVisible.value = false
    } else if (shortcutHelpVisible.value) {
      shortcutHelpVisible.value = false
    }
  }

  return {
    theme,
    sidebarCollapsed,
    searchDialogVisible,
    shortcutHelpVisible,
    toggleTheme,
    setTheme,
    toggleSidebar,
    toggleSearchDialog,
    toggleShortcutHelp,
    closeTopDialog
  }
})
```

## 十四、消息通知

### 14.1 通知Store

```typescript
// src/store/modules/notification.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getNotifications, getUnreadCount, markAsRead, markAllAsRead } from '@/api/system/notification'
import { useWebSocket } from '@/composables/useWebSocket'

export interface NotificationItem {
  id: string
  type: string
  title: string
  content: string
  read: boolean
  createdAt: string
  data?: Record<string, any>
}

export const useNotificationStore = defineStore('notification', () => {
  const notifications = ref<NotificationItem[]>([])
  const unreadCount = ref(0)
  const loading = ref(false)

  const unreadNotifications = computed(() =>
    notifications.value.filter(n => !n.read)
  )

  // 加载通知列表
  const loadNotifications = async (params?: { page?: number; pageSize?: number; type?: string }) => {
    loading.value = true
    try {
      const res = await getNotifications(params)
      notifications.value = res.data.items
    } finally {
      loading.value = false
    }
  }

  // 刷新未读数量
  const refreshUnreadCount = async () => {
    const res = await getUnreadCount()
    unreadCount.value = res.data.count
  }

  // 标记单条已读
  const markItemAsRead = async (id: string) => {
    await markAsRead(id)
    const item = notifications.value.find(n => n.id === id)
    if (item && !item.read) {
      item.read = true
      unreadCount.value = Math.max(0, unreadCount.value - 1)
    }
  }

  // 全部标记已读
  const markAllAsReadHandler = async () => {
    await markAllAsRead()
    notifications.value.forEach(n => (n.read = true))
    unreadCount.value = 0
  }

  // 监听WebSocket新消息
  const initWebSocketListener = () => {
    const { on } = useWebSocket()
    on('notification:new', (data: NotificationItem) => {
      notifications.value.unshift(data)
      unreadCount.value++
      // 桌面通知
      if (Notification.permission === 'granted') {
        new window.Notification(data.title, {
          body: data.content,
          icon: '/app-icon.svg'
        })
      }
    })
  }

  return {
    notifications,
    unreadCount,
    loading,
    unreadNotifications,
    loadNotifications,
    refreshUnreadCount,
    markItemAsRead,
    markAllAsRead: markAllAsReadHandler,
    initWebSocketListener
  }
})
```

## 十六、Redis缓存设计

### 16.1 菜单缓存

**缓存Key设计**：`qh:menu:user:{uid}`

| 字段 | 说明 |
|------|------|
| Key | `qh:menu:user:{uid}` |
| 类型 | String (JSON序列化) |
| TTL | 24小时 (86400秒) |
| 存储内容 | 用户可访问的菜单树结构JSON |
| 失效策略 | 菜单变更时删除所有 `qh:menu:user:*` Key |

```go
// internal/service/cache/menu_cache.go

package cache

import (
    "context"
    "encoding/json"
    "fmt"
    "time"
)

const (
    MenuCacheKeyPattern = "qh:menu:user:%d"
    MenuCacheTTL        = 24 * time.Hour
)

type MenuCache struct {
    rdb *redis.Client
}

// GetMenus 获取用户菜单缓存
func (c *MenuCache) GetMenus(ctx context.Context, uid int64) ([]MenuData, bool) {
    key := fmt.Sprintf(MenuCacheKeyPattern, uid)
    data, err := c.rdb.Get(ctx, key).Bytes()
    if err != nil {
        return nil, false
    }
    var menus []MenuData
    if err := json.Unmarshal(data, &menus); err != nil {
        return nil, false
    }
    return menus, true
}

// SetMenus 设置用户菜单缓存
func (c *MenuCache) SetMenus(ctx context.Context, uid int64, menus []MenuData) error {
    key := fmt.Sprintf(MenuCacheKeyPattern, uid)
    data, err := json.Marshal(menus)
    if err != nil {
        return err
    }
    return c.rdb.Set(ctx, key, data, MenuCacheTTL).Err()
}

// InvalidateUserMenus 删除指定用户菜单缓存
func (c *MenuCache) InvalidateUserMenus(ctx context.Context, uid int64) error {
    key := fmt.Sprintf(MenuCacheKeyPattern, uid)
    return c.rdb.Del(ctx, key).Err()
}

// InvalidateAllUserMenus 删除所有用户菜单缓存（菜单结构变更时调用）
func (c *MenuCache) InvalidateAllUserMenus(ctx context.Context) error {
    pattern := "qh:menu:user:*"
    iter := c.rdb.Scan(ctx, 0, pattern, 100).Iterator()
    for iter.Next(ctx) {
        c.rdb.Del(ctx, iter.Val())
    }
    return iter.Err()
}
```

### 16.2 权限缓存

**缓存Key设计**：`qh:permission:role:{rid}`

| 字段 | 说明 |
|------|------|
| Key | `qh:permission:role:{rid}` |
| 类型 | String (JSON序列化) |
| TTL | 12小时 (43200秒) |
| 存储内容 | 角色拥有的权限标识列表JSON |
| 失效策略 | 角色权限变更时删除对应Key |

```go
// internal/service/cache/permission_cache.go

package cache

import (
    "context"
    "encoding/json"
    "fmt"
    "time"
)

const (
    PermissionCacheKeyPattern = "qh:permission:role:%d"
    PermissionCacheTTL        = 12 * time.Hour
)

type PermissionCache struct {
    rdb *redis.Client
}

// GetPermissions 获取角色权限缓存
func (c *PermissionCache) GetPermissions(ctx context.Context, roleID int64) ([]string, bool) {
    key := fmt.Sprintf(PermissionCacheKeyPattern, roleID)
    data, err := c.rdb.Get(ctx, key).Bytes()
    if err != nil {
        return nil, false
    }
    var permissions []string
    if err := json.Unmarshal(data, &permissions); err != nil {
        return nil, false
    }
    return permissions, true
}

// SetPermissions 设置角色权限缓存
func (c *PermissionCache) SetPermissions(ctx context.Context, roleID int64, permissions []string) error {
    key := fmt.Sprintf(PermissionCacheKeyPattern, roleID)
    data, err := json.Marshal(permissions)
    if err != nil {
        return err
    }
    return c.rdb.Set(ctx, key, data, PermissionCacheTTL).Err()
}

// InvalidateRolePermissions 删除角色权限缓存
func (c *PermissionCache) InvalidateRolePermissions(ctx context.Context, roleID int64) error {
    key := fmt.Sprintf(PermissionCacheKeyPattern, roleID)
    return c.rdb.Del(ctx, key).Err()
}

// InvalidateAllPermissions 删除所有权限缓存（按钮权限变更时调用）
func (c *PermissionCache) InvalidateAllPermissions(ctx context.Context) error {
    pattern := "qh:permission:role:*"
    iter := c.rdb.Scan(ctx, 0, pattern, 100).Iterator()
    for iter.Next(ctx) {
        c.rdb.Del(ctx, iter.Val())
    }
    return iter.Err()
}
```

### 16.3 缓存失效触发时机

| 触发场景 | 失效Key | 说明 |
|---------|---------|------|
| 菜单新增/编辑/删除 | `qh:menu:user:*` | 菜单结构变更，清除所有用户菜单缓存 |
| 角色权限分配变更 | `qh:permission:role:{rid}` | 清除对应角色的权限缓存 |
| 用户角色变更 | `qh:menu:user:{uid}` + `qh:permission:role:{newRid}` | 清除用户菜单缓存及新角色权限缓存 |
| 按钮权限变更 | `qh:permission:role:*` | 清除所有角色权限缓存 |

```vue

<template>
  <el-tree-select
    v-model="selectedValue"
    :data="treeData"
    :props="treeProps"
    :placeholder="placeholder"

    :check-strictly="checkStrictly"

<script setup lang="ts">
import { ref, computed, onMounted, watch } from 'vue'

interface TreeNode {
  id: number
  name: string
  children?: TreeNode[]
  isLeaf?: boolean
}

const props = withDefaults(defineProps<{
  modelValue?: number | number[] | null
  /** 数据加载接口 */
  api?: () => Promise<any>
  /** 静态数据（不使用api时直接传入） */
  data?: TreeNode[]
  /** 是否多选 */
  multiple?: boolean
  /** 是否可清空 */
  clearable?: boolean
  /** 父子不关联 */
  checkStrictly?: boolean
  /** 是否懒加载 */
  lazy?: boolean
  /** 占位文本 */
  placeholder?: string
  /** 自定义props映射 */
  labelField?: string
  valueField?: string
  childrenField?: string
}>(), {
  clearable: true,
  checkStrictly: true,
  lazy: false,
  placeholder: '请选择分类',
  labelField: 'name',
  valueField: 'id',
  childrenField: 'children'
})

const emit = defineEmits(['update:modelValue', 'change'])

const selectedValue = computed({
  get: () => props.modelValue ?? undefined,
  set: (val) => emit('update:modelValue', val)
})

const treeData = ref<TreeNode[]>([])

const treeProps = computed(() => ({
  label: props.labelField,
  value: props.valueField,
  children: props.childrenField,
  isLeaf: 'isLeaf'
}))

// 加载树数据
const loadTreeData = async () => {
  if (props.data) {
    treeData.value = props.data
    return
  }
  if (props.api) {
    const res = await props.api()
    treeData.value = res.data
  }
}

// 懒加载节点
const loadNode = (node: any, resolve: (data: TreeNode[]) => void) => {
  if (node.level === 0) {
    loadTreeData().then(() => resolve(treeData.value))
    return
  }
  // 子节点已包含在父节点数据中
  resolve(node.data.children || [])
}

// 过滤节点
const filterNode = (value: string, data: TreeNode) => {
  if (!value) return true
  return data.name.includes(value)
}

// 值变更
const handleChange = (val: any) => {
  emit('change', val)
}

// 非懒加载模式下直接加载
if (!props.lazy) {
  watch(() => props.data, () => {
    if (props.data) treeData.value = props.data
  }, { immediate: true })

  if (props.api && !props.data) {
    onMounted(() => loadTreeData())
  }
}
</script>
```
