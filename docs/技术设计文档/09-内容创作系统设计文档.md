# 内容创作系统设计文档

**项目名称**：题小助 (QuestionHelper)
**版本**：V2.0
**日期**：2026-05-30
**模块**：内容创作系统

---

## 一、功能概述

| 功能 | 说明 |
|------|------|
| 创作者入驻 | 申请成为创作者 |
| 创作者功能 | 创建私有题目、提交公共审核 |
| 教师功能 | 创建班级、管理班级资源 |
| 班级创作者 | 在班级内创建资源 |
| 内容审核 | 公共资源审核流程 |
| 创作者等级 | 根据贡献度设置创作者等级 |
| 创作者协议 | 创作者规则说明 |
| 创作者作品集 | 作品集展示 |
| 内容版本管理 | 管理内容版本历史 |
| 内容标签 | 给内容打标签 |
| 内容收藏 | 收藏常用内容 |
| 内容预览 | 预览内容渲染效果 |
| 内容搜索 | 搜索内容 |
| 审核流程 | 完整审核流程（多级审核） |
| 审核通知 | 审核结果通知 |

## 二、数据库设计

### 2.1 内容审核表 (content_reviews)

```sql
CREATE TABLE `content_reviews` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `content_type` TINYINT NOT NULL COMMENT '内容类型:1=题目,2=分类,3=知识点,4=试卷',
  `content_id` BIGINT UNSIGNED NOT NULL COMMENT '内容ID',
  `applicant_id` BIGINT UNSIGNED NOT NULL COMMENT '申请人ID',
  `status` TINYINT DEFAULT 0 COMMENT '状态:0=待审核,1=已通过,2=已拒绝,3=已撤回',
  `reviewer_id` BIGINT UNSIGNED DEFAULT NULL COMMENT '审核人ID',
  `review_remark` VARCHAR(500) DEFAULT '' COMMENT '审核备注',
  `reviewed_at` DATETIME DEFAULT NULL COMMENT '审核时间',
  `submitted_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '提交时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_content` (`content_type`, `content_id`),
  KEY `idx_status` (`status`),
  KEY `idx_applicant` (`applicant_id`),
  KEY `idx_reviewer` (`reviewer_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='内容审核表';
```

### 2.2 创作者等级表 (creator_levels)

```sql
CREATE TABLE `creator_levels` (

  `min_points` INT NOT NULL DEFAULT 0 COMMENT '最低积分',
  `max_points` INT NOT NULL DEFAULT 0 COMMENT '最高积分',
  `privileges` JSON DEFAULT NULL COMMENT '等级权益(JSON)',
  `description` VARCHAR(255) DEFAULT '' COMMENT '等级描述',
  `sort` INT DEFAULT 0 COMMENT '排序',
  `status` TINYINT DEFAULT 1 COMMENT '状态:0=禁用,1=启用',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
```

### 2.14 审核流程模板表 (review_workflows)

```sql
CREATE TABLE `review_workflows` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(100) NOT NULL COMMENT '流程名称',
  `content_type` TINYINT NOT NULL COMMENT '适用内容类型:1=题目,2=分类,3=知识点,4=试卷',
  `steps` JSON NOT NULL COMMENT '审核步骤(JSON数组)',
  `is_default` TINYINT(1) DEFAULT 0 COMMENT '是否默认流程',
  `status` TINYINT DEFAULT 1 COMMENT '状态:0=禁用,1=启用',
  `created_by` BIGINT UNSIGNED DEFAULT NULL COMMENT '创建人ID',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_content_type` (`content_type`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='审核流程模板表';
```

**steps JSON 结构**：
```json
[
  {
    "step": 1,
    "name": "初审",
    "role": "content_manager",
    "timeout_hours": 24,
    "escalation_role": "admin"
  },
  {
    "step": 2,
    "name": "复审",
    "role": "admin",
    "timeout_hours": 48,
    "escalation_role": ""
  }
]
```

| 字段 | 类型 | 说明 |
|------|------|------|
| step | int | 步骤序号 |
| name | string | 步骤名称 |
| role | string | 审核角色 |
| timeout_hours | int | 超时时间(小时) |
| escalation_role | string | 超时转交角色(为空则不转交) |

### 2.18 审核意见回复表 (review_replies)

```sql
CREATE TABLE `review_replies` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `review_id` BIGINT UNSIGNED NOT NULL COMMENT '审核步骤记录ID(review_step_logs.id)',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '回复人ID',
  `content` TEXT NOT NULL COMMENT '回复内容',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_review` (`review_id`),
  KEY `idx_user` (`user_id`),
  KEY `idx_created` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='审核意见回复表';
```

**说明**：支持申请人与审核人之间就审核意见进行多轮对话，直到达成一致。每条回复关联一条审核步骤记录(`review_step_logs.id`)，双方可多次回复，所有回复按 `created_at` 正序排列。

```go
// ApplyCreator 申请创作者
func (s *ApplyService) ApplyCreator(userID uint, reason string) error {
    // 检查是否已有申请
    existing, _ := s.applyRepo.FindPendingByUserAndRole(userID, "creator")
    if existing != nil {
        return errors.New("已有待审核的申请")
    }

    // 检查是否已是创作者
    user, _ := s.userRepo.FindByID(userID)
    for _, role := range user.Roles {
        if role.Code == "creator" {
            return errors.New("已经是创作者")
        }
    }

    application := &model.RoleApplication{
        UserID:   userID,
        RoleCode: "creator",
        Reason:   reason,
        Status:   0,
    }

    return s.applyRepo.Create(application)
}

// ApproveApplication 审批通过
func (s *ApplyService) ApproveApplication(applicationID, reviewerID uint) error {
    application, err := s.applyRepo.FindByID(applicationID)
    if err != nil {
        return err
    }

    if application.Status != 0 {
        return errors.New("申请已处理")
    }

    // 更新申请状态
    now := time.Now()
    application.Status = 1
    application.ReviewerID = &reviewerID
    application.ReviewedAt = &now
    s.applyRepo.Update(application)

    // 分配角色
    role, _ := s.userRepo.FindRoleByCode(application.RoleCode)
    s.userRepo.AssignRole(application.UserID, role.ID)

    // 初始化创作者积分
    s.pointSvc.InitPoints(application.UserID)

    // 发送通知
    s.sendNotification(application.UserID, "角色申请已通过",
        fmt.Sprintf("您的%s申请已通过", application.RoleCode))

    return nil
}
```

```go
// service/creator/level_service.go

type LevelService struct {
    levelRepo *creator.LevelRepo
    pointRepo *creator.PointRepo
    pointLogRepo *creator.PointLogRepo
}

// LevelInfo 等级信息
type LevelInfo struct {
    ID       uint   `json:"id"`
    Name     string `json:"name"`
    Level    int    `json:"level"`
    Icon     string `json:"icon"`
    MinPoints int   `json:"min_points"`
    MaxPoints int   `json:"max_points"`
    Privileges json.RawMessage `json:"privileges"`
}

// PointInfo 积分信息
type PointInfo struct {
    TotalPoints     int        `json:"total_points"`
    AvailablePoints int        `json:"available_points"`
    Level           *LevelInfo `json:"level"`
    NextLevel       *LevelInfo `json:"next_level"`
    Progress        float64    `json:"progress"` // 当前等级进度百分比
}

// InitPoints 初始化创作者积分
func (s *LevelService) InitPoints(userID uint) error {
    cp := &model.CreatorPoints{
        UserID:          userID,
        TotalPoints:     0,
        AvailablePoints: 0,
        LevelID:         nil,
    }
    return s.pointRepo.Create(cp)
}

// AddPoints 增加积分
func (s *LevelService) AddPoints(userID uint, points int, pointType string, refType string, refID uint, desc string) error {
    cp, err := s.pointRepo.FindByUserID(userID)
    if err != nil {
        return err
    }

    cp.TotalPoints += points
    cp.AvailablePoints += points

    // 检查是否需要升级
    newLevel, err := s.levelRepo.FindByPoints(cp.TotalPoints)
    if err == nil && (cp.LevelID == nil || *cp.LevelID < newLevel.ID) {
        cp.LevelID = &newLevel.ID
    }

    s.pointRepo.Update(cp)

    // 记录积分变动
    log := &model.CreatorPointLog{
        UserID:      userID,
        Points:      points,
        Type:        pointType,
        RefType:     refType,
        RefID:       refID,
        Description: desc,
    }
    return s.pointLogRepo.Create(log)
}

// GetPointInfo 获取积分信息
func (s *LevelService) GetPointInfo(userID uint) (*PointInfo, error) {
    cp, err := s.pointRepo.FindByUserID(userID)
    if err != nil {
        return nil, err
    }

    info := &PointInfo{
        TotalPoints:     cp.TotalPoints,
        AvailablePoints: cp.AvailablePoints,
    }

    // 获取当前等级
    if cp.LevelID != nil {
        level, _ := s.levelRepo.FindByID(*cp.LevelID)
        info.Level = &LevelInfo{
            ID:        level.ID,
            Name:      level.Name,
            Level:     level.Level,
            Icon:      level.Icon,
            MinPoints: level.MinPoints,
            MaxPoints: level.MaxPoints,
        }

        // 获取下一等级
        nextLevel, err := s.levelRepo.FindByLevelNumber(level.Level + 1)
        if err == nil {
            info.NextLevel = &LevelInfo{
                ID:        nextLevel.ID,
                Name:      nextLevel.Name,
                Level:     nextLevel.Level,
                Icon:      nextLevel.Icon,
                MinPoints: nextLevel.MinPoints,
                MaxPoints: nextLevel.MaxPoints,
            }
            // 计算进度
            range_ := float64(nextLevel.MinPoints - level.MinPoints)
            if range_ > 0 {
                info.Progress = float64(cp.TotalPoints-level.MinPoints) / range_ * 100
            }
        } else {
            info.Progress = 100 // 已是最高等级
        }
    } else {
        // 无等级，获取最低等级
        firstLevel, _ := s.levelRepo.FindByLevelNumber(1)
        if firstLevel != nil {
            info.NextLevel = &LevelInfo{
                ID:        firstLevel.ID,
                Name:      firstLevel.Name,
                Level:     firstLevel.Level,
                MinPoints: firstLevel.MinPoints,
                MaxPoints: firstLevel.MaxPoints,
            }
            if firstLevel.MinPoints > 0 {
                info.Progress = float64(cp.TotalPoints) / float64(firstLevel.MinPoints) * 100
            }
        }
    }

    return info, nil
}

// GetPointLogs 获取积分变动记录
func (s *LevelService) GetPointLogs(userID uint, page, pageSize int) ([]model.CreatorPointLog, int64, error) {
    return s.pointLogRepo.FindByUserID(userID, page, pageSize)
}
```

```go
// service/content/search_service.go

type SearchService struct {
    questionRepo *question.QuestionRepo
    categoryRepo *question.CategoryRepo
    searchLogRepo *content.SearchLogRepo
}

// SearchRequest 搜索请求
type SearchRequest struct {
    Keyword    string `form:"keyword" binding:"required"`
    ContentType *uint `form:"content_type"`
    CategoryID  *uint `form:"category_id"`
    Difficulty  *uint `form:"difficulty"`
    TagID       *uint `form:"tag_id"`
    Sort        string `form:"sort"` // relevance, time, popularity
    Page        int    `form:"page,default=1"`
    PageSize    int    `form:"page_size,default=20"`
}

// SearchResult 搜索结果
type SearchResult struct {
    Items      []interface{} `json:"items"`
    Total      int64         `json:"total"`
    Page       int           `json:"page"`
    PageSize   int           `json:"page_size"`
    Keyword    string        `json:"keyword"`
    Took       int64         `json:"took"` // 搜索耗时(毫秒)
}

// Search 综合搜索
func (s *SearchService) Search(req *SearchRequest, userID uint) (*SearchResult, error) {
    startTime := time.Now()

    query := s.questionRepo.Query()

    // 关键词搜索
    if req.Keyword != "" {
        query = query.Where("(title LIKE ? OR content LIKE ? OR analysis LIKE ?)",
            "%"+req.Keyword+"%", "%"+req.Keyword+"%", "%"+req.Keyword+"%")
    }

    // 内容类型筛选
    if req.ContentType != nil {
        query = query.Where("type = ?", *req.ContentType)
    }

    // 分类筛选
    if req.CategoryID != nil {
        query = query.Where("category_id = ?", *req.CategoryID)
    }

    // 难度筛选
    if req.Difficulty != nil {
        query = query.Where("difficulty = ?", *req.Difficulty)
    }

    // 数据权限：只能看到公开、自己创建的、所在班级的内容
    query = query.Where(`
        visibility = 1
        OR (visibility = 2 AND creator_id = ?)
        OR (visibility = 3 AND class_id IN (
            SELECT class_id FROM class_members WHERE user_id = ?
        ))
    `, userID, userID)

    // 只搜索已发布内容
    query = query.Where("status = ?", 1)

    // 排序
    switch req.Sort {
    case "time":
        query = query.Order("created_at DESC")
    case "popularity":
        query = query.Order("view_count DESC")
    default:
        // 按相关度排序(默认)
        query = query.Order("CASE WHEN title LIKE ? THEN 0 ELSE 1 END, created_at DESC",
            "%"+req.Keyword+"%")
    }

    total, _ := query.Count()
    items, err := query.Offset((req.Page - 1) * req.PageSize).Limit(req.PageSize).Find()
    if err != nil {
        return nil, err
    }

    took := time.Since(startTime).Milliseconds()

    // 异步记录搜索日志
    go s.searchLogRepo.Create(&model.ContentSearchLog{
        UserID:      &userID,
        Keyword:     req.Keyword,
        ContentType: req.ContentType,
        ResultCount: int(total),
    })

    return &SearchResult{
        Items:    items,
        Total:    total,
        Page:     req.Page,
        PageSize: req.PageSize,
        Keyword:  req.Keyword,
        Took:     took,
    }, nil
}

// GetSuggestions 获取搜索建议
func (s *SearchService) GetSuggestions(keyword string, userID uint) ([]string, error) {
    if len(keyword) < 2 {
        return nil, nil
    }

    // 从搜索历史中获取建议
    historySuggestions, _ := s.searchLogRepo.FindPopularKeywords(keyword, 5)

    // 从题目标题中获取建议
    titleSuggestions, _ := s.questionRepo.FindTitleSuggestions(keyword, 5)

    // 合并去重
    seen := make(map[string]bool)
    result := make([]string, 0, 10)
    for _, s := range append(historySuggestions, titleSuggestions...) {
        if !seen[s] {
            seen[s] = true
            result = append(result, s)
        }
    }
    return result, nil
}
```

```go
type ReviewService struct {

// SubmitReview 提交审核
func (s *ReviewService) SubmitReview(contentType uint, contentID uint, userID uint) error {
    // 检查是否已有进行中的审核
    existing, _ := s.instanceRepo.FindActiveByContent(contentType, contentID)
    if existing != nil {
        return errors.New("已有进行中的审核")
    }

    // 获取默认审核流程
    workflow, err := s.workflowRepo.FindDefaultByContentType(contentType)
    if err != nil {
        return errors.New("未配置审核流程")
    }

    steps := []model.WorkflowStep{}
    json.Unmarshal(workflow.Steps, &steps)
    if len(steps) == 0 {
        return errors.New("审核流程步骤为空")
    }

    // 创建审核实例
    instance := &model.ReviewInstance{
        WorkflowID:  workflow.ID,
        ContentType: contentType,
        ContentID:   contentID,
        ApplicantID: userID,
        CurrentStep: 1,
        Status:      0,
    }
    if err := s.instanceRepo.Create(instance); err != nil {
        return err
    }

    // 创建第一步审核记录
    firstStep := steps[0]
    stepLog := &model.ReviewStepLog{
        InstanceID: instance.ID,
        Step:       1,
        StepName:   firstStep.Name,
        Status:     0,
    }

    if firstStep.TimeoutHours > 0 {
        timeout := time.Now().Add(time.Duration(firstStep.TimeoutHours) * time.Hour)
        stepLog.TimeoutAt = &timeout
    }

    s.stepLogRepo.Create(stepLog)

    // 发送通知给审核人
    s.notifService.NotifyReviewer(instance, firstStep)

    // 发送通知给申请人
    s.notifService.NotifyApplicantSubmit(userID, instance)

    return nil
}

// ApproveReview 审核通过
func (s *ReviewService) ApproveReview(instanceID uint, reviewerID uint, remark string) error {
    instance, err := s.instanceRepo.FindByID(instanceID)
    if err != nil {
        return errors.New("审核记录不存在")
    }
    if instance.Status != 0 {
        return errors.New("审核已处理")
    }

    // 更新当前步骤
    stepLog, _ := s.stepLogRepo.FindActiveByInstanceAndStep(instanceID, instance.CurrentStep)
    if stepLog == nil {
        return errors.New("审核步骤不存在")
    }

    now := time.Now()
    stepLog.Status = 1
    stepLog.ReviewerID = &reviewerID
    stepLog.Remark = remark
    stepLog.ReviewedAt = &now
    s.stepLogRepo.Update(stepLog)

    // 检查是否还有下一步
    workflow, _ := s.workflowRepo.FindByID(instance.WorkflowID)
    steps := []model.WorkflowStep{}
    json.Unmarshal(workflow.Steps, &steps)

    if int(instance.CurrentStep) >= len(steps) {
        // 最后一步，审核完成
        instance.Status = 1
        instance.CompletedAt = &now
        s.instanceRepo.Update(instance)

        // 更新内容状态为已发布
        s.publishContent(instance.ContentType, instance.ContentID)

        // 保存版本
        s.versionService.SaveVersion(instance.ContentType, instance.ContentID,
            nil, "审核通过", reviewerID)

        // 通知申请人审核通过
        s.notifService.NotifyApplicantPass(instance.ApplicantID, instance)
    } else {
        // 进入下一步
        instance.CurrentStep++
        s.instanceRepo.Update(instance)

        nextStep := steps[instance.CurrentStep-1]
        nextStepLog := &model.ReviewStepLog{
            InstanceID: instanceID,
            Step:       instance.CurrentStep,
            StepName:   nextStep.Name,
            Status:     0,
        }
        if nextStep.TimeoutHours > 0 {
            timeout := time.Now().Add(time.Duration(nextStep.TimeoutHours) * time.Hour)
            nextStepLog.TimeoutAt = &timeout
        }
        s.stepLogRepo.Create(nextStepLog)

        // 通知下一步审核人
        s.notifService.NotifyReviewer(instance, nextStep)
    }

    return nil
}

// RejectReview 审核拒绝

        return errors.New("审核已处理")

    // 更新当前步骤
    stepLog, _ := s.stepLogRepo.FindActiveByInstanceAndStep(instanceID, instance.CurrentStep)

    stepLog.Status = 2
    stepLog.ReviewerID = &reviewerID
    stepLog.Remark = remark
    stepLog.ReviewedAt = &now
    s.stepLogRepo.Update(stepLog)

    // 更新审核实例状态
    instance.Status = 2
    instance.CompletedAt = &now
    s.instanceRepo.Update(instance)

    // 通知申请人审核拒绝
    s.notifService.NotifyApplicantReject(instance.ApplicantID, instance, remark)

    return nil
}
```

```go
func (s *NotificationService) NotifyTimeout(instanceID uint) {

    // 短信/邮件提醒审核人
    reviewer, _ := s.userRepo.FindByID(*stepLog.ReviewerID)

        smsContent := fmt.Sprintf("【题小助】您有一条%s审核已超时，请尽快登录平台处理。", contentTypeName)
        s.smsService.Send(reviewer.Phone, smsContent)

        s.emailService.Send(reviewer.Email, emailSubject, emailBody)
    }

    // 同时通知管理员（站内+邮件）
    admins, _ := s.userRepo.FindByRole("admin")
    for _, admin := range admins {
        notif := &model.ReviewNotification{
            UserID:     admin.ID,
            InstanceID: instanceID,
            Type:       "timeout",
            Title:      "审核超时预警",
            Content:    fmt.Sprintf("有一条%s审核已超时，审核人ID：%d", contentTypeName, *stepLog.ReviewerID),

        // 邮件通知管理员

        emailBody := fmt.Sprintf("有一条%s审核已超时（内容ID: %d，审核人ID: %d），请及时跟进。", contentTypeName, instance.ContentID, *stepLog.ReviewerID)
        s.emailService.Send(admin.Email, emailSubject, emailBody)
    }
}
```

### 4.13 审核超时定时任务

```go
// service/review/timeout_task.go

type ReviewTimeoutTask struct {
    instanceRepo  *review.InstanceRepo
    stepLogRepo   *review.StepLogRepo
    notifService  *review.NotificationService
    workflowRepo  *review.WorkflowRepo
}

// CheckTimeouts 检查超时审核(每小时执行一次)
func (t *ReviewTimeoutTask) CheckTimeouts() {
    now := time.Now()

    // 查找所有超时的步骤
    timeoutSteps, err := t.stepLogRepo.FindTimeoutBefore(now)
    if err != nil {
        return
    }

    for _, step := range timeoutSteps {
        if step.Status != 0 { // 只处理待处理的
            continue
        }

        // 计算超时时长
        timeoutDuration := now.Sub(*step.TimeoutAt)

        // 第一阶段：超时即提醒(短信+邮件+站内)
        t.notifService.NotifyTimeout(step.InstanceID)

        // 第二阶段：超时24小时后再次提醒并通知管理员
        if timeoutDuration > 24*time.Hour {
            t.notifService.NotifyTimeoutEscalation(step.InstanceID, "超时24小时，请管理员介入")
        }

        // 第三阶段：超时48小时自动转交上级审核人
        if timeoutDuration > 48*time.Hour {
            t.escalateReviewer(step)
        }
    }
}

// escalateReviewer 超时自动转交审核人
func (t *ReviewTimeoutTask) escalateReviewer(step *model.ReviewStepLog) {
    instance, err := t.instanceRepo.FindByID(step.InstanceID)
    if err != nil {
        return
    }

    workflow, _ := t.workflowRepo.FindByID(instance.WorkflowID)
    steps := []model.WorkflowStep{}
    json.Unmarshal(workflow.Steps, &steps)

    // 当前步骤有上级角色则转交
    currentStepConfig := steps[step.Step-1]
    if currentStepConfig.EscalationRole != "" {
        escalationReviewers, _ := t.userRepo.FindByRole(currentStepConfig.EscalationRole)
        if len(escalationReviewers) > 0 {
            newReviewer := escalationReviewers[0]
            step.ReviewerID = &newReviewer.ID
            t.stepLogRepo.Update(step)
            t.notifService.NotifyReassign(newReviewer.ID, step.InstanceID)
        }
    }
}
```

## 五、Redis 缓存设计

### 5.1 缓存 Key 设计

| Key 模式 | 说明 | TTL | 数据类型 |
|----------|------|-----|----------|
| `creator:info:{userId}` | 创作者基本信息(等级、积分) | 30min | Hash |
| `creator:level:list` | 创作者等级列表 | 24h | String(JSON) |
| `content:detail:{type}:{id}` | 内容详情缓存 | 15min | String(JSON) |
| `content:preview:{type}:{id}` | 内容预览缓存 | 10min | String(JSON) |
| `review:status:{instanceId}` | 审核实例状态 | 5min | String |
| `review:pending:count:{userId}` | 用户待审核数量 | 5min | String |
| `search:hot:keywords` | 热搜关键词(Top 50) | 1h | ZSet |
| `search:suggestions:{prefix}` | 搜索建议缓存 | 10min | String(JSON) |
| `tag:list` | 标签列表 | 1h | String(JSON) |
| `share:token:{token}` | 分享链接信息 | 按分享设置 | Hash |
| `notification:unread:{userId}` | 未读通知数量 | 2min | String |

### 5.2 缓存更新策略

**写入策略（Cache-Aside）**：
- 写操作先更新数据库，再删除缓存
- 读操作先查缓存，未命中则查库并回填缓存

**缓存穿透防护**：
- 空值缓存：查询不存在的内容时缓存空值，TTL = 60s
- 布隆过滤器：对内容 ID 做前置校验

**缓存雪崩防护**：
- TTL 加随机偏移量(±10%)，避免同时过期
- 热点数据预加载：系统启动时预热等级列表、标签列表

**缓存一致性**：
- 内容编辑/删除后主动删除对应缓存 Key
- 审核状态变更后删除 `review:status:{instanceId}` 和 `content:detail:{type}:{id}`
- 创作者等级变更后删除 `creator:info:{userId}`

```go
// service/cache/creator_cache.go

type CreatorCache struct {
    rdb *redis.Client
}

const (
    CreatorInfoKey    = "creator:info:%d"
    CreatorLevelList  = "creator:level:list"
    ContentDetailKey  = "content:detail:%d:%d"
    ReviewStatusKey   = "review:status:%d"
    ShareTokenKey     = "share:token:%s"
)

// GetCreatorInfo 获取创作者信息缓存
func (c *CreatorCache) GetCreatorInfo(userID uint) (*dto.CreatorInfo, error) {
    key := fmt.Sprintf(CreatorInfoKey, userID)
    data, err := c.rdb.Get(ctx, key).Bytes()
    if err == redis.Nil {
        return nil, nil // 缓存未命中
    }
    if err != nil {
        return nil, err
    }
    var info dto.CreatorInfo
    json.Unmarshal(data, &info)
    return &info, nil
}

// SetCreatorInfo 设置创作者信息缓存
func (c *CreatorCache) SetCreatorInfo(userID uint, info *dto.CreatorInfo) error {
    key := fmt.Sprintf(CreatorInfoKey, userID)
    data, _ := json.Marshal(info)
    ttl := 30*time.Minute + time.Duration(rand.Intn(300))*time.Second // 随机偏移
    return c.rdb.Set(ctx, key, data, ttl).Err()
}

// DeleteCreatorInfo 删除创作者信息缓存
func (c *CreatorCache) DeleteCreatorInfo(userID uint) error {
    key := fmt.Sprintf(CreatorInfoKey, userID)
    return c.rdb.Del(ctx, key).Err()
}

// GetShareInfo 获取分享链接缓存
func (c *CreatorCache) GetShareInfo(token string) (*dto.ShareInfo, error) {
    key := fmt.Sprintf(ShareTokenKey, token)
    data, err := c.rdb.HGetAll(ctx, key).Result()
    if err != nil || len(data) == 0 {
        return nil, nil
    }
    // 解析分享信息...
    return &dto.ShareInfo{
        ContentType: data["content_type"],
        ContentID:   data["content_id"],
    }, nil
}

// InvalidateContentCache 内容变更时清除相关缓存
func (c *CreatorCache) InvalidateContentCache(contentType, contentID uint) {
    key := fmt.Sprintf(ContentDetailKey, contentType, contentID)
    c.rdb.Del(ctx, key)
}
```
