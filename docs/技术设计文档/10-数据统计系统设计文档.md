## 二、数据库设计

### 2.1 统计快照表 (statistics_snapshots)

```sql
CREATE TABLE `statistics_snapshots` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `date` DATE NOT NULL COMMENT '统计日期',
  `type` VARCHAR(20) NOT NULL COMMENT '统计类型:user/question/exam/class',
  `total_count` BIGINT DEFAULT 0 COMMENT '总数',
  `new_count` BIGINT DEFAULT 0 COMMENT '新增数',
  `active_count` BIGINT DEFAULT 0 COMMENT '活跃数',
  `data` JSON DEFAULT NULL COMMENT '扩展数据',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_date_type` (`date`, `type`),
  KEY `idx_date` (`date`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='统计快照表';
```

### 2.2 用户留存表 (user_retention)

```sql
CREATE TABLE `user_retention` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `register_date` DATE NOT NULL COMMENT '注册日期',
  `retention_day` INT NOT NULL COMMENT '留存天数(1/2/3/7/14/30)',
  `register_count` INT DEFAULT 0 COMMENT '注册用户数',
  `retention_count` INT DEFAULT 0 COMMENT '留存用户数',
  `retention_rate` DECIMAL(5,2) DEFAULT 0 COMMENT '留存率',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_register_day` (`register_date`, `retention_day`),
  KEY `idx_register_date` (`register_date`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户留存表';
```

### 2.3 用户流失表 (user_churn)

```sql
CREATE TABLE `user_churn` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `stat_date` DATE NOT NULL COMMENT '统计日期',
  `total_users` INT DEFAULT 0 COMMENT '总用户数',
  `churned_users` INT DEFAULT 0 COMMENT '流失用户数',
  `churn_rate` DECIMAL(5,2) DEFAULT 0 COMMENT '流失率',
  `churn_7d` INT DEFAULT 0 COMMENT '7天未活跃用户数',
  `churn_14d` INT DEFAULT 0 COMMENT '14天未活跃用户数',
  `churn_30d` INT DEFAULT 0 COMMENT '30天未活跃用户数',

```

### 2.4 用户行为事件表 (user_events)

```sql
CREATE TABLE `user_events` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `user_id` BIGINT UNSIGNED NOT NULL,
  `event_type` VARCHAR(50) NOT NULL COMMENT '事件类型:login/practice/exam/view_page/search/comment/collect',
  `event_action` VARCHAR(50) DEFAULT '' COMMENT '事件动作:start/complete/abandon/click/submit',
  `event_label` VARCHAR(200) DEFAULT '' COMMENT '事件标签',
  `event_value` DECIMAL(10,2) DEFAULT NULL COMMENT '事件值',
  `page_url` VARCHAR(500) DEFAULT '' COMMENT '页面URL',
  `referrer_url` VARCHAR(500) DEFAULT '' COMMENT '来源URL',
  `device_type` VARCHAR(20) DEFAULT '' COMMENT '设备类型:pc/mobile/miniapp',
  `platform` VARCHAR(20) DEFAULT '' COMMENT '平台:web/wechat/ios/android',
  `ip` VARCHAR(50) DEFAULT '' COMMENT 'IP地址',
  `extra_data` JSON DEFAULT NULL COMMENT '扩展数据',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_event_type` (`event_type`),
  KEY `idx_created_at` (`created_at`),
  KEY `idx_user_event` (`user_id`, `event_type`, `created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户行为事件表';
```

### 3.4 题目难度分析

#### 难度系数计算公式

$$P = \frac{R}{N}$$

其中：
- **P** = 难度系数（0~1之间）
- **R** = 答对该题的人数
- **N** = 总答题人数

#### 难度等级划分

| 难度系数范围 | 难度等级 | 说明 |
|-------------|---------|------|
| P >= 0.8 | 简单 | 大多数人能答对 |
| 0.6 <= P < 0.8 | 中等偏易 | |
| 0.4 <= P < 0.6 | 中等 | |
| 0.2 <= P < 0.4 | 中等偏难 | |
| P < 0.2 | 困难 | 大多数人答错 |

#### 实现代码

```go
// CalculateDifficulty 计算题目难度系数
func (s *StatisticsService) CalculateDifficulty(questionID uint) (*dto.DifficultyInfo, error) {
    var totalCount int64
    var correctCount int64

    // 总答题人数
    s.db.Model(&model.PracticeRecord{}).
        Where("question_id = ?", questionID).
       Count(&totalCount)

    // 答对人数
    s.db.Model(&model.PracticeRecord{}).
        Where("question_id = ? AND is_correct = ?", questionID, true).
        Count(&correctCount)

    if totalCount == 0 {
        return &dto.DifficultyInfo{
            QuestionID: questionID,
            P:          0,
            Level:      "未作答",
        }, nil
    }

    p := float64(correctCount) / float64(totalCount)

    level := "中等"
    switch {
    case p >= 0.8:
        level = "简单"
    case p >= 0.6:
        level = "中等偏易"
    case p >= 0.4:
        level = "中等"
    case p >= 0.2:
        level = "中等偏难"
    default:
        level = "困难"
    }

    return &dto.DifficultyInfo{
        QuestionID:   questionID,
        P:            p,
        Level:        level,
        TotalCount:   totalCount,
        CorrectCount: correctCount,
    }, nil
}

// GetDifficultyDistribution 获取题目难度分布
func (s *StatisticsService) GetDifficultyDistribution(categoryID uint) (*dto.DifficultyDistribution, error) {
    var results []struct {
        QuestionID  uint
        TotalCount  int64
        CorrectCount int64
    }

    query := s.db.Raw(`
        SELECT
            pr.question_id,
            COUNT(*) as total_count,
            SUM(CASE WHEN pr.is_correct = 1 THEN 1 ELSE 0 END) as correct_count
        FROM practice_records pr
        JOIN questions q ON q.id = pr.question_id
        WHERE q.deleted_at IS NULL
        GROUP BY pr.question_id
        HAVING total_count >= 10
    `)

    if categoryID > 0 {
        query = s.db.Raw(`
            SELECT
                pr.question_id,
                COUNT(*) as total_count,
                SUM(CASE WHEN pr.is_correct = 1 THEN 1 ELSE 0 END) as correct_count
            FROM practice_records pr
            JOIN questions q ON q.id = pr.question_id
            WHERE q.deleted_at IS NULL AND q.category_id = ?
            GROUP BY pr.question_id
            HAVING total_count >= 10
        `, categoryID)
    }

    query.Scan(&results)

    distribution := &dto.DifficultyDistribution{
        Easy:   0,
        Medium: 0,
        Hard:   0,
    }

    for _, r := range results {
        p := float64(r.CorrectCount) / float64(r.TotalCount)
        switch {
        case p >= 0.8:
            distribution.Easy++
        case p >= 0.4:
            distribution.Medium++
        default:
            distribution.Hard++
        }
    }

    distribution.Total = distribution.Easy + distribution.Medium + distribution.Hard
    return distribution, nil
}
```

### 3.5 题目区分度分析

#### 区分度计算公式

$$D = P_H - P_L$$

其中：
- **D** = 区分度（-1~1之间）
- **P_H** = 高分组（总分排名前27%）答对该题的正确率
- **P_L** = 低分组（总分排名后27%）答对该题的正确率

#### 区分度等级划分

| 区分度范围 | 等级 | 说明 |
|-----------|------|------|
| D >= 0.4 | 优秀 | 区分能力很强，题目质量好 |
| 0.3 <= D < 0.4 | 良好 | 区分能力较好 |
| 0.2 <= D < 0.3 | 一般 | 区分能力一般，可考虑修改 |
| D < 0.2 | 较差 | 区分能力弱，需要改进 |
| D < 0 | 反区分 | 题目存在问题，低分组反而答对更多 |

#### 实现代码

```go
// CalculateDiscrimination 计算题目区分度
func (s *StatisticsService) CalculateDiscrimination(questionID uint) (*dto.DiscriminationInfo, error) {
    // 获取所有答过该题的用户的总分（以该题所属考试或练习为基准）
    // 这里以最近一次包含该题的考试为例
    var examID uint
    s.db.Raw(`
        SELECT exam_id FROM exam_questions
        WHERE question_id = ? LIMIT 1
    `, questionID).Scan(&examID)

    if examID == 0 {
        // 如果不在考试中，使用练习正确率作为基准
        return s.calculateDiscriminationByPractice(questionID)
    }

    // 获取所有参加该考试的用户及其成绩
    var userScores []struct {
        UserID uint
        Score  float64
    }
    s.db.Raw(`
        SELECT user_id, score FROM exam_records
        WHERE exam_id = ? AND status = 2
        ORDER BY score DESC
    `, examID).Scan(&userScores)

    if len(userScores) < 10 {
        return &dto.DiscriminationInfo{
            QuestionID:    questionID,
            D:             0,
            Level:         "样本不足",
            SampleSize:    len(userScores),
        }, nil
    }

    // 按前27%和后27%分组
    groupSize := int(float64(len(userScores)) * 0.27)
    highGroup := userScores[:groupSize]
    lowGroup := userScores[len(userScores)-groupSize:]

    highGroupIDs := make([]uint, len(highGroup))
    lowGroupIDs := make([]uint, len(lowGroup))
    for i, u := range highGroup {
        highGroupIDs[i] = u.UserID
    }
    for i, u := range lowGroup {
        lowGroupIDs[i] = u.UserID
    }

    // 高分组正确率
    var highTotal, highCorrect int64
    s.db.Model(&model.PracticeRecord{}).
        Where("question_id = ? AND user_id IN ?", questionID, highGroupIDs).
        Count(&highTotal)
    s.db.Model(&model.PracticeRecord{}).
        Where("question_id = ? AND user_id IN ? AND is_correct = ?", questionID, highGroupIDs, true).
        Count(&highCorrect)

    // 低分组正确率
    var lowTotal, lowCorrect int64
    s.db.Model(&model.PracticeRecord{}).
        Where("question_id = ? AND user_id IN ?", questionID, lowGroupIDs).
        Count(&lowTotal)
    s.db.Model(&model.PracticeRecord{}).
        Where("question_id = ? AND user_id IN ? AND is_correct = ?", questionID, lowGroupIDs, true).
        Count(&lowCorrect)

    var pH, pL float64
    if highTotal > 0 {
        pH = float64(highCorrect) / float64(highTotal)
    }
    if lowTotal > 0 {
        pL = float64(lowCorrect) / float64(lowTotal)
    }

    d := pH - pL

    level := "一般"
    switch {
    case d >= 0.4:
        level = "优秀"
    case d >= 0.3:
        level = "良好"
    case d >= 0.2:
        level = "一般"
    case d >= 0:
        level = "较差"
    default:
        level = "反区分"
    }

    return &dto.DiscriminationInfo{
        QuestionID:   questionID,
        D:            d,
        Level:        level,
        PH:           pH,
        PL:           pL,
        HighGroupSize: groupSize,
        LowGroupSize:  groupSize,
        SampleSize:    len(userScores),
    }, nil
}
```

### 3.6 成绩预测

基于用户历史成绩数据，使用线性回归进行趋势预测。

#### 预测算法

采用简单线性回归模型：$y = a + bx$

其中：
- $y$ = 预测成绩
- $x$ = 考试序号
- $a$ = 截距
- $b$ = 斜率（成绩变化趋势）

#### 实现代码

```go
// PredictScore 成绩预测
func (s *StatisticsService) PredictScore(userID uint) (*dto.ScorePrediction, error) {
    var records []struct {
        ExamID    uint
        ExamName  string
        Score     float64
        ExamDate  time.Time
    }

    s.db.Raw(`
        SELECT er.exam_id, e.name as exam_name, er.score, e.created_at as exam_date
        FROM exam_records er
        JOIN exams e ON e.id = er.exam_id
        WHERE er.user_id = ? AND er.status = 2
        ORDER BY e.created_at ASC
    `, userID).Scan(&records)

    if len(records) < 3 {
        return &dto.ScorePrediction{
            HasEnoughData: false,
            Message:       "至少需要3次考试记录才能进行预测",
        }, nil
    }

    // 简单线性回归
    n := float64(len(records))
    var sumX, sumY, sumXY, sumX2 float64
    for i, r := range records {
        x := float64(i + 1)
        y := r.Score
        sumX += x
        sumY += y
        sumXY += x * y
        sumX2 += x * x
    }

    b := (n*sumXY - sumX*sumY) / (n*sumX2 - sumX*sumX)
    a := (sumY - b*sumX) / n

    // 预测下一次考试成绩
    nextX := float64(len(records) + 1)
    predicted := a + b*nextX

    // 限制预测值在0-100之间
    if predicted > 100 {
        predicted = 100
    }
    if predicted < 0 {
        predicted = 0
    }

    // 趋势判断
    trend := "平稳"
    if b > 0.5 {
        trend = "上升"
    } else if b < -0.5 {
        trend = "下降"
    }

    // 构建历史数据
    history := make([]dto.ScoreHistoryItem, len(records))
    for i, r := range records {
        history[i] = dto.ScoreHistoryItem{
            ExamID:   r.ExamID,
            ExamName: r.ExamName,
            Score:    r.Score,
            Date:     r.ExamDate.Format("2006-01-02"),
        }
    }

    // 计算拟合线上的值（用于图表展示）
    fittedLine := make([]dto.FittedPoint, len(records)+1)
    for i := 0; i <= len(records); i++ {
        x := float64(i + 1)
        fittedLine[i] = dto.FittedPoint{
            X:     i + 1,
            Value: a + b*x,
        }
    }

    return &dto.ScorePrediction{
        HasEnoughData: true,
        Trend:         trend,
        TrendSlope:    b,
        PredictedNext: predicted,
        Confidence:    calculateConfidence(records, a, b),
        History:       history,
        FittedLine:    fittedLine,
    }, nil
}

// calculateConfidence 计算预测置信度（基于R²）
func calculateConfidence(records []struct{ Score float64 }, a, b float64) float64 {
    n := float64(len(records))
    var sumY float64
    for _, r := range records {
        sumY += r.Score
    }
    meanY := sumY / n

    var ssRes, ssTot float64
    for i, r := range records {
        x := float64(i + 1)
        predicted := a + b*x
        ssRes += (r.Score - predicted) * (r.Score - predicted)
        ssTot += (r.Score - meanY) * (r.Score - meanY)
    }

    if ssTot == 0 {
        return 0
    }
    r2 := 1 - ssRes/ssTot
    if r2 < 0 {
        r2 = 0
    }
    return r2
}
```

```go
// CheckScoreAlert 检查成绩预警（考试提交后调用）
func (s *StatisticsService) CheckScoreAlert(userID uint, examID uint, score float64) error {
    var configs []model.ScoreAlertConfig
    s.db.Where("status = 1 AND (user_id = ? OR user_id IS NULL)", userID).Find(&configs)

    for _, config := range configs {
        // 检查冷却时间
        if config.LastTriggeredAt != nil {
            cooldown := time.Duration(config.CooldownHours) * time.Hour
            if time.Since(*config.LastTriggeredAt) < cooldown {
                continue
            }
        }

        triggered := false
        var message string

        switch config.AlertType {
        case "score_below":
            if score < config.Threshold {
                triggered = true
                message = fmt.Sprintf("您在最近考试中的成绩 %.1f 分低于预警阈值 %.1f 分", score, config.Threshold)
            }
        case "drop_rate":
            // 计算与上次成绩的下降幅度
            var lastScore float64
            s.db.Model(&model.ExamRecord{}).
                Where("user_id = ? AND exam_id != ? AND status = 2", userID, examID).
                Order("created_at DESC").
                Limit(1).
                Select("score").Scan(&lastScore)

            if lastScore > 0 {
                dropRate := (lastScore - score) / lastScore * 100
                if dropRate > config.Threshold {
                    triggered = true
                    message = fmt.Sprintf("您的成绩较上次下降了 %.1f%%，超过预警阈值 %.1f%%", dropRate, config.Threshold)
                }
            }
        }

        if triggered {
            // 更新触发时间
            s.db.Model(&config).Update("last_triggered_at", time.Now())

            // 发送通知
            notification := &model.Notification{
                UserID:  userID,
                Type:    "score_alert",
                Title:   "成绩预警通知",
                Content: message,
            }
            s.db.Create(notification)

            // 如果配置了邮件通知
            if config.NotifyType == "email" {
                go s.sendScoreAlertEmail(userID, message)
            }
        }
    }

    return nil
}
```

```go
// CalculateRetention 计算留存率(定时任务调用)
func (s *StatisticsService) CalculateRetention() error {
    retentionDays := []int{1, 2, 3, 7, 14, 30}

    for _, day := range retentionDays {
        statDate := time.Now().AddDate(0, 0, -day)

        // 统计当天注册用户数
        var registerCount int64
        s.db.Model(&model.User{}).
            Where("DATE(created_at) = ?", statDate.Format("2006-01-02")).
            Count(&registerCount)

        if registerCount == 0 {
            continue
        }

        // 统计留存用户数(注册后第N天仍有活跃行为)
        targetDate := statDate.AddDate(0, 0, day)
        var retentionCount int64
        s.db.Raw(`
            SELECT COUNT(DISTINCT u.id)
            FROM users u
            INNER JOIN user_events ue ON u.id = ue.user_id
            WHERE DATE(u.created_at) = ?
              AND ue.event_type = 'login'
              AND DATE(ue.created_at) = ?
        `, statDate.Format("2006-01-02"), targetDate.Format("2006-01-02")).Scan(&retentionCount)

        retentionRate := float64(retentionCount) / float64(registerCount) * 100

        // 写入留存表
        retention := &model.UserRetention{
            RegisterDate:  statDate,
            RetentionDay:  day,
            RegisterCount: int(registerCount),
            RetentionCount: int(retentionCount),
            RetentionRate:  retentionRate,
        }
        s.db.Where("register_date = ? AND retention_day = ?", statDate, day).
            Assign(retention).FirstOrCreate(&model.UserRetention{})
    }

    return nil
}

// CalculateChurn 计算流失统计(定时任务调用)
func (s *StatisticsService) CalculateChurn() error {
    today := time.Now()

    // 总用户数
    var totalUsers int64
    s.db.Model(&model.User{}).Where("deleted_at IS NULL").Count(&totalUsers)

    // 7天未活跃用户数
    var churn7d int64
    s.db.Raw(`
        SELECT COUNT(*)
        FROM users u
        WHERE u.deleted_at IS NULL
          AND u.id NOT IN (
              SELECT DISTINCT user_id FROM user_events
              WHERE event_type = 'login' AND created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)
          )
    `).Scan(&churn7d)

    // 14天未活跃用户数
    var churn14d int64
    s.db.Raw(`
        SELECT COUNT(*)
        FROM users u
        WHERE u.deleted_at IS NULL
          AND u.id NOT IN (
              SELECT DISTINCT user_id FROM user_events
              WHERE event_type = 'login' AND created_at >= DATE_SUB(NOW(), INTERVAL 14 DAY)
          )
    `).Scan(&churn14d)

    // 30天未活跃用户数(视为流失)
    var churn30d int64
    s.db.Raw(`
        SELECT COUNT(*)
        FROM users u
        WHERE u.deleted_at IS NULL
          AND u.id NOT IN (
              SELECT DISTINCT user_id FROM user_events
              WHERE event_type = 'login' AND created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
          )
    `).Scan(&churn30d)

    // 回流用户数(之前30天未活跃，今天又活跃了)
    var reactivatedUsers int64
    s.db.Raw(`
        SELECT COUNT(DISTINCT ue.user_id)
        FROM user_events ue
        WHERE ue.event_type = 'login'
          AND DATE(ue.created_at) = CURDATE()
          AND ue.user_id NOT IN (
              SELECT DISTINCT user_id FROM user_events
              WHERE event_type = 'login'
                AND created_at BETWEEN DATE_SUB(CURDATE(), INTERVAL 30 DAY) AND DATE_SUB(CURDATE(), INTERVAL 1 DAY)
          )
    `).Scan(&reactivatedUsers)

    churnRate := float64(churn30d) / float64(totalUsers) * 100

    churn := &model.UserChurn{
        StatDate:         today,
        TotalUsers:       int(totalUsers),
        ChurnedUsers:     int(churn30d),
        ChurnRate:        churnRate,
        Churn7d:          int(churn7d),
        Churn14d:         int(churn14d),
        Churn30d:         int(churn30d),
        ReactivatedUsers: int(reactivatedUsers),
    }

    return s.db.Where("stat_date = ?", today.Format("2006-01-02")).
        Assign(churn).FirstOrCreate(&model.UserChurn{})
}

// SyncSegmentMembers 同步分群成员
func (s *StatisticsService) SyncSegmentMembers(segmentID uint) error {
    var segment model.UserSegment
    if err := s.db.First(&segment, segmentID).Error; err != nil {
        return err
    }

    if segment.SegmentType != 2 || segment.Rules == nil {
        return nil
    }

    var rules dto.SegmentRules
    json.Unmarshal(segment.Rules, &rules)

    // 构建动态查询
    query := s.db.Model(&model.User{}).Where("deleted_at IS NULL")

    if rules.MinLoginDays > 0 {
        query = query.Where(`
            (SELECT COUNT(DISTINCT DATE(created_at)) FROM user_events
             WHERE user_id = users.id AND event_type = 'login'
             AND created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)) >= ?
        `, rules.MinLoginDays)
    }

    if rules.MinPracticeCount > 0 {
        query = query.Where(`
            (SELECT COUNT(*) FROM practice_sessions
             WHERE user_id = users.id) >= ?
        `, rules.MinPracticeCount)
    }

    if rules.RegisterDaysAgo > 0 {
        query = query.Where("created_at <= ?", time.Now().AddDate(0, 0, -rules.RegisterDaysAgo))
    }

    if rules.HasRole != "" {
        query = query.Joins("INNER JOIN user_roles ON user_roles.user_id = users.id").
            Joins("INNER JOIN roles ON roles.id = user_roles.role_id").
            Where("roles.code = ?", rules.HasRole)
    }

    var userIDs []uint
    query.Pluck("users.id", &userIDs)

    // 清空旧成员
    s.db.Where("segment_id = ?", segmentID).Delete(&model.UserSegmentMember{})

    // 批量插入新成员
    if len(userIDs) > 0 {
        members := make([]model.UserSegmentMember, len(userIDs))
        for i, uid := range userIDs {
            members[i] = model.UserSegmentMember{
                SegmentID: segmentID,
                UserID:    uid,
            }
        }
        s.db.CreateInBatches(members, 100)
    }

    // 更新用户数量
    s.db.Model(&segment).Update("user_count", len(userIDs))

    return nil
}

        SELECT page_url as path, COUNT(DISTINCT session_id) as sessions
        FROM user_page_views
        WHERE created_at BETWEEN ? AND ?
          AND session_id IN (
              SELECT session_id FROM user_page_views
              WHERE created_at BETWEEN ? AND ?
              GROUP BY session_id HAVING MIN(id)
          )
        GROUP BY page_url
        ORDER BY sessions DESC
        LIMIT ?
    `, startDate, endDate, startDate, endDate, limit).Scan(&analysis.EntryPages)

    s.db.Raw(`
        SELECT page_url as path, COUNT(DISTINCT session_id) as sessions
        FROM user_page_views
        WHERE created_at BETWEEN ? AND ?
          AND session_id IN (
              SELECT session_id FROM user_page_views
              WHERE created_at BETWEEN ? AND ?
              GROUP BY session_id HAVING MAX(id)
          )
        GROUP BY page_url
        ORDER BY sessions DESC
        LIMIT ?
    `, startDate, endDate, startDate, endDate, limit).Scan(&analysis.ExitPages)

    // 平均访问深度(每个会话的平均页面数)
    s.db.Raw(`
        SELECT AVG(page_count) FROM (
            SELECT session_id, COUNT(*) as page_count
            FROM user_page_views
            WHERE created_at BETWEEN ? AND ?
            GROUP BY session_id
        ) t
    `, startDate, endDate).Scan(&analysis.AvgPageDepth)

    // 平均停留时长
    s.db.Raw(`
        SELECT AVG(stay_duration) FROM user_page_views
        WHERE created_at BETWEEN ? AND ? AND stay_duration > 0 AND stay_duration < 3600

// GetFunnelStats 获取漏斗转化统计
func (s *StatisticsService) GetFunnelStats(funnelID uint, startDate, endDate time.Time) (*dto.FunnelStats, error) {
    var funnel model.ConversionFunnel
    if err := s.db.First(&funnel, funnelID).Error; err != nil {
        return nil, err
    }

    var steps []dto.FunnelStepConfig
    json.Unmarshal(funnel.Steps, &steps)

    var stats dto.FunnelStats
    stats.FunnelID = funnel.ID
    stats.FunnelName = funnel.Name
    stats.StartDate = startDate
    stats.EndDate = endDate

    previousCount := 0
    for i, step := range steps {
        var count int64
        s.db.Raw(`
            SELECT COUNT(DISTINCT user_id)
            FROM user_events
            WHERE event_type = ? AND event_action = ?
              AND created_at BETWEEN ? AND ?
        `, step.EventType, step.EventAction, startDate, endDate).Scan(&count)

        stepStat := dto.FunnelStepStat{
            StepIndex: i,
            StepName:  step.Name,
            Count:     int(count),
        }

        if i == 0 {
            stepStat.ConversionRate = 100.0
            previousCount = int(count)
        } else if previousCount > 0 {
            stepStat.ConversionRate = float64(count) / float64(previousCount) * 100
            previousCount = int(count)
        }

        stepStat.TotalRate = 0
        if stats.Steps != nil && len(stats.Steps) > 0 && stats.Steps[0].Count > 0 {
            stepStat.TotalRate = float64(count) / float64(stats.Steps[0].Count) * 100
        }

        stats.Steps = append(stats.Steps, stepStat)
    }

    return &stats, nil
}

// CheckAlerts 检查预警(定时任务调用)
func (s *StatisticsService) CheckAlerts() error {
    var rules []model.AlertRule
    s.db.Where("status = 1 AND deleted_at IS NULL").Find(&rules)

    for _, rule := range rules {
        var currentValue float64

        switch rule.Metric {
        case "user_count":
            s.db.Model(&model.User{}).Where("deleted_at IS NULL").Count((*int64)(&currentValue))
        case "login_count":
            var count int64
            s.db.Model(&model.UserEvent{}).
                Where("event_type = 'login' AND DATE(created_at) = CURDATE()").
                Count(&count)
            currentValue = float64(count)
        case "practice_count":
            var count int64
            s.db.Model(&model.PracticeSession{}).
                Where("DATE(created_at) = CURDATE()").
                Count(&count)
            currentValue = float64(count)
        case "exam_count":
            var count int64
            s.db.Model(&model.ExamRecord{}).
                Where("DATE(created_at) = CURDATE()").
                Count(&count)
            currentValue = float64(count)
        case "error_rate":
            s.db.Raw(`
                SELECT COALESCE(
                    (1 - AVG(accuracy)) * 100, 0
                ) FROM practice_sessions WHERE DATE(created_at) = CURDATE()
            `).Scan(&currentValue)
        }

        // 判断是否触发预警
        triggered := false
        switch rule.Condition {
        case "gt":
            triggered = currentValue > float64(rule.Threshold)
        case "lt":
            triggered = currentValue < float64(rule.Threshold)
        case "gte":
            triggered = currentValue >= float64(rule.Threshold)
        case "lte":
            triggered = currentValue <= float64(rule.Threshold)
        case "eq":
            triggered = currentValue == float64(rule.Threshold)
        }

        if triggered {
            // 创建预警记录
            record := &model.AlertRecord{
                RuleID:       rule.ID,
                RuleName:     rule.Name,
                Metric:       rule.Metric,
                CurrentValue: currentValue,
                Threshold:    float64(rule.Threshold),
                Message:      fmt.Sprintf("%s: 当前值 %.2f %s %.2f", rule.Name, currentValue, rule.Condition, rule.Threshold),
                Status:       1,
            }
            s.db.Create(record)

            // 更新规则的最近触发时间
            s.db.Model(&rule).Update("last_triggered_at", time.Now())

            // 发送通知
            s.sendAlertNotify(rule, record)
        }
    }

    return nil
}

// SendScheduledReports 发送定时报告(定时任务调用)
func (s *StatisticsService) SendScheduledReports() error {
    now := time.Now()

    // 获取需要发送的日报
    var dailySubs []model.DataSubscription
    s.db.Where("report_type = 'daily' AND status = 1 AND deleted_at IS NULL").
        Where("HOUR(send_time) = ?", now.Hour()).
        Find(&dailySubs)

    for _, sub := range dailySubs {
        s.generateAndSendReport(sub, "daily")
    }

    // 获取需要发送的周报(每周一发送)
    if now.Weekday() == time.Monday {
        var weeklySubs []model.DataSubscription
        s.db.Where("report_type = 'weekly' AND status = 1 AND deleted_at IS NULL").
            Where("send_day = ?", int(now.Weekday())).
            Where("HOUR(send_time) = ?", now.Hour()).
            Find(&weeklySubs)

        for _, sub := range weeklySubs {
            s.generateAndSendReport(sub, "weekly")
        }
    }

    // 获取需要发送的月报(每月1号发送)
    if now.Day() == 1 {
        var monthlySubs []model.DataSubscription
        s.db.Where("report_type = 'monthly' AND status = 1 AND deleted_at IS NULL").
            Where("send_day = 1").
            Where("HOUR(send_time) = ?", now.Hour()).
            Find(&monthlySubs)

        for _, sub := range monthlySubs {
            s.generateAndSendReport(sub, "monthly")
        }
    }

    return nil
}

// GetDataComparison 数据对比
func (s *StatisticsService) GetDataComparison(req *dto.ComparisonRequest) (*dto.ComparisonResult, error) {
    var result dto.ComparisonResult
    result.Metrics = req.Metrics

    for _, metric := range req.Metrics {
        var period1Data, period2Data float64

        switch metric {
        case "new_users":
            s.db.Model(&model.User{}).
                Where("created_at BETWEEN ? AND ?", req.Period1Start, req.Period1End).
                Count((*int64)(&period1Data))
            s.db.Model(&model.User{}).
                Where("created_at BETWEEN ? AND ?", req.Period2Start, req.Period2End).
                Count((*int64)(&period2Data))
        case "active_users":
            var count1, count2 int64
            s.db.Model(&model.UserEvent{}).
                Where("event_type = 'login' AND created_at BETWEEN ? AND ?", req.Period1Start, req.Period1End).
                Distinct("user_id").Count(&count1)
            s.db.Model(&model.UserEvent{}).
                Where("event_type = 'login' AND created_at BETWEEN ? AND ?", req.Period2Start, req.Period2End).
                Distinct("user_id").Count(&count2)
            period1Data = float64(count1)
            period2Data = float64(count2)
        case "practice_count":
            var count1, count2 int64
            s.db.Model(&model.PracticeSession{}).
                Where("created_at BETWEEN ? AND ?", req.Period1Start, req.Period1End).Count(&count1)
            s.db.Model(&model.PracticeSession{}).
                Where("created_at BETWEEN ? AND ?", req.Period2Start, req.Period2End).Count(&count2)
            period1Data = float64(count1)
            period2Data = float64(count2)
        case "exam_count":
            var count1, count2 int64
            s.db.Model(&model.ExamRecord{}).
                Where("created_at BETWEEN ? AND ?", req.Period1Start, req.Period1End).Count(&count1)
            s.db.Model(&model.ExamRecord{}).
                Where("created_at BETWEEN ? AND ?", req.Period2Start, req.Period2End).Count(&count2)
            period1Data = float64(count1)
            period2Data = float64(count2)
        case "avg_score":
            s.db.Model(&model.ExamRecord{}).
                Where("status = 2 AND created_at BETWEEN ? AND ?", req.Period1Start, req.Period1End).
                Select("COALESCE(AVG(score), 0)").Scan(&period1Data)
            s.db.Model(&model.ExamRecord{}).
                Where("status = 2 AND created_at BETWEEN ? AND ?", req.Period2Start, req.Period2End).
                Select("COALESCE(AVG(score), 0)").Scan(&period2Data)
        }

        change := 0.0
        if period1Data > 0 {
            change = (period2Data - period1Data) / period1Data * 100
        }

        result.Details = append(result.Details, dto.ComparisonDetail{
            Metric:     metric,
            Period1:    period1Data,
            Period2:    period2Data,
            Change:     change,
            ChangeType: getChangeType(change),
        })
    }

    return &result, nil
}

// getChangeType 获取变化类型

    if change > 0 {
        return "increase"
    } else if change < 0 {

    return "unchanged"
}

// RefreshStatistics 手动刷新统计数据
func (s *StatisticsService) RefreshStatistics(userID uint, req *dto.RefreshRequest) (*dto.RefreshResult, error) {
    result := &dto.RefreshResult{
        RefreshedAt: time.Now(),
        Details:     []dto.RefreshDetail{},
    }

    for _, dataType := range req.DataTypes {
        detail := dto.RefreshDetail{Type: dataType}

        switch dataType {
        case "dashboard":
            // 清除仪表盘缓存
            s.cache.Delete(context.Background(), "stats:dashboard")
            detail.Status = "success"
            detail.Message = "仪表盘数据已刷新"

        case "ranking":
            // 清除排行榜缓存
            s.cache.Delete(context.Background(), "stats:ranking:*")
            detail.Status = "success"
            detail.Message = "排行榜数据已刷新"

        case "practice_stats":
            // 重新计算用户练习统计
            s.recalculatePracticeStats(userID)
            s.cache.Delete(context.Background(), fmt.Sprintf("stats:practice:%d", userID))
            detail.Status = "success"
            detail.Message = "练习统计数据已刷新"

        case "score_stats":
            // 重新计算成绩统计
            s.recalculateScoreStats(userID)
            s.cache.Delete(context.Background(), fmt.Sprintf("stats:score:%d", userID))
            detail.Status = "success"
            detail.Message = "成绩统计数据已刷新"

        case "class_stats":
            // 刷新班级统计
            s.cache.DeletePattern(context.Background(), "stats:class:*")
            detail.Status = "success"
            detail.Message = "班级统计数据已刷新"

        default:
            detail.Status = "error"
            detail.Message = "不支持的数据类型: " + dataType
        }

        result.Details = append(result.Details, detail)
    }

    return result, nil
}
```

```go
// cache/statistics_cache.go

package cache

import (
    "context"
    "encoding/json"
    "fmt"
    "time"

    "github.com/redis/go-redis/v9"
)

type StatisticsCache struct {
    client *redis.Client
}

func NewStatisticsCache(client *redis.Client) *StatisticsCache {
    return &StatisticsCache{client: client}
}

// Get 获取缓存
func (c *StatisticsCache) Get(ctx context.Context, key string, dest interface{}) error {
    val, err := c.client.Get(ctx, key).Result()
    if err != nil {
        return err
    }
    return json.Unmarshal([]byte(val), dest)
}

// Set 设置缓存
func (c *StatisticsCache) Set(ctx context.Context, key string, value interface{}, ttl time.Duration) error {
    data, err := json.Marshal(value)
    if err != nil {
        return err
    }
    return c.client.Set(ctx, key, data, ttl).Err()
}

// Delete 删除缓存
func (c *StatisticsCache) Delete(ctx context.Context, key string) error {
    return c.client.Del(ctx, key).Err()
}

// DeletePattern 按模式删除缓存
func (c *StatisticsCache) DeletePattern(ctx context.Context, pattern string) error {
    iter := c.client.Scan(ctx, 0, pattern, 100).Iterator()
    for iter.Next(ctx) {
        c.client.Del(ctx, iter.Val())
    }
    return iter.Err()
}

// GetOrSet 获取缓存，不存在则计算并设置
func (c *StatisticsCache) GetOrSet(ctx context.Context, key string, dest interface{}, ttl time.Duration, compute func() (interface{}, error)) error {
    err := c.Get(ctx, key, dest)
    if err == nil {
        return nil
    }

    // 缓存未命中，计算数据
    value, err := compute()
    if err != nil {
        return err
    }

    // 写入缓存
    c.Set(ctx, key, value, ttl)

    // 赋值给目标
    data, _ := json.Marshal(value)
    json.Unmarshal(data, dest)
    return nil
}

// 预定义缓存 TTL
const (
    CacheTTLDashboard   = 5 * time.Minute
    CacheTTLRanking     = 10 * time.Minute
    CacheTTLUserStats   = 10 * time.Minute
    CacheTTLClassStats  = 5 * time.Minute
    CacheTTLQuestion    = 1 * time.Hour
    CacheTTLRetention   = 1 * time.Hour
    CacheTTLChurn       = 1 * time.Hour
    CacheTTLBehavior    = 5 * time.Minute
    CacheTTLShare       = 7 * 24 * time.Hour
    CacheTTLMobile      = 5 * time.Minute
)
```

```go
// GetDashboardWithCache 带缓存获取仪表盘数据
func (s *StatisticsService) GetDashboardWithCache() (*dto.DashboardInfo, error) {
    var dashboard dto.DashboardInfo
    ctx := context.Background()

    err := s.cache.GetOrSet(ctx, "stats:dashboard", &dashboard, CacheTTLDashboard, func() (interface{}, error) {
        return s.GetDashboard()
    })

    if err != nil {
        return nil, err
    }
    return &dashboard, nil
}

// GetRankingWithCache 带缓存获取排行榜
func (s *StatisticsService) GetRankingWithCache(rankType, page, pageSize int) ([]dto.RankInfo, error) {
    var ranks []dto.RankInfo
    ctx := context.Background()
    key := fmt.Sprintf("stats:ranking:%d:%d", rankType, page)

    err := s.cache.GetOrSet(ctx, key, &ranks, CacheTTLRanking, func() (interface{}, error) {
        return s.GetRanking(rankType, page, pageSize)
    })

    if err != nil {
        return nil, err
    }
    return ranks, nil
}

// OnExamSubmitted 考试提交后清除相关缓存
func (s *StatisticsService) OnExamSubmitted(userID uint, examID uint) {
    ctx := context.Background()
    // 清除用户成绩缓存
    s.cache.Delete(ctx, fmt.Sprintf("stats:score:%d", userID))
    s.cache.Delete(ctx, fmt.Sprintf("stats:score:trend:%d", userID))
    s.cache.Delete(ctx, fmt.Sprintf("stats:score:prediction:%d", userID))
    // 清除分数分布缓存
    s.cache.Delete(ctx, fmt.Sprintf("stats:score:distribution:%d", examID))
    // 清除排行榜缓存
    s.cache.DeletePattern(ctx, "stats:ranking:exam:*")
    // 清除仪表盘缓存
    s.cache.Delete(ctx, "stats:dashboard")
}

// OnPracticeCompleted 练习完成后清除相关缓存
func (s *StatisticsService) OnPracticeCompleted(userID uint) {
    ctx := context.Background()
    s.cache.Delete(ctx, fmt.Sprintf("stats:practice:%d", userID))
    s.cache.Delete(ctx, fmt.Sprintf("stats:practice:trend:%d:*", userID))
    s.cache.Delete(ctx, fmt.Sprintf("stats:practice:category:%d", userID))
    s.cache.DeletePattern(ctx, "stats:ranking:practice:*")
    s.cache.Delete(ctx, "stats:dashboard")
}
```

### 6.2 移动端统计 API

```go
// GetMobileOverview 获取移动端个人统计概览
func (s *StatisticsService) GetMobileOverview(userID uint) (*dto.MobileOverview, error) {
    var overview dto.MobileOverview
    ctx := context.Background()
    cacheKey := fmt.Sprintf("stats:mobile:overview:%d", userID)

    err := s.cache.GetOrSet(ctx, cacheKey, &overview, CacheTTLMobile, func() (interface{}, error) {
        var stats dto.MobileOverview

        // 总练习次数
        s.db.Model(&model.PracticeSession{}).Where("user_id = ?", userID).Count(&stats.TotalPractice)

        // 总做题数
        s.db.Model(&model.PracticeRecord{}).Where("user_id = ?", userID).Count(&stats.TotalQuestions)

        // 平均正确率
        s.db.Model(&model.PracticeSession{}).Where("user_id = ?", userID).
            Select("COALESCE(AVG(accuracy), 0)").Scan(&stats.AvgAccuracy)

        // 学习天数
        s.db.Raw(`
            SELECT COUNT(DISTINCT DATE(created_at))
            FROM practice_sessions
            WHERE user_id = ?
        `, userID).Scan(&stats.StudyDays)

        // 正确率变化（本周vs上周）
        var thisWeekAcc, lastWeekAcc float64
        s.db.Model(&model.PracticeSession{}).
            Where("user_id = ? AND created_at >= DATE_SUB(CURDATE(), INTERVAL 7 DAY)", userID).
            Select("COALESCE(AVG(accuracy), 0)").Scan(&thisWeekAcc)
        s.db.Model(&model.PracticeSession{}).
            Where("user_id = ? AND created_at >= DATE_SUB(CURDATE(), INTERVAL 14 DAY) AND created_at < DATE_SUB(CURDATE(), INTERVAL 7 DAY)", userID).
            Select("COALESCE(AVG(accuracy), 0)").Scan(&lastWeekAcc)
        stats.AccuracyChange = (thisWeekAcc - lastWeekAcc) * 100

        // 今日学习统计
        today := time.Now().Format("2006-01-02")
        s.db.Model(&model.PracticeSession{}).
            Where("user_id = ? AND DATE(created_at) = ?", userID, today).
            Count(&stats.TodayStats.PracticeCount)
        s.db.Model(&model.PracticeRecord{}).
            Where("user_id = ? AND DATE(created_at) = ?", userID, today).
            Count(&stats.TodayStats.QuestionCount)
        s.db.Model(&model.PracticeRecord{}).
            Where("user_id = ? AND DATE(created_at) = ? AND is_correct = ?", userID, today, true).
            Count(&stats.TodayStats.CorrectCount)
        s.db.Model(&model.PracticeSession{}).
            Where("user_id = ? AND DATE(created_at) = ?", userID, today).
            Select("COALESCE(SUM(duration), 0)").Scan(&stats.TodayStats.Duration)

        return stats, nil
    })

    if err != nil {
        return nil, err
    }
    return &overview, nil
}
```

## 七、数据分享

### 7.1 数据库表

```sql
CREATE TABLE `data_shares` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `token` VARCHAR(64) NOT NULL COMMENT '分享Token',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '创建人ID',
  `report_type` VARCHAR(50) NOT NULL COMMENT '报告类型:practice/score/class/custom',
  `report_data` JSON NOT NULL COMMENT '报告数据快照',
  `title` VARCHAR(200) NOT NULL COMMENT '报告标题',
  `description` VARCHAR(500) DEFAULT '' COMMENT '报告描述',
  `view_count` INT DEFAULT 0 COMMENT '查看次数',
  `status` TINYINT(1) DEFAULT 1 COMMENT '状态:0=失效,1=有效',
  `expire_at` DATETIME NOT NULL COMMENT '过期时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_token` (`token`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_expire_at` (`expire_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='数据分享表';
```

### 7.2 分享流程

1. 用户选择要分享的统计报告
2. 系统生成唯一 Token，将报告数据快照存入 `data_shares` 表
3. 返回分享链接（含 Token），有效期默认 7 天
4. 他人通过分享链接访问，系统根据 Token 查询报告数据并展示
5. 每次访问更新查看次数
