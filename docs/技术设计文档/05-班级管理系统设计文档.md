# 班级管理系统设计文档

**项目名称**：题小助 (QuestionHelper)
**版本**：V2.0
**日期**：2026-05-30
**模块**：班级管理系统

---

## 一、功能概述

| 功能 | 说明 |
|------|------|
| 班级创建 | 教师/管理员创建班级 |
| 成员管理 | 邀请码加入、成员管理 |
| 班级创作者 | 申请审批、权限管理 |
| 班级资源 | 班级专属题目、分类、知识点 |
| 班级功能 | 公告、作业、考试 |
| 班级归档 | 归档已结束班级（只读，保留历史数据） |
| 班级置顶 | 置顶重要班级，优先显示 |
| 班级标签/分类 | 给班级打标签便于分类管理 |
| 班级二维码 | 生成班级二维码，方便扫码加入 |
| 班级有效期 | 设置有效期，到期自动归档 |
| 成员分组 | 学习小组管理（创建分组、分配组长、组内管理） |
| 成员审核 | 加入班级需要教师审核 |
| 批量导入成员 | Excel 批量导入成员（学号、姓名、手机号） |
| 成员禁用 | 临时禁用成员权限（不影响其他班级） |
| 考勤管理 | 签到/签退（支持地理位置校验） |
| 学习计划 | 制定学习计划并追踪完成情况 |
| 班级文件 | 共享文件管理（上传、下载、删除） |
| 学生互评 | 作业互评（教师分配、学生评分） |
| 班级排名 | 成绩排名（支持按考试、按时间段排名） |
| 班级搜索 | 按关键词搜索班级 |
| 班级模板 | 班级配置模板管理，从模板快速创建班级 |
| 班级讨论 | 班级内发帖讨论，支持嵌套回复 |
| 作业提醒 | 作业截止前自动提醒未提交学生 |
| 作业模板 | 作业配置模板管理，快速创建作业 |
| 成绩预警 | 学生成绩低于阈值时自动预警提醒 |
| 学习报告 | 生成学生学习报告（成绩、进度、参与度） |
| 班级日志 | 记录班级内所有操作日志 |
| 班级资源版本管理 | 班级资源的版本历史记录与回溯 |
| 班级资源标签 | 给班级资源打标签便于分类 |
| 班级资源搜索 | 在班级资源库中搜索题目 |
| 班级资源审核 | 班级创作者创建的资源可配置是否需要教师审核 |
| 成员统计 | 成员活跃度、参与度统计 |
| Redis 缓存 | 班级信息、成员列表、邀请码等缓存设计 |
| 班级创作者权限配置 | 自定义班级创作者的权限范围 |
| 创作者统计 | 班级创作者的创建题目数、活跃度统计 |
| 班级考试发布 | 班级内发布考试 |

## 二、数据库设计

### 2.1 班级表 (classes)

```sql
CREATE TABLE `classes` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(100) NOT NULL COMMENT '班级名称',
  `description` VARCHAR(500) DEFAULT '' COMMENT '班级描述',
  `cover` VARCHAR(255) DEFAULT '' COMMENT '封面图',
  `code` VARCHAR(20) NOT NULL COMMENT '加入码',
  `creator_id` BIGINT UNSIGNED NOT NULL COMMENT '创建者ID',
  `member_count` INT DEFAULT 0 COMMENT '成员数量',
  `max_members` INT DEFAULT 500 COMMENT '最大成员数',
  `status` TINYINT DEFAULT 1 COMMENT '状态',
  `is_archived` TINYINT(1) DEFAULT 0 COMMENT '是否归档:0=正常,1=已归档',
  `archived_at` DATETIME DEFAULT NULL COMMENT '归档时间',
  `is_pinned` TINYINT(1) DEFAULT 0 COMMENT '是否置顶:0=未置顶,1=已置顶',
  `pinned_at` DATETIME DEFAULT NULL COMMENT '置顶时间',
  `expire_at` DATETIME DEFAULT NULL COMMENT '有效期截止时间, NULL表示永不过期',
  `is_auto_archived` TINYINT(1) DEFAULT 0 COMMENT '是否自动归档(有效期到期后)',
  `require_approval` TINYINT(1) DEFAULT 0 COMMENT '加入是否需要审核:0=不需要,1=需要',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `deleted_at` DATETIME DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_code` (`code`),
  KEY `idx_creator_id` (`creator_id`),
  KEY `idx_is_pinned` (`is_pinned`),
  KEY `idx_is_archived` (`is_archived`),
  KEY `idx_expire_at` (`expire_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='班级表';
```

```go
        Role:      3, // 管理员
        IsCreator: true,
        Status:    1,
    }

// JoinClass 加入班级
func (s *ClassService) JoinClass(classID uint, code string, userID uint) error {
    class, err := s.classRepo.FindByID(classID)
    if err != nil {
        return errors.New("班级不存在")
    }

    if class.IsArchived {
        return errors.New("班级已归档，无法加入")
    }

    if class.ExpireAt != nil && class.ExpireAt.Before(time.Now()) {
        return errors.New("班级已过期，无法加入")
    }

    if class.Code != code {
        return errors.New("加入码错误")
    }

    // 检查是否已是成员
    isMember, _ := s.memberRepo.IsMember(classID, userID)
    if isMember {
        return errors.New("已经是班级成员")
    }

    // 检查成员上限
    if class.MemberCount >= class.MaxMembers {
        return errors.New("班级成员已满")
    }

    // 判断是否需要审核
    if class.RequireApproval {
        return s.createJoinApplication(classID, userID, "")
    }

    return s.addMember(classID, userID)
}

// generateCode 生成6位随机加入码
func (s *ClassService) generateCode() string {
    const chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    code := make([]byte, 6)
    for i := range code {
        code[i] = chars[rand.Intn(len(chars))]
    }
    return string(code)
}
```

```go
// service/class/import_service.go

type ImportService struct {
    classRepo  *class.ClassRepo
    memberRepo *class.MemberRepo
    importRepo *class.ImportRecordRepo
}

// ImportMembersRequest 批量导入请求
type ImportMembersRequest struct {
    ClassID uint   `json:"class_id"`
    File    []byte `json:"-"` // Excel 文件内容
}

// ImportResult 导入结果
type ImportResult struct {
    Total      int            `json:"total"`
    Success    int            `json:"success"`
    Failed     int            `json:"failed"`
    Errors     []ImportError  `json:"errors,omitempty"`
}

// ImportError 导入错误
type ImportError struct {
    Row    int    `json:"row"`
    Reason string `json:"reason"`
}

// ImportMembers 批量导入成员
func (s *ImportService) ImportMembers(classID uint, file io.Reader, operatorID uint) (*ImportResult, error) {
    class, err := s.classRepo.FindByID(classID)
    if err != nil {
        return nil, errors.New("班级不存在")
    }

    if class.IsArchived {
        return nil, errors.New("班级已归档，无法导入成员")
    }

    // 使用 excelize 解析 Excel
    f, err := excelize.OpenReader(file)
    if err != nil {
        return nil, errors.New("文件格式错误")
    }

    defer f.Close()

    rows, err := f.GetRows(f.GetSheetName(0))
    if err != nil {
        return nil, errors.New("读取工作表失败")
    }

    result := &ImportResult{}

    // 从第2行开始（第1行为表头）
    for i := 1; i < len(rows); i++ {
        row := rows[i]
        result.Total++

        if len(row) < 2 {
            result.Failed++
            result.Errors = append(result.Errors, ImportError{
                Row:    i + 1,
                Reason: "数据不完整，至少需要学号和姓名",
            })
            continue
        }

        studentNo := strings.TrimSpace(row[0])
        name := strings.TrimSpace(row[1])
        phone := ""
        if len(row) > 2 {
            phone = strings.TrimSpace(row[2])
        }

        // 查找用户（按手机号或用户名）
        var user model.User
        var found bool
        if phone != "" {
            found = s.memberRepo.DB().Where("phone = ?", phone).First(&user).Error == nil
        }
        if !found && name != "" {
            found = s.memberRepo.DB().Where("nickname = ?", name).First(&user).Error == nil
        }

        if !found {
            result.Failed++
            result.Errors = append(result.Errors, ImportError{
                Row:    i + 1,
                Reason: fmt.Sprintf("未找到用户（学号:%s, 姓名:%s）", studentNo, name),
            })
            continue
        }

        // 检查是否已是成员
        isMember, _ := s.memberRepo.IsMember(classID, user.ID)
        if isMember {
            result.Failed++
            result.Errors = append(result.Errors, ImportError{
                Row:    i + 1,
                Reason: fmt.Sprintf("用户 %s 已是班级成员", name),
            })
            continue
        }

        // 检查成员上限
        if class.MemberCount >= class.MaxMembers {
            result.Failed++
            result.Errors = append(result.Errors, ImportError{
                Row:    i + 1,
                Reason: "班级成员已满",
            })
            break
        }

        // 添加成员
        member := &model.ClassMember{
            ClassID:   classID,
            UserID:    user.ID,
            Role:      1,
            StudentNo: studentNo,
            Status:    1,
        }

        if err := s.memberRepo.Create(member); err != nil {
            result.Failed++
            result.Errors = append(result.Errors, ImportError{
                Row:    i + 1,
                Reason: fmt.Sprintf("添加失败: %s", err.Error()),
            })
            continue
        }

        s.classRepo.IncrementMemberCount(classID)
        result.Success++
    }

    // 记录导入日志
    importRecord := &model.ClassImportRecord{
        ClassID:    classID,
        FileName:   "import.xlsx",
        Total:      result.Total,
        Success:    result.Success,
        Failed:     result.Failed,
        OperatorID: operatorID,
    }
    if len(result.Errors) > 0 {
        errorsJSON, _ := json.Marshal(result.Errors)
        importRecord.Errors = string(errorsJSON)
    }
    s.importRepo.Create(importRecord)

    return result, nil
}
```

```go
// CheckIn 学生签到
func (s *AttendanceService) CheckIn(attID uint, userID uint, req *dto.CheckInRequest) error {
    att, err := s.attRepo.FindByID(attID)
    if err != nil {
        return errors.New("考勤不存在")
    }

    now := time.Now()
    if now.Before(att.StartTime) {
        return errors.New("签到未开始")
    }
    if now.After(att.EndTime) {
        return errors.New("签到已结束")
    }

    // 地理位置校验（如果设置了考勤地点）
    if att.Latitude != nil && att.Longitude != nil && att.Radius > 0 {
        distance := calculateDistance(
            *att.Latitude, *att.Longitude,
            req.Latitude, req.Longitude,
        )
        if distance > float64(att.Radius) {
            return errors.New("不在考勤范围内")
        }
    }

    record, err := s.recordRepo.FindByAttendanceAndUser(attID, userID)
    if err != nil {
        return errors.New("考勤记录不存在")
    }

    // 判断是否迟到
    status := int8(1) // 正常
    if now.After(att.StartTime.Add(10 * time.Minute)) {
        status = 2 // 迟到
    }

    updates := map[string]interface{}{
        "check_in_at": now,
        "status":      status,
        "location":    req.Location,
        "latitude":    req.Latitude,
        "longitude":   req.Longitude,
    }

    return s.recordRepo.Update(record.ID, updates)
}

// CheckOut 学生签退
func (s *AttendanceService) CheckOut(attID uint, userID uint) error {
    att, err := s.attRepo.FindByID(attID)
    if err != nil {
        return errors.New("考勤不存在")
    }

    now := time.Now()

    record, err := s.recordRepo.FindByAttendanceAndUser(attID, userID)
    if err != nil {
        return errors.New("考勤记录不存在")
    }

    if record.CheckInAt == nil {
        return errors.New("尚未签到，无法签退")
    }

    // 判断是否早退
    if now.Before(att.EndTime) {
        record.Status = 3 // 早退
    }

    record.CheckOutAt = &now
    return s.recordRepo.Save(record)
}

// calculateDistance 使用 Haversine 公式计算两点间距离（米）
func calculateDistance(lat1, lon1, lat2, lon2 float64) float64 {
    const R = 6371000 // 地球半径（米）
    dLat := (lat2 - lat1) * math.Pi / 180
    dLon := (lon2 - lon1) * math.Pi / 180
    a := math.Sin(dLat/2)*math.Sin(dLat/2) +
        math.Cos(lat1*math.Pi/180)*math.Cos(lat2*math.Pi/180)*
            math.Sin(dLon/2)*math.Sin(dLon/2)
    c := 2 * math.Atan2(math.Sqrt(a), math.Sqrt(1-a))
    return R * c
}
```

```go
    if err != nil || plan.ClassID != classID {

    progress, _ := s.progressRepo.FindByPlanAndUser(planID, userID)

            if p.Status == 1 {
                completedCount++
            }
        }
        itemDetails = append(itemDetails, detail)
    }

    return &dto.StudyPlanDetail{
        Plan:           plan,
        Items:          itemDetails,
        TotalItems:     len(items),
        CompletedItems: completedCount,
    }, nil
```

```go
// AssignPeerReviews 分配互评（教师操作）
func (s *PeerReviewService) AssignPeerReviews(classID uint, homeworkID uint, req *dto.AssignPeerReviewRequest) error {
    // 获取所有已提交作业的学生
    submissions, err := s.submissionRepo.FindByHomeworkID(homeworkID)
    if err != nil || len(submissions) < 2 {
        return errors.New("已提交作业的学生不足，无法分配互评")
    }

    // 清除旧的分配
    s.reviewRepo.DeleteByHomeworkID(homeworkID)

    reviewCount := req.ReviewCount // 每人评价几份
    if reviewCount <= 0 {
        reviewCount = 2
    }

    var reviews []model.ClassPeerReview

    if req.AutoAssign {
        // 自动分配：随机分配，保证每人被评 N 次，每人评 N 份
        shuffled := shuffleSubmissions(submissions)

        for i, sub := range shuffled {
            for j := 1; j <= reviewCount; j++ {
                targetIdx := (i + j) % len(shuffled)
                reviews = append(reviews, model.ClassPeerReview{
                    HomeworkID: homeworkID,
                    ClassID:    classID,
                    ReviewerID: sub.UserID,
                    RevieweeID: shuffled[targetIdx].UserID,
                    Status:     0,
                })
            }
        }
    } else {
        // 手动分配
        for _, assign := range req.Assignments {
            reviews = append(reviews, model.ClassPeerReview{
                HomeworkID: homeworkID,
                ClassID:    classID,
                ReviewerID: assign.ReviewerID,
                RevieweeID: assign.RevieweeID,
                Status:     0,
            })
        }
    }

    return s.reviewRepo.BatchCreate(reviews)
}

// SubmitPeerReview 提交互评评分
func (s *PeerReviewService) SubmitPeerReview(reviewID uint, userID uint, req *dto.SubmitPeerReviewRequest) error {
    review, err := s.reviewRepo.FindByID(reviewID)

        return errors.New("互评记录不存在")

    if review.ReviewerID != userID {
        return errors.New("无权操作此互评")
    }

    if review.Status == 1 {
        return errors.New("已提交评价，不可重复提交")
    }

    review.Score = req.Score
    review.Content = req.Content
    review.Status = 1 // 已评价
    now := time.Now()
    review.ReviewedAt = &now

    return s.reviewRepo.Save(review)

// shuffleSubmissions 随机打乱提交列表
func shuffleSubmissions(submissions []model.HomeworkSubmission) []model.HomeworkSubmission {
    shuffled := make([]model.HomeworkSubmission, len(submissions))
    copy(shuffled, submissions)
    rand.Shuffle(len(shuffled), func(i, j int) {
        shuffled[i], shuffled[j] = shuffled[j], shuffled[i]
    })
    return shuffled
}
```

```go
// CalculateRanking 计算班级排名
func (s *RankingService) CalculateRanking(classID uint, rankType int8, examID *uint) error {
    // 获取班级所有学生
    members, _ := s.memberRepo.FindByClassID(classID, 1)

    var rankings []model.ClassRanking
    now := time.Now()

    if rankType == 1 && examID != nil {
        // 单次考试排名
        var scores []struct {
            UserID    uint
            TotalScore float64
        }
        s.examRecordRepo.DB().
            Table("exam_records er").
            Select("er.user_id, er.score as total_score").
            Joins("JOIN exam_publish ep ON er.exam_publish_id = ep.id").
            Where("ep.exam_id = ? AND er.status = ?", *examID, 2). // 已完成
            Where("er.user_id IN (?)", s.getMemberIDs(members)).
            Order("er.score DESC").
            Scan(&scores)

        for i, sc := range scores {
            rankings = append(rankings, model.ClassRanking{
                ClassID:     classID,
                UserID:      sc.UserID,
                ExamID:      examID,
                TotalScore:  sc.TotalScore,
                RankNum:     i + 1,
                RankType:    1,
                CalculatedAt: now,
            })
        }
    } else {
        // 综合排名（所有考试平均分）
        var results []struct {
            UserID   uint
            AvgScore float64
            ExamCount int
        }
        s.examRecordRepo.DB().
            Table("exam_records er").
            Select("er.user_id, AVG(er.score) as avg_score, COUNT(*) as exam_count").
            Joins("JOIN exam_publish ep ON er.exam_publish_id = ep.id").
            Where("ep.class_id = ? AND er.status = ?", classID, 2).
            Where("er.user_id IN (?)", s.getMemberIDs(members)).
            Group("er.user_id").
            Order("avg_score DESC").
            Scan(&results)

        for i, r := range results {
            rankings = append(rankings, model.ClassRanking{
                ClassID:     classID,
                UserID:      r.UserID,
                AvgScore:    r.AvgScore,
                ExamCount:   r.ExamCount,
                RankNum:     i + 1,
                RankType:    2,
                CalculatedAt: now,
            })
        }
    }

    // 清除旧排名并写入新排名
    s.rankingRepo.DeleteByClassAndType(classID, rankType, examID)
    return s.rankingRepo.BatchCreate(rankings)
}
```

```go
        Find(&classes)

    return &dto.PageResult{
        Items:    classes,
        Total:    total,
        Page:     req.Page,
        PageSize: req.PageSize,
    }, nil
}
```

```go
        "last_reply_at": &now,

// GetDiscussionDetail 获取讨论详情（含嵌套回复）
func (s *DiscussionService) GetDiscussionDetail(classID uint, discussionID uint) (*dto.DiscussionDetail, error) {
    discussion, err := s.discussionRepo.FindByID(discussionID)
    if err != nil || discussion.ClassID != classID {
        return nil, errors.New("讨论帖不存在")
    }

    // 获取一级回复
    replies, _ := s.replyRepo.FindTopLevelByDiscussion(discussionID)

    // 递归构建回复树
    var buildReplyTree func(parentID uint) []dto.ReplyNode
    buildReplyTree = func(parentID uint) []dto.ReplyNode {
        children, _ := s.replyRepo.FindByParentID(parentID)
        var nodes []dto.ReplyNode
        for _, child := range children {
            nodes = append(nodes, dto.ReplyNode{
                Reply:    child,
                Children: buildReplyTree(child.ID),
            })
        }
        return nodes
    }

    var replyNodes []dto.ReplyNode
    for _, r := range replies {
        replyNodes = append(replyNodes, dto.ReplyNode{
            Reply:    r,
            Children: buildReplyTree(r.ID),
        })
    }

    return &dto.DiscussionDetail{
        Discussion: discussion,
        Replies:    replyNodes,
    }, nil
}
```

```go
func (s *HomeworkReminderService) CheckAndRemind() {
    now := time.Now()

    // 查找截止前24小时内的作业
    deadline24h := now.Add(24 * time.Hour)
    var homeworks24h []model.Homework
    s.homeworkRepo.DB().
        Where("deadline > ? AND deadline <= ? AND deleted_at IS NULL", now, deadline24h).
        Find(&homeworks24h)

    for _, hw := range homeworks24h {
        s.remindUnsubmitted(hw, 24)
    }

    // 查找截止前2小时内的作业
    deadline2h := now.Add(2 * time.Hour)
    var homeworks2h []model.Homework
    s.homeworkRepo.DB().
        Where("deadline > ? AND deadline <= ? AND deleted_at IS NULL", now, deadline2h).
        Find(&homeworks2h)

    for _, hw := range homeworks2h {
        s.remindUnsubmitted(hw, 2)
    }
}

// remindUnsubmitted 提醒未提交的学生
func (s *HomeworkReminderService) remindUnsubmitted(hw model.Homework, hoursBefore int) {
    // 获取班级所有学生
    members, _ := s.memberRepo.FindByClassID(hw.ClassID, 1) // role=1 学生

    // 获取已提交的学生
    submissions, _ := s.submissionRepo.FindByHomeworkID(hw.ID)
    submittedMap := make(map[uint]bool)
    for _, sub := range submissions {
        submittedMap[sub.UserID] = true
    }

    // 向未提交的学生发送提醒
    for _, m := range members {
        if !submittedMap[m.UserID] {
            s.notifier.Send(m.UserID, notification.Notification{
                Type: "homework_remind",
                Title: fmt.Sprintf("作业提醒"),
                Content: fmt.Sprintf("您有一份作业「%s」将在%d小时后截止，请尽快提交", hw.Title, hoursBefore),
                Data: map[string]interface{}{
                    "class_id":    hw.ClassID,
                    "homework_id": hw.ID,
                },
            })
        }
    }
}
```

```go
// CheckAndCreateAlerts 考试成绩发布后自动检查并创建预警
func (s *ScoreAlertService) CheckAndCreateAlerts(classID uint, examID uint) {
    // 获取预警规则
    rules, _ := s.ruleRepo.FindByClassID(classID)
    if len(rules) == 0 {
        return // 未配置预警规则
    }

    // 获取学生成绩
    var scores []struct {
        UserID uint
        Score  float64
    }
    s.alertRepo.DB().
        Table("exam_records er").
        Select("er.user_id, er.score").
        Joins("JOIN exam_publish ep ON er.exam_publish_id = ep.id").
        Where("ep.exam_id = ? AND ep.class_id = ? AND er.status = 2", examID, classID).
        Scan(&scores)

    for _, sc := range scores {
        for _, rule := range rules {
            if sc.Score < float64(rule.Threshold) {
                alert := &model.ScoreAlert{
                    ClassID:   classID,
                    StudentID: sc.UserID,
                    ExamID:    examID,
                    Score:     sc.Score,
                    Threshold: rule.Threshold,
                    Status:    0,
                }
                s.alertRepo.Create(alert)

                // 自动通知学生和教师
                s.notifier.Send(sc.UserID, notification.Notification{
                    Type:    "score_alert",
                    Title:   "成绩预警",
                    Content: fmt.Sprintf("您在考试中的成绩 %.2f 低于预警阈值 %.2f，请注意", sc.Score, rule.Threshold),
                })
                s.alertRepo.DB().Model(alert).Update("status", 1) // 已通知
            }
        }
    }
}

// SetAlertRules 配置预警规则
func (s *ScoreAlertService) SetAlertRules(classID uint, req *dto.SetScoreAlertRulesRequest) error {
    // 清除旧规则
    s.ruleRepo.DeleteByClassID(classID)

    for _, rule := range req.Rules {
        s.ruleRepo.Create(&model.ScoreAlertRule{
            ClassID:   classID,
            SubjectID: rule.SubjectID,
            Threshold: rule.Threshold,
            NotifyTo:  rule.NotifyTo, // student, teacher, both
        })
    }
    return nil
}

// ConfirmAlert 确认预警

    }

    alert.Status = 2 // 已确认
    alert.ConfirmedBy = &userID
    alert.ConfirmedAt = &now
```

```go
// GenerateStudentReport 生成学生学习报告
func (s *ReportService) GenerateStudentReport(classID uint, userID uint) (*StudentReport, error) {
    isMember, _ := s.memberRepo.IsMember(classID, userID)
    if !isMember {
        return nil, errors.New("不是班级成员")
    }

    report := &StudentReport{
        UserID:      userID,
        ClassID:     classID,
        GeneratedAt: time.Now(),
    }

    // 获取成绩统计
    s.examRecordRepo.DB().
        Table("exam_records er").
        Select("COUNT(*) as exam_count, COALESCE(SUM(er.score),0) as total_score, COALESCE(AVG(er.score),0) as avg_score, COALESCE(MAX(er.score),0) as max_score, COALESCE(MIN(er.score),0) as min_score").
        Joins("JOIN exam_publish ep ON er.exam_publish_id = ep.id").
        Where("ep.class_id = ? AND er.user_id = ? AND er.status = 2", classID, userID).
        Scan(&report.ScoreStats)

    // 获取成绩趋势（最近10次考试）
    s.examRecordRepo.DB().
        Table("exam_records er").
        Select("er.score, er.created_at").
        Joins("JOIN exam_publish ep ON er.exam_publish_id = ep.id").
        Where("ep.class_id = ? AND er.user_id = ? AND er.status = 2", classID, userID).
        Order("er.created_at DESC").Limit(10).
        Scan(&report.ScoreStats.ScoreTrend)

    // 获取学习进度
    s.studyPlanRepo.DB().
        Table("class_study_plan_progress sp").
        Select("COUNT(*) as total_items, SUM(CASE WHEN sp.status = 1 THEN 1 ELSE 0 END) as completed_items").
        Joins("JOIN class_study_plan_items si ON sp.item_id = si.id").
        Joins("JOIN class_study_plans st ON si.plan_id = st.id").
        Where("st.class_id = ? AND sp.user_id = ?", classID, userID).
        Scan(&report.ProgressStats)
    if report.ProgressStats.TotalItems > 0 {
        report.ProgressStats.CompletionRate = float64(report.ProgressStats.CompletedItems) / float64(report.ProgressStats.TotalItems) * 100
    }

    // 获取作业提交率
    var totalHomework int64
    var submittedHomework int64
    s.submissionRepo.DB().Model(&model.Homework{}).Where("class_id = ? AND deleted_at IS NULL", classID).Count(&totalHomework)
    s.submissionRepo.DB().Model(&model.HomeworkSubmission{}).
        Joins("JOIN homeworks h ON homework_submissions.homework_id = h.id").
        Where("h.class_id = ? AND homework_submissions.user_id = ?", classID, userID).
        Count(&submittedHomework)
    if totalHomework > 0 {
        report.ParticipationStats.HomeworkSubmitRate = float64(submittedHomework) / float64(totalHomework) * 100
    }

    // 获取讨论参与度
    s.discussionRepo.DB().Model(&model.ClassDiscussion{}).
        Where("class_id = ? AND user_id = ? AND deleted_at IS NULL", classID, userID).
        Count(&report.ParticipationStats.DiscussionCount)

    return report, nil
}
```

```go
// service/class/exam_publish_service.go

// PublishExam 班级考试发布
func (s *ClassService) PublishExam(classID uint, req *dto.PublishExamRequest, userID uint) (*dto.ExamPublishInfo, error) {
    class, err := s.classRepo.FindByID(classID)
    if err != nil {
        return nil, errors.New("班级不存在")
    }

    if class.IsArchived {
        return nil, errors.New("班级已归档，无法发布考试")
    }

    isMember, _ := s.memberRepo.IsMember(classID, userID)
    if !isMember {
        return nil, errors.New("不是班级成员")
    }

    // 检查创作者权限
    permService := &CreatorPermissionService{permissionRepo: s.permissionRepo}
    if !permService.CheckPermission(classID, "create_exam") {
        return nil, errors.New("没有创建考试的权限")
    }

    exam := &model.Exam{
        ClassID:     classID,
        Title:       req.Title,
        Description: req.Description,
        StartTime:   req.StartTime,
        EndTime:     req.EndTime,
        Duration:    req.Duration,
        TotalScore:  req.TotalScore,
        CreatorID:   userID,
        Status:      1, // 已发布
    }

    if err := s.examRepo.Create(exam); err != nil {
        return nil, err
    }

    // 记录日志
    s.logService.RecordLog(classID, userID, "publish_exam", fmt.Sprintf("发布了考试「%s」", req.Title), "")

    // 发送通知给班级成员
    members, _ := s.memberRepo.FindByClassID(classID, 1)
    for _, m := range members {
        s.notifier.Send(m.UserID, notification.Notification{
            Type:    "exam_published",
            Title:   "新考试发布",
            Content: fmt.Sprintf("班级「%s」发布了新考试「%s」", class.Name, req.Title),
            Data: map[string]interface{}{
                "class_id": classID,
                "exam_id":  exam.ID,
            },
        })
    }

    return &dto.ExamPublishInfo{
        ExamID:    exam.ID,
        Title:     exam.Title,
        StartTime: exam.StartTime,
        EndTime:   exam.EndTime,
    }, nil
}
```

### 8.2 缓存更新策略

```go
// service/class/cache_service.go

type ClassCacheService struct {
    rdb *redis.Client
    ttl time.Duration
}

const (
    ClassInfoTTL        = 30 * time.Minute
    ClassMemberListTTL  = 10 * time.Minute
    ClassCodeTTL        = 24 * time.Hour
    ClassRankingTTL     = 10 * time.Minute
    ClassPermissionTTL  = 1 * time.Hour
)

// GetClassInfo 获取班级信息（优先缓存）
func (s *ClassCacheService) GetClassInfo(classID uint) (*model.Class, error) {
    key := fmt.Sprintf("class:info:%d", classID)

    // 尝试从缓存获取
    data, err := s.rdb.Get(ctx, key).Bytes()
    if err == nil {
        var class model.Class
        json.Unmarshal(data, &class)
        return &class, nil
    }

    // 缓存未命中，从数据库加载
    class, err := s.classRepo.FindByID(classID)
    if err != nil {
        return nil, err
    }

    // 写入缓存
    data, _ = json.Marshal(class)
    s.rdb.Set(ctx, key, data, ClassInfoTTL)

    return class, nil
}

// InvalidateClassInfo 清除班级信息缓存（班级信息变更时调用）
func (s *ClassCacheService) InvalidateClassInfo(classID uint) {
    s.rdb.Del(ctx, fmt.Sprintf("class:info:%d", classID))
}

// GetClassCodeMapping 通过邀请码获取班级ID
func (s *ClassCacheService) GetClassCodeMapping(code string) (uint, error) {
    key := fmt.Sprintf("class:code:%s", code)

    data, err := s.rdb.Get(ctx, key).Result()
    if err == nil {
        classID, _ := strconv.ParseUint(data, 10, 64)
        return uint(classID), nil
    }

    // 从数据库查找
    class, err := s.classRepo.FindByCode(code)
    if err != nil {
        return 0, err
    }

    s.rdb.Set(ctx, key, fmt.Sprintf("%d", class.ID), ClassCodeTTL)
    return class.ID, nil
}

// CacheRanking 缓存排名数据
func (s *ClassCacheService) CacheRanking(classID uint, rankType int, examID *uint, data interface{}) {
    key := fmt.Sprintf("class:ranking:%d:%d:%v", classID, rankType, examID)
    jsonData, _ := json.Marshal(data)
    s.rdb.Set(ctx, key, jsonData, ClassRankingTTL)
}

// GetRanking 获取缓存的排名数据
func (s *ClassCacheService) GetRanking(classID uint, rankType int, examID *uint) (string, error) {
    key := fmt.Sprintf("class:ranking:%d:%d:%v", classID, rankType, examID)
    return s.rdb.Get(ctx, key).Result()
}

// InvalidateRanking 清除排名缓存
func (s *ClassCacheService) InvalidateRanking(classID uint) {
    pattern := fmt.Sprintf("class:ranking:%d:*", classID)
    keys, _ := s.rdb.Keys(ctx, pattern).Result()
    if len(keys) > 0 {
        s.rdb.Del(ctx, keys...)
    }
}
```

### 8.3 缓存一致性保障

- **Write-Through**：写入数据库后立即更新/删除缓存
- **Cache-Aside**：读取时先查缓存，未命中则查数据库并回填缓存
- **主动失效**：班级信息变更、成员变动时主动清除相关缓存
- **定时刷新**：排名等计算密集型数据采用定时刷新策略（cron 触发）
